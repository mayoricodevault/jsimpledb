008package org.jsimpledb.index;
009
010import java.util.NavigableMap;
011import java.util.NavigableSet;
012
013import org.jsimpledb.tuple.Tuple2;
014import org.jsimpledb.tuple.Tuple3;
015
016/**
017 * An index on two fields.
018 *
019 * <p>
020 * Indexes are read-only and "live", always reflecting the current transaction state.
021 * </p>
022 *
023 * @param <V1> first indexed value type
024 * @param <V2> second indexed value type
025 * @param <T> index target type
026 * @see org.jsimpledb.index
027 */
028public interface Index2<V1, V2, T> {
029
030    /**
031     * View this index as a {@link NavigableSet} of tuples.
032     *
033     * @return {@link NavigableSet} of tuples containing indexed values and target value
034     */
035    NavigableSet<Tuple3<V1, V2, T>> asSet();
036
037    /**
038     * View this index as a {@link NavigableMap} of target values keyed by indexed value tuples.
039     *
040     * @return {@link NavigableMap} from indexed value tuple to the corresponding set of target objects
041     */
042    NavigableMap<Tuple2<V1, V2>, NavigableSet<T>> asMap();
043
044    /**
045     * View this index as a {@link NavigableMap} of {@link Index}s keyed by the first value.
046     *
047     * @return {@link NavigableMap} from first value to {@link Index}
048     */
049    NavigableMap<V1, Index<V2, T>> asMapOfIndex();
050
051    /**
052     * Get the prefix of this instance that only includes the first two indexed fields.
053     *
054     * @return prefix of this index
055     */
056    Index<V1, V2> asIndex();
057}
