023/**
024 * Reference paths.
025 *
026 * <p>
027 * A Reference path consists of:
028 * <ul>
029 *  <li>A {@linkplain #getStartType starting Java object type},</li>
030 *  <li>A final {@linkplain #getTargetType target object type} and {@linkplain #getTargetField target field}
031 *      within that type, and</li>
032 *  <li>An intermediate chain of zero or more {@linkplain #getReferenceFields reference fields} through which an
033 *      instance of the target type can be reached from an instance of the starting type.</li>
034 * </ul>
035 *
036 * <p>
037 * Reference paths can be described by the combination of (a) the starting Java object type, and (b) the path of reference
038 * fields and final target field in a {@link String} form consisting of field names separated by period ({@code "."}) characters.
039 * All of the fields except the last must be reference fields.
040 * </p>
041 *
042 * <p>
043 * For example, path {@code "parent.age"} starting from object type {@code Person} might refer to the age of
044 * the parent of a {@code Person}.
045 * </p>
046 *
047 * <p>
048 * When a complex field appears in a reference path, both the name of the complex field and the specific sub-field
049 * being traversed should appear, e.g., {@code "mymap.key.somefield"}. For set and list fields, the (only) sub-field is
050 * named {@code "element"}, while for map fields the sub-fields are named {@code "key"} and {@code "value"}.
051 * </p>
052 *
053 * <p>
054 * <b>Fields of Sub-Types</b>
055 * </p>
056 *
057 * <p>
058 * In some cases, a field may not exist in a Java object type, but it does exist in a some sub-type of that type. For example:
059 * </p>
060 *
061 * <pre>
062 * &#64;JSimpleClass(storageId = 10)
063 * public class Person {
064 *
065 *     &#64;JSimpleSetField(storageId = 11)
066 *     public abstract Set&lt;Person&gt; <b>getFriends</b>();
067 *
068 *     &#64;OnChange("friends.element.<b>name</b>")
069 *     private void friendNameChanged(SimpleFieldChange&lt;NamedPerson, String&gt; change) {
070 *         // ... do whatever
071 *     }
072 * }
073 *
074 * &#64;JSimpleClass(storageId = 20)
075 * public class NamedPerson extends Person {
076 *
077 *     &#64;JSimpleField(storageId = 21)
078 *     public abstract String <b>getName</b>();
079 *     public abstract void setName(String name);
080 * }
081 * </pre>
082 * Here the path {@code "friends.element.name"} is technically incorrect because {@code "friends.element"}
083 * has type {@code Person}, and {@code "name"} is a field of {@code NamedPerson}, not {@code Person}. However, this will
084 * still work as long as there is no ambiguity, i.e., in this example, there are no other sub-types of {@code Person}
085 * with a field named {@code "name"}. Note also in the example above the
086 * {@link org.jsimpledb.change.SimpleFieldChange} parameter to the method
087 * {@code friendNameChanged()} necessarily has generic type {@code NamedPerson}, not {@code Person}.
088 *
089 * <p>
090 * In cases where multiple sub-types of a common super-type type have fields with the same name but different storage IDs,
091 * the storage ID may be explicitly specified as a suffix, for example, {@code "name#123"}.
092 * </p>
093 *
094 * <p>
095 * Reference paths are created via {@link JSimpleDB#parseReferencePath JSimpleDB.parseReferencePath()}.
096 * </p>
097 *
098 * @see JSimpleDB#parseReferencePath JSimpleDB.parseReferencePath()
099 */
100public class ReferencePath {
101
102    final Class<?> startType;
103    final Class<?> targetType;
104    final JFieldInfo targetFieldInfo;
105    final JComplexFieldInfo targetSuperFieldInfo;
106    final ArrayList<JReferenceFieldInfo> referenceFieldInfos = new ArrayList<>();
107    final String path;
108
109    private final Logger log = LoggerFactory.getLogger(this.getClass());
110
111    /**
112     * Constructor.
113     *
114     * @param jdb {@link JSimpleDB} against which to resolve object and field names
115     * @param startType starting Java type for the path
116     * @param path dot-separated path of zero or more reference fields, followed by a target field
117     * @param lastIsSubField true if the last field can be a complex sub-field but not a complex field, false for the reverse,
118     *  or null for don't care
119     * @throws IllegalArgumentException if {@code jdb}, {@code startType}, or {@code path} is null
120     * @throws IllegalArgumentException if {@code path} is invalid
121     */
122    ReferencePath(JSimpleDB jdb, Class<?> startType, String path, Boolean lastIsSubField) {
123
124        // Sanity check
125        if (jdb == null)
126            throw new IllegalArgumentException("null jdb");
127        if (startType == null)
128            throw new IllegalArgumentException("null startType");
129        if (path == null)
130            throw new IllegalArgumentException("null path");
131        final String errorPrefix = "invalid path `" + path + "': ";
132        if (path.length() == 0)
133            throw new IllegalArgumentException(errorPrefix + "path is empty");
134        this.startType = startType;
135        this.path = path;
136
137        // Split the path into field names
138        final ArrayDeque<String> fieldNames = new ArrayDeque<>();
139        while (true) {
140            final int dot = path.indexOf('.');
141            if (dot == -1) {
142                if (path.length() == 0)
143                    throw new IllegalArgumentException(errorPrefix + "ends in `.'");
144                fieldNames.add(path);
145                break;
146            }
147            if (dot == 0)
148                throw new IllegalArgumentException(errorPrefix + "contains an empty path component");
149            fieldNames.add(path.substring(0, dot));
150            path = path.substring(dot + 1);
151        }
152
153        // Initialize loop state
154        Class<?> currentType = this.startType;
155        JFieldInfo fieldInfo = null;
156        JComplexFieldInfo superFieldInfo = null;
157        if (this.log.isTraceEnabled()) {
158            this.log.trace("RefPath: START: startType=" + this.startType + " path=" + fieldNames
159              + " lastIsSubField=" + lastIsSubField);
160        }
161
162        // Parse field names
163        final ArrayList<Integer> storageIds = new ArrayList<>();
164        assert !fieldNames.isEmpty();
165        while (!fieldNames.isEmpty()) {
166            final String fieldName = fieldNames.removeFirst();
167            String description = "field `" + fieldName + "' in type " + currentType;
168
169            // Get explicit storage ID, if any
170            final int hash = fieldName.indexOf('#');
171            int explicitStorageId = 0;
172            final String searchName;
173            if (hash != -1) {
174                try {
175                    explicitStorageId = Integer.parseInt(fieldName.substring(hash + 1));
176                } catch (NumberFormatException e) {
177                    throw new IllegalArgumentException(errorPrefix + "invalid field name `" + fieldName + "'");
178                }
179                searchName = fieldName.substring(0, hash);
180            } else
181                searchName = fieldName;
182
183            // Logging
184            if (this.log.isTraceEnabled())
185                this.log.trace("RefPath: [" + searchName + "] currentType=" + currentType + " storageId=" + explicitStorageId);
186
187            // Find all JFields matching 'fieldName' in some JClass whose type matches 'typeToken'
188            final HashMap<JClass<?>, JField> matchingFields = new HashMap<>();
189            for (JClass<?> jclass : jdb.jclasses.values()) {
190                if (!currentType.isAssignableFrom(jclass.type))
191                    continue;
192                final JField jfield = jclass.jfieldsByName.get(searchName);
193                if (jfield == null)
194                    continue;
195                if (explicitStorageId != 0 && jfield.storageId != explicitStorageId)
196                    continue;
197                matchingFields.put(jclass, jfield);
198            }
199
200            // Logging
201            if (this.log.isTraceEnabled())
202                this.log.trace("RefPath: matching fields: " + matchingFields);
203
204            // Check matching fields and verify they all have the same storage ID
205            if (matchingFields.isEmpty()) {
206                throw new IllegalArgumentException(errorPrefix + "there is no field named `" + searchName + "'"
207                  + (explicitStorageId != 0 ? " with storage ID " + explicitStorageId : "")
208                  + " in (any sub-type of) " + currentType);
209            }
210            final int fieldStorageId = matchingFields.values().iterator().next().storageId;
211            for (JField jfield : matchingFields.values()) {
212                if (jfield.storageId != fieldStorageId) {
213                    throw new IllegalArgumentException(errorPrefix + "there are multiple, non-equal fields named `"
214                      + fieldName + "' in sub-types of type " + currentType);
215                }
216            }
217            fieldInfo = jdb.getJFieldInfo(fieldStorageId, JFieldInfo.class);
218
219            // Get common supertype of all types containing the field
220            currentType = Util.findLowestCommonAncestorOfClasses(
221              Iterables.transform(matchingFields.keySet(), new JClassTypeFunction())).getRawType();
222
223            // Logging
224            if (this.log.isTraceEnabled())
225                this.log.trace("RefPath: updated currentType=" + currentType + " fieldType=" + fieldInfo.getTypeToken(currentType));
226
227            // Handle complex fields
228            superFieldInfo = null;
229            if (fieldInfo instanceof JComplexFieldInfo) {
230                final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)fieldInfo;
231
232                // Last field?
233                if (fieldNames.isEmpty()) {
234                    if (lastIsSubField != null && lastIsSubField) {
235                        throw new IllegalArgumentException(errorPrefix + "path may not end on complex " + description
236                          + "; a sub-field of must be specified");
237                    }
238                    break;
239                }
240                superFieldInfo = complexFieldInfo;
241
242                // Get sub-field
243                final String subFieldName = fieldNames.removeFirst();
244                description = "sub-field `" + subFieldName + "' of complex " + description;
245                try {
246                    fieldInfo = complexFieldInfo.getSubFieldInfo(subFieldName);
247                } catch (IllegalArgumentException e) {
248                    throw new IllegalArgumentException(errorPrefix + "invalid " + description + ": " + e.getMessage(), e);
249                }
250
251                // Update matching fields to match sub-field
252                for (Map.Entry<JClass<?>, JField> entry : matchingFields.entrySet())
253                    entry.setValue(((JComplexField)entry.getValue()).getSubField(subFieldName));
254
255                // Logging
256                if (this.log.isTraceEnabled())
257                    this.log.trace("RefPath: [" + searchName + "." + subFieldName + "] new matching fields=" + matchingFields);
258            }
259
260            // Last field?
261            if (fieldNames.isEmpty()) {
262                if (superFieldInfo != null && Boolean.FALSE.equals(lastIsSubField)) {
263                    throw new IllegalArgumentException(errorPrefix + "path may not end on " + description
264                      + "; specify the complex field itself instead");
265                }
266                break;
267            }
268
269            // Field is not last, so it must be a reference field
270            if (!(fieldInfo instanceof JReferenceFieldInfo))
271                throw new IllegalArgumentException(errorPrefix + description + " is not a reference field");
272            final JReferenceFieldInfo referenceFieldInfo = (JReferenceFieldInfo)fieldInfo;
273
274            // Add reference field to the reference field list
275            this.referenceFieldInfos.add(referenceFieldInfo);
276
277            // Advance through the reference, using the narrowest type possible
278            currentType = Util.findLowestCommonAncestor(Iterables.transform(
279               Iterables.transform(matchingFields.values(), new CastFunction<JReferenceField>(JReferenceField.class)),
280              new JReferenceFieldTypeFunction())).getRawType();
281
282            // Logging
283            if (this.log.isTraceEnabled())
284                this.log.trace("RefPath: step through reference, new currentType=" + currentType);
285        }
286
287        // Done
288        this.targetType = currentType;
289        this.targetFieldInfo = fieldInfo;
290        this.targetSuperFieldInfo = superFieldInfo;
291
292        // Logging
293        if (this.log.isTraceEnabled()) {
294            this.log.trace("RefPath: DONE: targetType=" + this.targetType + " targetFieldInfo=" + this.targetFieldInfo
295              + " targetSuperFieldInfo=" + this.targetSuperFieldInfo + " targetFieldType=" + this.getTargetFieldType()
296              + " references=" + this.referenceFieldInfos);
297        }
298    }
299
300    /**
301     * Get the Java type of the object at which this path starts.
302     *
303     * <p>
304     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
305     * equal the {@linkplain #getTargetType target object type}, or possibly a super-type if the target
306     * field exists only in a sub-type.
307     * </p>
308     *
309     * @return the Java type at which this reference path starts
310     */
311    public Class<?> getStartType() {
312        return this.startType;
313    }
314
315    /**
316     * Get the Java type of the object at which this path ends.
317     *
318     * <p>
319     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
320     * equal the Java type of the {@linkplain #getStartType starting object type}, or possibly a sub-type
321     * if the target field exists only in a sub-type.
322     * </p>
323     *
324     * @return the Java type at which this reference path ends
325     */
326    public Class<?> getTargetType() {
327        return this.targetType;
328    }
329
330    /**
331     * Get the Java type corresponding to the field at which this path ends.
332     *
333     * @return the type of the field at which this reference path ends
334     */
335    public TypeToken<?> getTargetFieldType() {
336        return this.targetFieldInfo.getTypeToken(this.targetType);
337    }
338
339    /**
340     * Get the storage ID associated with the target field in the {@linkplain #getTargetType target object type}.
341     *
342     * <p>
343     * This is just the storage ID of the last field in the path.
344     * </p>
345     *
346     * @return the storage ID of the field at which this reference path ends
347     */
348    public int getTargetField() {
349        return this.targetFieldInfo.storageId;
350    }
351
352    /**
353     * Get the storage ID associated with the complex field containing the {@linkplain #getTargetField target field}
354     * a sub-field, in the case that the target field is a sub-field of a complex field.
355     *
356     * @return target field's complex super-field storage ID, or zero if the target field is not a complex sub-field
357     */
358    public int getTargetSuperField() {
359        return this.targetSuperFieldInfo != null ? this.targetSuperFieldInfo.storageId : 0;
360    }
361
362    /**
363     * Get the storage IDs of the reference fields in this path.
364     *
365     * <p>
366     * The path may be empty, i.e., zero references are traversed in the path.
367     * </p>
368     *
369     * <p>
370     * Otherwise, the first field is a field in the {@linkplain #getStartType starting object type} and
371     * the last field is field in some object type that refers to the {@linkplain #getTargetType target object type}.
372     * </p>
373     *
374     * @return zero or more reference field storage IDs
375     */
376    public int[] getReferenceFields() {
377        final int[] storageIds = new int[this.referenceFieldInfos.size()];
378        for (int i = 0; i < storageIds.length; i++)
379            storageIds[i] = this.referenceFieldInfos.get(i).storageId;
380        return storageIds;
381    }
382
383    /**
384     * Get the {@link String} form of the path associated with this instance.
385     */
386    @Override
387    public String toString() {
388        return this.path;
389    }
390
391// Functions
392
393    private static class JReferenceFieldTypeFunction implements Function<JReferenceField, TypeToken<?>> {
394
395        @Override
396        public TypeToken<?> apply(JReferenceField jfield) {
397            return jfield.typeToken;
398        }
399    }
400
401    private static class JClassTypeFunction implements Function<JClass<?>, Class<?>> {
402
403        @Override
404        public Class<?> apply(JClass<?> jclass) {
405            return jclass.type;
406        }
407    }
