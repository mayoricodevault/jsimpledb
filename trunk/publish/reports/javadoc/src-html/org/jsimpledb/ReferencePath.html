008package org.jsimpledb;
009
010import com.google.common.reflect.TypeToken;
011
012import java.util.ArrayDeque;
013import java.util.ArrayList;
014import java.util.HashMap;
015
016/**
017 * Reference paths.
018 *
019 * <p>
020 * A Reference path consists of:
021 * <ul>
022 *  <li>A {@linkplain #getStartType starting Java object type},</li>
023 *  <li>A final {@linkplain #getTargetType target object type} and {@linkplain #getTargetField target field}
024 *      within that type, and</li>
025 *  <li>An intermediate chain of zero or more {@linkplain #getReferenceFields reference fields} through which an
026 *      instance of the target type can be reached from an instance of the starting type.</li>
027 * </ul>
028 * </p>
029 *
030 * <p>
031 * Reference paths can be described by the combination of (a) the starting Java object type, and (b) the path of reference
032 * fields and final target field in a {@link String} form consisting of field names separated by period ({@code "."}) characters.
033 * All of the fields except the last must be reference fields.
034 * </p>
035 *
036 * <p>
037 * For example, path {@code "parent.age"} starting from object type {@code Person} might refer to the age of
038 * the parent of a {@code Person}.
039 * </p>
040 *
041 * <p>
042 * When a complex field appears in a reference path, both the name of the complex field and the specific sub-field
043 * being traversed should appear, e.g., {@code "mymap.key.somefield"}. For set and list fields, the (only) sub-field is
044 * named {@code "element"}, while for map fields the sub-fields are named {@code "key"} and {@code "value"}.
045 * </p>
046 *
047 * <p>
048 * <b>Fields of Sub-Types</b>
049 * </p>
050 *
051 * <p>
052 * In some cases, a field may not exist in a Java object type, but it does exist in a some sub-type of that type. For example:
053 * </p>
054 *
055 * <p>
056 * <pre>
057 * &#64;JSimpleClass(storageId = 10)
058 * public class Person {
059 *
060 *     &#64;JSimpleSetField(storageId = 11)
061 *     public abstract Set&lt;Person&gt; <b>getFriends</b>();
062 *
063 *     &#64;OnChange("friends.element.<b>name</b>")
064 *     private void friendNameChanged(SimpleFieldChange&lt;NamedPerson, String&gt; change) {
065 *         // ... do whatever
066 *     }
067 * }
068 *
069 * &#64;JSimpleClass(storageId = 20)
070 * public class NamedPerson extends Person {
071 *
072 *     &#64;JSimpleField(storageId = 21)
073 *     public abstract String <b>getName</b>();
074 *     public abstract void setName(String name);
075 * }
076 * </pre>
077 * Here the path {@code "friends.element.name"} is technically incorrect because {@code "friends.element"}
078 * has type {@code Person}, and {@code "name"} is a field of {@code NamedPerson}, not {@code Person}. However, this will
079 * still work as long as there is no ambiguity, i.e., in this example, there are no other sub-types of {@code Person}
080 * with a field named {@code "name"}. Note also in the example above the
081 * {@link org.jsimpledb.change.SimpleFieldChange} parameter to the method
082 * {@code friendNameChanged()} necessarily has generic type {@code NamedPerson}, not {@code Person}.
083 * </p>
084 *
085 * <p>
086 * In cases where multiple sub-types of a common super-type type have fields with the same name but different storage IDs,
087 * the storage ID may be explicitly specified as a suffix, for example, {@code "name#123"}.
088 * </p>
089 */
090public class ReferencePath {
091
092    final TypeToken<?> startType;
093    final TypeToken<?> targetType;
094    final JField targetField;
095    final JComplexField targetSuperField;
096    final ArrayList<JReferenceField> referenceFields = new ArrayList<>();
097
098    /**
099     * Constructor.
100     *
101     * @param jdb {@link JSimpleDB} against which to resolve object and field names
102     * @param startType starting Java type for the path
103     * @param path dot-separated path of zero or more reference fields, followed by a target field
104     * @param lastIsSubField true if the last field can be a complex sub-field but not a complex field, false for the reverse,
105     *  or null for don't care
106     * @throws IllegalArgumentException if {@code jdb}, {@code startType}, or {@code path} is null
107     * @throws IllegalArgumentException if {@code path} is invalid
108     */
109    ReferencePath(JSimpleDB jdb, TypeToken<?> startType, String path, Boolean lastIsSubField) {
110
111        // Sanity check
112        if (jdb == null)
113            throw new IllegalArgumentException("null jdb");
114        if (startType == null)
115            throw new IllegalArgumentException("null startType");
116        if (path == null)
117            throw new IllegalArgumentException("null path");
118        final String errorPrefix = "invalid path `" + path + "': ";
119        if (path.length() == 0)
120            throw new IllegalArgumentException(errorPrefix + "path is empty");
121        this.startType = startType;
122
123        // Split the path into field names
124        final ArrayDeque<String> fieldNames = new ArrayDeque<>();
125        while (true) {
126            final int dot = path.indexOf('.');
127            if (dot == -1) {
128                if (path.length() == 0)
129                    throw new IllegalArgumentException(errorPrefix + "ends in `.'");
130                fieldNames.add(path);
131                break;
132            }
133            if (dot == 0)
134                throw new IllegalArgumentException(errorPrefix + "contains an empty path component");
135            fieldNames.add(path.substring(0, dot));
136            path = path.substring(dot + 1);
137        }
138
139        // Initialize loop state
140        TypeToken<?> currentType = this.startType;
141        JField currentField = null;
142        JComplexField currentSuperField = null;
143
144        // Parse field names
145        final ArrayList<Integer> storageIds = new ArrayList<>();
146        while (!fieldNames.isEmpty()) {
147            final String fieldName = fieldNames.removeFirst();
148            String description = "field `" + fieldName + "' in type " + currentType;
149
150            // Get explicit storage ID, if any
151            final int hash = fieldName.indexOf('#');
152            int explicitStorageId = 0;
153            final String searchName;
154            if (hash != -1) {
155                try {
156                    explicitStorageId = Integer.parseInt(fieldName.substring(hash + 1));
157                } catch (NumberFormatException e) {
158                    throw new IllegalArgumentException(errorPrefix + "invalid field name `" + fieldName + "'");
159                }
160                searchName = fieldName.substring(0, hash);
161            } else
162                searchName = fieldName;
163
164            // Find all JFields matching 'fieldName' in some JClass whose type matches 'typeToken'
165            final HashMap<JClass<?>, JField> matchingFields = new HashMap<>();
166            for (JClass<?> jclass : jdb.jclasses.values()) {
167                if (!currentType.isAssignableFrom(jclass.typeToken))
168                    continue;
169                final JField jfield = jclass.jfieldsByName.get(searchName);
170                if (jfield == null)
171                    continue;
172                if (explicitStorageId != 0 && jfield.storageId != explicitStorageId)
173                    continue;
174                matchingFields.put(jclass, jfield);
175            }
176
177            // None found?
178            if (matchingFields.isEmpty()) {
179                throw new IllegalArgumentException(errorPrefix + "there is no field named `" + searchName + "'"
180                  + (explicitStorageId != 0 ? " with storage ID " + explicitStorageId : "")
181                  + " in (any sub-type of) " + currentType);
182            }
183
184            // Get the field; if multiple fields are found, verify they all are really the same field
185            currentField = matchingFields.values().iterator().next();
186            for (JField otherField : matchingFields.values()) {
187                if (otherField.storageId != currentField.storageId) {
188                    throw new IllegalArgumentException(errorPrefix + "there are multiple different fields named `"
189                      + fieldName + "' in sub-types of type " + currentType);
190                }
191            }
192            currentSuperField = null;
193
194            // Get the lowest common super-type among the matching JClass's
195            for (JClass<?> jclass : matchingFields.keySet()) {
196                boolean commonSupertype = true;
197                for (JClass<?> other : matchingFields.keySet()) {
198                    if (!jclass.typeToken.isAssignableFrom(other.typeToken)) {
199                        commonSupertype = false;
200                        break;
201                    }
202                }
203                if (commonSupertype) {
204                    currentType = jclass.typeToken;
205                    break;
206                }
207            }
208
209            // Handle complex fields
210            if (currentField instanceof JComplexField) {
211                final JComplexField complexField = (JComplexField)currentField;
212
213                // Last field?
214                if (fieldNames.isEmpty()) {
215                    if (lastIsSubField != null && lastIsSubField) {
216                        throw new IllegalArgumentException(errorPrefix + "path may not end on complex " + description
217                          + "; a sub-field of must be specified");
218                    }
219                    break;
220                }
221                currentSuperField = complexField;
222
223                // Get sub-field
224                final String subFieldName = fieldNames.removeFirst();
225                description = "sub-field `" + subFieldName + "' of complex " + description;
226                try {
227                    currentField = complexField.getSubField(subFieldName);
228                } catch (IllegalArgumentException e) {
229                    throw new IllegalArgumentException(errorPrefix + "invalid " + description + ": " + e.getMessage(), e);
230                }
231            }
232
233            // Last field?
234            if (fieldNames.isEmpty()) {
235                if (currentField.parent instanceof JComplexField && lastIsSubField != null && !lastIsSubField) {
236                    throw new IllegalArgumentException(errorPrefix + "path may not end on " + description
237                      + "; specify the complex field itself instead");
238                }
239                break;
240            }
241
242            // Field is not last, so it must be a reference field
243            if (!(currentField instanceof JReferenceField))
244                throw new IllegalArgumentException(errorPrefix + description + " is not a reference field");
245            final JReferenceField referenceField = (JReferenceField)currentField;
246
247            // Add reference field to the reference field list
248            this.referenceFields.add(referenceField);
249
250            // Advance through the reference
251            currentType = referenceField.typeToken;
252        }
253
254        // Done
255        this.targetType = currentType;
256        this.targetField = currentField;
257        this.targetSuperField = currentSuperField;
258    }
259
260    /**
261     * Get the Java type at which this path starts.
262     *
263     * <p>
264     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
265     * equal the {@linkplain #getTargetType target object type}, or possibly a super-type if the target
266     * field exists only in a sub-type.
267     * </p>
268     */
269    public TypeToken<?> getStartType() {
270        return this.startType;
271    }
272
273    /**
274     * Get the Java type at which this path ends.
275     *
276     * <p>
277     * If there are zero {@linkplain #getReferenceFields reference fields} in this path, then this will
278     * equal the Java type of the {@linkplain #getStartType starting object type}, or possibly a sub-type
279     * if the target field exists only in a sub-type.
280     * </p>
281     */
282    public TypeToken<?> getTargetType() {
283        return this.targetType;
284    }
285
286    /**
287     * Get the storage ID associated with the target field in the {@linkplain #getTargetType target object type}.
288     *
289     * <p>
290     * This is just the storage ID of the last field in the path.
291     * </p>
292     */
293    public int getTargetField() {
294        return this.targetField.storageId;
295    }
296
297    /**
298     * Get the storage ID associated with the complex field containing the {@linkplain #getTargetField target field}
299     * a sub-field, in the case that the target field is a sub-field of a complex field.
300     *
301     * @return target field's complex super-field storage ID, or zero if the target field is not a complex sub-field
302     */
303    public int getTargetSuperField() {
304        return this.targetSuperField != null ? this.targetSuperField.storageId : 0;
305    }
306
307    /**
308     * Get the storage IDs of the reference fields in this path.
309     *
310     * <p>
311     * The path may be empty, i.e., zero references are traversed in the path.
312     * </p>
313     *
314     * <p>
315     * Otherwise, the first field is a field in the {@linkplain #getStartType starting object type} and
316     * the last field is field in some object type that refers to the {@linkplain #getTargetType target object type}.
317     * </p>
318     *
319     * @return zero or more reference field storage IDs
320     */
321    public int[] getReferenceFields() {
322        final int[] storageIds = new int[this.referenceFields.size()];
323        for (int i = 0; i < storageIds.length; i++)
324            storageIds[i] = this.referenceFields.get(i).storageId;
325        return storageIds;
326    }
327
328    /**
329     * Get the {@link String} form of the path associated with this instance.
330     */
331    @Override
332    public String toString() {
333        final StringBuilder buf = new StringBuilder();
334        for (JReferenceField referenceField : this.referenceFields)
335            this.appendField(buf, referenceField);
336        this.appendField(buf, this.targetField);
337        return buf.toString();
338    }
339
340    private void appendField(StringBuilder buf, JField field) {
341        if (buf.length() > 0)
342            buf.append('.');
343        if (field instanceof JSimpleField) {
344            final JSimpleField simpleField = (JSimpleField)field;
345            if (simpleField.parent instanceof JComplexField) {
346                final JComplexField parent = (JComplexField)simpleField.parent;
347                buf.append(parent.name).append('.').append(parent.getSubFieldName(simpleField));
348                return;
349            }
350        }
351        buf.append(field.name);
352    }
353}
