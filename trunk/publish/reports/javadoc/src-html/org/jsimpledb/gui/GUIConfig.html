008package org.jsimpledb.gui;
009
010import org.jsimpledb.JSimpleDB;
011
012/**
013 * GUI configuration object.
014 */
015public interface GUIConfig {
016
017    /**
018     * Get the {@link JSimpleDB}.
019     */
020    JSimpleDB getJSimpleDB();
021
022    /**
023     * Get a short description of the database.
024     */
025    String getDatabaseDescription();
026
027    /**
028     * Determine the schema version associated with the {@link JSimpleDB}.
029     */
030    int getSchemaVersion();
031
032    /**
033     * Determine whether we are allowed to create a new schema version.
034     */
035    boolean isAllowNewSchema();
036
037    /**
038     * Determine whether the underlying database is read-only.
039     */
040    boolean isReadOnly();
041
042    /**
043     * Determine whether verbose mode is enabled.
044     */
045    boolean isVerbose();
046
047    /**
048     * Get any custom {@link org.jsimpledb.parse.func.AbstractFunction} classes.
049     */
050    Iterable<Class<?>> getFunctionClasses();
051}
