008package org.jsimpledb.gui;
009
010import com.vaadin.annotations.PreserveOnRefresh;
011import com.vaadin.annotations.Push;
012import com.vaadin.annotations.Theme;
013import com.vaadin.server.ExternalResource;
014import com.vaadin.server.Sizeable;
015import com.vaadin.server.ThemeResource;
016import com.vaadin.server.VaadinRequest;
017import com.vaadin.shared.ui.MarginInfo;
018import com.vaadin.ui.Alignment;
019import com.vaadin.ui.Component;
020import com.vaadin.ui.HorizontalLayout;
021import com.vaadin.ui.Label;
022import com.vaadin.ui.Link;
023import com.vaadin.ui.UI;
024import com.vaadin.ui.VerticalLayout;
025
026import org.jsimpledb.Version;
027import org.slf4j.Logger;
028import org.slf4j.LoggerFactory;
029
030/**
031 * Superclass of the various {@link UI}s that constitute the GUI.
032 */
033@PreserveOnRefresh
034@Push
035@SuppressWarnings("serial")
036@Theme("jsdb")
037public abstract class AbstractUI extends UI {
038
039    private static final float UPPER_BAR_HEIGHT = 44;
040
041    protected final Logger log = LoggerFactory.getLogger(this.getClass());
042
043    private final VerticalLayout rootLayout = new VerticalLayout();
044
045// Vaadin lifecycle
046
047    @Override
048    public void init(VaadinRequest request) {
049        this.getPage().setTitle(this.getTitle());
050        this.setContent(this.rootLayout);
051        this.rootLayout.setSpacing(true);
052        this.rootLayout.setSizeFull();
053        this.rootLayout.setMargin(new MarginInfo(false, true, true, true));
054        this.rootLayout.addComponent(this.buildRootUpperBar());
055        this.rootLayout.addComponent(new HorizontalLine(3));
056        final Component lowerPanel = this.buildLowerPanel();
057        this.rootLayout.addComponent(lowerPanel);
058        this.rootLayout.setExpandRatio(lowerPanel, 1.0f);
059        this.rootLayout.setComponentAlignment(lowerPanel, Alignment.MIDDLE_CENTER);
060    }
061
062// Layout construction
063
064    protected String getTitle() {
065        return "JSimpleDB";
066    }
067
068    protected Component getTopRightLabel() {
069        return null;
070    }
071
072    protected Component buildRootUpperBar() {
073
074        // Logo
075        final Link logo = new Link(null, new ExternalResource(MainUI.URI_PATH));
076        logo.setIcon(new ThemeResource("img/jsimpledb-logo-48x48.png"));
077        final HorizontalLayout logoLayout = new HorizontalLayout();
078        logoLayout.addStyleName("jsdb-upper-bar-company-logo-layout");
079        logoLayout.setWidth(48, Sizeable.Unit.PIXELS);
080        logoLayout.addComponent(logo);
081        logoLayout.setComponentAlignment(logo, Alignment.BOTTOM_LEFT);
082
083        // Title
084        final SizedLabel titleLabel = new SizedLabel(this.getTitle());
085        titleLabel.addStyleName("jsdb-title");
086
087        // Sequence parts
088        final HorizontalLayout layout = new HorizontalLayout();
089        layout.setSpacing(true);
090        layout.setWidth("100%");
091        layout.setHeight(UPPER_BAR_HEIGHT, Sizeable.Unit.PIXELS);
092        layout.addComponent(logoLayout);
093        layout.setComponentAlignment(logoLayout, Alignment.BOTTOM_LEFT);
094        layout.addComponent(titleLabel);
095        layout.setExpandRatio(titleLabel, 1.0f);
096        layout.setComponentAlignment(titleLabel, Alignment.BOTTOM_CENTER);
097        final Component topRightLabel = this.getTopRightLabel();
098        if (topRightLabel != null) {
099            layout.addComponent(topRightLabel);
100            layout.setComponentAlignment(topRightLabel, Alignment.BOTTOM_RIGHT);
101        }
102        return layout;
103    }
104
105    protected Component buildLowerPanel() {
106
107        // Add main panel
108        final VerticalLayout layout = new VerticalLayout();
109        layout.setSizeFull();
110        layout.setMargin(false);
111        layout.setSpacing(true);
112        final Component mainPanel = this.buildMainPanel();
113        mainPanel.setSizeFull();
114        layout.addComponent(mainPanel);
115        layout.setComponentAlignment(mainPanel, Alignment.TOP_CENTER);
116        layout.setExpandRatio(mainPanel, 1);
117
118        // Add footer containing links (admin only) and version label
119        final HorizontalLayout footerLayout = new HorizontalLayout();
120        footerLayout.setWidth("100%");
121        footerLayout.setSpacing(true);
122        final Label spacer = new Label();
123        footerLayout.addComponent(spacer);
124        footerLayout.setExpandRatio(spacer, 1);
125        footerLayout.addComponent(new SizedLabel("JSimpleDB Version " + Version.JSIMPLEDB_VERSION));
126        layout.addComponent(footerLayout);
127
128        // Done
129        return layout;
130    }
131
132    protected abstract Component buildMainPanel();
133}
