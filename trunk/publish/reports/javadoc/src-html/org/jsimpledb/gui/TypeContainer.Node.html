008package org.jsimpledb.gui;
009
010import com.google.common.base.Function;
011import com.google.common.collect.Lists;
012import com.vaadin.data.Container;
013import com.vaadin.shared.ui.label.ContentMode;
014
015import java.util.ArrayList;
016import java.util.Collection;
017import java.util.Collections;
018import java.util.List;
019
020import org.dellroad.stuff.vaadin7.ProvidesProperty;
021import org.dellroad.stuff.vaadin7.SimpleKeyedContainer;
022import org.jsimpledb.JClass;
023import org.jsimpledb.JSimpleDB;
024import org.slf4j.Logger;
025import org.slf4j.LoggerFactory;
026
027/**
028 * Hierarical container that contains the tree of all Java model classes in the database schema.
029 */
030@SuppressWarnings("serial")
031public class TypeContainer extends SimpleKeyedContainer<Class<?>, TypeContainer.Node> implements Container.Hierarchical {
032
033    public static final String NAME_PROPERTY = "name";
034    public static final String STORAGE_ID_PROPERTY = "storageId";
035    public static final String TYPE_PROPERTY = "type";
036
037    protected final Logger log = LoggerFactory.getLogger(this.getClass());
038
039    private final JSimpleDB jdb;
040    private final Class<?> type;
041    private final ArrayList<Node> rootList = new ArrayList<>();
042
043    /**
044     * Constructor.
045     */
046    public TypeContainer(JSimpleDB jdb) {
047        this(jdb, null);
048    }
049
050    /**
051     * Constructor.
052     */
053    public TypeContainer(JSimpleDB jdb, Class<?> type) {
054        super(Node.class);
055        this.jdb = jdb;
056        this.type = type;
057    }
058
059    @Override
060    public void connect() {
061        super.connect();
062        this.reload();
063    }
064
065    @Override
066    public Class<?> getKeyFor(Node node) {
067        return node.getType();
068    }
069
070    public void reload() {
071
072        // Node set
073        final ArrayList<Node> nodes = new ArrayList<>();
074
075        // Create root node for type if it doesn't correspond to a JClass
076        final Class<?> topType = this.type != null ? this.type : Object.class;
077        boolean needsTop = this.type == null;
078        if (!needsTop) {
079            try {
080                this.jdb.getJClass(type);
081            } catch (IllegalArgumentException e) {
082                needsTop = true;
083            }
084        }
085        if (needsTop)
086            nodes.add(new Node(topType));
087
088        // Create Node's for each JClass
089        for (JClass<?> jclass : this.jdb.getJClasses().values()) {
090            if (topType.isAssignableFrom(jclass.getType()))
091                nodes.add(new Node(jclass));
092        }
093
094        // Determine parents by inspecting Java types
095        for (Node node1 : nodes) {
096            final Class<?> type1 = node1.getType();
097            for (Node node2 : nodes) {
098                if (node2 == node1)
099                    continue;
100                final Class<?> type2 = node2.getType();
101                if (type2.isAssignableFrom(type1)) {
102                    final Node previous = node1.getParent();
103                    if (previous == null || previous.getType().isAssignableFrom(type2))
104                        node1.setParent(node2);
105                }
106            }
107        }
108
109        // Derive children
110        for (Node node1 : nodes) {
111            for (Node node2 : nodes) {
112                if (node2 == node1.getParent())
113                    node2.getChilds().add(node1);
114            }
115        }
116
117        // Derive roots
118        this.rootList.clear();
119        for (Node node : nodes) {
120            if (node.getParent() == null)
121                this.rootList.add(node);
122        }
123
124        // Load container
125        this.load(nodes);
126    }
127
128// Node
129
130    public static class Node {
131
132        public static final Function<Node, Class<?>> TYPE_FUNCTION = new Function<Node, Class<?>>() {
133            @Override
134            public Class<?> apply(Node node) {
135                return node.getType();
136            }
137        };
138
139        private final Class<?> type;
140        private final JClass<?> jclass;
141        private final ArrayList<Node> childs = new ArrayList<>();
142
143        private Node parent;
144
145        public Node(JClass<?> jclass) {
146            this(jclass.getType(), jclass);
147        }
148
149        public Node(Class<?> type) {
150            this(type, null);
151        }
152
153        private Node(Class<?> type, JClass<?> jclass) {
154            if (type == null)
155                throw new IllegalArgumentException("null type");
156            this.type = type;
157            this.jclass = jclass;
158        }
159
160        public Class<?> getType() {
161            return this.type;
162        }
163
164        public Node getParent() {
165            return this.parent;
166        }
167        public void setParent(Node parent) {
168            this.parent = parent;
169        }
170
171        public List<Node> getChilds() {
172            return this.childs;
173        }
174
175        @ProvidesProperty(TypeContainer.NAME_PROPERTY)
176        public String propertyName() {
177            return this.jclass != null ? this.jclass.getName() : this.type.getSimpleName();
178        }
179
180        @ProvidesProperty(TypeContainer.STORAGE_ID_PROPERTY)
181        public Integer propertyStorageId() {
182            return this.jclass != null ? this.jclass.getStorageId() : null;
183        }
184
185        @ProvidesProperty(TypeContainer.TYPE_PROPERTY)
186        public SizedLabel proeprtyType() {
187            return new SizedLabel("<code>" + this.getType().toString() + "</code>", ContentMode.HTML);
188        }
189    }
190
191// Container.Hierarchical methods
192
193    @Override
194    public Collection<Class<?>> getChildren(Object itemId) {
195        final Node node = this.getJavaObject(itemId);
196        return node != null ? Lists.transform(node.getChilds(), Node.TYPE_FUNCTION) : Collections.<Class<?>>emptySet();
197    }
198
199    @Override
200    public Class<?> getParent(Object itemId) {
201        final Node child = this.getJavaObject(itemId);
202        if (child == null)
203            return null;
204        final Node parent = child.getParent();
205        if (parent == null)
206            return null;
207        return parent.getType();
208    }
209
210    @Override
211    public Collection<Class<?>> rootItemIds() {
212        return Lists.transform(this.rootList, Node.TYPE_FUNCTION);
213    }
214
215    @Override
216    public boolean areChildrenAllowed(Object itemId) {
217        final Node node = this.getJavaObject(itemId);
218        return node != null && !node.getChilds().isEmpty();
219    }
220
221    @Override
222    public boolean isRoot(Object itemId) {
223        final Node node = this.getJavaObject(itemId);
224        return node != null && node.getParent() == null;
225    }
226
227    @Override
228    public boolean hasChildren(Object itemId) {
229        final Node node = this.getJavaObject(itemId);
230        return node != null && !node.getChilds().isEmpty();
231    }
232
233// Mutators are not supported
234
235    @Override
236    public boolean setParent(Object itemId, Object newParentId) {
237        throw new UnsupportedOperationException();
238    }
239
240    @Override
241    public boolean setChildrenAllowed(Object itemId, boolean areChildrenAllowed) {
242        throw new UnsupportedOperationException();
243    }
244
245    @Override
246    public boolean removeItem(Object itemId) {
247        throw new UnsupportedOperationException();
248    }
249}
