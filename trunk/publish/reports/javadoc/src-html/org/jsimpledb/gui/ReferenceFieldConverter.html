008package org.jsimpledb.gui;
009
010import com.vaadin.data.util.converter.Converter;
011
012import java.util.Locale;
013
014import org.jsimpledb.JObject;
015import org.jsimpledb.JSimpleDB;
016import org.jsimpledb.core.ObjId;
017
018/**
019 * Vaadin {@link Converter} for that converts between {@link ObjId}'s and {@link JObject}s.
020 */
021@SuppressWarnings("serial")
022public class ReferenceFieldConverter implements Converter<ObjId, JObject> {
023
024    private final JSimpleDB jdb;
025
026    public ReferenceFieldConverter(JSimpleDB jdb) {
027        if (jdb == null)
028            throw new IllegalArgumentException("null jdb");
029        this.jdb = jdb;
030    }
031
032    @Override
033    public Class<ObjId> getPresentationType() {
034        return ObjId.class;
035    }
036
037    @Override
038    public Class<JObject> getModelType() {
039        return JObject.class;
040    }
041
042    @Override
043    public ObjId convertToPresentation(JObject jobj, Class<? extends ObjId> targetType, Locale locale) {
044        if (jobj == null)
045            return null;
046        return jobj.getObjId();
047    }
048
049    @Override
050    public JObject convertToModel(ObjId id, Class<? extends JObject> targetType, Locale locale) {
051        if (id == null)
052            return null;
053        return this.jdb.getJObject(id);
054    }
055}
