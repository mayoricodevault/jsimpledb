008package org.jsimpledb.gui;
009
010import com.vaadin.data.Property;
011import com.vaadin.server.Sizeable;
012import com.vaadin.shared.ui.label.ContentMode;
013import com.vaadin.ui.Button;
014import com.vaadin.ui.Component;
015import com.vaadin.ui.HorizontalLayout;
016import com.vaadin.ui.HorizontalSplitPanel;
017import com.vaadin.ui.Label;
018import com.vaadin.ui.Notification;
019import com.vaadin.ui.VerticalLayout;
020
021import org.dellroad.stuff.spring.RetryTransaction;
022import org.jsimpledb.JObject;
023import org.jsimpledb.JTransaction;
024import org.jsimpledb.core.ObjId;
025import org.jsimpledb.core.ObjIdSet;
026import org.jsimpledb.parse.ParseSession;
027import org.springframework.transaction.annotation.Transactional;
028
029/**
030 * Edits an object reference.
031 */
032@SuppressWarnings("serial")
033public class ObjectEditor extends HorizontalLayout {
034
035    private final JTransaction dest;
036    private final ParseSession session;
037    private final String name;
038    private final Class<?> type;
039    private final Property<JObject> property;
040    private final JObjectContainer.RefLabelPropertyDef refLabelPropertyDef = new JObjectContainer.RefLabelPropertyDef();
041
042    private final SmallButton changeButton = new SmallButton("Change...", new Button.ClickListener() {
043        @Override
044        public void buttonClick(Button.ClickEvent event) {
045            ObjectEditor.this.change();
046        }
047    });
048    private final SmallButton nullButton = new SmallButton("Null", new Button.ClickListener() {
049        @Override
050        public void buttonClick(Button.ClickEvent event) {
051            ObjectEditor.this.setNull();
052        }
053    });
054
055    private Component refLabel = new Label();
056
057    /**
058     * Conveinence constructor. Sets {@code allowNull} to true.
059     */
060    public ObjectEditor(JTransaction dest, ParseSession session, String name, Class<?> type, Property<JObject> property) {
061        this(dest, session, name, type, property, true);
062    }
063
064    /**
065     * Constructor.
066     *
067     * @param dest destination transaction for the chosen object
068     * @param session session for expression parsing
069     * @param name name of the property
070     * @param type type restriction, or null for none
071     * @param property reference property to edit
072     * @param allowNull whether null value is allowed
073     */
074    public ObjectEditor(JTransaction dest, ParseSession session,
075      String name, Class<?> type, Property<JObject> property, boolean allowNull) {
076
077        // Initialize
078        if (dest == null)
079            throw new IllegalArgumentException("null dest");
080        if (session == null)
081            throw new IllegalArgumentException("null session");
082        if (name == null)
083            throw new IllegalArgumentException("null name");
084        if (property == null)
085            throw new IllegalArgumentException("null property");
086        this.dest = dest;
087        this.session = session;
088        this.name = name;
089        this.type = type;
090        this.property = property;
091
092        // Layout parts
093        this.setMargin(false);
094        this.setSpacing(true);
095        this.addComponent(this.refLabel);
096        this.updateRefLabel();
097        this.addComponent(new Label("\u00a0\u00a0"));
098        this.addComponent(this.changeButton);
099        if (allowNull)
100            this.addComponent(this.nullButton);
101
102        // Listen for property value and read-only status changes
103        if (property instanceof Property.ValueChangeNotifier) {
104            ((Property.ValueChangeNotifier)property).addValueChangeListener(new Property.ValueChangeListener() {
105                @Override
106                public void valueChange(Property.ValueChangeEvent event) {
107                    ObjectEditor.this.updateRefLabel();
108                }
109            });
110        }
111        if (property instanceof Property.ReadOnlyStatusChangeNotifier) {
112            ((Property.ReadOnlyStatusChangeNotifier)property).addReadOnlyStatusChangeListener(
113              new Property.ReadOnlyStatusChangeListener() {
114                @Override
115                public void readOnlyStatusChange(Property.ReadOnlyStatusChangeEvent event) {
116                    ObjectEditor.this.updateRefLabel();
117                }
118            });
119        }
120    }
121
122    // Update reference label component to reflect new choice
123    private void updateRefLabel() {
124        final JObject jobj = this.property.getValue();
125        final boolean readOnly = this.property.isReadOnly();
126        final Component newLabel = jobj != null ?
127          this.refLabelPropertyDef.extract(jobj) : new SizedLabel("<i>Null</i>&#160;", ContentMode.HTML);
128        this.replaceComponent(this.refLabel, newLabel);
129        this.refLabel = newLabel;
130        this.changeButton.setEnabled(!readOnly);
131        this.nullButton.setEnabled(!readOnly && jobj != null);
132    }
133
134    // Handle change button click
135    private void change() {
136        this.new ChangeWindow().show();
137    }
138
139    // Handle null button click
140    private void setNull() {
141        this.property.setValue(null);
142    }
143
144// ChangeWindow
145
146    public class ChangeWindow extends ConfirmWindow {
147
148        private final ObjectChooser objectChooser;
149
150        ChangeWindow() {
151            super(ObjectEditor.this.getUI(), "Select " + ObjectEditor.this.name);
152            this.setWidth(800, Sizeable.Unit.PIXELS);
153            this.setHeight(500, Sizeable.Unit.PIXELS);
154            this.objectChooser = new ObjectChooser(ObjectEditor.this.dest.getJSimpleDB(),
155              ObjectEditor.this.session, ObjectEditor.this.type, false);
156        }
157
158        @Override
159        protected void addContent(VerticalLayout layout) {
160            final HorizontalSplitPanel objectPanel = this.objectChooser.getObjectPanel();
161            objectPanel.setHeight(200, Sizeable.Unit.PIXELS);
162            objectPanel.setSplitPosition(40);
163            layout.addComponent(objectPanel);
164            layout.addComponent(this.objectChooser.getShowForm());
165        }
166
167        @Override
168        @RetryTransaction
169        @Transactional("jsimpledbGuiTransactionManager")
170        protected boolean execute() {
171            final ObjId id = this.objectChooser.getObjId();
172            if (id == null)
173                return true;
174            final JObject jobj = JTransaction.getCurrent().getJObject(id);
175            if (!jobj.exists()) {
176                Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
177                return false;
178            }
179            try {
180                ObjectEditor.this.property.setValue(jobj.copyTo(ObjectEditor.this.dest, id, new ObjIdSet()));
181            } catch (Exception e) {
182                Notification.show("Error: " + e, null, Notification.Type.ERROR_MESSAGE);
183            }
184            return true;
185        }
186    }
187}
