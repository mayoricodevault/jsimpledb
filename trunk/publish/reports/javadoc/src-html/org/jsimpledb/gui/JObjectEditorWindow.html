098    /**
099     * Constructor for editing an existing object.
100     *
101     * @param ui associated {@link UI}
102     * @param session parse session for {@link JObjectChooser}
103     * @param jobj object to edit; should be contained in a snapshot transaction
104     * @param titleComponent title for edit panel
105     */
106    @SuppressWarnings("unchecked")
107    public JObjectEditorWindow(UI ui, ParseSession session, JClass<?> jclass, JObject jobj, Component titleComponent) {
108        super(ui, (jobj != null ? "Edit " : "New ") + jclass.getName());
109        this.setWidth(600, Sizeable.Unit.PIXELS);
110        this.setHeight(450, Sizeable.Unit.PIXELS);
111        this.jclass = jclass;
112        this.jobj = jobj != null ? jobj : this.doCreateForEdit();
113        this.create = jobj == null;
114        this.session = session;
115        this.titleComponent = titleComponent;
116
117        // Create PropertysetItem to hold our editable properties
118        final PropertysetItem item = new PropertysetItem();
119
120        // Introspect for any @FieldBuilder.* annotations
121        this.fieldMap.putAll(new FieldBuilder().buildBeanPropertyFields(this.jobj));
122
123        // Build fields and components for all remaining database properties
124        final SortedMap<String, JField> jfieldMap = jclass.getJFieldsByName();
125        for (Map.Entry<String, JField> entry : jfieldMap.entrySet()) {
126            final String fieldName = entry.getKey();
127            final JField jfield = entry.getValue();
128
129            // If a Field already exists for this database field, just use it, otherwise build one
130            Field<?> field = this.fieldMap.get(fieldName);
131            if (field == null) {
132                field = this.buildFieldField(fieldName, jfield);
133                this.fieldMap.put(fieldName, field);
134            }
135
136            // Build an appropriate Vaadin Property for the field
137            item.addItemProperty(fieldName, this.buildFieldProperty(this.jobj, jfield));
138
139            // Set the field's caption
140            field.setCaption(this.buildCaption(jfield.getName(), !(field instanceof CheckBox)));
141        }
142
143        // Connect fields to properties via FieldGroup
144        this.fieldGroup.setItemDataSource(item);
145
146        // Bind fields into FieldGroup
147        for (Map.Entry<String, Field<?>> entry : this.fieldMap.entrySet())
148            this.fieldGroup.bind(entry.getValue(), entry.getKey());
149    }
150
151    @Override
152    protected void addContent(VerticalLayout layout) {
153        if (this.titleComponent != null)
154            layout.addComponent(this.titleComponent);
155        final FormLayout formLayout = new FormLayout();
156        for (Field<?> field : this.fieldMap.values())
157            formLayout.addComponent(field);
158        layout.addComponent(formLayout);
159    }
160
161    @Override
162    protected boolean execute() {
163
164        // Commit fields in field group
165        try {
166            this.fieldGroup.commit();
167        } catch (FieldGroup.CommitException e) {
168            Notification.show("Invalid value(s)", null, Notification.Type.WARNING_MESSAGE);
169            throw new RuntimeException(e);
170        }
171
172        // Commit transaction, if possible
173        try {
174            return this.writeBack();
175        } catch (UnexpectedRollbackException e) {
176            this.log.debug("ignoring UnexpectedRollbackException presumably caused by validation failure");
177            return false;
178        }
179    }
180
181    @RetryTransaction
182    @Transactional("jsimpledbGuiTransactionManager")
183    private JObject doCreateForEdit() {
184        return (JObject)JTransaction.getCurrent().getSnapshotTransaction().create(this.jclass);
185    }
186
187    @RetryTransaction
188    @Transactional("jsimpledbGuiTransactionManager")
189    protected boolean writeBack() {
190
191        // Find/create target object in current transaction
192        final JTransaction jtx = JTransaction.getCurrent();
193        final ObjId id = this.jobj.getObjId();
194        final JObject target = this.create ? (JObject)jtx.create(this.jclass) : jtx.getJObject(id);
195
196        // Verify object still exists when editing
197        if (!this.create && !target.exists()) {
198            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
199            return true;
200        }
201
202        // Copy fields
203        this.jobj.copyTo(jtx, target.getObjId(), new ObjIdSet());
204
205        // Run validation queue
206        try {
207            jtx.validate();
208        } catch (ValidationException e) {
209            Notification.show("Validation failed", e.getMessage(), Notification.Type.ERROR_MESSAGE);
210            jtx.getTransaction().setRollbackOnly();
211            return false;
212        }
213
214        // Broadcast update event after successful commit
215        if (this.changePublisher != null) {
216            if (create)
217                this.changePublisher.publishChangeOnCommit(new ObjectCreate<Object>(target));
218            else
219                this.changePublisher.publishChangeOnCommit(target);
220        }
221
222        // Show notification after successful commit
223        final VaadinSession vaadinSession = VaadinUtil.getCurrentSession();
224        jtx.getTransaction().addCallback(new Transaction.CallbackAdapter() {
225            @Override
226            public void afterCommit() {
227                VaadinUtil.invoke(vaadinSession, new Runnable() {
228                    @Override
229                    public void run() {
230                        Notification.show((JObjectEditorWindow.this.create ? "Created" : "Updated") + " object " + id);
231                    }
232                });
233            }
234        });
235        return true;
236    }

240    private Field<?> buildFieldField(String fieldName, JField jfield) {
241        return jfield.visit(new JFieldSwitchAdapter<Field<?>>() {
242            @Override
243            public Field<?> caseJSimpleField(JSimpleField jfield) {
244                final boolean allowNull = jfield.getGetter().getAnnotation(NotNull.class) == null
245                  && !jfield.getTypeToken().isPrimitive();
246                return new SimpleFieldFieldBuilder(JObjectEditorWindow.this.jobj.getTransaction(),
247                  jfield, JObjectEditorWindow.this.session, allowNull).buildField();
248            }
249            @Override
250            public Field<?> caseJCounterField(JCounterField jfield) {
251                final TextField field = new TextField();
252                field.setWidth("100%");
253                field.setNullSettingAllowed(false);
254                field.setConverter(new SimpleFieldConverter<Long>(JObjectEditorWindow.this.jclass.getJSimpleDB()
255                  .getDatabase().getFieldTypeRegistry().getFieldType(TypeToken.of(long.class))));
256                return field;
257            }
258            @Override
259            public Field<?> caseJSetField(JSetField jfield) {
260                //return new SetFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
261                return new PlaceHolderField(jfield);                                       // TODO
262            }
263            @Override
264            public Field<?> caseJListField(JListField jfield) {
265                //return new ListFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
266                return new PlaceHolderField(jfield);                                       // TODO
267            }
268            @Override
269            public Field<?> caseJMapField(JMapField jfield) {
270                //return new MapFieldFieldBuilder(jfield, JObjectEditorWindow.this.session).buildField();
271                return new PlaceHolderField(jfield);                                       // TODO
272            }
273        });
274    }

276    private Property<?> buildFieldProperty(final JObject jobj, JField jfield) {
277        return jfield.visit(new JFieldSwitchAdapter<Property<?>>() {
278            @Override
279            @SuppressWarnings({ "unchecked", "rawtypes" })
280            public Property<?> caseJSimpleField(JSimpleField jfield) {
281                return new MethodProperty(jfield.getTypeToken().getRawType(), jobj, jfield.getGetter(), jfield.getSetter());
282            }
283            @Override
284            public Property<?> caseJCounterField(JCounterField jfield) {
285                return new CounterProperty(jfield.getValue(jobj));
286            }
287            @Override
288            @SuppressWarnings("rawtypes")
289            public Property<?> caseJSetField(JSetField jfield) {
290                return new CollectionProperty(jfield.getValue(jobj));
291            }
292            @Override
293            @SuppressWarnings("rawtypes")
294            public Property<?> caseJListField(JListField jfield) {
295                return new CollectionProperty(jfield.getValue(jobj));
296            }
297            @Override
298            public Property<?> caseJMapField(JMapField jfield) {
299                return new ObjectProperty<Void>(null, Void.class);                          // TODO
300            }
301        });
302    }
