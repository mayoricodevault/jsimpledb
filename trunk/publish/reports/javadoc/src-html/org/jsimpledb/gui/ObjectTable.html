008package org.jsimpledb.gui;
009
010import com.vaadin.ui.DefaultFieldFactory;
011import com.vaadin.ui.Table;
012
013import org.jsimpledb.JSimpleDB;
014import org.jsimpledb.Session;
015
016/**
017 * Table showing all objects of a certain type, backed by an {@link ObjectContainer}.
018 */
019@SuppressWarnings("serial")
020public class ObjectTable extends AbstractTable<JObjectContainer> {
021
022    private final JSimpleDB jdb;
023    private final JObjectContainer container;
024    private final Session session;
025    private final boolean showFields;
026
027    public ObjectTable(JSimpleDB jdb, JObjectContainer container, Session session) {
028        this(jdb, container, session, true);
029    }
030
031    public ObjectTable(JSimpleDB jdb, JObjectContainer container, Session session, boolean showFields) {
032        if (jdb == null)
033            throw new IllegalArgumentException("null jdb");
034        if (container == null)
035            throw new IllegalArgumentException("null container");
036        if (session == null)
037            throw new IllegalArgumentException("null session");
038        this.jdb = jdb;
039        this.container = container;
040        this.session = session;
041        this.showFields = showFields;
042        this.setSelectable(true);
043        this.setImmediate(true);
044        this.setSizeFull();
045    }
046
047    @Override
048    protected JObjectContainer buildContainer() {
049        return this.container;
050    }
051
052    @Override
053    protected void configureColumns() {
054
055        // Add columns
056        this.setColumnCollapsingAllowed(true);
057        for (String fieldName : this.getContainer().getOrderedPropertyNames()) {
058            String title = DefaultFieldFactory.createCaptionByPropertyId(fieldName);
059            Table.Align align = Table.Align.CENTER;
060            int width = 120;
061            boolean showField = this.showFields;
062            switch (fieldName) {
063            case JObjectContainer.REFERENCE_LABEL_PROPERTY:
064                title = "Label";
065                width = 120;
066                showField = true;
067                break;
068            case JObjectContainer.OBJECT_ID_PROPERTY:
069                title = "ID";
070                width = 120;
071                break;
072            case JObjectContainer.TYPE_PROPERTY:
073                title = "Type";
074                width = 80;
075                break;
076            case JObjectContainer.VERSION_PROPERTY:
077                title = "Version";
078                width = 40;
079                break;
080            default:
081                break;
082            }
083            this.addColumn(fieldName, title, width, align);
084            this.setColumnExpandRatio(fieldName, width / 120.0f);
085            this.setColumnCollapsed(fieldName, !showField);
086        }
087
088        // Adjust columns
089        this.setColumnCollapsingAllowed(true);
090        this.setColumnCollapsed(JObjectContainer.OBJECT_ID_PROPERTY, true);
091    }
092}
