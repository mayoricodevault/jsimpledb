008package org.jsimpledb.gui;
009
010import com.google.common.collect.Iterables;
011
012import java.util.Collections;
013
014import org.dellroad.stuff.vaadin7.VaadinApplicationListener;
015import org.dellroad.stuff.vaadin7.VaadinConfigurable;
016import org.jsimpledb.JObject;
017import org.jsimpledb.JSimpleDB;
018import org.jsimpledb.parse.ParseContext;
019import org.jsimpledb.parse.ParseException;
020import org.jsimpledb.parse.ParseSession;
021import org.jsimpledb.parse.expr.EvalException;
022import org.jsimpledb.parse.expr.ExprParser;
023import org.jsimpledb.parse.expr.Node;
024import org.jsimpledb.util.CastFunction;
025import org.springframework.beans.factory.annotation.Autowired;
026import org.springframework.beans.factory.annotation.Qualifier;
027import org.springframework.context.event.ApplicationEventMulticaster;
028
029/**
030 * {@link JObjectContainer} whose contents are determined by a Java expression.
031 * Listens for {@link DataChangeEvent}s broadcast by an autowired {@link ApplicationEventMulticaster}.
032 */
033@SuppressWarnings("serial")
034@VaadinConfigurable(preConstruction = true)
035public class ExprQueryJObjectContainer extends JObjectContainer {
036
037    private final ParseSession session;
038
039    private DataChangeListener dataChangeListener;
040    private String contentExpression;
041
042    @Autowired(required = false)
043    @Qualifier("jsimpledbGuiEventMulticaster")
044    private ApplicationEventMulticaster eventMulticaster;
045
046    /**
047     * Constructor.
048     */
049    public <T> ExprQueryJObjectContainer(JSimpleDB jdb, ParseSession session) {
050        this(jdb, null, session);
051    }
052
053    /**
054     * Constructor.
055     *
056     * @param type type restriction, or null for no restriction
057     */
058    public <T> ExprQueryJObjectContainer(JSimpleDB jdb, Class<T> type, ParseSession session) {
059        super(jdb, type);
060        if (session == null)
061            throw new IllegalArgumentException("null session");
062        this.session = session;
063    }
064
065    /**
066     * Configure the expression that returns this container's content. Container will automatically reload.
067     */
068    public void setContentExpression(String contentExpression) {
069        this.contentExpression = contentExpression;
070        this.reload();
071    }
072
073    @Override
074    protected void doInTransaction(final Runnable action) {
075        this.session.perform(new ParseSession.Action() {
076            @Override
077            public void run(ParseSession session) {
078                action.run();
079            }
080        });
081    }
082
083    @Override
084    protected Iterable<? extends JObject> queryForObjects() {
085
086        // Any content?
087        if (this.contentExpression == null)
088            return Collections.<JObject>emptySet();
089
090        // Parse expression
091        final ParseContext ctx = new ParseContext(this.contentExpression);
092        final Node node = new ExprParser().parse(this.session, ctx, false);
093        ctx.skipWhitespace();
094        if (!ctx.isEOF())
095            throw new ParseException(ctx, "syntax error");
096
097        // Evaluate parsed expression
098        final Object content = node.evaluate(this.session).get(this.session);
099        if (!(content instanceof Iterable)) {
100            throw new EvalException("expression must evaluate to an Iterable; found "
101              + (content != null ? content : "null") + " instead");
102        }
103
104        // Reload container with results of expression
105        return Iterables.transform((Iterable<?>)content, new CastFunction<JObject>(JObject.class));
106    }
107
108// Connectable
109
110    @Override
111    public void connect() {
112        super.connect();
113        if (this.eventMulticaster != null) {
114            this.dataChangeListener = new DataChangeListener();
115            this.dataChangeListener.register();
116        }
117    }
118
119    @Override
120    public void disconnect() {
121        if (this.dataChangeListener != null) {
122            this.dataChangeListener.unregister();
123            this.dataChangeListener = null;
124        }
125        super.disconnect();
126    }
127
128// DataChangeListener
129
130    private class DataChangeListener extends VaadinApplicationListener<DataChangeEvent> {
131
132        DataChangeListener() {
133            super(ExprQueryJObjectContainer.this.eventMulticaster);
134            this.setAsynchronous(true);
135        }
136
137        @Override
138        protected void onApplicationEventInternal(DataChangeEvent event) {
139            ExprQueryJObjectContainer.this.handleChange(event.getChange());
140        }
141    }
142}
