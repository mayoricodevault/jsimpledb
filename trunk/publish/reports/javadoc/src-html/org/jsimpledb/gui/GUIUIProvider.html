008package org.jsimpledb.gui;
009
010import com.vaadin.server.UIClassSelectionEvent;
011import com.vaadin.server.UIProvider;
012import com.vaadin.ui.UI;
013
014/**
015 * {@link UIProvider} for this GUI.
016 */
017@SuppressWarnings("serial")
018public class GUIUIProvider extends UIProvider {
019
020    @Override
021    public Class<? extends UI> getUIClass(UIClassSelectionEvent event) {
022
023        // Get requested URI path
024        String pathInfo = event.getRequest().getPathInfo();
025        if (pathInfo != null)
026            pathInfo = pathInfo.substring(1);
027
028        // Look for well-known patterns... TODO
029
030        // Default to main root
031        return MainUI.class;
032    }
033}
