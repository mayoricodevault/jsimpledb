008package org.jsimpledb.gui;
009
010import org.jsimpledb.change.Change;
011import org.jsimpledb.change.ChangeCopier;
012import org.springframework.context.ApplicationEvent;
013
014/**
015 * Spring application event that notifies about a {@link Change} to the database.
016 */
017@SuppressWarnings("serial")
018public class DataChangeEvent extends ApplicationEvent {
019
020    private final Change<?> change;
021
022    /**
023     * Constructor. Assumes there is a current transaction open.
024     *
025     * @param source event source
026     * @param change change information; will be copied out of the current transaction with a {@link ChangeCopier}
027     */
028    public DataChangeEvent(Object source, Change<?> change) {
029        super(source);
030        if (change == null)
031            throw new IllegalArgumentException("null change");
032        this.change = change.visit(new ChangeCopier(true));
033    }
034
035    /**
036     * Get the change that occurred, already copied out of the transaction with a {@link ChangeCopier}.
037     * Related objects are also copied.
038     */
039    public Change<?> getChange() {
040        return this.change;
041    }
042}
