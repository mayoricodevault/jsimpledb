008package org.jsimpledb.gui;
009
010import com.google.common.reflect.TypeToken;
011
012import com.vaadin.ui.Table;
013import com.vaadin.ui.TreeTable;
014
015/**
016 * Table showing object types.
017 */
018@SuppressWarnings("serial")
019public class TypeTable extends TreeTable {
020
021    public TypeTable(TypeContainer container) {
022        super(null, container);
023
024        this.setSelectable(true);
025        this.setImmediate(true);
026        this.setSizeFull();
027        this.setHierarchyColumn(TypeContainer.NAME_PROPERTY);
028        this.setAnimationsEnabled(true);
029
030        this.addColumn(TypeContainer.NAME_PROPERTY, "Type", 140, Table.Align.LEFT);
031        this.addColumn(TypeContainer.STORAGE_ID_PROPERTY, "SID", 40, Table.Align.CENTER);
032        this.addColumn(TypeContainer.TYPE_PROPERTY, "Java Type", 250, Table.Align.CENTER);
033
034        this.setColumnExpandRatio(TypeContainer.NAME_PROPERTY, 1.0f);
035
036        this.setVisibleColumns(TypeContainer.NAME_PROPERTY, TypeContainer.STORAGE_ID_PROPERTY, TypeContainer.TYPE_PROPERTY);
037        this.setColumnCollapsingAllowed(true);
038        this.setColumnCollapsed(TypeContainer.STORAGE_ID_PROPERTY, true);
039        this.setColumnCollapsed(TypeContainer.TYPE_PROPERTY, true);
040    }
041
042    protected void addColumn(String property, String name, int width, Table.Align alignment) {
043        this.setColumnHeader(property, name);
044        this.setColumnWidth(property, width);
045        if (alignment != null)
046            this.setColumnAlignment(property, alignment);
047    }
048
049    public TypeContainer getContainer() {
050        return (TypeContainer)this.getContainerDataSource();
051    }
052
053// Vaadin lifecycle
054
055    @Override
056    public void attach() {
057        super.attach();
058        this.getContainer().connect();
059
060        // Expand all root nodes
061        for (TypeToken<?> typeToken : this.getContainer().rootItemIds())
062            this.setCollapsed(typeToken, false);
063    }
064
065    @Override
066    public void detach() {
067        this.getContainer().disconnect();
068        super.detach();
069    }
070}
