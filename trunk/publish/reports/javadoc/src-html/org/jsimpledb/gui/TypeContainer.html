030/**
031 * Hierarical container that contains the tree of all Java model classes in the database schema.
032 */
033@SuppressWarnings("serial")
034public class TypeContainer extends SimpleKeyedContainer<Class<?>, TypeContainer.Node> implements Container.Hierarchical {
035
036    public static final String NAME_PROPERTY = "name";
037    public static final String STORAGE_ID_PROPERTY = "storageId";
038    public static final String TYPE_PROPERTY = "type";
039
040    protected final Logger log = LoggerFactory.getLogger(this.getClass());
041
042    private final JSimpleDB jdb;
043    private final Class<?> rootType;
044    private final ArrayList<Node> rootList = new ArrayList<>();
045
046    /**
047     * Constructor.
048     *
049     * @param jdb underlying database
050     */
051    public TypeContainer(JSimpleDB jdb) {
052        this(jdb, null);
053    }
054
055    /**
056     * Constructor.
057     *
058     * @param jdb underlying database
059     * @param type type restrictions, or null for none
060     */
061    public TypeContainer(JSimpleDB jdb, Class<?> type) {
062        super(Node.class);
063        if (jdb == null)
064            throw new IllegalArgumentException("null jdb");
065        this.jdb = jdb;
066
067        // Get the types of all JClasses assignable to the given type, and use lowest common ancestor as the "top type"
068        final HashSet<Class<?>> types = new HashSet<>();
069        for (JClass<?> jclass : this.jdb.getJClasses().values()) {
070            if (type == null || type.isAssignableFrom(jclass.getType()))
071                types.add(jclass.getType());
072        }
073        this.rootType = !types.isEmpty() ? Util.findLowestCommonAncestorOfClasses(types).getRawType() : Object.class;
074    }
075
076    @Override
077    public void connect() {
078        super.connect();
079        this.reload();
080    }
081
082    @Override
083    public Class<?> getKeyFor(Node node) {
084        return node.getType();
085    }
086
087    public Class<?> getRootType() {
088        return this.rootType;
089    }
090
091    public void reload() {
092
093        // Node set
094        final ArrayList<Node> nodes = new ArrayList<>();
095
096        // Create Node's for each JClass
097        boolean addedRoot = false;
098        for (JClass<?> jclass : this.jdb.getJClasses().values()) {
099            if (this.rootType.isAssignableFrom(jclass.getType())) {
100                nodes.add(new Node(jclass));
101                if (jclass.getType() == this.rootType)
102                    addedRoot = true;
103            }
104        }
105
106        // Ensure there is a root node
107        if (!addedRoot)
108            nodes.add(new Node(this.rootType));
109
110        // Sort by name
111        Collections.sort(nodes, new Comparator<Node>() {
112            @Override
113            public int compare(Node node1, Node node2) {
114                return node1.propertyName().compareTo(node2.propertyName());
115            }
116        });
117
118        // Determine parents by inspecting Java types
119        for (Node node1 : nodes) {
120            final Class<?> type1 = node1.getType();
121            for (Node node2 : nodes) {
122                if (node2 == node1)
123                    continue;
124                final Class<?> type2 = node2.getType();
125                if (type2.isAssignableFrom(type1)) {
126                    final Node previous = node1.getParent();
127                    if (previous == null || previous.getType().isAssignableFrom(type2))
128                        node1.setParent(node2);
129                }
130            }
131        }
132
133        // Derive children
134        for (Node node1 : nodes) {
135            for (Node node2 : nodes) {
136                if (node2 == node1.getParent())
137                    node2.getChilds().add(node1);
138            }
139        }
140
141        // Derive roots
142        this.rootList.clear();
143        for (Node node : nodes) {
144            if (node.getParent() == null)
145                this.rootList.add(node);
146        }
147
148        // Load container
149        this.load(nodes);
150    }
151
152// Node
153
154    public static class Node {
155
156        public static final Function<Node, Class<?>> TYPE_FUNCTION = new Function<Node, Class<?>>() {
157            @Override
158            public Class<?> apply(Node node) {
159                return node.getType();
160            }
161        };
162
163        private final Class<?> type;
164        private final JClass<?> jclass;
165        private final ArrayList<Node> childs = new ArrayList<>();
166
167        private Node parent;
168
169        public Node(JClass<?> jclass) {
170            this(jclass.getType(), jclass);
171        }
172
173        public Node(Class<?> type) {
174            this(type, null);
175        }
176
177        private Node(Class<?> type, JClass<?> jclass) {
178            if (type == null)
179                throw new IllegalArgumentException("null type");
180            this.type = type;
181            this.jclass = jclass;
182        }
183
184        public Class<?> getType() {
185            return this.type;
186        }
187
188        public Node getParent() {
189            return this.parent;
190        }
191        public void setParent(Node parent) {
192            this.parent = parent;
193        }
194
195        public List<Node> getChilds() {
196            return this.childs;
197        }
198
199        @ProvidesProperty(TypeContainer.NAME_PROPERTY)
200        public String propertyName() {
201            return this.jclass != null ? this.jclass.getName() : this.type.getSimpleName();
202        }
203
204        @ProvidesProperty(TypeContainer.STORAGE_ID_PROPERTY)
205        public Integer propertyStorageId() {
206            return this.jclass != null ? this.jclass.getStorageId() : null;
207        }
208
209        @ProvidesProperty(TypeContainer.TYPE_PROPERTY)
210        public SizedLabel proeprtyType() {
211            return new SizedLabel("<code>" + this.getType().toString() + "</code>", ContentMode.HTML);
212        }
213    }
214
215// Container.Hierarchical methods
216
217    @Override
218    public Collection<Class<?>> getChildren(Object itemId) {
219        final Node node = this.getJavaObject(itemId);
220        return node != null ? Lists.transform(node.getChilds(), Node.TYPE_FUNCTION) : Collections.<Class<?>>emptySet();
221    }
222
223    @Override
224    public Class<?> getParent(Object itemId) {
225        final Node child = this.getJavaObject(itemId);
226        if (child == null)
227            return null;
228        final Node parent = child.getParent();
229        if (parent == null)
230            return null;
231        return parent.getType();
232    }
233
234    @Override
235    public Collection<Class<?>> rootItemIds() {
236        return Lists.transform(this.rootList, Node.TYPE_FUNCTION);
237    }
238
239    @Override
240    public boolean areChildrenAllowed(Object itemId) {
241        final Node node = this.getJavaObject(itemId);
242        return node != null && !node.getChilds().isEmpty();
243    }
244
245    @Override
246    public boolean isRoot(Object itemId) {
247        final Node node = this.getJavaObject(itemId);
248        return node != null && node.getParent() == null;
249    }
250
251    @Override
252    public boolean hasChildren(Object itemId) {
253        final Node node = this.getJavaObject(itemId);
254        return node != null && !node.getChilds().isEmpty();
255    }
256
257// Mutators are not supported
258
259    @Override
260    public boolean setParent(Object itemId, Object newParentId) {
261        throw new UnsupportedOperationException();
262    }
263
264    @Override
265    public boolean setChildrenAllowed(Object itemId, boolean areChildrenAllowed) {
266        throw new UnsupportedOperationException();
267    }
268
269    @Override
270    public boolean removeItem(Object itemId) {
271        throw new UnsupportedOperationException();
272    }
273}
