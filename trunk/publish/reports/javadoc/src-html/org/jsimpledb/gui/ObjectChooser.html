031/**
032 * Contains widgets that allow choosing an object reference. Supports searching and filtering.
033 */
034@SuppressWarnings({ "serial", "deprecation" })
035public class ObjectChooser implements Property.ValueChangeNotifier {
036
037    protected final Logger log = LoggerFactory.getLogger(this.getClass());
038
039    private final Button showButton = new Button("Show", new Button.ClickListener() {
040        @Override
041        public void buttonClick(Button.ClickEvent event) {
042            ObjectChooser.this.showButtonClicked();
043        }
044    });
045
046    private final JSimpleDB jdb;
047    private final ParseSession session;
048    private final boolean showFields;
049    private final TypeContainer typeContainer;
050    private final ObjectContainer objectContainer;
051    private final HashSet<Property.ValueChangeListener> listeners = new HashSet<>();
052
053    private final HorizontalSplitPanel splitPanel = new HorizontalSplitPanel();
054    private final FormLayout showForm = new FormLayout();
055    private final TypeTable typeTable;
056    private final TextArea exprField = new TextArea();
057    private final ComboBox sortComboBox = new ComboBox();
058
059    private ObjectTable objectTable;
060    private SortKeyContainer sortKeyContainer;
061
062    /**
063     * Constructor.
064     *
065     * @param jdb database
066     * @param session session for evaluating expressions
067     * @param type type restriction, or null for none
068     * @param showFields true to show all object fields, false for just reference label
069     */
070    public ObjectChooser(JSimpleDB jdb, ParseSession session, Class<?> type, boolean showFields) {
071        if (jdb == null)
072            throw new IllegalArgumentException("null jdb");
073        if (session == null)
074            throw new IllegalArgumentException("null session");
075        this.jdb = jdb;
076        this.session = session;
077        this.showFields = showFields;
078        this.typeContainer = new TypeContainer(this.jdb, type);
079        this.typeTable = new TypeTable(this.typeContainer);
080        this.objectContainer = new ObjectContainer(this.jdb, type, this.session);
081
082        // Build object panel
083        this.splitPanel.setWidth("100%");
084        this.splitPanel.setHeight(300, Sizeable.Unit.PIXELS);
085        this.splitPanel.setFirstComponent(this.typeTable);
086        this.splitPanel.setSecondComponent(new SizedLabel(" "));
087        this.splitPanel.setSplitPosition(20);
088
089        // Build show form
090        this.showForm.setMargin(false);
091        this.showForm.setWidth("100%");
092
093        // Add "Show all by" combo box
094        final HorizontalLayout sortLayout = new HorizontalLayout();
095        sortLayout.setCaption("Show all by:");
096        sortLayout.setSpacing(true);
097        sortLayout.setMargin(false);
098        this.sortComboBox.setNullSelectionAllowed(false);
099        this.sortComboBox.setNewItemsAllowed(false);
100        this.sortComboBox.setTextInputAllowed(false);
101        this.sortComboBox.setFilteringMode(FilteringMode.OFF);
102        this.sortComboBox.setItemCaptionMode(AbstractSelect.ItemCaptionMode.PROPERTY);
103        this.sortComboBox.setItemCaptionPropertyId(SortKeyContainer.DESCRIPTION_PROPERTY);
104        this.sortComboBox.setImmediate(true);
105        sortLayout.addComponent(this.sortComboBox);
106        this.showForm.addComponent(sortLayout);
107
108        // Add show/expression field
109        this.exprField.setCaption("Expression:");
110        this.exprField.setRows(4);
111        this.exprField.setWidth("100%");
112        this.exprField.addStyleName("jsdb-fixed-width");
113        this.showForm.addComponent(this.exprField);
114        this.showForm.addComponent(this.showButton);
115
116        // Listen to type selections
117        this.typeTable.addValueChangeListener(new Property.ValueChangeListener() {
118            @Override
119            public void valueChange(Property.ValueChangeEvent event) {
120                ObjectChooser.this.selectType((TypeToken<?>)event.getProperty().getValue(), false);
121            }
122        });
123
124        // Listen to "Show all by" selection
125        this.sortComboBox.addValueChangeListener(new Property.ValueChangeListener() {
126            @Override
127            public void valueChange(Property.ValueChangeEvent event) {
128                ObjectChooser.this.selectSort((SortKeyContainer.SortKey)event.getProperty().getValue());
129            }
130        });
131
132        // Populate table
133        this.selectType(TypeToken.of(type != null ? type : Object.class), true);
134    }
135
136    /**
137     * Get the object container for this instance.
138     */
139    public ObjectContainer getObjectContainer() {
140        return this.objectContainer;
141    }
142
143    /**
144     * Get the current type restriction, if any.
145     */
146    public Class<?> getType() {
147        return this.objectContainer.getType();
148    }
149
150    /**
151     * Get the type table.
152     */
153    public TypeTable getTypeTable() {
154        return this.typeTable;
155    }
156
157    /**
158     * Get the component containing the type chooser and object panel.
159     */
160    public HorizontalSplitPanel getObjectPanel() {
161        return this.splitPanel;
162    }
163
164    /**
165     * Get the currently selected object, if any.
166     */
167    public ObjId getObjId() {
168        return this.objectTable != null ? (ObjId)this.objectTable.getValue() : null;
169    }
170
171    /**
172     * Get the form with the sort chooser and expression text area.
173     */
174    public FormLayout getShowForm() {
175        return this.showForm;
176    }
177
178    /**
179     * Get the {@link JClass} corresponding to the currently selected type, if any.
180     */
181    public JClass<?> getJClass() {
182        final Class<?> type = this.objectContainer.getType();
183        if (type == null)
184            return null;
185        try {
186            return this.jdb.getJClass(type);
187        } catch (IllegalArgumentException e) {
188            return null;
189        }
190    }
191
192// Property.ValueChangeNotifier
193
194    @Override
195    public void addValueChangeListener(Property.ValueChangeListener listener) {
196        if (listener == null)
197            throw new IllegalArgumentException("null listener");
198        this.listeners.add(listener);
199        if (this.objectTable != null)
200            this.objectTable.addValueChangeListener(listener);
201    }
202
203    @Override
204    public void removeValueChangeListener(Property.ValueChangeListener listener) {
205        if (listener == null)
206            return;
207        this.listeners.remove(listener);
208        if (this.objectTable != null)
209            this.objectTable.removeValueChangeListener(listener);
210    }
211
212    @Override
213    public void addListener(Property.ValueChangeListener listener) {
214        this.addValueChangeListener(listener);
215    }
216
217    @Override
218    public void removeListener(Property.ValueChangeListener listener) {
219        this.removeValueChangeListener(listener);
220    }
221
222// GUI Updates
223
224    // Invoked when a type is clicked on
225    private void selectType(TypeToken<?> typeToken, boolean force) {
226
227        // Anything to do?
228        if (typeToken == null)
229            return;
230        if (!this.setNewType(!typeToken.equals(TypeToken.of(Object.class)) ? typeToken.getRawType() : null, force))
231            return;
232
233        // Rebuild the sort combobox
234        final JClass<?> jclass = this.getJClass();
235        this.sortKeyContainer = jclass != null ?
236          new SortKeyContainer(this.jdb, jclass) : new SortKeyContainer(this.jdb, this.objectContainer.getType());
237        this.sortComboBox.setContainerDataSource(this.sortKeyContainer);
238
239        // Default to sorting by object ID
240        this.sortComboBox.setValue(this.sortKeyContainer.new ObjectIdSortKey());
241        this.selectSort((SortKeyContainer.SortKey)this.sortComboBox.getValue());
242        this.showObjects();
243    }
244
245    // Invoked when a sort choice is made
246    private void selectSort(SortKeyContainer.SortKey sortKey) {
247        if (sortKey == null)
248            return;
249        this.exprField.setValue(sortKey.getExpression(null, false));        // TODO: starting point and sort order
250        this.showObjects();
251    }
252
253    // Invoked when "Show" button is clicked
254    private void showButtonClicked() {
255        this.setNewType(null, false);
256        this.typeTable.setValue(null);
257        this.showObjects();
258    }
259
260    private void showObjects() {
261        this.objectContainer.setContentExpression(this.exprField.getValue());
262    }
263
264    private boolean setNewType(Class<?> type, boolean force) {
265        final Class<?> currentType = this.objectContainer.getType();
266        if (!force && (currentType != null ? currentType.equals(type) : type == null))
267            return false;
268        if (this.objectTable != null) {
269            for (Property.ValueChangeListener listener : this.listeners)
270                this.objectTable.removeValueChangeListener(listener);
271            this.splitPanel.removeComponent(this.objectTable);
272        }
273        this.objectContainer.setType(type);
274        this.objectTable = new ObjectTable(this.jdb, this.objectContainer, this.session, this.showFields);
275        for (Property.ValueChangeListener listener : this.listeners)
276            this.objectTable.addValueChangeListener(listener);
277        this.splitPanel.setSecondComponent(this.objectTable);
278        this.objectTable.setValue(null);
279        final Property.ValueChangeEvent event = new Property.ValueChangeEvent() {
280            @Override
281            public Property getProperty() {
282                return ObjectChooser.this.objectTable;
283            }
284        };
285        for (Property.ValueChangeListener listener : new HashSet<Property.ValueChangeListener>(this.listeners)) {
286            try {
287                listener.valueChange(event);
288            } catch (Exception e) {
289                this.log.error("exception thrown by value change listener", e);
290            }
291        }
292        return true;
293    }
