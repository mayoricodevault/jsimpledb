118    public MainPanel(GUIConfig guiConfig) {
119        this.guiConfig = guiConfig;
120        this.jdb = this.guiConfig.getJSimpleDB();
121
122        // Setup parse session
123        this.session = new ParseSession(this.jdb) {
124            @Override
125            protected void reportException(Exception e) {
126                Notification.show("Error: " + e.getMessage(), null, Notification.Type.ERROR_MESSAGE);
127            }
128        };
129        this.session.setReadOnly(this.guiConfig.isReadOnly());
130        this.session.setSchemaModel(this.jdb.getSchemaModel());
131        this.session.setSchemaVersion(this.guiConfig.getSchemaVersion());
132        this.session.setAllowNewSchema(this.guiConfig.isAllowNewSchema());
133        for (Class<?> cl : this.guiConfig.getFunctionClasses()) {
134            try {
135                this.session.registerFunction(cl);
136            } catch (IllegalArgumentException e) {
137                this.log.warn("failed to register function " + cl + ": " + e.getMessage());
138            }
139        }
140
141        // Setup object chooser
142        this.objectChooser = new ObjectChooser(this.jdb, this.session, null, true);
143
144        // Listen to object selections
145        this.objectChooser.addValueChangeListener(new Property.ValueChangeListener() {
146            @Override
147            public void valueChange(Property.ValueChangeEvent event) {
148                MainPanel.this.selectObject(MainPanel.this.objectChooser.getObjId());
149            }
150        });
151    }

194    // Invoked when an object is clicked on
195    protected void selectObject(ObjId id) {
196
197        // New button
198        this.newButton.setEnabled(this.objectChooser.getJClass() != null);
199
200        // Update buttons
201        if (id == null) {
202            this.editButton.setEnabled(false);
203            this.deleteButton.setEnabled(false);
204            this.upgradeButton.setEnabled(false);
205        } else {
206            this.editButton.setEnabled(true);
207            this.deleteButton.setEnabled(true);
208            this.upgradeButton.setEnabled(this.canUpgrade(id));
209        }
210    }

214    private void editButtonClicked(ObjId id) {
215        this.log.info("editing object " + id);
216
217        // Copy object
218        final JObject jobj = this.doCopyForEdit(id);
219        if (jobj == null) {
220            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
221            return;
222        }
223
224        // Ensure type is known
225        final int storageId = id.getStorageId();
226        final JClass<?> jclass;
227        try {
228            jclass = this.jdb.getJClass(storageId);
229        } catch (UnknownTypeException e) {
230            Notification.show("Can't edit object " + id + " having unknown type",
231              e.getMessage(), Notification.Type.WARNING_MESSAGE);
232            return;
233        }
234
235        // Open window
236        new EditWindow(jobj, jclass, false).show();
237    }

239    @RetryTransaction
240    @Transactional("jsimpledbGuiTransactionManager")
241    private JObject doCopyForEdit(ObjId id) {
242        JObject jobj = JTransaction.getCurrent().getJObject(id);
243        if (!jobj.exists())
244            return null;
245        return jobj.copyOut((String[])null);
246    }

251        final JClass<?> jclass = this.objectChooser.getJClass();

267// Delete

269    private void deleteButtonClicked(ObjId id) {
270        this.log.info("deleting object " + id);
271        final boolean deleted;
272        try {
273            deleted = this.doDelete(id);
274        } catch (DeletedObjectException e) {
275            Notification.show("Object " + e.getId() + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
276            return;
277        } catch (ReferencedObjectException e) {
278            Notification.show("Object " + id + " is referenced by " + e.getReferrer(),
279              e.getMessage(), Notification.Type.ERROR_MESSAGE);
280            return;
281        }
282        if (deleted)
283            Notification.show("Removed object " + id);
284        else
285            Notification.show("Could not delete object " + id, null, Notification.Type.WARNING_MESSAGE);
286        this.selectObject(null);
287    }

289    @RetryTransaction
290    @Transactional("jsimpledbGuiTransactionManager")
291    private boolean doDelete(ObjId id) {
292        final JObject jobj = JTransaction.getCurrent().getJObject(id);
293        final boolean deleted = jobj.delete();
294        if (deleted && this.changePublisher != null)
295            this.changePublisher.publishChangeOnCommit(new ObjectDelete<Object>(jobj));
296        return deleted;
297    }

299// Upgrade
300
301    private void upgradeButtonClicked(ObjId id) {
302        final int newVersion = this.jdb.getActualVersion();
303        this.log.info("upgrading object " + id + " to schema version " + newVersion);
304        final int oldVersion = this.doUpgrade(id);
305        switch (oldVersion) {
306        case -1:
307            Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
308            break;
309        case 0:
310            Notification.show("Object " + id + " was already upgraded", null, Notification.Type.WARNING_MESSAGE);
311            break;
312        default:
313            Notification.show("Upgraded object " + id + " version from " + oldVersion + " to " + newVersion);
314            break;
315        }
316    }
317
318    @RetryTransaction
319    @Transactional("jsimpledbGuiTransactionManager")
320    private int doUpgrade(ObjId id) {
321        final JObject jobj = JTransaction.getCurrent().getJObject(id);
322        final int oldVersion;
323        try {
324            oldVersion = jobj.getSchemaVersion();
325        } catch (DeletedObjectException e) {
326            return -1;
327        }
328        final boolean upgraded = jobj.upgrade();
329        if (upgraded && this.changePublisher != null)
330            this.changePublisher.publishChangeOnCommit(jobj);
331        return upgraded ? oldVersion : 0;
332    }
333
334    @RetryTransaction
335    @Transactional("jsimpledbGuiTransactionManager")
336    private boolean canUpgrade(ObjId id) {
337        final JObject jobj = JTransaction.getCurrent().getJObject(id);
338        return jobj.exists() && jobj.getSchemaVersion() != this.jdb.getActualVersion();
339    }

343    public class EditWindow extends ConfirmWindow {
344
345        private final JObject jobj;
346        private final JClass<?> jclass;
347        private final boolean create;
348        private final FieldGroup fieldGroup = new FieldGroup();
349        private final LinkedHashMap<String, Editor> editorMap = new LinkedHashMap<>();
350
351        EditWindow(JObject jobj, JClass<?> jclass, boolean create) {
352            super(MainPanel.this.getUI(), (create ? "New" : "Edit") + " " + jclass.getName());
353            this.setWidth(600, Sizeable.Unit.PIXELS);
354            this.setHeight(450, Sizeable.Unit.PIXELS);
355            this.jobj = jobj;
356            this.jclass = jclass;
357            this.create = create;
358
359            // Introspect for any @FieldBuilder.* annotations
360            for (Map.Entry<String, AbstractField<?>> entry : new FieldBuilder().buildBeanPropertyFields(this.jobj).entrySet()) {
361                final String fieldName = entry.getKey();
362                final AbstractField<?> field = entry.getValue();
363                this.editorMap.put(fieldName, new Editor(field));
364            }
365
366            // Build editors for all remaining database properties
367            final SortedMap<String, JField> jfieldMap = jclass.getJFieldsByName();
368            for (Map.Entry<String, JField> entry : jfieldMap.entrySet()) {
369                final String fieldName = entry.getKey();
370                if (this.editorMap.containsKey(fieldName))
371                    continue;
372                final JField jfield = entry.getValue();
373                final Editor editor = jfield instanceof JSimpleField ?
374                  this.buildSimpleFieldEditor((JSimpleField)jfield) : new Editor(new Label("TODO: " + jfield));       // TODO
375                editorMap.put(fieldName, editor);
376            }
377
378            // Create BeanItem exposing fields' properties
379            this.fieldGroup.setItemDataSource(new BeanItem<JObject>(this.jobj,
380              Maps.filterValues(this.editorMap, new Predicate<Editor>() {
381                @Override
382                public boolean apply(Editor editor) {
383                    return editor.getField() != null;
384                }
385            }).keySet()));
386
387            // Bind fields into FieldGroup
388            for (Map.Entry<String, Editor> entry : this.editorMap.entrySet()) {
389                final Editor editor = entry.getValue();
390                if (editor.getField() != null)
391                    this.fieldGroup.bind(editor.getField(), entry.getKey());
392            }
393        }
394
395        @Override
396        protected void addContent(VerticalLayout layout) {
397            if (!this.create) {
398                layout.addComponent((Component)MainPanel.this.objectChooser.getObjectContainer().getContainerProperty(
399                  this.jobj.getObjId(), JObjectContainer.REFERENCE_LABEL_PROPERTY).getValue());
400            }
401            final FormLayout formLayout = new FormLayout();
402            for (Editor editor : this.editorMap.values())
403                formLayout.addComponent(editor.getComponent());
404            layout.addComponent(formLayout);
405        }
406
407        @Override
408        protected boolean execute() {
409
410            // Commit fields in field group
411            try {
412                this.fieldGroup.commit();
413            } catch (FieldGroup.CommitException e) {
414                Notification.show("Invalid value(s)", null, Notification.Type.WARNING_MESSAGE);
415                throw new RuntimeException(e);
416            }
417
418            // Commit transaction, if possible
419            try {
420                return this.writeBack();
421            } catch (UnexpectedRollbackException e) {
422                MainPanel.this.log.debug("ignoring UnexpectedRollbackException presumably caused by validation failure");
423                return false;
424            }
425        }
426
427        @RetryTransaction
428        @Transactional("jsimpledbGuiTransactionManager")
429        protected boolean writeBack() {
430            final JTransaction jtx = JTransaction.getCurrent();
431
432            // Find/create object
433            final ObjId id = this.jobj.getObjId();
434            final JObject target = this.create ? (JObject)jtx.create(this.jclass) : jtx.getJObject(id);
435
436            // Verify object still exists when editing
437            if (!this.create && !target.exists()) {
438                Notification.show("Object " + id + " no longer exists", null, Notification.Type.WARNING_MESSAGE);
439                return true;
440            }
441
442            // Copy fields
443            this.jobj.copyTo(jtx, target.getObjId(), new ObjIdSet());
444
445            // Run validation queue
446            try {
447                jtx.validate();
448            } catch (ValidationException e) {
449                Notification.show("Validation failed", e.getMessage(), Notification.Type.ERROR_MESSAGE);
450                jtx.getTransaction().setRollbackOnly();
451                return false;
452            }
453
454            // Broadcast update event after successful commit
455            if (MainPanel.this.changePublisher != null) {
456                if (create)
457                    MainPanel.this.changePublisher.publishChangeOnCommit(new ObjectCreate<Object>(target));
458                else
459                    MainPanel.this.changePublisher.publishChangeOnCommit(target);
460            }
461
462            // Show notification after successful commit
463            final VaadinSession vaadinSession = VaadinUtil.getCurrentSession();
464            jtx.getTransaction().addCallback(new Transaction.CallbackAdapter() {
465                @Override
466                public void afterCommit() {
467                    VaadinUtil.invoke(vaadinSession, new Runnable() {
468                        @Override
469                        public void run() {
470                            Notification.show((EditWindow.this.create ? "Created" : "Updated") + " object " + id);
471                        }
472                    });
473                }
474            });
475            return true;
476        }

480        protected Editor buildSimpleFieldEditor(JSimpleField jfield) {
481
482            // Get field info
483            final boolean allowNull = jfield.getGetter().getAnnotation(NotNull.class) == null
484              && !jfield.getGetter().getReturnType().isPrimitive();
485
486            // Get the property we want to edit
487            final Property<?> property = this.buildProperty(jfield);
488
489            // Build editor
490            final Editor editor = this.buildSimpleFieldEditor(jfield, property, allowNull);
491            editor.getComponent().setCaption(this.buildCaption(jfield.getName(), !(editor.getComponent() instanceof CheckBox)));
492            return editor;
493        }
494
495        @SuppressWarnings("unchecked")
496        protected Editor buildSimpleFieldEditor(JSimpleField jfield, Property<?> property, boolean allowNull) {
497
498            // Get property type
499            final Class<?> propertyType = jfield.getGetter().getReturnType();
500
501            // Use object choosers for references
502            if (jfield instanceof JReferenceField) {
503                final JReferenceField refField = (JReferenceField)jfield;
504                final ObjectEditor objectEditor = new ObjectEditor(this.jobj.getTransaction(),
505                  MainPanel.this.session, refField.getGetter().getReturnType(), (Property<JObject>)property, allowNull);
506                return new Editor(objectEditor);
507            }
508
509            // Use ComboBox for Enum's
510            if (Enum.class.isAssignableFrom(propertyType)) {
511                final EnumComboBox comboBox = this.createEnumComboBox(propertyType.asSubclass(Enum.class), allowNull);
512                comboBox.setPropertyDataSource(property);
513                comboBox.setInputPrompt("Null");
514                return new Editor(comboBox);
515            }
516
517            // Use DatePicker for dates
518            if (Date.class.isAssignableFrom(propertyType)) {
519                final PopupDateField dateField = new PopupDateField();
520                dateField.setResolution(Resolution.SECOND);
521                dateField.setPropertyDataSource(property);
522                return new Editor(dateField, allowNull ? this.addNullCheckbox(dateField) : dateField);
523            }
524
525            // Use CheckBox for booleans
526            if (propertyType == boolean.class || propertyType == Boolean.class) {
527                final CheckBox checkBox = new CheckBox();
528                checkBox.setPropertyDataSource(property);
529                return new Editor(checkBox, allowNull ? this.addNullCheckbox(checkBox) : checkBox);
530            }
531
532            // Use text field for all other field types
533            final TextField textField = new TextField();
534            textField.setWidth("100%");
535            textField.setNullRepresentation("");
536            final FieldType<?> fieldType = this.getFieldType(jfield);
537            textField.setConverter(this.buildSimpleFieldConverter(fieldType));
538            textField.setPropertyDataSource(property);
539            return new Editor(textField, allowNull ? this.addNullCheckbox(textField) : textField);
540        }

542        private HorizontalLayout addNullCheckbox(final AbstractField<?> field) {
543
544            // Set up checkbox
545            final CheckBox checkBox = new CheckBox("Null");
546            checkBox.addValueChangeListener(new Property.ValueChangeListener() {
547                @Override
548                public void valueChange(Property.ValueChangeEvent event) {
549                    final boolean setNull = checkBox.getValue();
550                    field.setEnabled(!setNull);
551                    if (setNull)
552                        field.setValue(null);
553                }
554            });
555
556            // Build layout
557            final HorizontalLayout layout = new HorizontalLayout() {
558                @Override
559                public void attach() {
560                    super.attach();
561                    final Property<?> property = field.getPropertyDataSource();
562                    final boolean isNull = property.getValue() == null;
563                    checkBox.setValue(isNull);
564                    field.setEnabled(!isNull);
565                }
566            };
567            layout.setSpacing(true);
568            layout.setMargin(false);
569            layout.addComponent(field);
570            layout.addComponent(checkBox);
571            return layout;
572        }

574        @SuppressWarnings({ "unchecked", "rawtypes" })
575        private Property<?> buildProperty(JSimpleField jfield) {
576            return new MethodProperty(jfield.getGetter().getReturnType(), this.jobj, jfield.getGetter(), jfield.getSetter());
577        }

580        private <T extends Enum> EnumComboBox createEnumComboBox(Class<T> enumType, boolean allowNull) {
581            return new EnumComboBox(enumType, allowNull);
582        }

584        private <T> SimpleFieldConverter<T> buildSimpleFieldConverter(FieldType<T> fieldType) {
585            return new SimpleFieldConverter<T>(fieldType);

587
588        private FieldType<?> getFieldType(JSimpleField jfield) {
589            return MainPanel.this.jdb.getDatabase().getFieldTypeRegistry().getFieldType(jfield.getTypeName());
590        }

592        private String buildCaption(String fieldName, boolean includeColon) {
593            return DefaultFieldFactory.createCaptionByPropertyId(fieldName) + (includeColon ? ":" : "");
594        }

597    private static class Editor {
598
599        private final AbstractField<?> field;
600        private final Component component;
601
602        Editor(Component component) {
603            this(null, component);
604        }
605
606        Editor(AbstractField<?> field) {
607            this(field, field);
608        }
609
610        Editor(AbstractField<?> field, Component component) {
611            if (component == null)
612                throw new IllegalArgumentException("null component");
613            this.field = field;
614            this.component = component;
615        }
616
617        public AbstractField<?> getField() {
618            return this.field;
619        }
620
621        public Component getComponent() {
622            return this.component;
623        }
624    }
