008package org.jsimpledb.gui;
009
010import org.dellroad.stuff.vaadin7.VaadinConfigurable;
011import org.jsimpledb.JObject;
012import org.jsimpledb.JTransaction;
013import org.jsimpledb.change.Change;
014import org.jsimpledb.change.SimpleFieldChange;
015import org.jsimpledb.core.ObjId;
016import org.jsimpledb.core.Transaction;
017import org.springframework.beans.factory.annotation.Autowired;
018import org.springframework.beans.factory.annotation.Qualifier;
019import org.springframework.context.event.ApplicationEventMulticaster;
020import org.springframework.stereotype.Component;
021
022/**
023 * Publishes change notifications.
024 */
025@Component
026@VaadinConfigurable
027public class ChangePublisher {
028
029    @Autowired
030    @Qualifier("jsimpledbGuiEventMulticaster")
031    private ApplicationEventMulticaster eventMulticaster;
032
033    /**
034     * Publish the given {@link Change} if the current transaction is successful.
035     */
036    public void publishChangeOnCommit(Change<?> change) {
037        JTransaction.getCurrent().getTransaction().addCallback(
038          new PublishChangeCallback(this.eventMulticaster, new DataChangeEvent(this, change)));
039    }
040
041    /**
042     * Publish that the specified object has changed if the current transaction is successful.
043     */
044    public void publishChangeOnCommit(JObject jobj) {
045
046        // We rely on the fact that publishing a change in any property reloads all properties for that object
047        this.publishChangeOnCommit(new SimpleFieldChange<Object, ObjId>(jobj,
048          Integer.MAX_VALUE, ObjectContainer.OBJECT_ID_PROPERTY, jobj.getObjId(), jobj.getObjId()));
049    }
050
051// PublishChangeCallback - notifies the rest of the application when a data instance has been added/removed/changed
052
053    private static class PublishChangeCallback extends Transaction.CallbackAdapter {
054
055        private final ApplicationEventMulticaster eventMulticaster;
056        private final DataChangeEvent event;
057
058        PublishChangeCallback(ApplicationEventMulticaster eventMulticaster, DataChangeEvent event) {
059            if (eventMulticaster == null)
060                throw new IllegalArgumentException("null eventMulticaster");
061            if (event == null)
062                throw new IllegalArgumentException("null event");
063            this.eventMulticaster = eventMulticaster;
064            this.event = event;
065        }
066
067        @Override
068        public void afterCommit() {
069            this.eventMulticaster.multicastEvent(this.event);
070        }
071
072        @Override
073        public boolean equals(Object obj) {
074            if (obj == null || obj.getClass() != this.getClass())
075                return false;
076            final PublishChangeCallback that = (PublishChangeCallback)obj;
077            return this.event.equals(that.event);
078        }
079
080        @Override
081        public int hashCode() {
082            return this.event.hashCode();
083        }
084    }
085}
