058/**
059 * A transaction associated with a {@link JSimpleDB} instance.
060 *
061 * <p>
062 * Commonly used methods in this class can be divided into the following categories:
063 * </p>
064 *
065 * <p>
066 * <b>Transaction Meta-Data</b>
067 * <ul>
068 *  <li>{@link #getJSimpleDB getJSimpleDB()} - Get the associated {@link JSimpleDB} instance</li>
069 *  <li>{@link #getTransaction} - Get the core API {@link Transaction} underlying this instance</li>
070 * </ul>
071 * </p>
072 *
073 * <p>
074 * <b>Transaction Lifecycle</b>
075 * <ul>
076 *  <li>{@link #commit commit()} - Commit transaction</li>
077 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
078 *  <li>{@link #getCurrent getCurrent()} - Get the {@link JTransaction} instance associated with the current thread</li>
079 *  <li>{@link #setCurrent setCurrent()} - Set the {@link JTransaction} instance associated with the current thread</li>
080 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
081 * </ul>
082 * </p>
083 *
084 * <p>
085 * <b>Object Access</b>
086 * <ul>
087 *  <li>{@link #getJObject(ObjId) getJObject()} - Get the Java model object corresponding to a specific database object ID</li>
088 *  <li>{@link #create(Class) create()} - Create a new database object</li>
089 *  <li>{@link #getAll getAll()} - Get all database objects that are instances of a given Java type</li>
090 *  <li>{@link #queryVersion queryVersion()} - Get database objects grouped according to their schema versions</li>
091 * </ul>
092 * </p>
093 *
094 * <p>
095 * <b>Validation</b>
096 * <ul>
097 *  <li>{@link #validate validate()} - Validate all objects in the validation queue</li>
098 *  <li>{@link #resetValidationQueue} - Clear the validation queue</li>
099 * </ul>
100 * </p>
101 *
102 * <p>
103 * <b>Index Queries</b>
104 * <ul>
105 *  <li>{@link #queryIndex(Class, String, Class) queryIndex()}
106 *      - Access any indexed field's index for containing objects</li>
107 *  <li>{@link #queryListFieldEntries(Class, String, Class) queryListFieldEntries()}
108 *      - Access an indexed list field's index for {@link ListIndexEntry}s</li>
109 *  <li>{@link #queryMapFieldKeyEntries(Class, String, Class, Class) queryMapFieldKeyEntries()}
110 *      - Access an indexed map field's key index for {@link MapKeyIndexEntry}s</li>
111 *  <li>{@link #queryMapFieldKeyEntries(Class, String, Class, Class) queryMapFieldKeyEntries()}
112 *      - Access an indexed map field's value index for {@link MapValueIndexEntry}s</li>
113 * </ul>
114 * </p>
115 *
116 * <p>
117 * <b>Reference Inversion</b>
118 * <ul>
119 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
120 *      of objects through a specified reference path</li>
121 *  </li>
122 * </ul>
123 * </p>
124 *
125 * <p>
126 * <b>Snapshot Transactions</b>
127 * <ul>
128 *  <li>{@link #getSnapshotTransaction getSnapshotTransaction()} - Get the default {@link SnapshotJTransaction}</li>
129 *  <li>{@link #copyTo(JTransaction, JObject, ObjId, String[]) copyTo()} - Copy an object
130 *      and its related objects into another transaction</li>
131 *  <li>{@link #copyTo(JTransaction, Iterable) copyTo()} - Copy explicitly specified objects into another transaction</li>
132 * </ul>
133 * </p>
134 *
135 * <p>
136 * <b>Lower Layer Access</b>
137 * <ul>
138 *  <li>{@link #getKey(JObject) getKey()} - Get the {@link org.jsimpledb.kv.KVDatabase} key prefix for a specific object</li>
139 *  <li>{@link #getKey(JObject, String) getKey()} - Get the {@link org.jsimpledb.kv.KVDatabase}
140 *      key for a specific field in a specific object</li>
141 * </ul>
142 * </p>
143 *
144 * <p>
145 * The remaining methods in this class are normally only used by generated Java model object subclasses.
146 * Instead of using these methods directly, using the appropriately annotated Java model object method
147 * or {@link JObject} interface method is recommended.
148 * </p>
149 *
150 * <p>
151 * <b>Java Model Object Methods</b>
152 * <ul>
153 *  <li>{@link #readSimpleField readSimpleField()} - Read the value of a simple field</li>
154 *  <li>{@link #writeSimpleField writeSimpleField()} - Write the value of a simple field</li>
155 *  <li>{@link #readCounterField readCounterField()} - Access a {@link Counter} field</li>
156 *  <li>{@link #readSetField readSetField()} - Access a set field</li>
157 *  <li>{@link #readListField readListField()} - Access a list field</li>
158 *  <li>{@link #readMapField readMapField()} - Access a map field</li>
159 *  <li>{@link #querySimpleField(int, Class) querySimpleField()} - Query a simple field index by storage ID</li>
160 *  <li>{@link #queryListFieldEntries(int, Class) queryListFieldEntries()} - Query a list field entry index by storage ID</li>
161 *  <li>{@link #queryMapFieldKeyEntries(int, Class) queryMapFieldKeyEntries()}
162 *      - Query a map field key entry index by storage ID</li>
163 *  <li>{@link #queryMapFieldKeyEntries(int, Class) queryMapFieldKeyEntries()}
164 *      - Query a map field value entry index by storage ID</li>
165 * </ul>
166 * </p>
167 *
168 * <p>
169 * <b>{@link JObject} Methods</b>

202    /**
203     * Constructor.
204     *
205     * @throws IllegalArgumentException if any parameter is null
206     */
207    JTransaction(JSimpleDB jdb, Transaction tx, ValidationMode validationMode) {
208
209        // Initialization
210        if (jdb == null)
211            throw new IllegalArgumentException("null jdb");
212        if (tx == null)
213            throw new IllegalArgumentException("null tx");
214        if (validationMode == null)
215            throw new IllegalArgumentException("null validationMode");
216        this.jdb = jdb;
217        this.tx = tx;
218        this.validationMode = validationMode;
219
220        // Register listeners
221        this.tx.addCreateListener(this.internalCreateListener);
222        this.tx.addDeleteListener(this.internalDeleteListener);
223        this.tx.addVersionChangeListener(this.internalVersionChangeListener);
224        for (JClass<?> jclass : this.jdb.jclasses.values()) {
225            for (OnChangeScanner<?>.MethodInfo info : jclass.onChangeMethods) {
226                if (this instanceof SnapshotJTransaction && !info.getAnnotation().snapshotTransactions())
227                    continue;
228                final OnChangeScanner<?>.ChangeMethodInfo changeInfo = (OnChangeScanner<?>.ChangeMethodInfo)info;
229                changeInfo.registerChangeListener(this);
230            }
231        }
232        if (validationMode == ValidationMode.AUTOMATIC) {
233            for (JClass<?> jclass : this.jdb.jclasses.values()) {
234                for (JField jfield : jclass.jfields.values()) {
235                    if (jfield.requiresValidation)
236                        jfield.registerChangeListener(this.tx, new int[0], this.validationListener);
237                }
238            }
239        }
240    }

408            throw new IllegalArgumentException("jobj is not an instance of " + refPath.targetType); // should never happen
409        return this.tx.getKey(jobj.getObjId(), refPath.targetField.storageId);

429    /**
430     * Copy the specified object into the specified destination transaction. The destination object will
431     * be created if necessary. {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
432     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly;
433     * however, the annotation must have {@code snapshotTransactions = true}.
434     *
435     * <p>
436     * Circular references are handled properly: if an object is encountered more than once, it is not copied again.
437     * Does nothing if this instance and {@code dest} are the same instance and {@code srcId} and {@code dstId} are the same.
438     * </p>
439     *
440     * <p>
441     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
442     * must be identical in both transactions.
443     * </p>
444     *
445     * <p>
446     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
447     * deadlock could result.
448     * </p>
449     *
450     * <p>
451     * This method is typically only used by generated classes; normally, {@link JObject#copyIn}, {@link JObject#copyOut},
452     * or {@link JObject#copyTo} would be used instead.
453     * </p>
454     *
455     * @param dest destination transaction
456     * @param srcObj source object
457     * @param dstId target object ID, or null for the object ID of {@code srcObj}
458     * @param refPaths zero or more reference paths that refer to additional objects to be copied
459     * @return the copied object, i.e., the object having ID {@code dstId} in {@code dest}
460     * @throws DeletedObjectException if {@code srcObj} does not exist in this transaction
461     * @throws org.jsimpledb.core.SchemaMismatchException if the schema corresponding to {@code srcObj}'s object's version
462     *  is not identical in this instance and {@code dest} (as well for any referenced objects)
463     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
464     * @throws ReadOnlyTransactionException if {@code dest}'s underlying transaction
465     *  is {@linkplain Transaction#setReadOnly set read-only}
466     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
467     * @throws IllegalArgumentException if any parameter is null
468     * @see JObject#copyTo JObject.copyTo()
469     * @see JObject#copyOut JObject.copyOut()
470     * @see JObject#copyIn JObject.copyIn()
471     * @see #copyTo(JTransaction, Iterable)
472     */
473    public JObject copyTo(JTransaction dest, JObject srcObj, ObjId dstId, String... refPaths) {
474
475        // Sanity check
476        if (dest == null)
477            throw new IllegalArgumentException("null destination transaction");
478        if (srcObj == null)
479            throw new IllegalArgumentException("null srcObj");
480
481        // Handle possible re-entrant object cache load
482        srcObj.getTransaction().getJObjectCache().registerJObject(srcObj);
483
484        // Get source and dest ID
485        final ObjId srcId = srcObj.getObjId();
486        if (dstId == null)
487            dstId = srcId;
488
489        // Check trivial case
490        if (this.tx == dest.tx && srcId.equals(dstId))
491            return dest.getJObject(dstId);
492
493        // Copy "related objects" if a null reference path is given
494        final Iterable<? extends JObject> relatedObjects = refPaths == null ? srcObj.getRelatedObjects() : null;
495        if (refPaths == null)
496            refPaths = new String[0];
497
498        // Parse paths
499        final TypeToken<?> startType = this.jdb.getJClass(srcId.getStorageId()).typeToken;
500        final HashSet<ReferencePath> paths = new HashSet<>(refPaths.length);
501        for (String refPath : refPaths) {
502
503            // Parse refernce path
504            if (refPath == null)
505                throw new IllegalArgumentException("null refPath");
506            final ReferencePath path = this.jdb.parseReferencePath(startType, refPath, null);
507
508            // Verify target field is a reference field; convert a complex target field into its reference sub-field(s)
509            final String lastFieldName = refPath.substring(refPath.lastIndexOf('.') + 1);
510            final JField targetField = this.jdb.jfields.get(path.getTargetField());
511            if (targetField instanceof JComplexField) {
512                final JComplexField superField = (JComplexField)targetField;
513                boolean foundReferenceSubField = false;
514                for (JSimpleField subField : superField.getSubFields()) {
515                    if (subField instanceof JReferenceField) {
516                        paths.add(this.jdb.parseReferencePath(startType,
517                          refPath + "." + superField.getSubFieldName(subField), true));
518                        foundReferenceSubField = true;
519                    }
520                }
521                if (!foundReferenceSubField) {
522                    throw new IllegalArgumentException("the last field `" + lastFieldName
523                      + "' of path `" + refPath + "' does not contain any reference sub-fields");
524                }
525            } else {
526                if (!(targetField instanceof JReferenceField)) {
527                    throw new IllegalArgumentException("the last field `" + lastFieldName
528                      + "' of path `" + path + "' is not a reference field");
529                }
530                paths.add(path);
531            }
532        }
533
534        // Ensure object is copied even when there are zero reference paths
535        final HashSet<ObjId> seen = new HashSet<>();
536        this.copyTo(seen, dest, srcId, dstId, true, new ArrayDeque<JReferenceField>());
537
538        // Copy related objects (if any) if refPaths was null
539        if (relatedObjects != null) {
540            for (JObject jobj : Iterables.filter(relatedObjects, JObject.class)) {
541                final ObjId id = jobj.getObjId();
542                this.copyTo(seen, dest, id, id, false, new ArrayDeque<JReferenceField>());
543            }
544        }
545
546        // Recurse over each reference path
547        for (ReferencePath path : paths) {
548            final int[] storageIds = path.getReferenceFields();
549
550            // Convert reference path, including final target field, into a list of JReferenceFields
551            final ArrayDeque<JReferenceField> fields = new ArrayDeque<>(storageIds.length + 1);
552            for (int storageId : storageIds)
553                fields.add((JReferenceField)this.jdb.jfields.get(storageId));
554            fields.add((JReferenceField)this.jdb.jfields.get(path.getTargetField()));
555
556            // Recurse over this path
557            this.copyTo(seen, dest, srcId, dstId, false/*doesn't matter*/, fields);
558        }
559
560        // Done
561        return dest.getJObject(dstId);
562    }

564    /**
565     * Copy the objects in the specified {@link Iterable} into the specified destination transaction.
566     * Destination objects will be created if necessary, and {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
567     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly;
568     * however, the annotation must have {@code snapshotTransactions = true}.
569     *
570     * <p>
571     * If an object is encountered more than once, it is not copied again.
572     * Does nothing if this instance and {@code dest} are the same instance.
573     * </p>
574     *
575     * <p>
576     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
577     * must be identical in both transactions.
578     * </p>
579     *
580     * <p>
581     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
582     * deadlock could result.
583     * </p>
584     *
585     * @param dest destination transaction
586     * @param jobjs {@link Iterable} returning the objects to copy; null values are ignored
587     * @throws DeletedObjectException if an object in {@code jobjs} does not exist in this transaction
588     * @throws org.jsimpledb.core.SchemaMismatchException if the schema version corresponding to an object in
589     *  {@code jobjs} is not identical in this instance and {@code dest}
590     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
591     * @throws ReadOnlyTransactionException if {@code dest}'s underlying transaction
592     *  is {@linkplain Transaction#setReadOnly set read-only}
593     * @throws IllegalArgumentException if {@code dest} or {@code jobjs} is null
594     * @see #copyTo(JTransaction, JObject, ObjId, String[])
595     */
596    public void copyTo(JTransaction dest, Iterable<? extends JObject> jobjs) {
597
598        // Sanity check
599        if (dest == null)
600            throw new IllegalArgumentException("null dest");
601        if (jobjs == null)
602            throw new IllegalArgumentException("null jobjs");
603
604        // Check trivial case
605        if (this.tx == dest.tx)
606            return;
607
608        // Copy objects
609        final HashSet<ObjId> seen = new HashSet<>();
610        final ArrayDeque<JReferenceField> emptyFields = new ArrayDeque<>();
611        for (JObject jobj : jobjs) {
612
613            // Get next object
614            if (jobj == null)
615                continue;
616
617            // Handle possible re-entrant object cache load
618            jobj.getTransaction().getJObjectCache().registerJObject(jobj);
619
620            // Copy object
621            final ObjId id = jobj.getObjId();
622            this.copyTo(seen, dest, id, id, true, emptyFields);
623        }
624    }
625
626    void copyTo(Set<ObjId> seen, JTransaction dest, ObjId srcId, ObjId dstId, boolean required, Deque<JReferenceField> fields) {
627
628        // Already copied this object?
629        if (!seen.add(dstId))
630            return;
631
632        // Copy current instance
633        try {
634            this.tx.copy(srcId, dstId, dest.tx);
635        } catch (DeletedObjectException e) {
636            if (required)
637                throw e;
638        }
639
640        // Recurse through the next reference field in the path
641        if (fields.isEmpty())
642            return;
643        final JReferenceField jfield = fields.removeFirst();
644        if (jfield.parent instanceof JComplexField) {
645            final JComplexField superField = (JComplexField)jfield.parent;
646            superField.copyRecurse(seen, this, dest, srcId, jfield, fields);
647        } else {
648            assert jfield instanceof JReferenceField;
649            final ObjId referrent = (ObjId)this.tx.readSimpleField(srcId, jfield.storageId, false);
650            if (referrent != null)
651                this.copyTo(seen, dest, referrent, referrent, false, fields);
652        }
653    }

1041     *
1042     * @param startType starting Java type for the path
1043     * @param path dot-separated path of one or more reference fields
1044     * @param targetObjects target objects

1051        if (targetObjects == null)
1052            throw new IllegalArgumentException("null targetObjects");
1053        final ReferencePath refPath = this.jdb.parseReferencePath(TypeToken.of(startType), path, true);
1054        final int targetField = refPath.getTargetField();

1059        final int[] refs = Ints.concat(refPath.getReferenceFields(), new int[] { targetField });
1060        final NavigableSet<ObjId> ids = this.tx.invertReferencePath(refs,
1061          Iterables.transform(targetObjects, this.referenceConverter));
1062        return (NavigableSet<T>)new ConvertedNavigableSet<JObject, ObjId>(ids, this.referenceConverter);

1385     *
1386     * <p>
1387     * Prior to actual commit, if this transaction was created with a validation mode other than {@link ValidationMode#DISABLED},
1388     * {@linkplain #validate validation} of outstanding objects in the validation queue is performed.
1389     * </p>

1402

1413        try {
1414            this.validate();
1415        } catch (ValidationException e) {
1416            this.tx.rollback();

1418        }

1420        // Commit

1422    }

1507    private void doValidate() {
1508        while (true) {
1509
1510            // Get next object
1511            final ObjId id;
1512            synchronized (this) {
1513                final Iterator<ObjId> i = this.validationQueue.iterator();
1514                try {
1515                    id = i.next();
1516                } catch (NoSuchElementException e) {
1517                    return;
1518                }
1519                i.remove();
1520            }
1521
1522            // Does it still exist?
1523            if (!this.tx.exists(id))
1524                continue;
1525
1526            // Do JSR 303 validation
1527            final JObject jobj = this.getJObject(id);
1528            final Set<ConstraintViolation<JObject>> violations = ValidationUtil.validate(jobj);
1529            if (!violations.isEmpty()) {
1530                throw new ValidationException(jobj, violations, "validation error for object " + id + " of type `"
1531                  + this.jdb.jclasses.get(id.getStorageId()).name + "':\n" + ValidationUtil.describe(violations));
1532            }
1533
1534            // Do @Validate validation
1535            final JClass<?> jclass = this.jdb.getJClass(id.getStorageId());
1536            for (ValidateScanner<?>.MethodInfo info : jclass.validateMethods)
1537                Util.invoke(info.getMethod(), jobj);
1538        }
1539    }

1613          int oldVersion, int newVersion, Map<Integer, Object> oldFieldValues) {
1614            Object jobj = null;
1615            final SchemaVersion oldSchema = JTransaction.this.tx.getSchema().getVersion(oldVersion);
1616            for (OnVersionChangeScanner<T>.MethodInfo info : jclass.onVersionChangeMethods) {
1617                final OnVersionChange annotation = info.getAnnotation();
1618                final Method method = info.getMethod();
1619
1620                // Check old & new version numbers
1621                if ((annotation.oldVersion() != 0 && annotation.oldVersion() != oldVersion)
1622                  || (annotation.newVersion() != 0 && annotation.newVersion() != newVersion))
1623                    continue;
1624
1625                // Get Java model object
1626                if (jobj == null)
1627                    jobj = JTransaction.this.getJObject(id);
1628
1629                // Convert old field values so ObjId's become JObjects
1630                final TreeMap<Integer, Object> convertedValues = new TreeMap<>();
1631                for (Map.Entry<Integer, Object> entry : oldFieldValues.entrySet()) {
1632                    final int storageId = entry.getKey();
1633                    Object oldValue = entry.getValue();
1634
1635                    // Convert old field value as needed; we only convert ObjId -> JObject
1636                    final Field<?> field = oldSchema.getSchemaItem(storageId, Field.class);
1637                    oldValue = JTransaction.this.convertOldValue(field, oldValue);
1638
1639                    // Update value
1640                    convertedValues.put(storageId, oldValue);
1641                }
1642
1643                // Invoke method
1644                switch ((annotation.oldVersion() != 0 ? 2 : 0) + (annotation.newVersion() != 0 ? 1 : 0)) {
1645                case 0:
1646                    Util.invoke(method, jobj, oldVersion, newVersion, convertedValues);
1647                    break;
1648                case 1:
1649                    Util.invoke(method, jobj, oldVersion, convertedValues);
1650                    break;
1651                case 2:
1652                    Util.invoke(method, jobj, newVersion, convertedValues);
1653                    break;
1654                case 3:
1655                default:
1656                    Util.invoke(method, jobj, convertedValues);
1657                    break;
1658                }
1659            }
1660        }
1661    }

1670    // Convert an old value from core database in prior schema version
1671    @SuppressWarnings({"unchecked", "rawtypes"})
1672    private Object convertOldValue(Field<?> field, Object value) {
1673        if (value == null)
1674            return null;
1675        final Converter converter = field.visit(new FieldSwitchAdapter<Converter>() {
1676
1677            @Override
1678            public Converter caseReferenceField(ReferenceField field) {
1679                return JTransaction.this.referenceConverter;
1680            }
1681
1682            @Override
1683            public <E> Converter caseSetField(SetField<E> field) {
1684                return field.getElementField() instanceof ReferenceField ?
1685                  new NavigableSetConverter(JTransaction.this.referenceConverter) : null;
1686            }
1687
1688            @Override
1689            public <E> Converter caseListField(ListField<E> field) {
1690                return field.getElementField() instanceof ReferenceField ?
1691                  new ListConverter(JTransaction.this.referenceConverter) : null;
1692            }
1693
1694            @Override
1695            public <K, V> Converter caseMapField(MapField<K, V> field) {
1696                if (field.getKeyField() instanceof ReferenceField || field.getValueField() instanceof ReferenceField) {
1697                    final Converter keyConverter = field.getKeyField() instanceof ReferenceField ?
1698                      JTransaction.this.referenceConverter : Converter.identity();
1699                    final Converter valueConverter = field.getValueField() instanceof ReferenceField ?
1700                      JTransaction.this.referenceConverter : Converter.identity();
1701                    return new NavigableMapConverter(keyConverter, valueConverter);
1702                }
1703                return null;
1704            }
1705
1706            @Override
1707            public <T> Converter caseField(Field<T> field) {
1708                return null;
1709            }
1710        });
1711        if (converter != null) {
1712            try {
1713                value = converter.reverse().convert(value);
1714            } catch (UnmatchedEnumException e) {
1715                // ignore - give them the EnumValue object instead
1716            }
1717        }
1718        return value;
1719    }
