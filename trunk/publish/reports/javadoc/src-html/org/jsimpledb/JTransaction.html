070/**
071 * A transaction associated with a {@link JSimpleDB} instance.
072 *
073 * <p>
074 * Commonly used methods in this class can be divided into the following categories:
075 * </p>
076 *
077 * <p>
078 * <b>Transaction Meta-Data</b>
079 * <ul>
080 *  <li>{@link #getJSimpleDB getJSimpleDB()} - Get the associated {@link JSimpleDB} instance</li>
081 *  <li>{@link #getTransaction} - Get the core API {@link Transaction} underlying this instance</li>
082 * </ul>
083 * </p>
084 *
085 * <p>
086 * <b>Transaction Lifecycle</b>
087 * <ul>
088 *  <li>{@link #commit commit()} - Commit transaction</li>
089 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
090 *  <li>{@link #getCurrent getCurrent()} - Get the {@link JTransaction} instance associated with the current thread</li>
091 *  <li>{@link #setCurrent setCurrent()} - Set the {@link JTransaction} instance associated with the current thread</li>
092 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
093 * </ul>
094 * </p>
095 *
096 * <p>
097 * <b>Object Access</b>
098 * <ul>
099 *  <li>{@link #getJObject(ObjId) getJObject()} - Get the Java model object corresponding to a specific database object ID</li>
100 *  <li>{@link #create(Class) create()} - Create a new database object</li>
101 *  <li>{@link #getAll getAll()} - Get all database objects that are instances of a given Java type</li>
102 *  <li>{@link #queryVersion queryVersion()} - Get database objects grouped according to their schema versions</li>
103 * </ul>
104 * </p>
105 *
106 * <p>
107 * <b>Validation</b>
108 * <ul>
109 *  <li>{@link #validate validate()} - Validate all objects in the validation queue</li>
110 *  <li>{@link #resetValidationQueue} - Clear the validation queue</li>
111 * </ul>
112 * </p>
113 *
114 * <p>
115 * <b>Index Queries</b>
116 * <ul>
117 *  <li>{@link #queryIndex(Class, String, Class) queryIndex()}
118 *      - Access the index associated with a simple field</li>
119 *  <li>{@link #queryListElementIndex queryListElementIndex()}
120 *      - Access the composite index associated with a list field that includes corresponding list indicies</li>
121 *  <li>{@link #queryMapValueIndex queryMapValueIndex()}
122 *      - Access the composite index associated with a map value field that includes corresponding map keys</li>
123 *  <li>{@link #queryCompositeIndex(Class, String, Class, Class) queryCompositeIndex()}
124 *      - Access a composite index defined on two fields</li>
125 *  <li>{@link #queryCompositeIndex(Class, String, Class, Class, Class) queryCompositeIndex()}
126 *      - Access a composite index defined on three fields</li>
127 *  <!-- COMPOSITE-INDEX -->
128 * </ul>
129 * </p>
130 *
131 * <p>
132 * <b>Reference Inversion</b>
133 * <ul>
134 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
135 *      of objects through a specified reference path</li>
136 *  </li>
137 * </ul>
138 * </p>
139 *
140 * <p>
141 * <b>Snapshot Transactions</b>
142 * <ul>
143 *  <li>{@link #getSnapshotTransaction getSnapshotTransaction()} - Get the default in-memory transaction
144 *      associated with this transaction</li>
145 *  <li>{@link #copyTo(JTransaction, JObject, ObjId, ObjIdSet, String[]) copyTo()} - Copy an object into another transaction</li>
146 *  <li>{@link #copyTo(JTransaction, ObjIdSet, Iterable) copyTo()} - Copy explicitly specified objects into another transaction</li>

150 * <p>

218    /**
219     * Constructor.
220     *
221     * @throws IllegalArgumentException if any parameter is null
222     */
223    JTransaction(JSimpleDB jdb, Transaction tx, ValidationMode validationMode) {
224
225        // Initialization
226        if (jdb == null)
227            throw new IllegalArgumentException("null jdb");
228        if (tx == null)
229            throw new IllegalArgumentException("null tx");
230        if (validationMode == null)
231            throw new IllegalArgumentException("null validationMode");
232        this.jdb = jdb;
233        this.tx = tx;
234        this.validationMode = validationMode;
235
236        // Register listeners for @OnCreate
237        if (this.jdb.hasOnCreateMethods
238          || (validationMode == ValidationMode.AUTOMATIC && this.jdb.anyJClassRequiresInitialValidation))
239            this.tx.addCreateListener(this.internalCreateListener);
240
241        // Register listeners for @OnDelete
242        if (this.jdb.hasOnDeleteMethods)
243            this.tx.addDeleteListener(this.internalDeleteListener);
244
245        // Register listeners for @OnChange
246        for (JClass<?> jclass : this.jdb.jclasses.values()) {
247            for (OnChangeScanner<?>.MethodInfo info : jclass.onChangeMethods) {
248                if (this instanceof SnapshotJTransaction && !info.getAnnotation().snapshotTransactions())
249                    continue;
250                final OnChangeScanner<?>.ChangeMethodInfo changeInfo = (OnChangeScanner<?>.ChangeMethodInfo)info;
251                changeInfo.registerChangeListener(this);
252            }
253        }
254
255        // Register field change listeners to trigger validation of corresponding JSR 303 constraints
256        if (validationMode == ValidationMode.AUTOMATIC) {
257            for (JFieldInfo jfieldInfo : this.jdb.jfieldInfos.values()) {
258                if (jfieldInfo.isRequiresValidation())
259                    jfieldInfo.registerChangeListener(this.tx, new int[0], null, this.validationListener);
260            }
261        }
262
263        // Register listeners for @OnVersionChange
264        if (this.jdb.hasOnVersionChangeMethods)
265            this.tx.addVersionChangeListener(this.internalVersionChangeListener);
266    }

331        return new ConvertedNavigableSet<T, ObjId>(ids, new ReferenceConverter<T>(this, type));

401        return this.tx.getKey(jobj.getObjId(), refPath.targetFieldInfo.storageId);
402    }

421    /**
422     * Copy the specified object into the specified destination transaction. If the target object already exists, it's
423     * schema version will be updated to match the source object if necessary, otherwise it will be created.
424     * {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange},
425     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
426     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly
427     * (however, for create and change notifications in {@code dest}, these annotations must have
428     * {@code snapshotTransactions = true} if {@code dest} is a {@link SnapshotJTransaction}).
429     *
430     * <p>
431     * Circular references are handled properly: if an object is encountered more than once, it is not copied again.
432     * The {@code seen} parameter tracks which objects have already been copied. For a "fresh" copy operation, pass a newly
433     * created instance; for a copy operation that is a continuation of a previous copy, the {@code seen} may be reused.
434     * </p>
435     *
436     * <p>
437     * Does nothing if this instance and {@code dest} are the same instance and {@code srcId} and {@code dstId} are equal.
438     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
439     * must be identical in both transactions.
440     * </p>
441     *
442     * <p>
443     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
444     * deadlock could result.
445     * </p>
446     *
447     * <p>
448     * This method is typically only used by generated classes; normally, {@link JObject#copyIn JObject.copyIn()},
449     * {@link JObject#copyOut JObject.copyOut()}, or {@link JObject#copyTo JObject.copyTo()} would be used instead.
450     * </p>
451     *
452     * @param dest destination transaction
453     * @param srcObj source object
454     * @param dstId target object ID, or null for the object ID of {@code srcObj}
455     * @param seen tracks which indirectly referenced objects have already been copied
456     * @param refPaths zero or more reference paths that refer to additional objects to be copied
457     * @return the copied object, i.e., the object having ID {@code dstId} in {@code dest}
458     * @throws DeletedObjectException if {@code srcObj} does not exist in this transaction
459     * @throws org.jsimpledb.core.SchemaMismatchException if the schema corresponding to {@code srcObj}'s object's version
460     *  is not identical in this instance and {@code dest} (as well for any referenced objects)
461     * @throws TypeNotInSchemaVersionException if the current schema version does not contain the source object's type
462     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
463     * @throws ReadOnlyTransactionException if {@code dest}'s underlying transaction
464     *  is {@linkplain Transaction#setReadOnly set read-only}
465     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
466     * @throws IllegalArgumentException if any parameter is null
467     * @see JObject#copyTo JObject.copyTo()
468     * @see JObject#copyOut JObject.copyOut()
469     * @see JObject#copyIn JObject.copyIn()
470     * @see #copyTo(JTransaction, ObjIdSet, Iterable)
471     */
472    public JObject copyTo(JTransaction dest, JObject srcObj, ObjId dstId, ObjIdSet seen, String... refPaths) {
473
474        // Sanity check
475        if (dest == null)
476            throw new IllegalArgumentException("null destination transaction");
477        if (srcObj == null)
478            throw new IllegalArgumentException("null srcObj");
479        if (seen == null)
480            throw new IllegalArgumentException("null seen");
481        if (refPaths == null)
482            throw new IllegalArgumentException("null refPaths");
483
484        // Handle possible re-entrant object cache load
485        srcObj.getTransaction().getJObjectCache().registerJObject(srcObj);
486
487        // Get source and dest ID
488        final ObjId srcId = srcObj.getObjId();
489        if (dstId == null)
490            dstId = srcId;
491
492        // Check trivial case
493        if (this.tx == dest.tx && srcId.equals(dstId))
494            return dest.getJObject(dstId);
495
496        // Parse paths
497        final Class<?> startType = this.jdb.getJClass(srcId).type;
498        final HashSet<ReferencePath> paths = new HashSet<>(refPaths.length);
499        for (String refPath : refPaths) {
500
501            // Parse reference path
502            if (refPath == null)
503                throw new IllegalArgumentException("null refPath");
504            final ReferencePath path = this.jdb.parseReferencePath(startType, refPath, null);
505
506            // Verify target field is a reference field; convert a complex target field into its reference sub-field(s)
507            final String lastFieldName = refPath.substring(refPath.lastIndexOf('.') + 1);
508            final JFieldInfo targetFieldInfo = this.jdb.jfieldInfos.get(path.getTargetField());
509            if (targetFieldInfo instanceof JComplexFieldInfo) {
510                final JComplexFieldInfo superFieldInfo = (JComplexFieldInfo)targetFieldInfo;
511                boolean foundReferenceSubFieldInfo = false;
512                for (JSimpleFieldInfo subFieldInfo : superFieldInfo.getSubFieldInfos()) {
513                    if (subFieldInfo instanceof JReferenceFieldInfo) {
514                        paths.add(this.jdb.parseReferencePath(startType,
515                          refPath + "." + superFieldInfo.getSubFieldInfoName(subFieldInfo), true));
516                        foundReferenceSubFieldInfo = true;
517                    }
518                }
519                if (!foundReferenceSubFieldInfo) {
520                    throw new IllegalArgumentException("the last field `" + lastFieldName
521                      + "' of path `" + refPath + "' does not contain any reference sub-fields");
522                }
523            } else {
524                if (!(targetFieldInfo instanceof JReferenceFieldInfo)) {
525                    throw new IllegalArgumentException("the last field `" + lastFieldName
526                      + "' of path `" + path + "' is not a reference field");
527                }
528                paths.add(path);
529            }
530        }
531
532        // Ensure object is copied even when there are zero reference paths
533        this.copyTo(seen, dest, srcId, dstId, true, new ArrayDeque<Integer>());
534
535        // Recurse over each reference path
536        for (ReferencePath path : paths) {
537            final int[] storageIds = path.getReferenceFields();
538
539            // Convert reference path, including final target field, into a list of JReferenceFields
540            final ArrayDeque<Integer> fields = new ArrayDeque<>(storageIds.length + 1);
541            fields.addAll(Ints.asList(storageIds));
542            fields.add(path.getTargetField());
543
544            // Recurse over this path
545            this.copyTo(seen, dest, srcId, dstId, false/*doesn't matter*/, fields);
546        }
547
548        // Done
549        return dest.getJObject(dstId);
550    }

552    /**
553     * Copy the objects in the specified {@link Iterable} into the specified destination transaction.
554     * If a target object already exists, it's schema version will be updated to match the source object if necessary,
555     * otherwise it will be created.
556     * {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange},
557     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
558     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly
559     * (however, for create and change notifications in {@code dest}, these annotations must have
560     * {@code snapshotTransactions = true} if {@code dest} is a {@link SnapshotJTransaction}).
561     *
562     * <p>
563     * Circular references are handled properly: if an object is encountered more than once, it is not copied again.
564     * The {@code seen} set tracks which objects have already been copied. For a "fresh" copy operation, pass a newly
565     * created instance; for a copy operation that is a continuation of a previous copy, the {@code seen} may be reused.
566     * </p>
567     *
568     * <p>
569     * If an object is encountered more than once, it is not copied again.
570     * Does nothing if this instance and {@code dest} are the same instance.
571     * </p>
572     *
573     * <p>
574     * Does nothing if this instance and {@code dest} are the same instance and {@code srcId} and {@code dstId} are equal.
575     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
576     * must be identical in both transactions.
577     * </p>
578     *
579     * <p>
580     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
581     * deadlock could result.
582     * </p>
583     *
584     * @param dest destination transaction
585     * @param jobjs {@link Iterable} returning the objects to copy; null values are ignored
586     * @param seen tracks which objects have already been copied
587     * @throws DeletedObjectException if an object in {@code jobjs} does not exist in this transaction
588     * @throws org.jsimpledb.core.SchemaMismatchException if the schema version corresponding to an object in
589     *  {@code jobjs} is not identical in this instance and {@code dest}
590     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
591     * @throws ReadOnlyTransactionException if {@code dest}'s underlying transaction
592     *  is {@linkplain Transaction#setReadOnly set read-only}
593     * @throws IllegalArgumentException if {@code dest} or {@code jobjs} is null
594     * @see #copyTo(JTransaction, JObject, ObjId, ObjIdSet, String[])
595     */
596    public void copyTo(JTransaction dest, ObjIdSet seen, Iterable<? extends JObject> jobjs) {
597
598        // Sanity check
599        if (dest == null)
600            throw new IllegalArgumentException("null dest");
601        if (seen == null)
602            throw new IllegalArgumentException("null seen");
603        if (jobjs == null)
604            throw new IllegalArgumentException("null jobjs");
605
606        // Check trivial case
607        if (this.tx == dest.tx)
608            return;
609
610        // Copy objects
611        for (JObject jobj : jobjs) {
612
613            // Get next object
614            if (jobj == null)
615                continue;
616
617            // Handle possible re-entrant object cache load
618            jobj.getTransaction().getJObjectCache().registerJObject(jobj);
619
620            // Copy object
621            final ObjId id = jobj.getObjId();
622            this.copyTo(seen, dest, id, id, true, new ArrayDeque<Integer>());
623        }
624    }
625
626    void copyTo(ObjIdSet seen, JTransaction dest, ObjId srcId, ObjId dstId, boolean required, Deque<Integer> fields) {
627
628        // Already copied this object?
629        if (!seen.add(dstId))
630            return;
631
632        // Copy current instance
633        try {
634            this.tx.copy(srcId, dstId, dest.tx);
635        } catch (DeletedObjectException e) {
636            if (required)
637                throw e;
638        }
639
640        // Recurse through the next reference field in the path
641        if (fields.isEmpty())
642            return;
643        final int storageId = fields.removeFirst();
644        final JReferenceFieldInfo referenceFieldInfo = this.jdb.getJFieldInfo(storageId, JReferenceFieldInfo.class);
645        final int parentStorageId = referenceFieldInfo.getParentStorageId();
646        if (parentStorageId != 0) {
647            final JComplexFieldInfo parentInfo = this.jdb.getJFieldInfo(parentStorageId, JComplexFieldInfo.class);
648            parentInfo.copyRecurse(seen, this, dest, srcId, storageId, fields);
649        } else {
650            assert referenceFieldInfo instanceof JReferenceFieldInfo;
651            final ObjId referrent = (ObjId)this.tx.readSimpleField(srcId, storageId, false);
652            if (referrent != null)
653                this.copyTo(seen, dest, referrent, referrent, false, fields);
654        }
655    }

1051     * @param path dot-separated path of one or more reference fields
1052     * @param targetObjects target objects

1062        final int targetField = refPath.getTargetField();
1063        try {
1064            this.jdb.getJFieldInfo(targetField, JReferenceFieldInfo.class);
1065        } catch (UnknownFieldException e) {

1067            throw new IllegalArgumentException("last field `" + fieldName + "' of path `" + path + "' is not a reference field", e);
1068        }
1069        final int[] refs = Ints.concat(refPath.getReferenceFields(), new int[] { targetField });
1070        final NavigableSet<ObjId> ids = this.tx.invertReferencePath(refs,
1071          Iterables.transform(targetObjects, new ReferenceConverter<JObject>(this, JObject.class)));
1072        return new ConvertedNavigableSet<T, ObjId>(ids, new ReferenceConverter<T>(this, startType));

1194     * Query an index by storage ID. For storage ID's corresponding to simple fields, this method returns an
1195     * {@link Index}, except for list element and map value fields, for which an {@link Index2} is returned.
1196     * For storage ID's corresponding to composite indexes, this method returns an {@link Index2}, {@link Index3},
1197     * etc. as appropriate.
1198     *
1199     * <p>
1200     * This method exists mainly for the convenience of programmatic tools, etc.
1201     * </p>
1202     *
1203     * @param storageId indexed {@link JSimpleField}'s storage ID
1204     * @return read-only, real-time view of the fields' values and the objects having those values in the fields
1205     * @throws IllegalArgumentException if {@code storageId} does not correspond to an indexed field or composite index
1206     * @throws StaleTransactionException if this transaction is no longer usable
1207     */
1208    @SuppressWarnings({"unchecked", "rawtypes"})
1209    public Object queryIndex(int storageId) {
1210
1211        // Look for a composite index
1212        final JCompositeIndexInfo indexInfo = this.jdb.jcompositeIndexInfos.get(storageId);
1213        if (indexInfo != null) {
1214            final Converter<JObject, ObjId> targetConverter = new ReferenceConverter<JObject>(this, JObject.class);
1215            switch (indexInfo.jfieldInfos.size()) {
1216            case 2:
1217            {
1218                final Converter<?, ?> value1Converter = this.getReverseConverter(indexInfo.jfieldInfos.get(0));
1219                final Converter<?, ?> value2Converter = this.getReverseConverter(indexInfo.jfieldInfos.get(1));
1220                return new ConvertedIndex2(this.tx.queryCompositeIndex2(indexInfo.storageId),
1221                  value1Converter, value2Converter, targetConverter);
1222            }
1223            case 3:
1224            {
1225                final Converter<?, ?> value1Converter = this.getReverseConverter(indexInfo.jfieldInfos.get(0));
1226                final Converter<?, ?> value2Converter = this.getReverseConverter(indexInfo.jfieldInfos.get(1));
1227                final Converter<?, ?> value3Converter = this.getReverseConverter(indexInfo.jfieldInfos.get(2));
1228                return new ConvertedIndex3(this.tx.queryCompositeIndex3(indexInfo.storageId),
1229                  value1Converter, value2Converter, value3Converter, targetConverter);
1230            }
1231            // COMPOSITE-INDEX
1232            default:
1233                throw new RuntimeException("internal error");
1234            }
1235        }
1236
1237        // Must be an indexed field
1238        final JFieldInfo someFieldInfo = this.jdb.jfieldInfos.get(storageId);
1239        if (someFieldInfo == null)
1240            throw new IllegalArgumentException("no composite index or simple indexed field exists with storage ID " + storageId);
1241        if (!(someFieldInfo instanceof JSimpleFieldInfo) || !((JSimpleFieldInfo)someFieldInfo).isIndexed()) {
1242            throw new IllegalArgumentException("storage ID " + storageId + " does not correspond to an indexed simple field (found "
1243              + someFieldInfo + " instead)");
1244        }
1245
1246        // Build the appropriate index for the field
1247        final JSimpleFieldInfo fieldInfo = (JSimpleFieldInfo)someFieldInfo;
1248        final Converter<?, ?> valueConverter = this.getReverseConverter(fieldInfo);
1249        final ReferenceConverter<JObject> referenceConverter = new ReferenceConverter<JObject>(this, JObject.class);
1250        final int parentStorageId = fieldInfo.getParentStorageId();
1251        final JComplexFieldInfo parentInfo = parentStorageId != 0 ?
1252          this.jdb.getJFieldInfo(parentStorageId, JComplexFieldInfo.class) : null;
1253        if (parentInfo instanceof JListFieldInfo) {
1254            return new ConvertedIndex2(this.tx.queryListElementIndex(fieldInfo.storageId),
1255              valueConverter, referenceConverter, Converter.identity());
1256        } else if (parentInfo instanceof JMapFieldInfo
1257          && ((JMapFieldInfo)parentInfo).getSubFieldInfoName(fieldInfo).equals(MapField.VALUE_FIELD_NAME)) {
1258            final JMapFieldInfo mapFieldInfo = (JMapFieldInfo)parentInfo;
1259            final JSimpleFieldInfo keyFieldInfo = mapFieldInfo.getKeyFieldInfo();
1260            final Converter<?, ?> keyConverter = this.getReverseConverter(keyFieldInfo);
1261            return new ConvertedIndex2(this.tx.queryMapValueIndex(fieldInfo.storageId),
1262              valueConverter, referenceConverter, keyConverter);
1263        } else
1264            return new ConvertedIndex(this.tx.queryIndex(fieldInfo.storageId), valueConverter, referenceConverter);
1265    }

1337    /**
1338     * Commit this transaction.
1339     *
1340     * <p>
1341     * Prior to actual commit, if this transaction was created with a validation mode other than {@link ValidationMode#DISABLED},
1342     * {@linkplain #validate validation} of outstanding objects in the validation queue is performed.
1343     * </p>
1344     *
1345     * <p>
1346     * If a {@link ValidationException} is thrown, the transaction is no longer usable. To perform validation and leave
1347     * the transaction open, invoke {@link #validate} prior to commit.
1348     * </p>
1349     *
1350     * @throws StaleTransactionException if this transaction is no longer usable
1351     * @throws org.jsimpledb.kv.RetryTransactionException from {@link org.jsimpledb.kv.KVTransaction#commit KVTransaction.commit()}
1352     * @throws ValidationException if a validation error is detected
1353     * @throws IllegalStateException if this method is invoked re-entrantly from within a validation check
1354     */
1355    public synchronized void commit() {
1356
1357        // Sanity check
1358        if (!this.tx.isValid())
1359            throw new StaleTransactionException(this.tx);
1360        synchronized (this) {
1361            if (this.committing)
1362                throw new IllegalStateException("commit() invoked re-entrantly");
1363            this.committing = true;
1364        }
1365
1366        // Do validation
1367        try {
1368            this.validate();
1369        } catch (ValidationException e) {
1370            this.tx.rollback();
1371            throw e;
1372        }
1373
1374        // Commit
1375        this.tx.commit();
1376    }

1398     * This method may be called at any time prior to {@link #commit} to
1399     * process and clear the queue of validatable objects.
1400     *
1401     * <p>
1402     * If validation fails, validation stops, all remaining unvalidated objects are left on the validation queue,
1403     * and a {@link ValidationException} is thrown. The transaction will remain usable.
1404     * </p>
1405     *
1406     * <p>
1407     * <b>Note:</b> if the this transaction was created with {@link ValidationMode#DISABLED}, then this method does nothing.
1408     * </p>
1409     *
1410     * @throws org.jsimpledb.kv.RetryTransactionException from {@link org.jsimpledb.kv.KVTransaction#commit KVTransaction.commit()}
1411     * @throws ValidationException if a validation error is detected
1412     * @throws IllegalStateException if transaction commit is already in progress
1413     * @throws StaleTransactionException if this transaction is no longer usable
1414     *
1415     * @see JObject#revalidate
1416     */
1417    public void validate() {
1418
1419        // Sanity check
1420        if (!this.tx.isValid())
1421            throw new StaleTransactionException(this.tx);
1422
1423        // Check validation mode
1424        if (this.validationMode == ValidationMode.DISABLED)
1425            return;
1426
1427        // Do validation
1428        this.performAction(new Runnable() {
1429            @Override
1430            public void run() {
1431                JTransaction.this.doValidate();
1432            }
1433        });
1434    }

1436    /**
1437     * Invoke the given {@link Runnable} with this instance as the {@linkplain #getCurrent current transaction}.
1438     *
1439     * <p>
1440     * If another instance is currently associated with the current thread, it is set aside for the duration of
1441     * {@code action}'s execution, and then restored when {@code action} is complete.
1442     * </p>
1443     *
1444     * @param action action to perform
1445     * @throws IllegalArgumentException if {@code action} is null
1446     */
1447    public void performAction(Runnable action) {
1448        if (action == null)
1449            throw new IllegalArgumentException("null action");
1450        final JTransaction previous = CURRENT.get();
1451        CURRENT.set(this);
1452        try {
1453            action.run();
1454        } finally {
1455            CURRENT.set(previous);
1456        }
1457    }

1461    private void doValidate() {
1462        while (true) {
1463
1464            // Get next object
1465            final ObjId id;
1466            synchronized (this) {
1467                final Iterator<ObjId> i = this.validationQueue.iterator();
1468                try {
1469                    id = i.next();
1470                } catch (NoSuchElementException e) {
1471                    return;
1472                }
1473                i.remove();
1474            }
1475
1476            // Does it still exist?
1477            if (!this.tx.exists(id))
1478                continue;
1479
1480            // Do JSR 303 validation
1481            final JObject jobj = this.getJObject(id);
1482            final Set<ConstraintViolation<JObject>> violations = ValidationUtil.validate(jobj);
1483            if (!violations.isEmpty()) {
1484                throw new ValidationException(jobj, violations, "validation error for object " + id + " of type `"
1485                  + this.jdb.jclasses.get(id.getStorageId()).name + "':\n" + ValidationUtil.describe(violations));
1486            }
1487
1488            // Do @Validate validation
1489            final JClass<?> jclass = this.jdb.getJClass(id);
1490            for (ValidateScanner<?>.MethodInfo info : jclass.validateMethods)
1491                Util.invoke(info.getMethod(), jobj);
1492        }
1493    }
