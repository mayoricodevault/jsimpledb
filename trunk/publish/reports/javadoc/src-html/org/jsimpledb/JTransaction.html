076    /**
077     * Constructor.
078     *
079     * @throws IllegalArgumentException if any parameter is null
080     */
081    JTransaction(JSimpleDB jdb, Transaction tx, ValidationMode validationMode) {
082
083        // Initialization
084        if (jdb == null)
085            throw new IllegalArgumentException("null jdb");
086        if (tx == null)
087            throw new IllegalArgumentException("null tx");
088        if (validationMode == null)
089            throw new IllegalArgumentException("null validationMode");
090        this.jdb = jdb;
091        this.tx = tx;
092        this.validationMode = validationMode;
093
094        // Register listeners
095        this.tx.addCreateListener(this);
096        this.tx.addDeleteListener(this);
097        this.tx.addVersionChangeListener(this);
098        for (JClass<?> jclass : this.jdb.jclasses.values()) {
099            for (OnChangeScanner<?>.MethodInfo info : jclass.onChangeMethods) {
100                final OnChangeScanner<?>.ChangeMethodInfo changeInfo = (OnChangeScanner<?>.ChangeMethodInfo)info;
101                changeInfo.registerChangeListener(this);
102            }
103        }
104        if (validationMode == ValidationMode.AUTOMATIC) {
105            for (JClass<?> jclass : this.jdb.jclasses.values()) {
106                for (JField jfield : jclass.jfields.values()) {
107                    if (jfield.requiresValidation)
108                        jfield.registerChangeListener(this.tx, new int[0], this.validationListener);
109                }
110            }
111        }
112    }

155     * Get all instances of the given type. The ordering of the returned set is based on the object IDs.
156     *
157     * @param type any Java type with annotated Java object model sub-types
158     * @return read-only view of all instances of {@code type}
159     */
160    @SuppressWarnings("unchecked")
161    public <T> NavigableSet<T> getAll(Class<T> type) {
162        final TypeToken<T> typeToken = TypeToken.of(type);
163        final ArrayList<NavigableSet<ObjId>> sets = new ArrayList<>();
164        for (JClass<?> jclass : this.jdb.jclasses.values()) {
165            if (typeToken.isAssignableFrom(jclass.typeToken))
166                sets.add(this.tx.getAll(jclass.storageId));
167        }
168        return sets.isEmpty() ? NavigableSets.<T>empty() :
169          (NavigableSet<T>)new ConvertedNavigableSet<JObject, ObjId>(NavigableSets.union(sets), this.referenceConverter);
170    }

193    /**
194     * Get the {@code byte[]} key in the underlying key/value store corresponding to the specified field in the specified object.
195     *
196     * <p>
197     * Notes:
198     * <ul>
199     *  <li>Complex fields utilize mutiple keys; the return value is the common prefix of all such keys.</li>
200     *  <li>The {@link org.jsimpledb.kv.KVDatabase} should not be modified directly, otherwise behavior is undefined</li>
201     * </ul>
202     * </p>
203     *
204     * @param jobj Java model object
205     * @param path a {@linkplain ReferencePath reference path} with zero intermediate reference fields
206     * @return the {@link org.jsimpledb.kv.KVDatabase} key of the field in the specified object
207     * @throws IllegalArgumentException if {@code path} is invalid
208     * @throws IllegalArgumentException if {@code path} contains one or more intermediate references
209     * @throws IllegalArgumentException if the target of {@code path} is a sub-field of a complex field
210     * @throws IllegalArgumentException if {@code jobj} does not contain the specified field
211     * @throws IllegalArgumentException if either parameter is null
212     */
213    public byte[] getKey(JObject jobj, String path) {
214        if (jobj == null)
215            throw new IllegalArgumentException("null jobj");
216        final TypeToken<?> startType = this.jdb.getJClass(jobj.getObjId().getStorageId()).typeToken;
217        final ReferencePath refPath = this.jdb.parseReferencePath(startType, path, false);
218        if (refPath.getReferenceFields().length > 0)
219            throw new IllegalArgumentException("path `" + path + "' contains one or more reference fields");
220        if (!refPath.targetType.getRawType().isInstance(jobj))
221            throw new IllegalArgumentException("jobj is not an instance of " + refPath.targetType);
222        return this.tx.getKey(jobj.getObjId(), refPath.targetField.storageId);
223    }

238    /**
239     * Copy the specified object into the specified destination transaction.
240     *
241     * <p>
242     * This method is typically only used by generated classes.
243     * </p>
244     *
245     * @param dest destination transaction
246     * @param jobj object to copy
247     * @param refPaths zero or more reference paths that refer to additional objects to be copied
248     * @return the copied version of this instance
249     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
250     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
251     * @throws IllegalArgumentException if any parameter is null
252     * @see JObject#copyOut JObject.copyOut()
253     * @see JObject#copyIn JObject.copyIn()
254     * @see JObject#copyTo JObject.copyTo()
255     */
256    public JObject copyTo(JTransaction dest, JObject jobj, String... refPaths) {
257
258        // Sanity check
259        if (dest == null)
260            throw new IllegalArgumentException("null destination transaction");
261        if (jobj == null)
262            throw new IllegalArgumentException("null jobj");
263        if (refPaths == null)
264            throw new IllegalArgumentException("null refPaths");
265        if (this.tx == dest.tx)
266            return jobj;
267        final ObjId id = jobj.getObjId();
268
269        // Parse paths
270        final TypeToken<?> startType = this.jdb.getJClass(id.getStorageId()).typeToken;
271        final HashSet<ReferencePath> paths = new HashSet<>(refPaths.length);
272        for (String refPath : refPaths) {
273
274            // Parse refernce path
275            if (refPath == null)
276                throw new IllegalArgumentException("null refPath");
277            final ReferencePath path = this.jdb.parseReferencePath(startType, refPath, null);
278
279            // Verify target field is a reference field; convert a complex target field into its reference sub-field(s)
280            final String lastFieldName = refPath.substring(refPath.lastIndexOf('.') + 1);
281            final JField targetField = this.jdb.jfields.get(path.getTargetField());
282            if (targetField instanceof JComplexField) {
283                final JComplexField superField = (JComplexField)targetField;
284                boolean foundReferenceSubField = false;
285                for (JSimpleField subField : superField.getSubFields()) {
286                    if (subField instanceof JReferenceField) {
287                        paths.add(this.jdb.parseReferencePath(startType,
288                          refPath + "." + superField.getSubFieldName(subField), true));
289                        foundReferenceSubField = true;
290                    }
291                }
292                if (!foundReferenceSubField) {
293                    throw new IllegalArgumentException("the last field `" + lastFieldName
294                      + "' of path `" + refPath + "' does not contain any reference sub-fields");
295                }
296            } else {
297                if (!(targetField instanceof JReferenceField)) {
298                    throw new IllegalArgumentException("the last field `" + lastFieldName
299                      + "' of path `" + path + "' is not a reference field");
300                }
301                paths.add(path);
302            }
303        }
304
305        // Ensure object is copied when there are zero reference paths
306        final HashSet<ObjId> seen = new HashSet<>();
307        if (paths.isEmpty())
308            this.copyTo(seen, dest, id, new ArrayDeque<JReferenceField>());
309
310        // Recurse over each reference path
311        for (ReferencePath path : paths) {
312            final int[] storageIds = path.getReferenceFields();
313
314            // Convert reference path, including final target field, into a list of JReferenceFields
315            final ArrayDeque<JReferenceField> fields = new ArrayDeque<>(storageIds.length + 1);
316            for (int storageId : storageIds)
317                fields.add((JReferenceField)this.jdb.jfields.get(storageId));
318            fields.add((JReferenceField)this.jdb.jfields.get(path.getTargetField()));
319
320            // Recurse over this path
321            this.copyTo(seen, dest, id, fields);
322        }
323
324        // Done
325        return dest.getJObject(id);
326    }
327
328    void copyTo(Set<ObjId> seen, JTransaction dest, ObjId id, Deque<JReferenceField> fields) {
329
330        // Already copied this object?
331        if (!seen.add(id))
332            return;
333
334        // Copy current instance
335        this.tx.copyTo(id, dest.tx);
336
337        // Recurse through the next reference field in the path
338        if (fields.isEmpty())
339            return;
340        final JReferenceField jfield = fields.removeFirst();
341        if (jfield.parent instanceof JComplexField) {
342            final JComplexField superField = (JComplexField)jfield.parent;
343            superField.copyRecurse(seen, this, dest, id, jfield, fields);
344        } else {
345            final ObjId referrent = (ObjId)this.tx.readSimpleField(id, jfield.storageId, false);
346            if (referrent != null)
347                this.copyTo(seen, dest, referrent, fields);
348        }
349    }

462
463    void revalidate(Collection<? extends ObjId> ids) {
464        if (!this.tx.isValid())
465            throw new StaleTransactionException(this.tx);
466        if (this.validationMode == ValidationMode.DISABLED)
467            return;
468        synchronized (this) {
469            if (this.committing)
470                throw new IllegalStateException("commit() has already been invoked");
471            this.validationQueue.addAll(ids);
472        }
473    }

581    /**
582     * Find all objects that refer to any object in the given target set through the specified path of references.
583     *
584     * @param startType starting Java type for the path
585     * @param path dot-separated path of zero or more reference fields, followed by a final reference field
586     * @param targetObjects target objects
587     * @return set of objects that refer to any of the {@code targetObjects} via the {@code path} from {@code startType}
588     * @throws org.jsimpledb.core.UnknownFieldException if {@code path} is invalid
589     * @throws IllegalArgumentException if any parameter is null
590     */
591    @SuppressWarnings("unchecked")
592    public <T> NavigableSet<T> invertReferencePath(Class<T> startType, String path, Iterable<? extends JObject> targetObjects) {
593        if (targetObjects == null)
594            throw new IllegalArgumentException("null targetObjects");
595        final ReferencePath refPath = this.jdb.parseReferencePath(TypeToken.of(startType), path, true);
596        final int targetField = refPath.getTargetField();
597        if (!(this.jdb.jfields.get(targetField) instanceof JReferenceField)) {
598            final String fieldName = path.substring(path.lastIndexOf('.') + 1);
599            throw new IllegalArgumentException("last field `" + fieldName + "' of path `" + path + "' is not a reference field");
600        }
601        final int[] refs = Ints.concat(refPath.getReferenceFields(), new int[] { targetField });
602        final NavigableSet<ObjId> ids = this.tx.invertReferencePath(refs,
603          Iterables.transform(targetObjects, this.referenceConverter));
604        return (NavigableSet<T>)new ConvertedNavigableSet<JObject, ObjId>(ids, this.referenceConverter);
605    }

609    /**
610     * Query a simple field index for {@link JObject}s.
611     *
612     * <p>
613     * This returns the map returned by {@link Transaction#querySimpleField} with {@link ObjId}s converted into {@link JObject}s.
614     * </p>
615     *
616     * <p>
617     * Used by generated {@link org.jsimpledb.annotation.IndexQuery &#64;IndexQuery} override methods
618     * that return index maps whose values are sets of Java model objects (i.e., not other index entry types).
619     * </p>
620     */
621    @SuppressWarnings({"unchecked", "rawtypes"})
622    public NavigableMap<?, NavigableSet<?>> querySimpleField(int storageId) {
623        Converter<?, ?> keyConverter = this.jdb.getJField(storageId, JSimpleField.class).getConverter(this);
624        keyConverter = keyConverter != null ? keyConverter.reverse() : Converter.identity();
625        final NavigableSetConverter<JObject, ObjId> valueConverter = new NavigableSetConverter(this.referenceConverter);

708    /**
709     * Commit this transaction.
710     *
711     * <p>
712     * Prior to actual commit, if this transaction was created with a validation mode other than {@link ValidationMode#DISABLED},
713     * {@linkplain #validate validation} of outstanding objects in the validation queue is performed.
714     * </p>
715     *
716     * <p>
717     * If a {@link ValidationException} is thrown, the transaction is no longer usable. To perform validation and leave
718     * the transaction open, invoke {@link #validate} prior to commit.
719     * </p>
720     *
721     * @throws StaleTransactionException if this transaction is no longer usable
722     * @throws org.jsimpledb.kv.RetryTransactionException from {@link org.jsimpledb.kv.KVTransaction#commit KVTransaction.commit()}
723     * @throws ValidationException if a validation error is detected
724     * @throws IllegalStateException if this method is invoked re-entrantly from within a validation check
725     */
726    public synchronized void commit() {
727
728        // Sanity check
729        if (!this.tx.isValid())
730            throw new StaleTransactionException(this.tx);
731        synchronized (this) {
732            if (this.committing)
733                throw new IllegalStateException("commit() invoked re-entrantly");
734            this.committing = true;
735        }
736
737        // Do validation
738        try {
739            this.validate();
740        } catch (ValidationException e) {
741            this.tx.rollback();
742            throw e;
743        }
744
745        // Commit
746        this.tx.commit();
747    }

749    /**
750     * Roll back this transaction.
751     *
752     * @throws StaleTransactionException if this transaction is no longer usable
753     */
754    public void rollback() {
755        this.tx.rollback();
756    }

767    /**
768     * Perform validation checks on all objects currently in the validation queue.
769     * This method may be called at any time prior to {@link #commit} to
770     * process and clear the queue of validatable objects.
771     *
772     * <p>
773     * If validation fails, validation stops, all remaining unvalidated objects are left on the validation queue,
774     * and a {@link ValidationException} is thrown. The transaction will remain usable.
775     * </p>
776     *
777     * <p>
778     * <b>Note:</b> if the this transaction was created with {@link ValidationMode#DISABLED}, then this method does nothing.
779     * </p>
780     *
781     * @throws org.jsimpledb.kv.RetryTransactionException from {@link org.jsimpledb.kv.KVTransaction#commit KVTransaction.commit()}
782     * @throws ValidationException if a validation error is detected
783     * @throws IllegalStateException if transaction commit is already in progress
784     * @throws StaleTransactionException if this transaction is no longer usable
785     *
786     * @see JObject#revalidate
787     */
788    public void validate() {
789
790        // Sanity check
791        if (!this.tx.isValid())
792            throw new StaleTransactionException(this.tx);
793
794        // Check validation mode
795        if (this.validationMode == ValidationMode.DISABLED)
796            return;
797
798        // Do validation
799        this.runWithCurrent(new Runnable() {
800            @Override
801            public void run() {
802                JTransaction.this.doValidate();
803            }
804        });
805    }
806
807// Internal methods
808
809    private void doValidate() {
810        while (true) {
811
812            // Get next object
813            final ObjId id;
814            synchronized (this) {
815                final Iterator<ObjId> i = this.validationQueue.iterator();
816                try {
817                    id = i.next();
818                } catch (NoSuchElementException e) {
819                    return;
820                }
821                i.remove();
822            }
823
824            // Does it still exist?
825            if (!this.tx.exists(id))
826                continue;
827
828            // Do JSR 303 validation
829            final JObject jobj = this.getJObject(id);
830            final Set<ConstraintViolation<JObject>> violations = ValidationUtil.validate(jobj);
831            if (!violations.isEmpty()) {
832                throw new ValidationException(jobj, violations, "validation error for object " + id + " of type `"
833                  + this.jdb.jclasses.get(id.getStorageId()).name + "':\n" + ValidationUtil.describe(violations));
834            }
835
836            // Do @Validate validation
837            final JClass<?> jclass = this.jdb.getJClass(id.getStorageId());
838            for (ValidateScanner<?>.MethodInfo info : jclass.validateMethods)
839                Util.invoke(info.getMethod(), jobj);
840        }
841    }

843    private void runWithCurrent(Runnable action) {
844        final JTransaction previous = CURRENT.get();
845        CURRENT.set(this);
846        try {
847            action.run();
848        } finally {
849            CURRENT.set(previous);
850        }
851    }

895    // This method exists solely to bind the generic type parameters
896    private <T> void doOnVersionChange(JClass<T> jclass, ObjId id,
897      int oldVersion, int newVersion, Map<Integer, Object> oldFieldValues) {
898        Object jobj = null;
899        final SchemaVersion oldSchema = this.tx.getSchema().getVersion(oldVersion);
900        for (OnVersionChangeScanner<T>.MethodInfo info : jclass.onVersionChangeMethods) {
901            final OnVersionChange annotation = info.getAnnotation();
902            final Method method = info.getMethod();
903
904            // Check old & new version numbers
905            if ((annotation.oldVersion() != 0 && annotation.oldVersion() != oldVersion)
906              || (annotation.newVersion() != 0 && annotation.newVersion() != newVersion))
907                continue;
908
909            // Get Java model object
910            if (jobj == null)
911                jobj = this.getJObject(id);
912
913            // Convert old field values so ObjId's become JObjects
914            final TreeMap<Integer, Object> convertedValues = new TreeMap<>();
915            for (Map.Entry<Integer, Object> entry : oldFieldValues.entrySet()) {
916                final int storageId = entry.getKey();
917                Object oldValue = entry.getValue();
918
919                // Convert old field value as needed; we only convert ObjId -> JObject
920                final Field<?> field = oldSchema.getSchemaItem(storageId, Field.class);
921                oldValue = this.convertOldValue(field, oldValue);
922
923                // Update value
924                convertedValues.put(storageId, oldValue);
925            }
926
927            // Invoke method
928            switch ((annotation.oldVersion() != 0 ? 2 : 0) + (annotation.newVersion() != 0 ? 1 : 0)) {
929            case 0:
930                Util.invoke(method, jobj, oldVersion, newVersion, convertedValues);
931                break;
932            case 1:
933                Util.invoke(method, jobj, oldVersion, convertedValues);
934                break;
935            case 2:
936                Util.invoke(method, jobj, newVersion, convertedValues);
937                break;
938            case 3:
939            default:
940                Util.invoke(method, jobj, convertedValues);
941                break;
942            }
943        }
944    }

947    private <X, Y> Y convert(Converter<X, Y> converter, Object value) {
948        try {
949            return converter != null ? converter.convert((X)value) : (Y)value;
950        } catch (UnmatchedEnumException e) {
951            this.log.warn("unable to convert enum value that has disappeared, returning null", e);
952            return null;
953        }
954    }

956    // Convert an old value from core database in prior schema version
957    @SuppressWarnings({"unchecked", "rawtypes"})
958    private Object convertOldValue(Field<?> field, Object value) {
959        if (value == null)
960            return null;
961        Converter converter = null;
962        if (field instanceof ReferenceField)
963            converter = this.referenceConverter;
964        else if (field instanceof SetField && ((SetField)field).getElementField() instanceof ReferenceField)
965            converter = new NavigableSetConverter(this.referenceConverter);
966        else if (field instanceof ListField && ((SetField)field).getElementField() instanceof ReferenceField)
967            converter = new ListConverter(this.referenceConverter);
968        else if (field instanceof MapField
969          && (((MapField)field).getKeyField() instanceof ReferenceField
970           || ((MapField)field).getValueField() instanceof ReferenceField)) {
971            final MapField<?, ?> mapField = (MapField<?, ?>)field;
972            final Converter keyConverter = mapField.getKeyField() instanceof ReferenceField ?
973              this.referenceConverter : Converter.identity();
974            final Converter valueConverter = mapField.getValueField() instanceof ReferenceField ?
975              this.referenceConverter : Converter.identity();
976            converter = new NavigableMapConverter(keyConverter, valueConverter);
977        }
978        if (converter != null)
979            value = converter.reverse().convert(value);
980        return value;
981    }
