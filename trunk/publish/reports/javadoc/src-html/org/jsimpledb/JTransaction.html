072/**
073 * A transaction associated with a {@link JSimpleDB} instance.
074 *
075 * <p>
076 * Commonly used methods in this class can be divided into the following categories:
077 * </p>
078 *
079 * <p>
080 * <b>Transaction Meta-Data</b>
081 * <ul>
082 *  <li>{@link #getJSimpleDB getJSimpleDB()} - Get the associated {@link JSimpleDB} instance</li>
083 *  <li>{@link #getTransaction} - Get the core API {@link Transaction} underlying this instance</li>
084 * </ul>
085 *
086 * <p>
087 * <b>Transaction Lifecycle</b>
088 * <ul>
089 *  <li>{@link #commit commit()} - Commit transaction</li>
090 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
091 *  <li>{@link #getCurrent getCurrent()} - Get the {@link JTransaction} instance associated with the current thread</li>
092 *  <li>{@link #setCurrent setCurrent()} - Set the {@link JTransaction} instance associated with the current thread</li>
093 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
094 * </ul>
095 *
096 * <p>
097 * <b>Object Access</b>
098 * <ul>
099 *  <li>{@link #getJObject(ObjId) getJObject()} - Get the Java model object corresponding to a specific database object ID</li>
100 *  <li>{@link #create(Class) create()} - Create a new database object</li>
101 *  <li>{@link #getAll getAll()} - Get all database objects that are instances of a given Java type</li>
102 *  <li>{@link #queryVersion queryVersion()} - Get database objects grouped according to their schema versions</li>
103 * </ul>
104 *
105 * <p>
106 * <b>Validation</b>
107 * <ul>
108 *  <li>{@link #validate validate()} - Validate all objects in the validation queue</li>
109 *  <li>{@link #resetValidationQueue} - Clear the validation queue</li>
110 * </ul>
111 *
112 * <p>
113 * <b>Index Queries</b>
114 * <ul>
115 *  <li>{@link #queryIndex(Class, String, Class) queryIndex()}
116 *      - Access the index associated with a simple field</li>
117 *  <li>{@link #queryListElementIndex queryListElementIndex()}
118 *      - Access the composite index associated with a list field that includes corresponding list indicies</li>
119 *  <li>{@link #queryMapValueIndex queryMapValueIndex()}
120 *      - Access the composite index associated with a map value field that includes corresponding map keys</li>
121 *  <li>{@link #queryCompositeIndex(Class, String, Class, Class) queryCompositeIndex()}
122 *      - Access a composite index defined on two fields</li>
123 *  <li>{@link #queryCompositeIndex(Class, String, Class, Class, Class) queryCompositeIndex()}
124 *      - Access a composite index defined on three fields</li>
125 *  <li>{@link #queryCompositeIndex(Class, String, Class, Class, Class, Class) queryCompositeIndex()}
126 *      - Access a composite index defined on four fields</li>
127 *  <!-- COMPOSITE-INDEX -->
128 * </ul>
129 *
130 * <p>
131 * <b>Reference Inversion</b>
132 * <ul>
133 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
134 *      of objects through a specified reference path</li>
135 * </ul>
136 *
137 * <p>
138 * <b>Snapshot Transactions</b>
139 * <ul>
140 *  <li>{@link #getSnapshotTransaction getSnapshotTransaction()} - Get the default in-memory transaction
141 *      associated with this transaction</li>
142 *  <li>{@link #copyTo(JTransaction, JObject, ObjId, CopyState, String[]) copyTo()}
143 *      - Copy an object into another transaction</li>
144 *  <li>{@link #copyTo(JTransaction, CopyState, Iterable) copyTo()}
145 *      - Copy explicitly specified objects into another transaction</li>
146 * </ul>
147 *
148 * <p>
149 * <b>Lower Layer Access</b>
150 * <ul>
151 *  <li>{@link #getKey(JObject) getKey()} - Get the {@link org.jsimpledb.kv.KVDatabase} key prefix for a specific object</li>
152 *  <li>{@link #getKey(JObject, String) getKey()} - Get the {@link org.jsimpledb.kv.KVDatabase}
153 *      key for a specific field in a specific object</li>
154 * </ul>
155 *
156 * <p>
157 * The remaining methods in this class are normally only used by generated Java model object subclasses.
158 * Instead of using these methods directly, using the appropriately annotated Java model object method
159 * or {@link JObject} interface method is recommended.
160 * </p>
161 *
162 * <p>
163 * <b>Java Model Object Methods</b>
164 * <ul>
165 *  <li>{@link #readSimpleField readSimpleField()} - Read the value of a simple field</li>
166 *  <li>{@link #writeSimpleField writeSimpleField()} - Write the value of a simple field</li>
167 *  <li>{@link #readCounterField readCounterField()} - Access a {@link Counter} field</li>
168 *  <li>{@link #readSetField readSetField()} - Access a set field</li>
169 *  <li>{@link #readListField readListField()} - Access a list field</li>
170 *  <li>{@link #readMapField readMapField()} - Access a map field</li>
171 * </ul>
172 *
173 * <p>
174 * <b>{@link JObject} Methods</b>
175 * <ul>
176 *  <li>{@link #delete delete()} - Delete an object from this transaction</li>
177 *  <li>{@link #exists exists()} - Test whether an object exists in this transaction</li>
178 *  <li>{@link #recreate recreate()} - Recreate an object in this transaction</li>
179 *  <li>{@link #revalidate revalidate()} - Add an object to the validation queue</li>
180 *  <li>{@link #getSchemaVersion getSchemaVersion()} - Get this schema version of an object</li>

214    /**
215     * Constructor.
216     *
217     * @throws IllegalArgumentException if any parameter is null
218     */
219    JTransaction(JSimpleDB jdb, Transaction tx, ValidationMode validationMode) {
220
221        // Initialization
222        if (jdb == null)
223            throw new IllegalArgumentException("null jdb");
224        if (tx == null)
225            throw new IllegalArgumentException("null tx");
226        if (validationMode == null)
227            throw new IllegalArgumentException("null validationMode");
228        this.jdb = jdb;
229        this.tx = tx;
230        this.validationMode = validationMode;
231
232        // Register listeners for @OnCreate and validation on creation
233        if (this.jdb.hasOnCreateMethods
234          || (validationMode == ValidationMode.AUTOMATIC && this.jdb.anyJClassRequiresValidation))
235            this.tx.addCreateListener(this.internalCreateListener);
236
237        // Register listeners for @OnDelete
238        if (this.jdb.hasOnDeleteMethods)
239            this.tx.addDeleteListener(this.internalDeleteListener);
240
241        // Register listeners for @OnChange
242        for (JClass<?> jclass : this.jdb.jclasses.values()) {
243            for (OnChangeScanner<?>.MethodInfo info : jclass.onChangeMethods) {
244                if (this instanceof SnapshotJTransaction && !info.getAnnotation().snapshotTransactions())
245                    continue;
246                final OnChangeScanner<?>.ChangeMethodInfo changeInfo = (OnChangeScanner<?>.ChangeMethodInfo)info;
247                changeInfo.registerChangeListener(this);
248            }
249        }
250
251        // Register field change listeners to trigger validation of corresponding JSR 303 and uniqueness constraints
252        if (validationMode == ValidationMode.AUTOMATIC) {
253            for (JFieldInfo jfieldInfo : this.jdb.jfieldInfos.values()) {
254                if (jfieldInfo.isRequiresValidation())
255                    jfieldInfo.registerChangeListener(this.tx, new int[0], null, this.validationListener);
256            }
257        }
258
259        // Register listeners for @OnVersionChange and validation on upgrade
260        if (this.jdb.hasOnVersionChangeMethods
261          || (validationMode == ValidationMode.AUTOMATIC && this.jdb.anyJClassRequiresValidation))
262            this.tx.addVersionChangeListener(this.internalVersionChangeListener);
263    }

400            throw new IllegalArgumentException("null jobj");
401        final Class<?> type = this.jdb.getJClass(jobj.getObjId()).type;
402        final ReferencePath refPath = this.jdb.parseReferencePath(type, fieldName, false);
403        if (refPath.getReferenceFields().length > 0)
404            throw new IllegalArgumentException("invalid field name `" + fieldName + "'");
405        if (!refPath.targetType.isInstance(jobj))
406            throw new IllegalArgumentException("jobj is not an instance of " + refPath.targetType); // should never happen
407        return this.tx.getKey(jobj.getObjId(), refPath.targetFieldInfo.storageId);
408    }

428    /**
429     * Copy the specified object into the specified destination transaction. If the target object already exists, it's
430     * schema version will be updated to match the source object if necessary, otherwise it will be created.
431     * {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange},
432     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
433     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly
434     * (however, for create and change notifications in {@code dest}, these annotations must have
435     * {@code snapshotTransactions = true} if {@code dest} is a {@link SnapshotJTransaction}).
436     *
437     * <p>
438     * Circular references are handled properly: if an object is encountered more than once, it is not copied again.
439     * The {@code copyState} parameter can be used to keep track of objects that have already been copied and/or traversed
440     * along some reference path (however, if an object is marked as copied in {@code copyState} and is traversed, but does not
441     * actually already exist in {@code dest}, an exception is thrown).
442     * For a "fresh" copy operation, pass a newly created {@code CopyState}; for a copy operation that is a continuation
443     * of a previous copy, {@code copyState} may be reused.
444     * </p>
445     *
446     * <p>
447     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
448     * must be identical in both transactions.
449     * </p>
450     *
451     * <p>
452     * Does nothing if this instance and {@code dest} are the same instance and {@code srcObj}'s object ID is {@code dstId}.
453     * </p>
454     *
455     * <p>
456     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
457     * deadlock could result.
458     * </p>
459     *
460     * <p>
461     * This method is typically only used by generated classes; normally, {@link JObject#copyIn JObject.copyIn()},
462     * {@link JObject#copyOut JObject.copyOut()}, or {@link JObject#copyTo JObject.copyTo()} would be used instead.
463     * </p>
464     *
465     * @param dest destination transaction
466     * @param srcObj source object
467     * @param dstId target object ID, or null for the object ID of {@code srcObj}
468     * @param copyState tracks which objects have already been copied and traversed
469     * @param refPaths zero or more reference paths that refer to additional objects to be copied (including intermediate objects)
470     * @return the copied object, i.e., the object having ID {@code dstId} in {@code dest}
471     * @throws DeletedObjectException if {@code srcObj} does not exist in this transaction
472     * @throws org.jsimpledb.core.DeletedObjectException if an object in {@code copyState} is traversed but does not actually exist
473     * @throws org.jsimpledb.core.SchemaMismatchException if the schema corresponding to {@code srcObj}'s object's version
474     *  is not identical in this instance and {@code dest} (as well for any referenced objects)
475     * @throws TypeNotInSchemaVersionException if the current schema version does not contain the source object's type
476     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
477     * @throws org.jsimpledb.core.ReadOnlyTransactionException if {@code dest}'s underlying transaction
478     *  is {@linkplain Transaction#setReadOnly set read-only}
479     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
480     * @throws IllegalArgumentException if any parameter is null
481     * @see JObject#copyTo JObject.copyTo()
482     * @see JObject#copyOut JObject.copyOut()
483     * @see JObject#copyIn JObject.copyIn()
484     * @see #copyTo(JTransaction, CopyState, Iterable)
485     */
486    public JObject copyTo(JTransaction dest, JObject srcObj, ObjId dstId, CopyState copyState, String... refPaths) {
487
488        // Sanity check
489        if (dest == null)
490            throw new IllegalArgumentException("null destination transaction");
491        if (srcObj == null)
492            throw new IllegalArgumentException("null srcObj");
493        if (copyState == null)
494            throw new IllegalArgumentException("null copyState");
495        if (refPaths == null)
496            throw new IllegalArgumentException("null refPaths");
497
498        // Handle possible re-entrant object cache load
499        srcObj.getTransaction().getJObjectCache().registerJObject(srcObj);
500
501        // Get source and dest ID
502        final ObjId srcId = srcObj.getObjId();
503        if (dstId == null)
504            dstId = srcId;
505
506        // Check trivial case
507        if (this.tx == dest.tx && srcId.equals(dstId))
508            return dest.getJObject(dstId);
509
510        // Parse paths
511        final Class<?> startType = this.jdb.getJClass(srcId).type;
512        final LinkedHashSet<ReferencePath> paths = new LinkedHashSet<>(refPaths.length);
513        for (String refPath : refPaths) {
514
515            // Parse reference path
516            if (refPath == null)
517                throw new IllegalArgumentException("null refPath");
518            final ReferencePath path = this.jdb.parseReferencePath(startType, refPath, null);
519
520            // Verify target field is a reference field; convert a complex target field into its reference sub-field(s)
521            final String lastFieldName = refPath.substring(refPath.lastIndexOf('.') + 1);
522            final JFieldInfo targetFieldInfo = this.jdb.jfieldInfos.get(path.getTargetField());
523            if (targetFieldInfo instanceof JComplexFieldInfo) {
524                final JComplexFieldInfo superFieldInfo = (JComplexFieldInfo)targetFieldInfo;
525                boolean foundReferenceSubFieldInfo = false;
526                for (JSimpleFieldInfo subFieldInfo : superFieldInfo.getSubFieldInfos()) {
527                    if (subFieldInfo instanceof JReferenceFieldInfo) {
528                        paths.add(this.jdb.parseReferencePath(startType,
529                          refPath + "." + superFieldInfo.getSubFieldInfoName(subFieldInfo), true));
530                        foundReferenceSubFieldInfo = true;
531                    }
532                }
533                if (!foundReferenceSubFieldInfo) {
534                    throw new IllegalArgumentException("the last field `" + lastFieldName
535                      + "' of path `" + refPath + "' does not contain any reference sub-fields");
536                }
537            } else {
538                if (!(targetFieldInfo instanceof JReferenceFieldInfo)) {
539                    throw new IllegalArgumentException("the last field `" + lastFieldName
540                      + "' of path `" + path + "' is not a reference field");
541                }
542                paths.add(path);
543            }
544        }
545
546        // Ensure object is copied even when there are zero reference paths
547        if (paths.isEmpty())
548            this.copyTo(copyState, dest, srcId, dstId, true, 0, new int[0]);
549
550        // Recurse over each reference path
551        for (ReferencePath path : paths) {
552            this.copyTo(copyState, dest, srcId, dstId, false/*doesn't matter*/,
553              0, Ints.concat(path.getReferenceFields(), new int[] { path.getTargetField() }));
554        }
555
556        // Done
557        return dest.getJObject(dstId);
558    }

560    /**
561     * Copy the objects in the specified {@link Iterable} into the specified destination transaction.
562     * If a target object already exists, it's schema version will be updated to match the source object if necessary,
563     * otherwise it will be created.
564     * {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange},
565     * {@link org.jsimpledb.annotation.OnCreate &#64;OnCreate} and
566     * {@link org.jsimpledb.annotation.OnCreate &#64;OnChange} notifications will be delivered accordingly
567     * (however, for create and change notifications in {@code dest}, these annotations must have
568     * {@code snapshotTransactions = true} if {@code dest} is a {@link SnapshotJTransaction}).
569     *
570     * <p>
571     * The {@code copyState} parameter tracks which objects that have already been copied. For a "fresh" copy operation,
572     * pass a newly created {@code CopyState}; for a copy operation that is a continuation of a previous copy,
573     * the previous {@link CopyState} may be reused.
574     * </p>
575     *
576     * <p>
577     * This instance and {@code dest} must be compatible in that for any schema versions encountered, those schema versions
578     * must be identical in both transactions.
579     * </p>
580     *
581     * <p>
582     * Does nothing if this instance and {@code dest} are the same instance.
583     * </p>
584     *
585     * <p>
586     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
587     * deadlock could result.
588     * </p>
589     *
590     * @param dest destination transaction
591     * @param jobjs {@link Iterable} returning the objects to copy; null values are ignored
592     * @param copyState tracks which objects have already been copied
593     * @throws DeletedObjectException if an object in {@code jobjs} does not exist in this transaction
594     * @throws org.jsimpledb.core.SchemaMismatchException if the schema version corresponding to an object in
595     *  {@code jobjs} is not identical in this instance and {@code dest}
596     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
597     * @throws org.jsimpledb.core.ReadOnlyTransactionException if {@code dest}'s underlying transaction
598     *  is {@linkplain Transaction#setReadOnly set read-only}
599     * @throws IllegalArgumentException if {@code dest} or {@code jobjs} is null
600     * @see #copyTo(JTransaction, JObject, ObjId, CopyState, String[])
601     */
602    public void copyTo(JTransaction dest, CopyState copyState, Iterable<? extends JObject> jobjs) {
603
604        // Sanity check
605        if (dest == null)
606            throw new IllegalArgumentException("null dest");
607        if (copyState == null)
608            throw new IllegalArgumentException("null copyState");
609        if (jobjs == null)
610            throw new IllegalArgumentException("null jobjs");
611
612        // Check trivial case
613        if (this.tx == dest.tx)
614            return;
615
616        // Copy objects
617        for (JObject jobj : jobjs) {
618
619            // Get next object
620            if (jobj == null)
621                continue;
622
623            // Handle possible re-entrant object cache load
624            jobj.getTransaction().getJObjectCache().registerJObject(jobj);
625
626            // Copy object
627            final ObjId id = jobj.getObjId();
628            this.copyTo(copyState, dest, id, id, true, 0, new int[0]);
629        }
630    }

632    void copyTo(CopyState copyState, JTransaction dest, ObjId srcId, ObjId dstId, boolean required, int fieldIndex, int[] fields) {
633
634        // Copy current instance unless already copied
635        if (copyState.markCopied(dstId)) {
636            try {
637                this.tx.copy(srcId, dstId, dest.tx);
638            } catch (DeletedObjectException e) {
639                if (required)
640                    throw e;
641            }
642        }
643
644        // Any more fields to traverse?
645        if (fieldIndex == fields.length)
646            return;
647
648        // Have we already traversed the path?
649        final int[] pathSuffix = fieldIndex == 0 ? fields : Arrays.copyOfRange(fields, fieldIndex, fields.length);
650        if (!copyState.markTraversed(srcId, pathSuffix))
651            return;
652
653        // Recurse through the next reference field in the path
654        final int storageId = fields[fieldIndex++];
655        final JReferenceFieldInfo referenceFieldInfo = this.jdb.getJFieldInfo(storageId, JReferenceFieldInfo.class);
656        final int parentStorageId = referenceFieldInfo.getParentStorageId();
657        if (parentStorageId != 0) {
658            final JComplexFieldInfo parentInfo = this.jdb.getJFieldInfo(parentStorageId, JComplexFieldInfo.class);
659            parentInfo.copyRecurse(copyState, this, dest, srcId, storageId, fieldIndex, fields);
660        } else {
661            assert referenceFieldInfo instanceof JReferenceFieldInfo;
662            final ObjId referrent = (ObjId)this.tx.readSimpleField(srcId, storageId, false);
663            if (referrent != null)
664                this.copyTo(copyState, dest, referrent, referrent, false, fieldIndex, fields);
665        }
666    }

1084     * Find all objects that refer to any object in the given target set through the specified path of references.
1085     *
1086     * @param startType starting Java type for the path
1087     * @param path dot-separated path of one or more reference fields
1088     * @param targetObjects target objects
1089     * @param <T> starting Java type
1090     * @return set of objects that refer to any of the {@code targetObjects} via the {@code path} from {@code startType}
1091     * @throws UnknownFieldException if {@code path} contains an unknown field
1092     * @throws IllegalArgumentException if {@code path} is invalid, e.g., does not end on a reference field
1093     * @throws IllegalArgumentException if any parameter is null
1094     */
1095    public <T> NavigableSet<T> invertReferencePath(Class<T> startType, String path, Iterable<? extends JObject> targetObjects) {
1096        if (targetObjects == null)
1097            throw new IllegalArgumentException("null targetObjects");
1098        final ReferencePath refPath = this.jdb.parseReferencePath(startType, path, true);
1099        final int targetField = refPath.getTargetField();
1100        try {
1101            this.jdb.getJFieldInfo(targetField, JReferenceFieldInfo.class);

1557    @SuppressWarnings("unchecked")
1558    private void doValidate() {
1559        while (true) {
1560
1561            // Get next object
1562            final ObjId id;
1563            synchronized (this) {
1564                final Iterator<ObjId> i = this.validationQueue.iterator();
1565                try {
1566                    id = i.next();
1567                } catch (NoSuchElementException e) {
1568                    return;
1569                }
1570                i.remove();
1571            }
1572
1573            // Does it still exist?
1574            if (!this.tx.exists(id))
1575                continue;
1576
1577            // Get object and verify type exists in current schema (if not, the remaining validation is unneccessary)
1578            final JObject jobj = this.getJObject(id);
1579            final JClass<?> jclass = this.jdb.jclasses.get(id.getStorageId());
1580            if (jclass == null)
1581                return;
1582
1583            // Do JSR 303 validation
1584            final Set<ConstraintViolation<JObject>> violations = ValidationUtil.validate(jobj);
1585            if (!violations.isEmpty()) {
1586                throw new ValidationException(jobj, violations, "validation error for object " + id + " of type `"
1587                  + this.jdb.jclasses.get(id.getStorageId()).name + "':\n" + ValidationUtil.describe(violations));
1588            }
1589
1590            // Do @Validate validation
1591            for (ValidateScanner<?>.MethodInfo info : jclass.validateMethods)
1592                Util.invoke(info.getMethod(), jobj);
1593
1594            // Do uniqueness validation
1595            for (JSimpleField jfield : Iterables.filter(jclass.jfields.values(), JSimpleField.class)) {
1596
1597                // Does this field have a uniqueness constraint?
1598                if (!jfield.unique)
1599                    continue;
1600                assert jfield.indexed;
1601
1602                // Get field's (core API) value
1603                final Object value = this.tx.readSimpleField(id, jfield.storageId, false);
1604
1605                // Compare to excluded value list
1606                if (jfield.uniqueExcludes != null
1607                  && Collections.binarySearch(jfield.uniqueExcludes, value, (Comparator<Object>)jfield.fieldType) >= 0)
1608                    continue;
1609
1610                // Seach for other objects with the same value in the field and report violation if any are found
1611                final ArrayList<ObjId> conflictors = new ArrayList<>(MAX_UNIQUE_CONFLICTORS);
1612                for (ObjId conflictor : this.tx.queryIndex(jfield.storageId).asMap().get(value)) {
1613                    if (conflictor.equals(id))                          // ignore jobj's own index entry
1614                        continue;
1615                    conflictors.add(conflictor);
1616                    if (conflictors.size() >= MAX_UNIQUE_CONFLICTORS)
1617                        break;
1618                }
1619                if (!conflictors.isEmpty()) {
1620                    throw new ValidationException(jobj, "uniqueness constraint on " + jfield + " failed: field value "
1621                      + value + " is also shared by object(s) " + conflictors);
1622                }
1623            }
1624        }
1625    }

1698
1699        @Override
1700        public void onVersionChange(Transaction tx, ObjId id, int oldVersion, int newVersion, Map<Integer, Object> oldFieldValues) {
1701            final JClass<?> jclass;
1702            try {
1703                jclass = JTransaction.this.jdb.getJClass(id);
1704            } catch (TypeNotInSchemaVersionException e) {
1705                return;                                             // object type does not exist in our schema
1706            }
1707            this.doOnVersionChange(jclass, id, oldVersion, newVersion, oldFieldValues);
1708        }
1709
1710        // This method exists solely to bind the generic type parameters
1711        private <T> void doOnVersionChange(JClass<T> jclass, ObjId id,
1712          int oldVersion, int newVersion, Map<Integer, Object> oldFieldValues) {
1713
1714            // Enqueue for revalidation
1715            if (validationMode == ValidationMode.AUTOMATIC && jclass.requiresValidation)
1716                JTransaction.this.revalidate(Collections.singleton(id));
1717
1718            // Skip the rest if there are no @OnChange methods
1719            if (jclass.onVersionChangeMethods.isEmpty())
1720                return;
1721
1722            // Get old object type info
1723            final Schema oldSchema = JTransaction.this.tx.getSchemas().getVersion(oldVersion);
1724            final ObjType objType = oldSchema.getObjType(id.getStorageId());
1725
1726            // The object that was upgraded
1727            JObject jobj = null;
1728
1729            // Convert old field values from core API objects to JDB layer objects
1730            final Map<Integer, Object> oldValuesByStorageId = Maps.transformEntries(oldFieldValues,
1731              new Maps.EntryTransformer<Integer, Object, Object>() {
1732                @Override
1733                public Object transformEntry(Integer storageId, Object oldValue) {
1734                    return JTransaction.this.convertCoreValue(objType.getField(storageId), oldValue);
1735                }
1736            });
1737
1738            // Build alternate version of old values map that is keyed by field name instead of storage ID
1739            final Map<String, Object> oldValuesByName = Maps.transformValues(objType.getFieldsByName(),
1740              new Function<Field<?>, Object>() {
1741                @Override
1742                public Object apply(Field<?> field) {
1743                    return oldValuesByStorageId.get(field.getStorageId());
1744                }
1745            });
1746
1747            // Invoke listener methods
1748            for (OnVersionChangeScanner<T>.MethodInfo info0 : jclass.onVersionChangeMethods) {
1749                final OnVersionChangeScanner<T>.VersionChangeMethodInfo info
1750                  = (OnVersionChangeScanner<T>.VersionChangeMethodInfo)info0;
1751
1752                // Get Java model object
1753                if (jobj == null)
1754                    jobj = JTransaction.this.getJObject(id);
1755
1756                // Invoke method
1757                info.invoke(jobj, oldVersion, newVersion, oldValuesByStorageId, oldValuesByName);
1758            }
1759        }
1760    }

1779    /**
1780     * Builds a {@link Converter} for any core API {@link Field} that converts, in the forward direction, core API values
1781     * into {@link JSimpleDB} values, to the extent possible. In the case of reference and enum fields, the
1782     * original Java type may no longer be available; if not, values are converted to {@link UntypedJObject}
1783     * or left as {@link org.jsimpledb.core.EnumValue}s.
1784     *
1785     * <p>
1786     * Returns null if no conversion is necessary.
1787     * </p>
1788     */
1789    @SuppressWarnings({ "unchecked", "rawtypes" })
1790    class CoreValueConverterBuilder extends FieldSwitchAdapter<Converter<?, ?>> {
1791
1792        // We can only convert EnumValue -> Enum if the Enum type is known and matches the old field's original type
1793        @Override
1794        public Converter<?, ?> caseEnumField(EnumField field) {
1795            final Class<? extends Enum<?>> enumType = field.getFieldType().getEnumType();
1796            return enumType != null ? EnumConverter.createEnumConverter(enumType).reverse() : null;
1797        }
1798
1799        @Override
1800        public Converter<?, ?> caseReferenceField(ReferenceField field) {
1801            return new ReferenceConverter<JObject>(JTransaction.this, JObject.class).reverse();
1802        }
1803
1804        @Override
1805        public <E> Converter<?, ?> caseSetField(SetField<E> field) {
1806            final Converter elementConverter = field.getElementField().visit(this);
1807            return elementConverter != null ? new NavigableSetConverter(elementConverter) : null;
1808        }
1809
1810        @Override
1811        public <E> Converter<?, ?> caseListField(ListField<E> field) {
1812            final Converter elementConverter = field.getElementField().visit(this);
1813            return elementConverter != null ? new ListConverter(elementConverter) : null;
1814        }
1815
1816        @Override
1817        public <K, V> Converter<?, ?> caseMapField(MapField<K, V> field) {
1818            Converter keyConverter = field.getKeyField().visit(this);
1819            Converter valueConverter = field.getValueField().visit(this);
1820            if (keyConverter != null || valueConverter != null) {
1821                if (keyConverter == null)
1822                    keyConverter = Converter.identity();
1823                if (valueConverter == null)
1824                    valueConverter = Converter.identity();
1825                return new NavigableMapConverter(keyConverter, valueConverter);
1826            }
1827            return null;
1828        }
1829
1830        @Override
1831        public <T> Converter caseField(Field<T> field) {
1832            return null;
1833        }
1834    }
