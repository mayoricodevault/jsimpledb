008package org.jsimpledb.spring;
009
010import org.springframework.beans.factory.xml.NamespaceHandlerSupport;
011
012/**
013 * Spring {@link org.springframework.beans.factory.xml.NamespaceHandler NamespaceHandler}
014 * for the <code>jsimpledb</code> XML namespace.
015 *
016 * <p>
017 * This class adds support for the <code>&lt;jsimpledb:jsimpledb&gt;</code>, <code>&lt;jsimpledb:scan-classes&gt;</code> and
018 * <code>&lt;jsimpledb:scan-field-types&gt;</code> XML tags.
019 * </p>
020 *
021 * @see org.jsimpledb.spring
022 */
023public class JSimpleDBNamespaceHandler extends NamespaceHandlerSupport {
024
025    public static final String JSIMPLEDB_NAMESPACE_URI = "http://jsimpledb.googlecode.com/schema/jsimpledb";
026
027    public static final String JSIMPLEDB_TAG = "jsimpledb";
028    public static final String SCAN_CLASSES_TAG = "scan-classes";
029    public static final String SCAN_FIELD_TYPES_TAG = "scan-field-types";
030
031    @Override
032    public void init() {
033        this.registerBeanDefinitionParser(JSIMPLEDB_TAG, new JSimpleDBBeanDefinitionParser());
034        this.registerBeanDefinitionParser(SCAN_CLASSES_TAG, new ScanClassesBeanDefinitionParser());
035        this.registerBeanDefinitionParser(SCAN_FIELD_TYPES_TAG, new ScanFieldTypesBeanDefinitionParser());
036    }
037}
