008package org.jsimpledb.spring;
009
010import java.io.File;
011import java.io.IOException;
012import java.io.InputStream;
013
014import org.dellroad.stuff.io.StreamRepository;
015import org.jsimpledb.kv.simple.XMLKVDatabase;
016import org.springframework.core.io.FileSystemResource;
017import org.springframework.core.io.Resource;
018
019/**
020 * {@link XMLKVDatabase} that adds support for loading the default initial content from a Spring {@link Resource}.
021 *
022 * @see XMLKVDatabase
023 */
024public class SpringXMLKVDatabase extends XMLKVDatabase {
025
026    private Resource initialContentResource;
027
028// Constructors
029
030    /**
031     * Normal constructor. Uses a {@link org.dellroad.stuff.io.FileStreamRepository} backed by the specified file.
032     *
033     * @param file persistent XML file
034     * @throws IllegalArgumentException if {@code file} is null
035     */
036    public SpringXMLKVDatabase(File file) {
037        super(file);
038    }
039
040    /**
041     * Constructor allowing storage in any user-supplied {@link StreamRepository}.
042     *
043     * @param repository XML file storage
044     * @throws IllegalArgumentException if {@code file} is null
045     */
046    public SpringXMLKVDatabase(StreamRepository repository) {
047        super(repository);
048    }
049
050    /**
051     * Configure the {@link Resource} containing default initial content for an uninitialized database. This method is invoked
052     * by {@link #getInitialContent} when, on the first load, the backing XML file is not found.
053     *
054     * @param initialContentResource resource containing default initial XML database content, or null for none
055     */
056    public void setInitialContentResource(Resource initialContentResource) {
057        this.initialContentResource = initialContentResource;
058    }
059
060    /**
061     * Configure the {@link Resource} containing default initial content for an uninitialized database from the specified file.
062     *
063     * @param initialContentFile file containing default initial XML database content, or null for none
064     */
065    @Override
066    public void setInitialContentFile(File initialContentFile) {
067        this.setInitialContentResource(initialContentFile != null ? new FileSystemResource(initialContentFile) : null);
068    }
069
070    /**
071     * Get the initial content for an uninitialized database. This method is invoked when, on the first load,
072     * the backing XML file is not found. It should return a stream that reads initial content for the database,
073     * if any, otherwise null.
074     *
075     * <p>
076     * The implementation in {@link SpringXMLKVDatabase} returns an {@link InputStream} acquired from the {@link Resource}
077     * configured by {@link #setInitialContentResource setInitialContentResource()}, if any, otherwise null.
078     * </p>
079     *
080     * @return default initial XML database content, or null for none
081     */
082    @Override
083    protected InputStream getInitialContent() throws IOException {
084        return this.initialContentResource != null ? this.initialContentResource.getInputStream() : null;
085    }
086}
