033/**
034 * JSimpleDB implementation of Spring's
035 * {@link org.springframework.transaction.PlatformTransactionManager PlatformTransactionManager} interface,
036 * allowing methods annotated with Spring's {@link org.springframework.transaction.annotation.Transactional &#64;Transactional}
037 * annotation to perform transactions on a {@link JSimpleDB} database.
038 *
039 * <p>
040 * Properly integrates with {@link JTransaction#getCurrent JTransaction.getCurrent()} to participate in
041 * existing transactions when appropriate.
042 * </p>
043 *
044 * @see org.jsimpledb.spring
045 */
046@SuppressWarnings("serial")
047public class JSimpleDBTransactionManager extends AbstractPlatformTransactionManager
048  implements ResourceTransactionManager, InitializingBean {
049
050    /**
051     * The default {@link ValidationMode} to use for transactions ({@link ValidationMode#AUTOMATIC}).
052     */
053    public static final ValidationMode DEFAULT_VALIDATION_MODE = ValidationMode.AUTOMATIC;
054
055    /**
056     * The configured {@link JSimpleDB} from which transactions are created.
057     */
058    protected transient JSimpleDB jdb;
059
060    /**
061     * Whether a new schema version is allowed. Default true.
062     */
063    protected boolean allowNewSchema = true;
064
065    /**
066     * The {@link ValidationMode} to use for transactions.
067     */
068    protected ValidationMode validationMode = DEFAULT_VALIDATION_MODE;
069
070    private boolean validateBeforeCommit = true;
071
072    public void afterPropertiesSet() throws Exception {
073        if (this.jdb == null)
074            throw new Exception("no JSimpleDB configured");
075    }
076
077    /**
078     * Configure the {@link JSimpleDB} that this instance will operate on.
079     *
080     * <p>
081     * Required property.
082     * </p>
083     *
084     * @param jdb associated database
085     */
086    public void setJSimpleDB(JSimpleDB jdb) {
087        this.jdb = jdb;
088    }
089
090    /**
091     * Configure whether a new schema version may be created.
092     *
093     * <p>
094     * Default value is false.
095     * </p>
096     *
097     * @param allowNewSchema whether to allow recording new schema versions
098     */
099    public void setAllowNewSchema(boolean allowNewSchema) {
100        this.allowNewSchema = allowNewSchema;
101    }
102
103    /**
104     * Configure the {@link ValidationMode} to use for transactions.
105     *
106     * <p>
107     * Default value is {@link ValidationMode#AUTOMATIC}.
108     * </p>
109     *
110     * @param validationMode validation mode for transactions
111     */
112    public void setValidationMode(ValidationMode validationMode) {
113        this.validationMode = validationMode != null ? validationMode : DEFAULT_VALIDATION_MODE;
114    }
115
116    /**
117     * Configure whether to invoke {@link JTransaction#validate} just prior to commit (and prior to any
118     * synchronization callbacks). This also causes validation to be performed at the end of each inner
119     * transaction that is participating in an outer transaction.
120     * If set to false, validation still occurs, but only when the outermost transaction commits.
121     *
122     * <p>
123     * Default true.
124     * </p>
125     *
126     * @param validateBeforeCommit whether to validate after inner transactions
127     */
128    public void setValidateBeforeCommit(boolean validateBeforeCommit) {
129        this.validateBeforeCommit = validateBeforeCommit;
130    }

132    @Override
133    public Object getResourceFactory() {
134        return this.jdb;
135    }

137    @Override
138    protected Object doGetTransaction() {
139        return new TxWrapper(this.getCurrent());
140    }
141
142    @Override
143    protected boolean isExistingTransaction(Object txObj) {
144        return ((TxWrapper)txObj).getJTransaction() != null;
145    }
146
147    @Override
148    protected void doBegin(Object txObj, TransactionDefinition txDef) {
149
150        // Sanity check
151        final TxWrapper tx = (TxWrapper)txObj;
152        if (tx.getJTransaction() != null)
153            throw new TransactionUsageException("there is already a transaction associated with the current thread");
154
155        // Create JSimpleDB transaction
156        final JTransaction jtx;
157        try {
158            jtx = this.jdb.createTransaction(this.allowNewSchema, this.validationMode);
159        } catch (DatabaseException e) {
160            throw new CannotCreateTransactionException("error creating new JSimpleDB transaction", e);
161        }
162
163        // Configure JSimpleDB transaction and bind to current thread; but if we fail, roll it back
164        boolean succeeded = false;
165        try {
166            this.configureTransaction(jtx, txDef);
167            JTransaction.setCurrent(jtx);
168            succeeded = true;
169        } catch (DatabaseException e) {
170            throw new CannotCreateTransactionException("error configuring JSimpleDB transaction", e);
171        } finally {
172            if (!succeeded) {
173                try {
174                    jtx.rollback();
175                } catch (DatabaseException e) {
176                    // ignore
177                }
178            }
179        }
180
181        // Done
182        tx.setJTransaction(jtx);
183    }
184
185    /**
186     * Suspend the current transaction.
187     */
188    @Override
189    protected Object doSuspend(Object txObj) {
190
191        // Sanity check
192        final JTransaction jtx = ((TxWrapper)txObj).getJTransaction();
193        if (jtx == null)
194            throw new TransactionUsageException("no JTransaction exists in the provided transaction object");
195        if (jtx != this.getCurrent())
196            throw new TransactionUsageException("the provided transaction object contains the wrong JTransaction");
197
198        // Suspend it
199        if (this.logger.isTraceEnabled())
200            this.logger.trace("suspending current JSimpleDB transaction " + jtx);
201        JTransaction.setCurrent(null);
202
203        // Done
204        return jtx;
205    }
206
207    /**
208     * Resume a previously suspended transaction.
209     */
210    @Override
211    protected void doResume(Object txObj, Object suspendedResources) {
212
213        // Sanity check
214        if (this.getCurrent() != null)
215            throw new TransactionUsageException("there is already a transaction associated with the current thread");
216
217        // Resume transaction
218        final JTransaction jtx = (JTransaction)suspendedResources;
219        if (this.logger.isTraceEnabled())
220            this.logger.trace("resuming JSimpleDB transaction " + jtx);
221        JTransaction.setCurrent(jtx);
222    }
223
224    /**
225     * Configure a new transaction.
226     *
227     * <p>
228     * The implementation in {@link JSimpleDBTransactionManager} sets the transaction's timeout and read-only properties.
229     * </p>
230     *
231     * @param jtx transaction to configure
232     * @param txDef transaction definition
233     * @throws DatabaseException if an error occurs
234     */
235    protected void configureTransaction(JTransaction jtx, TransactionDefinition txDef) {
236
237        // Set name
238        //jtx.setName(txDef.getName());
239
240        // Set read-only
241        jtx.getTransaction().setReadOnly(txDef.isReadOnly());
242
243        // Set lock timeout
244        final int timeout = this.determineTimeout(txDef);
245        if (timeout != TransactionDefinition.TIMEOUT_DEFAULT) {
246            try {
247                jtx.getTransaction().setTimeout(timeout * 1000L);
248            } catch (UnsupportedOperationException e) {
249                if (this.logger.isDebugEnabled())
250                    this.logger.debug("setting non-default timeout of " + timeout + "sec not supported by underlying transaction");
251            }
252        }
253    }
254
255    @Override
256    protected void prepareForCommit(DefaultTransactionStatus status) {
257
258        // Get transaction
259        final JTransaction jtx = ((TxWrapper)status.getTransaction()).getJTransaction();
260        if (jtx == null)
261            throw new NoTransactionException("no current JTransaction exists");
262
263        // Validate
264        if (this.validateBeforeCommit) {
265            if (this.logger.isTraceEnabled())
266                this.logger.trace("triggering validation prior to commit of JSimpleDB transaction " + jtx);
267            jtx.validate();
268        }
269    }
270
271    @Override
272    protected void doCommit(DefaultTransactionStatus status) {
273
274        // Get transaction
275        final JTransaction jtx = ((TxWrapper)status.getTransaction()).getJTransaction();
276        if (jtx == null)
277            throw new NoTransactionException("no current JTransaction exists");
278
279        // Commit
280        try {
281            if (this.logger.isTraceEnabled())
282                this.logger.trace("committing JSimpleDB transaction " + jtx);
283            jtx.commit();
284        } catch (RetryTransactionException e) {
285            throw new PessimisticLockingFailureException("transaction must be retried", e);
286        } catch (StaleTransactionException e) {
287            throw new TransactionTimedOutException("transaction is no longer usable", e);
288        } catch (DatabaseException e) {
289            throw new TransactionSystemException("error committing transaction", e);
290        } finally {
291            JTransaction.setCurrent(null);          // transaction is no longer usable
292        }
293    }
294
295    @Override
296    protected void doRollback(DefaultTransactionStatus status) {
297
298        // Get transaction
299        final JTransaction jtx = ((TxWrapper)status.getTransaction()).getJTransaction();
300        if (jtx == null)
301            throw new NoTransactionException("no current JTransaction exists");
302
303        // Rollback
304        try {
305            if (this.logger.isTraceEnabled())
306                this.logger.trace("rolling back JSimpleDB transaction " + jtx);
307            jtx.rollback();
308        } catch (StaleTransactionException e) {
309            throw new TransactionTimedOutException("transaction is no longer usable", e);
310        } catch (DatabaseException e) {
311            throw new TransactionSystemException("error committing transaction", e);
312        } finally {
313            JTransaction.setCurrent(null);          // transaction is no longer usable
314        }
315    }
316
317    @Override
318    protected void doSetRollbackOnly(DefaultTransactionStatus status) {
319
320        // Get transaction
321        final JTransaction jtx = ((TxWrapper)status.getTransaction()).getJTransaction();
322        if (jtx == null)
323            throw new NoTransactionException("no current JTransaction exists");
324
325        // Set rollback only
326        if (this.logger.isTraceEnabled())
327            this.logger.trace("marking JSimpleDB transaction " + jtx + " for rollback-only");
328        jtx.getTransaction().setRollbackOnly();
329    }
330
331    @Override
332    protected void doCleanupAfterCompletion(Object txObj) {
333        JTransaction.setCurrent(null);
334    }
335
336    @Override
337    protected void registerAfterCompletionWithExistingTransaction(Object txObj, List<TransactionSynchronization> synchronizations) {
338
339        // Get transaction
340        final JTransaction jtx = ((TxWrapper)txObj).getJTransaction();
341        if (jtx == null)
342            throw new NoTransactionException("no current JTransaction exists");
343
344        // Add synchronizations
345        final Transaction tx = jtx.getTransaction();
346        for (TransactionSynchronization synchronization : synchronizations)
347            tx.addCallback(new TransactionSynchronizationCallback(synchronization));
348    }
349
350    /**
351     * Like {@link JTransaction#getCurrent}, but returns null instead of throwing {@link IllegalStateException}.
352     *
353     * @return the transaction associated with the current thread, or null if there is none
354     */
355    protected JTransaction getCurrent() {
356        try {
357            return JTransaction.getCurrent();
358        } catch (IllegalStateException e) {
359            return null;
360        }
361    }

365    private static class TxWrapper implements SmartTransactionObject {
366
367        private JTransaction jtx;
368
369        TxWrapper(JTransaction jtx) {
370            this.jtx = jtx;
371        }
372
373        public JTransaction getJTransaction() {
374            return this.jtx;
375        }
376        public void setJTransaction(JTransaction jtx) {
377            this.jtx = jtx;
378        }
379
380        @Override
381        public boolean isRollbackOnly() {
382            return this.jtx != null && this.jtx.getTransaction().isRollbackOnly();
383        }
384
385        @Override
386        public void flush() {
387        }
388    }

392    /**
393     * Adapter class that wraps a Spring {@link TransactionSynchronization} in the
394     * {@link org.jsimpledb.core.Transaction.Callback} interface.
395     */
396    public static class TransactionSynchronizationCallback implements Transaction.Callback {
397
398        protected final TransactionSynchronization synchronization;
399
400        /**
401         * Constructor.
402         *
403         * @param synchronization transaction callback
404         * @throws IllegalArgumentException if {@code synchronization} is null
405         */
406        public TransactionSynchronizationCallback(TransactionSynchronization synchronization) {
407            if (synchronization == null)
408                throw new IllegalArgumentException("null synchronization");
409            this.synchronization = synchronization;
410        }
411
412        @Override
413        public void beforeCommit(boolean readOnly) {
414            this.synchronization.beforeCommit(readOnly);
415        }
416
417        @Override
418        public void beforeCompletion() {
419            this.synchronization.beforeCompletion();
420        }
421
422        @Override
423        public void afterCommit() {
424            this.synchronization.afterCommit();
425        }
426
427        @Override
428        public void afterCompletion(boolean committed) {
429            this.synchronization.afterCompletion(committed ?
430              TransactionSynchronization.STATUS_COMMITTED : TransactionSynchronization.STATUS_ROLLED_BACK);
431        }
432
433        @Override
434        public boolean equals(Object obj) {
435            if (obj == this)
436                return true;
437            if (obj == null || obj.getClass() != this.getClass())
438                return false;
439            final TransactionSynchronizationCallback that = (TransactionSynchronizationCallback)obj;
440            return this.synchronization.equals(that.synchronization);
441        }
442
443        @Override
444        public int hashCode() {
445            return this.synchronization.hashCode();
446        }
447    }
