008package org.jsimpledb.cli;
009
010import java.util.ArrayList;
011import java.util.HashMap;
012import java.util.HashSet;
013import java.util.List;
014import java.util.regex.Matcher;
015
016import org.jsimpledb.util.ParseContext;
017
018/**
019 * Parses a command with optional flags.
020 */
021public class ParamParser {
022
023    private final int minParams;
024    private final int maxParams;
025    private final String name;
026    private final String usage;
027    private final List<String> options = new ArrayList<>();
028    private final HashSet<String> takesArgument = new HashSet<>();
029
030    private final HashMap<String, String> flags = new HashMap<>();
031    private final List<String> params = new ArrayList<>();
032
033    public ParamParser(int minParams, int maxParams, String usage, String... options) {
034        this.minParams = minParams;
035        this.maxParams = maxParams;
036        this.usage = usage;
037        final int space = usage.indexOf(' ');
038        this.name = space != -1 ? usage.substring(0, space) : usage;
039        for (String option : options) {
040            if (option.endsWith("@")) {
041                option = option.substring(0, option.length() - 1);
042                this.takesArgument.add(option);
043            }
044            this.options.add(option);
045        }
046    }
047
048    public ParamParser parse(ParseContext ctx) {
049
050        // Parse option flags
051        this.parseOptions(ctx);
052
053        // Parse parameters
054        this.parseParams(ctx);
055        if (this.params.size() < this.minParams || this.params.size() > this.maxParams)
056            throw new ParseException(ctx, "Usage: " + this.usage);
057
058        // Done
059        return this;
060    }
061
062    protected void parseOptions(ParseContext ctx) {
063        this.flags.clear();
064        while (true) {
065            final Matcher matcher = ctx.tryPattern("(\\s*)(-[^\\s;]+)");
066            if (matcher == null)
067                break;
068            final String whitespace = matcher.group(1);
069            final String flag = matcher.group(2);
070            if (flag.equals("--"))
071                break;
072            if (!this.options.contains(flag)) {
073                throw new ParseException(ctx, "unknown flag `" + flag + "' given to the `" + this.name + "' command")
074                  .addCompletions(Util.complete(this.options, flag, whitespace != null));
075            }
076            String arg = "";
077            if (this.takesArgument.contains(flag)) {
078                final Matcher argMatcher = ctx.tryPattern("\\s*([^\\s;]+)");
079                if (argMatcher == null)
080                    throw new ParseException(ctx, "flag `" + flag + "' for command `"
081                      + this.name + "' requires an argument but none was given");
082                arg = argMatcher.group(1);
083            }
084            this.flags.put(flag, arg);
085        }
086    }
087
088    protected void parseParams(ParseContext ctx) {
089        this.params.clear();
090        while (true) {
091            final Matcher matcher = ctx.tryPattern("\\s*([^\\s;]+)");
092            if (matcher == null)
093                break;
094            this.params.add(matcher.group(1));
095        }
096    }
097
098    public boolean hasFlag(String flag) {
099        return this.flags.containsKey(flag);
100    }
101
102    public String getFlag(String flag) {
103        return this.flags.get(flag);
104    }
105
106    public List<String> getParams() {
107        return this.params;
108    }
109
110    public String getParam(int index) {
111        return this.params.get(index);
112    }
113}
