008package org.jsimpledb.cli.func;
009
010import org.jsimpledb.cli.CliSession;
011import org.jsimpledb.parse.ParseSession;
012import org.jsimpledb.parse.expr.Value;
013import org.jsimpledb.parse.func.SimpleFunction;
014
015/**
016 * CLI extension of {@link SimpleFunction}.
017 */
018public abstract class SimpleCliFunction extends SimpleFunction {
019
020// Constructors
021
022    /**
023     * Constructor.
024     *
025     * @param name function name
026     * @param minArgs minimum number of arguments (inclusive)
027     * @param maxArgs maximum number of arguments (inclusive)
028     */
029    protected SimpleCliFunction(String name, int minArgs, int maxArgs) {
030        super(name, minArgs, maxArgs);
031    }
032
033    @Override
034    protected final Value apply(ParseSession session, Value[] params) {
035        return this.apply((CliSession)session, params);
036    }
037
038    /**
039     * Apply this function to the given values.
040     *
041     * @param session CLI session
042     * @param params parsed parameters; will already be checked between {@link #getMinArgs} and {@link #getMaxArgs}
043     * @return value returned by this function
044     * @throws RuntimeException if there is an error
045     */
046    protected abstract Value apply(CliSession session, Value[] params);
047}
