008package org.jsimpledb.cli;
009
010import org.dellroad.stuff.string.ParseContext;
011import org.jsimpledb.core.ObjId;
012import org.jsimpledb.core.Transaction;
013import org.jsimpledb.schema.SchemaObject;
014
015public class ListCommand extends Command {
016
017    public ListCommand(AggregateCommand parent) {
018        super(parent, "list");
019    }
020
021    @Override
022    public Action parse(Session session, ParseContext ctx) throws ParseException {
023        final SchemaObject schemaObject = Util.parseSchemaObject(session, ctx, this.getFullName() + " type");
024        return new TransactionAction() {
025            @Override
026            public void run(Session session) throws Exception {
027                final Transaction tx = session.getTransaction();
028                int count = 0;
029                for (ObjId id : tx.getAll(schemaObject.getStorageId())) {
030                    if (count >= session.getLineLimit()) {
031                        session.getConsole().println("...");
032                        break;
033                    }
034                    final Util.ObjInfo info = Util.getObjInfo(tx, id);
035                    session.getConsole().println(info != null ? info.toString() : id + " DELETED?");
036                    count++;
037                }
038            }
039        };
040    }
041
042    @Override
043    public String getHelpSummary() {
044        return "Lists objects of a given type";
045    }
046}
