008package org.jsimpledb.cli.cmd;
009
010import java.util.Map;
011
012import org.jsimpledb.cli.CliSession;
013import org.jsimpledb.core.Schema;
014import org.jsimpledb.parse.ParseContext;
015import org.jsimpledb.schema.SchemaModel;
016
017@Command
018public class ShowAllSchemasCommand extends AbstractCommand {
019
020    public ShowAllSchemasCommand() {
021        super("show-all-schemas -x:xml");
022    }
023
024    @Override
025    public String getHelpSummary() {
026        return "Shows all schema versions recorded in the database";
027    }
028
029    @Override
030    public String getHelpDetail() {
031        return "If the `-x' flag is provided, the XML representation of each schema version is included.";
032    }
033
034    @Override
035    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
036        final boolean xml = params.containsKey("xml");
037        return new CliSession.Action() {
038            @Override
039            public void run(CliSession session) throws Exception {
040                for (Map.Entry<Integer, Schema> entry : session.getTransaction().getSchemas().getVersions().entrySet()) {
041                    final int number = entry.getKey();
042                    final SchemaModel model = entry.getValue().getSchemaModel();
043                    if (xml) {
044                        session.getWriter().println("=== Schema version " + number + " ===\n"
045                          + model.toString().replaceAll("^<.xml[^>]+>\\n", ""));
046                    } else
047                        session.getWriter().println(number);
048                }
049            }
050        };
051    }
052}
