048    /**
049     * Parse an object type name.
050     */
051    public static SchemaObject parseSchemaObject(Session session, ParseContext ctx, String usage) throws ParseException {
052        final NameIndex nameIndex = session.getNameIndex();
053        SortedSet<String> typeNames = nameIndex != null ? nameIndex.getSchemaObjectNames() : new TreeSet<String>();
054        String typeName = null;
055        try {
056            typeName = ctx.matchPrefix("([^\\s]+)\\s*$").group(1);
057            if (!typeNames.contains(typeName))
058                throw new IllegalArgumentException();
059        } catch (IllegalArgumentException e) {
060            if (typeName != null)
061                typeNames = Sets.filter(typeNames, new PrefixPredicate(typeName));
062            throw new ParseException(ctx, "Usage: " + usage).addCompletions(typeNames);
063        }
064        return nameIndex.getSchemaObject(typeName);
065    }

067    /**
068     * Parse an object ID.
069     */
070    public static ObjId parseObjId(Session session, ParseContext ctx, String usage) throws ParseException {
071
072        // Get parameter
073        final String param = ctx.getInput().trim();
074
075        // Attempt to parse it whole
076        try {
077            return new ObjId(param);
078        } catch (IllegalArgumentException e) {
079            // parse failed
080        }
081
082        // Param must look like a hex ID
083        if (!param.matches("[0-9A-Fa-f]{0,16}"))
084            throw new ParseException(ctx, usage);
085
086        // Get corresponding min & max ObjId (both inclusive)
087        final char[] paramChars = param.toCharArray();
088        final char[] idChars = new char[16];
089        System.arraycopy(paramChars, 0, idChars, 0, paramChars.length);
090        Arrays.fill(idChars, paramChars.length, idChars.length, '0');
091        final String minString = new String(idChars);
092        ObjId id;
093        try {
094            id = new ObjId(minString);
095        } catch (IllegalArgumentException e) {
096            if (!minString.startsWith("00"))
097                throw new ParseException(ctx, usage);
098            id = null;
099        }
100        final ObjId min = id;
101        Arrays.fill(idChars, paramChars.length, idChars.length, 'f');
102        try {
103            id = new ObjId(new String(idChars));
104        } catch (IllegalArgumentException e) {
105            id = null;
106        }
107        final ObjId max = id;
108
109        // Find object IDs in the range
110        final ArrayList<String> completions = new ArrayList<>();
111        session.perform(new TransactionAction() {
112            @Override
113            public void run(Session session) throws Exception {
114                final Transaction tx = session.getTransaction();
115                final TreeSet<Integer> storageIds = new TreeSet<>();
116                final ArrayList<NavigableSet<ObjId>> idSets = new ArrayList<>();
117                for (SchemaVersion schemaVersion : tx.getSchema().getSchemaVersions().values()) {
118                    for (SchemaItem schemaItem : schemaVersion.getSchemaItemMap().values()) {
119                        if (!(schemaItem instanceof ObjType))
120                            continue;
121                        final int storageId = schemaItem.getStorageId();
122                        if ((min != null && storageId < min.getStorageId()) || (max != null && storageId > max.getStorageId()))
123                            continue;
124                        NavigableSet<ObjId> idSet = tx.getAll(storageId);
125                        if (min != null) {
126                            try {
127                                idSet = idSet.tailSet(min, true);
128                            } catch (IllegalArgumentException e) {
129                                // ignore
130                            }
131                        }
132                        if (max != null) {
133                            try {
134                                idSet = idSet.headSet(max, true);
135                            } catch (IllegalArgumentException e) {
136                                // ignore
137                            }
138                        }
139                        idSets.add(idSet);
140                    }
141                }
142                int count = 0;
143                for (ObjId id : NavigableSets.union(idSets)) {
144                    completions.add(id.toString());
145                    count++;
146                    if (count >= session.getLineLimit() + 1)
147                        break;
148                }
149            }
150        });
151
152        // Throw exception with completions
153        throw new ParseException(ctx, usage).addCompletions(completions);
154    }

158    /**
159     * Object meta-data.
160     */
161    public static class ObjInfo {
162
163        private final ObjId id;
164        private final SchemaVersion schemaVersion;
165        private final ObjType type;
166
167        public ObjInfo(Transaction tx, ObjId id) {
168            this.id = id;
169            this.schemaVersion = tx.getSchema().getVersion(tx.getSchemaVersion(id));
170            this.type = this.schemaVersion.getSchemaItem(id.getStorageId(), ObjType.class);
171        }
172
173        public ObjId getObjId() {
174            return this.id;
175        }
176
177        public SchemaVersion getSchemaVersion() {
178            return this.schemaVersion;
179        }
180
181        public ObjType getObjType() {
182            return this.type;
183        }
184
185        @Override
186        public String toString() {
187            return this.id + " type " + this.type.getName() + "#" + this.type.getStorageId()
188              + " version " + this.schemaVersion.getVersionNumber();
189        }
190    }
