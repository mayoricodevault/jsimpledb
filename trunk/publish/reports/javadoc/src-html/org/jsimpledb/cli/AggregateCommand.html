008package org.jsimpledb.cli;
009
010import com.google.common.base.Function;
011import com.google.common.collect.Iterables;
012
013import java.util.ArrayList;
014import java.util.List;
015
016import org.dellroad.stuff.string.ParseContext;
017
018/**
019 * A command that consists of sub-commands.
020 */
021public abstract class AggregateCommand extends Command {
022
023    private final ArrayList<Command> subCommands = new ArrayList<>();
024
025    public AggregateCommand(String name) {
026        super(name);
027    }
028
029    public AggregateCommand(AggregateCommand parent, String name) {
030        super(parent, name);
031    }
032
033    public AggregateCommand(String prefix, String name) {
034        super(prefix, name);
035    }
036
037    public List<Command> getSubCommands() {
038        return this.subCommands;
039    }
040
041    @Override
042    public Action parse(Session session, ParseContext ctx) throws ParseException {
043        String subCommandName = null;
044        try {
045            subCommandName = ctx.matchPrefix("([^\\s]+)\\s*").group(1);
046        } catch (IllegalArgumentException e) {
047            // ignore
048        }
049        if (subCommandName != null) {
050            for (Command subCommand : this.getSubCommands()) {
051                if (subCommand.getName().equals(subCommandName))
052                    return subCommand.parse(session, ctx);
053            }
054        }
055
056        // Build error message
057        String errmsg = "";
058        if (this.getName().length() > 0) {
059            errmsg += "usage: " + this.getFullName() + " [";
060            for (int i = 0; i < this.subCommands.size(); i++) {
061                errmsg += i > 0 ? " | " : " ";
062                errmsg += this.subCommands.get(i).getName();
063            }
064            errmsg += " ]";
065        } else
066            errmsg = "unknown command `" + this.getFullName(subCommandName) + "'; type `help' for help.";
067
068        // Throw exception
069        throw new ParseException(ctx, errmsg).addCompletions(Iterables.transform(
070          Iterables.filter(Iterables.transform(this.getSubCommands(), Command.NAME_FUNCTION), new PrefixPredicate(subCommandName)),
071          new PrefixFunction(subCommandName)));
072    }
073
074// PrefixFunction
075
076    private class PrefixFunction implements Function<String, String> {
077
078        private final String prefix;
079
080        PrefixFunction(String prefix) {
081            this.prefix = prefix;
082        }
083
084        @Override
085        public String apply(String commandName) {
086            return (this.prefix != null ? commandName.substring(prefix.length()) : commandName) + " ";
087        }
088    }
089}
