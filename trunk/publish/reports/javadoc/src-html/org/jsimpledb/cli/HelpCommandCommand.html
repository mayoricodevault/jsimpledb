008package org.jsimpledb.cli;
009
010import java.io.PrintWriter;
011
012import org.dellroad.stuff.string.ParseContext;
013
014/**
015 * Help command for a "leaf" (non-aggregate) command.
016 */
017public class HelpCommandCommand extends Command implements Action {
018
019    private final Command command;
020
021    public HelpCommandCommand(Command command) {
022        super("help " + command.getPrefix(), command.getName());
023        this.command = command;
024    }
025
026    @Override
027    public Action parse(Session session, ParseContext ctx) throws ParseException {
028        // ignore any extra parameters
029        return this;
030    }
031
032    @Override
033    public String getHelpSummary() {
034        return "Displays information about the `" + this.command.getFullName() + "' command";
035    }
036
037// Action
038
039    @Override
040    public void run(Session session) throws Exception {
041        final PrintWriter out = new PrintWriter(session.getConsole().getOutput());
042        out.println(this.command.getFullName() + " - " + this.command.getHelpDetail());
043    }
044}
