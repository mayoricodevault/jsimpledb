033/**
034 * Parses command line parameters, including optional flags, based on a specification string.
035 *
036 * <p>
037 * The specification string contains whitespace-separated parameter specifications; see {@link Param} for syntax.
038 * </p>
039 */
040public class ParamParser implements Parser<Map<String, Object>> {
041
042    private final LinkedHashSet<Param> optionFlags = new LinkedHashSet<>();
043    private final ArrayList<Param> params = new ArrayList<>();
044
045    public ParamParser(String spec) {
046        if (spec.length() > 0) {
047            for (String pspec : spec.split("\\s+")) {
048                final Param param = new Param(pspec);
049                if (param.isOption())
050                    this.optionFlags.add(param);
051                else
052                    this.params.add(param);
053            }
054        }
055    }
056
057    /**
058     * Build a usage summary string.
059     */
060    public String getUsage(String commandName) {
061        final StringBuilder buf = new StringBuilder(commandName);
062        for (Param param : this.optionFlags) {
063            buf.append(" [").append(param.getOptionFlag());
064            if (param.getTypeName() != null)
065                buf.append(' ').append(param.getName());
066            buf.append(']');
067        }
068        for (Param param : this.params) {
069            buf.append(' ');
070            if (param.getMin() == 0)
071                buf.append('[');
072            buf.append(param.getName());
073            if (param.getMax() > 1)
074                buf.append(" ...");
075            if (param.getMin() == 0)
076                buf.append(']');
077        }
078        return buf.toString();
079    }
080
081    /**
082     * Get option flags.
083     */
084    public Set<Param> getOptionFlags() {
085        return this.optionFlags;
086    }
087
088    /**
089     * Get regular parameters, in expected order.
090     */
091    public List<Param> getParameters() {
092        return this.params;
093    }
094
095    /**
096     * Convert parameter spec type name into a {@link Parser}.
097     *
098     * <p>
099     * The implementation in {@link ParamParser} supports the Java primitive types, and
100     * {@code word} for a {@link String} containing one or more non-whitespace characters.
101     * Subclasses should override as required to add additional supported types.
102     * </p>
103     *
104     * @param typeName name of type
105     * @throws IllegalArgumentException if {@code typeName} is unknown
106     */
107    protected Parser<?> getParser(String typeName) {
108        if (typeName == null)
109            throw new IllegalArgumentException("null typeName");
110        if (typeName.equals("word"))
111            return new WordParser("parameter");
112        for (Primitive<?> prim : Primitive.values()) {
113            if (prim == Primitive.VOID)
114                continue;
115            if (typeName.equals(prim.getName()))
116                return this.createPrimitiveParser(prim);
117        }
118        throw new IllegalArgumentException("unknown parameter type `" + typeName + "'");
119    }
120
121    // This method exists solely to bind the generic type parameters
122    private <T> Parser<T> createPrimitiveParser(final Primitive<T> prim) {
123        return new Parser<T>() {
124            @Override
125            public T parse(ParseSession session, ParseContext ctx, boolean complete) {
126                try {
127                    return prim.parseValue(ctx.matchPrefix("[^\\s]+").group());
128                } catch (IllegalArgumentException e) {
129                    throw new ParseException(ctx, "invalid " + prim.getName() + " value", e);
130                }
131            }
132        };
133    }
134
135    /**
136     * Parse command line parameters.
137     *
138     * @param session associated session
139     * @param ctx parse context positioned at whitespace preceeding parameters (if any)
140     * @param complete true if we're only determining completions
141     * @throws ParseException if parse fails
142     */
143    @Override
144    public Map<String, Object> parse(ParseSession session, ParseContext ctx, boolean complete) {
145
146        // Store results here
147        final HashMap<String, Object> values = new HashMap<String, Object>();
148
149        // First parse options
150        boolean needSpace = !this.params.isEmpty() && this.params.get(0).getMin() > 0;
151        while (true) {
152            new SpaceParser(needSpace).parse(ctx, complete);
153            needSpace = false;
154            if (ctx.getInput().matches("(?s)^--([\\s;].*)?$")) {
155                ctx.setIndex(ctx.getIndex() + 2);
156                needSpace = !this.params.isEmpty() && this.params.get(0).getMin() > 0;
157                break;
158            }
159            final Matcher matcher = ctx.tryPattern("(-[^\\s;]+)");
160            if (matcher == null)
161                break;
162            final String option = matcher.group(1);
163            final Param param = Iterables.find(this.optionFlags, new Predicate<Param>() {
164                @Override
165                public boolean apply(Param param) {
166                    return option.equals(param.getOptionFlag());
167                }
168            }, null);
169            if (param == null) {
170                throw new ParseException(ctx, "unrecognized option `" + option + "'").addCompletions(
171                  ParseUtil.complete(Iterables.transform(this.optionFlags, new Function<Param, String>() {
172                    @Override
173                    public String apply(Param param) {
174                        return param.getOptionFlag();
175                    }
176                }), option));
177            }
178            final Parser<?> parser = param.getParser();
179            if (parser != null) {
180                new SpaceParser(true).parse(ctx, complete);
181                values.put(param.getName(), parser.parse(session, ctx, complete));
182            } else
183                values.put(param.getName(), true);
184            needSpace = !this.params.isEmpty() && this.params.get(0).getMin() > 0;
185        }
186
187        // Next parse parameters
188        for (Param param : this.params) {
189            final ArrayList<Object> paramValues = new ArrayList<>();
190            final String typeName = param.getTypeName();
191            final Parser<?> parser = param.getParser();
192            while (paramValues.size() < param.getMax()) {
193                new SpaceParser(needSpace).parse(ctx, complete);
194                needSpace = false;
195                if (!ctx.getInput().matches("(?s)^[^\\s;].*$")) {
196                    if (complete) {
197                        parser.parse(session, new ParseContext(""), true);      // calculate completions from empty string
198                        throw new ParseException(ctx, "");                      // should never get here
199                    }
200                    break;
201                }
202                paramValues.add(parser.parse(session, ctx, complete));
203                needSpace = paramValues.size() < param.getMin();
204            }
205            if (paramValues.size() < param.getMin()) {
206                final ParseException e = new ParseException(ctx, "missing `" + param.getName() + "' parameter");
207                if (complete) {
208                    try {
209                        parser.parse(session, new ParseContext(""), true);
210                    } catch (ParseException e2) {
211                        e.addCompletions(e2.getCompletions());
212                    }
213                }
214                throw e;
215            }
216            if (param.getMax() > 1)
217                values.put(param.getName(), Arrays.asList(paramValues.toArray()));
218            else if (!paramValues.isEmpty())
219                values.put(param.getName(), paramValues.get(0));
220        }
221
222        // Check for trailing garbage
223        new SpaceParser().parse(ctx, complete);
224        if (!ctx.getInput().matches("(?s)^(;.*)?$"))
225            throw new ParseException(ctx);
226
227        // Done
228        return values;
229    }

231// Param
232
233    /**
234     * Represents one parsed parameter specification.
235     *
236     * <p>
237     * {@link String} form is {@code -flag:name:type}, where the {@code -flag} is optional and indicates
238     * an option flag, {@code name} is the name of the flag or parameter, and {@code type} is optional as well:
239     * if missing, it indicates either an argumment-less option flag or a "word" type ({@link String} that is a
240     * sequence of one or more non-whitespace characters). Otherwise {@code type} is the name of the parameter type.
241     * </p>
242     *
243     * <p>
244     * Non-option parameters may have a {@code ?} suffix if optional,
245     * or a {@code +}, or {@code *} suffix if repeatable, in which case the result is a {@link List}.
246     * </p>
247     *
248     * <p>
249     * Spec string syntax examples:
250     * <ul>
251     *  <li><code>-v:foo</code> - boolean flag named {@code foo}</li>
252     *  <li><code>-v:foo:int</code> - integer flag named {@code foo}</li>
253     *  <li><code>foo</code> - string (word) parameter</li>
254     *  <li><code>foo:int</code> - {@code int} parameter</li>
255     *  <li><code>foo?</code> - optional final parameter</li>
256     *  <li><code>foo*</code> - array of zero or more final parameters</li>
257     *  <li><code>foo+</code> - array of one or more final parameters</li>
258     *  <li><code>foo:int+</code> - array of one or more final {@code int} parameters</li>
259     * </ul>
260     * </p>
261     */
262    public class Param {
263
264        private final String optionFlag;
265        private final String name;
266        private final String typeName;
267        private final Parser<?> parser;
268        private final int min;
269        private final int max;
270
271        public Param(String spec) {
272
273            // Sanity check
274            if (spec == null)
275                throw new IllegalArgumentException("null spec");
276
277            // Apply pattern
278            final Pattern pattern = Pattern.compile("((-[^\\s:]+):)?([^-][^\\s:?+*]*)(:([^\\s?+*]+))?([?+*])?");
279            final Matcher matcher = pattern.matcher(spec);
280            if (!matcher.matches())
281                throw new IllegalArgumentException("invalid parameter spec `" + spec + "'");
282
283            // Get components
284            this.optionFlag = matcher.group(2);
285            this.name = matcher.group(3);
286            this.typeName = matcher.group(5);
287            final String repeat = matcher.group(6);
288            if (repeat == null) {
289                this.min = 1;
290                this.max = 1;
291            } else if (repeat.charAt(0) == '?') {
292                this.min = 0;
293                this.max = 1;
294            } else if (repeat.charAt(0) == '*') {
295                this.min = 0;
296                this.max = Integer.MAX_VALUE;
297            } else if (repeat.charAt(0) == '+') {
298                this.min = 1;
299                this.max = Integer.MAX_VALUE;
300            } else
301                throw new IllegalArgumentException("invalid parameter spec `" + spec + "'");
302
303            // Get parser
304            this.parser = this.typeName != null ?
305              ParamParser.this.getParser(typeName) : !this.isOption() ? new WordParser("parameter") : null;
306        }
307
308        public String getOptionFlag() {
309            return this.optionFlag;
310        }
311
312        public boolean isOption() {
313            return this.optionFlag != null;
314        }
315
316        public String getName() {
317            return this.name;
318        }
319
320        public String getTypeName() {
321            return this.typeName;
322        }
323
324        public int getMin() {
325            return this.min;
326        }
327
328        public int getMax() {
329            return this.max;
330        }
331
332        public Parser<?> getParser() {
333            return this.parser;
334        }
335
336        public String toString() {
337            return (this.optionFlag != null ? this.optionFlag + ":" : "")
338              + this.name
339              + (this.typeName != null ? ":" + this.typeName : "")
340              + (this.min == 0 && this.max == 1 ? "?" :
341               this.min == 0 && this.max == Integer.MAX_VALUE ? "*" :
342               this.min == 1 && this.max == Integer.MAX_VALUE ? "+" : "");
343        }
344    }
345}
