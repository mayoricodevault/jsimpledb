008package org.jsimpledb.cli.cmd;
009
010import java.util.Map;
011
012import org.jsimpledb.ValidationMode;
013import org.jsimpledb.cli.CliSession;
014import org.jsimpledb.parse.EnumNameParser;
015import org.jsimpledb.parse.ParseContext;
016import org.jsimpledb.parse.Parser;
017
018@Command
019public class SetValidationModeCommand extends AbstractCommand {
020
021    public SetValidationModeCommand() {
022        super("set-validation-mode mode:mode");
023    }
024
025    @Override
026    public String getHelpSummary() {
027        return "Sets the validation mode for JSimpleDB transactions";
028    }
029
030    @Override
031    public String getHelpDetail() {
032        return "Sets the validation mode for JSimpleDB transactions, one of `disabled', `manual', or `automatic' (the default)."
033          + " This setting is only used when the CLI is running in JSimpleDB mode.";
034    }
035
036    @Override
037    protected Parser<?> getParser(String typeName) {
038        return "mode".equals(typeName) ? new EnumNameParser<ValidationMode>(ValidationMode.class) : super.getParser(typeName);
039    }
040
041    @Override
042    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
043        final ValidationMode mode = (ValidationMode)params.get("mode");
044        return new CliSession.Action() {
045            @Override
046            public void run(CliSession session) throws Exception {
047                session.setValidationMode(mode);
048                session.getWriter().println("Set validation mode to " + mode);
049                if (session.getJSimpleDB() == null)
050                    session.getWriter().println("Warning: no JSimpleDB instance was created, so validation mode will not be used");
051            }
052        };
053    }
054}
