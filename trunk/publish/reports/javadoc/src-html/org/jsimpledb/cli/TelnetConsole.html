020/**
021 * A {@link Console} connected to a telnet client via a {@link Socket}.
022 *
023 * <p>
024 * This class can be used to implement an embedded CLI console within an application, accessed via telnet. For example:
025 * <pre>
026 *      // Accept a new connection
027 *      final Socket socket = ...
028 *
029 *      // Set up telnet CLI console
030 *      final TelnetConsole console = TelnetConsole.create(database, socket);
031 *      console.getSession().registerStandardFunctions();
032 *      console.getSession().registerStandardCommands();
033 *
034 *      // Run the console
035 *      console.run();
036 * </pre>
037 */
038public final class TelnetConsole extends Console {
039
040    /**
041     * Internal constructor.
042     */
043    private TelnetConsole(Database db, JSimpleDB jdb, InputStream input, OutputStream output,
044      jline.Terminal terminal, String encoding, String appName) throws IOException {
045        super(db, jdb, input, output, terminal, encoding, appName);
046    }
047
048    /**
049     * Simplified factory method for core API CLI mode.
050     *
051     * @param db core API {@link Database}
052     * @param socket socket connected to telnet client
053     * @return new telnet console instance
054     * @throws IOException if an I/O error occurs
055     */
056    public static TelnetConsole create(Database db, Socket socket) throws IOException {
057        return TelnetConsole.create(db, null, socket.getInputStream(), socket.getOutputStream(), null, null);
058    }
059
060    /**
061     * Simplified factory method for {@link JSimpleDB} API CLI mode.
062     *
063     * @param jdb {@link JSimpleDB} database
064     * @param socket socket connected to telnet client
065     * @return new telnet console instance
066     * @throws IOException if an I/O error occurs
067     */
068    public static TelnetConsole create(JSimpleDB jdb, Socket socket) throws IOException {
069        return TelnetConsole.create(null, jdb, socket.getInputStream(), socket.getOutputStream(), null, null);
070    }
071
072    /**
073     * Generic factory method.
074     *
075     * @param db core API {@link Database}; must be null if and only if {@code jdb} is not null
076     * @param jdb {@link JSimpleDB} database; must be null if and only if {@code db} is not null
077     * @param input console input
078     * @param output console output
079     * @param encoding character encoding for {@code terminal}, or null for default
080     * @param appName JLine application name, or null for none
081     * @return new telnet console instance
082     * @throws IOException if an I/O error occurs
083     * @throws IllegalArgumentException if {@code db} and {@code jdb} are both null or both not null
084     */
085    public static TelnetConsole create(Database db, JSimpleDB jdb, InputStream input, OutputStream output,
086      String encoding, String appName) throws IOException {
087
088        // Set up nvt4j; ignore the initial clear & reposition
089        final nvt4j.impl.Terminal nvt4jTerminal = new nvt4j.impl.Terminal(input, output) {
090
091            private boolean cleared;
092            private boolean moved;
093
094            @Override
095            public void clear() throws IOException {
096                if (this.cleared)
097                    super.clear();
098                this.cleared = true;
099            }
100
101            @Override
102            public void move(int row, int col) throws IOException {
103                if (this.moved)
104                    super.move(row, col);
105                this.moved = true;
106            }
107        };
108        nvt4jTerminal.put(nvt4j.impl.Terminal.AUTO_WRAP_ON);
109        nvt4jTerminal.setCursor(true);
110
111        // Have JLine do input & output through telnet terminal
112        final InputStream jlineInput = new InputStream() {
113            @Override
114            public int read() throws IOException {
115                return nvt4jTerminal.get();
116            }
117        };
118        final OutputStream jlineOutput = new OutputStream() {
119            @Override
120            public void write(int value) throws IOException {
121                nvt4jTerminal.put(value);
122            }
123        };
124
125        // Build console
126        return new TelnetConsole(db, jdb, jlineInput, jlineOutput,
127          new TelnetTerminal(TerminalFactory.get(), nvt4jTerminal), encoding, appName);
128    }

132    /**
133     * Wrapper class for a JLine {@link jline.Terminal} that forwards all methods to the wrapped delegate.
134     */
135    public static class ForwardingTerminal implements jline.Terminal {
136
137        protected final jline.Terminal terminal;
138
139        public ForwardingTerminal(jline.Terminal terminal) {
140            if (terminal == null)
141                throw new IllegalArgumentException("null terminal");
142            this.terminal = terminal;
143        }
144
145        @Override
146        public void init() throws Exception {
147            this.terminal.init();
148        }
149
150        @Override
151        public void restore() throws Exception {
152            this.terminal.restore();
153        }
154
155        @Override
156        public void reset() throws Exception {
157            this.terminal.reset();
158        }
159
160        @Override
161        public boolean isSupported() {
162            return this.terminal.isSupported();
163        }
164
165        @Override
166        public int getWidth() {
167            return this.terminal.getWidth();
168        }
169
170        @Override
171        public int getHeight() {
172            return this.terminal.getHeight();
173        }
174
175        @Override
176        public boolean isAnsiSupported() {
177            return this.terminal.isAnsiSupported();
178        }
179
180        @Override
181        public OutputStream wrapOutIfNeeded(OutputStream out) {
182            return this.terminal.wrapOutIfNeeded(out);
183        }
184
185        @Override
186        public InputStream wrapInIfNeeded(InputStream in) throws IOException {
187            return this.terminal.wrapInIfNeeded(in);
188        }
189
190        @Override
191        public boolean hasWeirdWrap() {
192            return this.terminal.hasWeirdWrap();
193        }
194
195        @Override
196        public boolean isEchoEnabled() {
197            return this.terminal.isEchoEnabled();
198        }
199
200        @Override
201        public void setEchoEnabled(boolean enabled) {
202            this.terminal.setEchoEnabled(enabled);
203        }
204
205        @Override
206        public String getOutputEncoding() {
207            return this.terminal.getOutputEncoding();
208        }
209    }

213    /**
214     * JLine {@link jline.Terminal} that gets window size information from an associated nvt4j {@link nvt4j.impl.Terminal}.
215     */
216    public static class TelnetTerminal extends ForwardingTerminal {
217
218        private final nvt4j.impl.Terminal nvt;
219
220        public TelnetTerminal(jline.Terminal terminal, nvt4j.impl.Terminal nvt) {
221            super(terminal);
222            if (nvt == null)
223                throw new IllegalArgumentException("null nvt");
224            this.nvt = nvt;
225        }
226
227        @Override
228        public int getWidth() {
229            final int width = this.nvt.getColumns();
230            return width > 0 ? width : super.getWidth();
231        }
232
233        @Override
234        public int getHeight() {
235            final int height = this.nvt.getRows();
236            return height > 0 ? height : super.getHeight();
237        }
238

240    }
