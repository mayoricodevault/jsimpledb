008package org.jsimpledb.cli;
009
010import java.io.BufferedInputStream;
011import java.io.File;
012import java.io.FileDescriptor;
013import java.io.FileInputStream;
014import java.io.InputStream;
015import java.util.ArrayDeque;
016import java.util.Arrays;
017
018import org.dellroad.stuff.main.MainClass;
019import org.jsimpledb.core.Database;
020import org.jsimpledb.kv.KVDatabase;
021import org.jsimpledb.kv.fdb.FoundationKVDatabase;
022import org.jsimpledb.kv.simple.SimpleKVDatabase;
023import org.jsimpledb.kv.simple.XMLKVDatabase;
024import org.jsimpledb.schema.SchemaModel;
025import org.jsimpledb.util.ByteUtil;
026
027/**
028 * CLI main entry point.
029 */
030public class Main extends MainClass {
031
032    private static final int KV_MEM = 0;
033    private static final int KV_FDB = 1;
034    private static final int KV_XML = 2;
035
036    private int kvType = KV_MEM;
037    private String fdbClusterFile;
038    private File xmlFile;
039    private byte[] keyPrefix;
040    private File schemaFile;
041    private int schemaVersion;
042    private boolean newSchema;
043    private boolean verbose;
044    private boolean readOnly;
045
046    @Override
047    public int run(String[] args) throws Exception {
048
049        // Parse command line
050        File outputFile = null;
051        final ArrayDeque<String> params = new ArrayDeque<String>(Arrays.asList(args));
052        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
053            final String option = params.removeFirst();
054            if (option.equals("-h") || option.equals("--help")) {
055                this.usageMessage();
056                return 0;
057            } else if (option.equals("-ro") || option.equals("--read-only"))
058                this.readOnly = true;
059            else if (option.equals("-v") || option.equals("--verbose"))
060                this.verbose = true;
061            else if (option.equals("--schema")) {
062                if (params.isEmpty())
063                    this.usageError();
064                this.schemaFile = new File(params.removeFirst());
065            } else if (option.equals("--version")) {
066                if (params.isEmpty())
067                    this.usageError();
068                final String vstring = params.removeFirst();
069                try {
070                    this.schemaVersion = Integer.parseInt(vstring);
071                    if (this.schemaVersion < 0)
072                        throw new IllegalArgumentException("schema version is negative");
073                } catch (Exception e) {
074                    System.err.println(this.getName() + ": invalid schema version `" + vstring + "': " + e.getMessage());
075                    this.usageError();
076                }
077            } else if (option.equals("--new-schema"))
078                this.newSchema = true;
079            else if (option.equals("--mem"))
080                this.kvType = KV_MEM;
081            else if (option.equals("--prefix")) {
082                if (params.isEmpty())
083                    this.usageError();
084                final String value = params.removeFirst();
085                try {
086                    this.keyPrefix = ByteUtil.parse(value);
087                } catch (IllegalArgumentException e) {
088                    this.keyPrefix = value.getBytes("UTF-8");
089                }
090            } else if (option.equals("--fdb")) {
091                if (params.isEmpty())
092                    this.usageError();
093                this.kvType = KV_FDB;
094                this.fdbClusterFile = params.removeFirst();
095                if (!new File(this.fdbClusterFile).exists())
096                    System.err.println(this.getName() + ": file `" + this.fdbClusterFile + "' does not exist");
097            } else if (option.equals("--xml")) {
098                if (params.isEmpty())
099                    this.usageError();
100                this.kvType = KV_XML;
101                this.xmlFile = new File(params.removeFirst());
102                if (!this.xmlFile.exists()) {
103                    System.err.println(this.getName() + ": file `" + this.xmlFile + "' does not exist");
104                    return 1;
105                }
106            } else if (option.equals("--"))
107                break;
108            else {
109                System.err.println(this.getName() + ": unknown option `" + option + "'");
110                this.usageError();
111            }
112        }
113        switch (params.size()) {
114        case 0:
115            break;
116        default:
117            this.usageError();
118            return 1;
119        }
120        if (this.kvType != KV_FDB && this.keyPrefix != null) {
121            System.err.println(this.getName() + ": option `--prefix' is only valid in combination with `--fdb'");
122            this.usageError();
123        }
124
125        // Create KV database
126        final KVDatabase kvdb;
127        switch (this.kvType) {
128        case KV_MEM:
129            kvdb = new SimpleKVDatabase();
130            break;
131        case KV_FDB:
132        {
133            final FoundationKVDatabase fdb = new FoundationKVDatabase();
134            fdb.setClusterFilePath(this.fdbClusterFile);
135            fdb.setKeyPrefix(this.keyPrefix);
136            fdb.start();
137            kvdb = fdb;
138            break;
139        }
140        case KV_XML:
141            kvdb = new XMLKVDatabase(this.xmlFile);
142            break;
143        default:
144            throw new RuntimeException("internal error");
145        }
146
147        // Set up console
148        final Console console = new Console(new Database(kvdb), new FileInputStream(FileDescriptor.in), System.out);
149        console.setHistoryFile(new File(new File(System.getProperty("user.home")), ".jsimpledb_history"));
150        console.getSession().setReadOnly(this.readOnly);
151        console.getSession().setVerbose(this.verbose);
152        console.getSession().setSchemaVersion(this.schemaVersion);
153        console.getSession().setAllowNewSchema(this.newSchema);
154        if (this.schemaFile != null) {
155            try {
156                final InputStream input = new BufferedInputStream(new FileInputStream(this.schemaFile));
157                try {
158                    console.getSession().setSchemaModel(SchemaModel.fromXML(input));
159                } finally {
160                    input.close();
161                }
162            } catch (Exception e) {
163                System.err.println(this.getName() + ": can't load schema from `" + this.schemaFile + "': " + e.getMessage());
164                return 1;
165            }
166        }
167
168        // Run console
169        console.run();
170
171        // Shut down KV database
172        switch (this.kvType) {
173        case KV_FDB:
174            ((FoundationKVDatabase)kvdb).stop();
175            break;
176        default:
177            break;
178        }
179
180        // Done
181        return 0;
182    }
183
184    protected String getName() {
185        return "jsimpledb";
186    }
187
188    @Override
189    protected void usageMessage() {
190        System.err.println("Usage:");
191        System.err.println("  " + this.getName() + " [options]");
192        System.err.println("Options:");
193        System.err.println("  --fdb file        Use FoundationDB with specified cluster file");
194        System.err.println("  --mem             Use an empty in-memory database (default)");
195        System.err.println("  --prefix prefix   FoundationDB key prefix (hex or string)");
196        System.err.println("  --read-only       Disallow database modifications");
197        System.err.println("  --new-schema      Allow recording of a new database schema version");
198        System.err.println("  --schema file     Load database schema from XML file");
199        System.err.println("  --xml file        Use the specified XML flat file database");
200        System.err.println("  --version num     Specify database schema version (default highest recorded)");
201        System.err.println("  -h, --help        Show this help message");
202        System.err.println("  -v, --verbose     Show verbose error messages");
203    }
204
205    public static void main(String[] args) throws Exception {
206        new Main().doMain(args);
207    }
208}
