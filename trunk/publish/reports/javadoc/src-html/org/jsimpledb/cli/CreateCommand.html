008package org.jsimpledb.cli;
009
010import java.util.TreeSet;
011
012import org.jsimpledb.core.ObjId;
013import org.jsimpledb.core.Transaction;
014import org.jsimpledb.util.ParseContext;
015
016public class CreateCommand extends AbstractCommand {
017
018    public CreateCommand() {
019        super("create");
020    }
021
022    @Override
023    public String getUsage() {
024        return this.name + " [-c count] [-v version] type";
025    }
026
027    @Override
028    public String getHelpSummary() {
029        return "create new object instance(s)";
030    }
031
032    @Override
033    public String getHelpDetail() {
034        return "Creates a new instance of the specified type and outputs it. Use `-v' to force a specific object version"
035        + " and `-c' to create multiple instances.";
036    }
037
038    @Override
039    public Action parseParameters(Session session, ParseContext ctx) {
040
041        // Parse options
042        final ParamParser parser = new ParamParser(1, 1, this.getUsage(), "-c@", "-v@").parse(ctx);
043        final String vflag = parser.getFlag("-v");
044        final int version;
045        if (vflag != null) {
046            try {
047                version = Integer.parseInt(vflag);
048                if (version < 0)
049                    throw new IllegalArgumentException("version is negative");
050            } catch (IllegalArgumentException e) {
051                throw new ParseException(ctx, "invalid version `" + vflag + "'");
052            }
053        } else
054            version = 0;
055        final String cflag = parser.getFlag("-c");
056        final int count;
057        if (cflag != null) {
058            try {
059                count = Integer.parseInt(vflag);
060                if (count < 0)
061                    throw new IllegalArgumentException("count is negative");
062            } catch (IllegalArgumentException e) {
063                throw new ParseException(ctx, "invalid count `" + cflag + "'");
064            }
065        } else
066            count = 1;
067
068        // Parse type
069        final int storageId = Util.parseTypeName(session, ctx, parser.getParam(0));
070
071        // Return action that creates instance(s) and pushes them onto the stack
072        return new Action() {
073            @Override
074            public void run(Session session) throws Exception {
075                CreateCommand.this.push(session, new ObjectChannel(session) {
076                    @Override
077                    public TreeSet<ObjId> getItems(Session session) {
078                        final Transaction tx = session.getTransaction();
079                        final TreeSet<ObjId> set = new TreeSet<ObjId>();
080                        for (int i = 0; i < count; i++)
081                            set.add(version != 0 ? tx.create(storageId, version) : tx.create(storageId));
082                        session.getWriter().println("Created " + count + " object" + (count != 1 ? "s" : ""));
083                        return set;
084                    }
085                });
086            }
087        };
088    }
089}
