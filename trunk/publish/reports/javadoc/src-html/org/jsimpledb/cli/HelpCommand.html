008package org.jsimpledb.cli;
009
010import java.io.PrintWriter;
011
012import org.dellroad.stuff.string.ParseContext;
013
014/**
015 * Help command for an {@link AggregateCommand}.
016 */
017public class HelpCommand extends AggregateCommand implements Action {
018
019    private final AggregateCommand target;
020
021    public HelpCommand(RootCommand target) {
022        super("help");
023        this.target = target;
024    }
025
026    public HelpCommand(AggregateCommand target) {
027        super("help" + (target.getPrefix() != null ? " " + target.getPrefix() : ""), target.getName());
028        this.target = target;
029    }
030
031    public void addSubCommands() {
032        this.getSubCommands().clear();
033        for (Command command : this.target.getSubCommands()) {
034            final Command helpCommand;
035            if (command instanceof HelpCommand)
036                continue;
037            if (command instanceof AggregateCommand)
038                helpCommand = new HelpCommand((AggregateCommand)command);
039            else if (!command.getHelpDetail().equals(command.getHelpSummary()))
040                helpCommand = new HelpCommandCommand(command);
041            else
042                continue;
043            this.getSubCommands().add(helpCommand);
044        }
045        for (Command command : this.getSubCommands()) {
046            if (command instanceof HelpCommand)
047                ((HelpCommand)command).addSubCommands();
048        }
049    }
050
051    @Override
052    public Action parse(Session session, ParseContext ctx) throws ParseException {
053        if (ctx.getInput().length() != 0)
054            return super.parse(session, ctx);
055        return this;
056    }
057
058    @Override
059    public String getHelpSummary() {
060        return this.getPrefix() == null ?
061          "Displays information about the specified command" :
062          "Displays information about the `" + this.getName() + "' command";
063    }
064
065// Action
066
067    @Override
068    public void run(Session session) throws Exception {
069        final PrintWriter out = new PrintWriter(session.getConsole().getOutput());
070        out.println("Available commands:");
071        for (Command command : this.target.getSubCommands())
072            out.println(String.format("%8s - %s", command.getFullName(), command.getHelpSummary()));
073        if (!this.getSubCommands().isEmpty()) {
074            out.println("Additional topics available under `" + this.getFullName() + "':");
075            for (Command command : this.getSubCommands())
076                out.println("    " + command.getFullName());
077        }
078    }
079}
