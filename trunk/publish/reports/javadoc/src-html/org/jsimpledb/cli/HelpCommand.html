008package org.jsimpledb.cli;
009
010import java.io.PrintWriter;
011import java.util.SortedMap;
012
013import org.jsimpledb.util.ParseContext;
014
015public class HelpCommand extends AbstractCommand {
016
017    public HelpCommand() {
018        super("help");
019    }
020
021    @Override
022    public String getUsage() {
023        return this.name + " [command]";
024    }
025
026    @Override
027    public String getHelpSummary() {
028        return "display help information";
029    }
030
031    @Override
032    public String getHelpDetail() {
033        return "Displays help information about a command.";
034    }
035
036    @Override
037    public Action parseParameters(Session session, ParseContext ctx) {
038        final ParamParser parser = new ParamParser(0, 1, this.getUsage()).parse(ctx);
039
040        // Get all commands
041        final SortedMap<String, AbstractCommand> commandMap = CommandParser.getCommands();
042
043        // Find command specified, if any
044        final AbstractCommand command;
045        if (parser.getParams().isEmpty())
046            command = null;
047        else {
048            final String commandName = parser.getParam(0);
049            if ((command = commandMap.get(commandName)) == null) {
050                throw new ParseException(ctx, "unknown command `" + commandName + "'")
051                  .addCompletions(Util.complete(commandMap.keySet(), commandName, false));
052            }
053        }
054
055        // Return help action
056        return new Action() {
057            @Override
058            public void run(Session session) throws Exception {
059                final PrintWriter writer = session.getWriter();
060                if (command == null) {
061                    writer.println("Available commands:");
062                    for (AbstractCommand c : commandMap.values())
063                        writer.println(String.format("%24s - %s", c.getName(), c.getHelpSummary()));
064                } else {
065                    writer.println("Usage: " + command.getUsage());
066                    writer.println(command.getHelpDetail());
067                }
068            }
069        };
070    }
071}
