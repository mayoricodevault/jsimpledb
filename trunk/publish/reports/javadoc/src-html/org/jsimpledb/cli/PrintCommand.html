008package org.jsimpledb.cli;
009
010import java.io.IOException;
011import java.util.Arrays;
012import java.util.Map;
013
014import org.dellroad.stuff.string.ParseContext;
015import org.jsimpledb.core.CounterField;
016import org.jsimpledb.core.Field;
017import org.jsimpledb.core.ListField;
018import org.jsimpledb.core.MapField;
019import org.jsimpledb.core.ObjId;
020import org.jsimpledb.core.SetField;
021import org.jsimpledb.core.SimpleField;
022import org.jsimpledb.core.Transaction;
023
024public class PrintCommand extends Command {
025
026    public PrintCommand(AggregateCommand parent) {
027        super(parent, "print");
028    }
029
030    @Override
031    public Action parse(Session session, ParseContext ctx) throws ParseException {
032        final ObjId id = Util.parseObjId(session, ctx, this.getFullName() + " object-id");
033        return new TransactionAction() {
034            @Override
035            public void run(Session session) throws Exception {
036                PrintCommand.this.printObject(session, id);
037            }
038        };
039    }
040
041    @Override
042    public String getHelpSummary() {
043        return "Prints an object's schema version and fields";
044    }
045
046    private void printObject(Session session, ObjId id) throws IOException {
047
048        // Verify object exists
049        final Transaction tx = session.getTransaction();
050        final Util.ObjInfo info = Util.getObjInfo(tx, id);
051        if (info == null) {
052            session.getConsole().println("object " + id + " does not exist");
053            return;
054        }
055
056        // Display object meta-data
057        session.getConsole().println("object " + info);
058
059        // Calculate indent amount
060        int nameFieldSize = 0;
061        for (Field<?> field : info.getObjType().getFields().values())
062            nameFieldSize = Math.max(nameFieldSize, field.getName().length());
063        final char[] ichars = new char[nameFieldSize + 3];
064        Arrays.fill(ichars, ' ');
065        final String indent = new String(ichars);
066        final String eindent = indent + "  ";
067
068        // Display fields
069        for (Field<?> field : info.getObjType().getFields().values()) {
070            session.getConsole().print(String.format("%" + nameFieldSize + "s = ", field.getName()));
071            if (field instanceof SimpleField)
072                session.getConsole().println(this.readSimpleFieldStringValue(tx, (SimpleField<?>)field, id));
073            else if (field instanceof CounterField)
074                session.getConsole().println("" + tx.readCounterField(id, field.getStorageId(), false));
075            else if (field instanceof SetField) {
076                final SimpleField<?> elementField = ((SetField<?>)field).getElementField();
077                session.getConsole().println("{");
078                int count = 0;
079                for (Object elem : tx.readSetField(id, field.getStorageId())) {
080                    if (count >= session.getLineLimit()) {
081                        session.getConsole().println(eindent + "...");
082                        break;
083                    }
084                    session.getConsole().println(eindent + this.getSimpleFieldStringValue(elementField, elem));
085                    count++;
086                }
087                session.getConsole().println(indent + "}");
088            } else if (field instanceof ListField) {
089                final SimpleField<?> elementField = ((ListField<?>)field).getElementField();
090                session.getConsole().println("{");
091                int count = 0;
092                for (Object elem : tx.readListField(id, field.getStorageId())) {
093                    if (count >= session.getLineLimit()) {
094                        session.getConsole().println(eindent + "...");
095                        break;
096                    }
097                    session.getConsole().println(eindent + "[" + count + "]" + this.getSimpleFieldStringValue(elementField, elem));
098                    count++;
099                }
100                session.getConsole().println(indent + "}");
101            } else if (field instanceof MapField) {
102                final SimpleField<?> keyField = ((MapField<?, ?>)field).getKeyField();
103                final SimpleField<?> valueField = ((MapField<?, ?>)field).getValueField();
104                session.getConsole().println("{");
105                int count = 0;
106                for (Map.Entry<?, ?> entry : tx.readMapField(id, field.getStorageId()).entrySet()) {
107                    if (count >= session.getLineLimit()) {
108                        session.getConsole().println(eindent + "...");
109                        break;
110                    }
111                    session.getConsole().println(eindent + this.getSimpleFieldStringValue(keyField, entry.getKey())
112                      + " -> " + this.getSimpleFieldStringValue(valueField, entry.getValue()));
113                    count++;
114                }
115                session.getConsole().println(indent + "}");
116            } else
117                throw new RuntimeException("internal error");
118        }
119    }
120
121    // This method exists solely to bind the generic type parameters
122    @SuppressWarnings("unchecked")
123    private <T> String readSimpleFieldStringValue(Transaction tx, SimpleField<T> field, ObjId id) {
124        return field.getFieldType().toString((T)tx.readSimpleField(id, field.getStorageId(), false));
125    }
126
127    // This method exists solely to bind the generic type parameters
128    @SuppressWarnings("unchecked")
129    private <T> String getSimpleFieldStringValue(SimpleField<T> field, Object value) {
130        return field.getFieldType().toString((T)value);
131    }
132}
