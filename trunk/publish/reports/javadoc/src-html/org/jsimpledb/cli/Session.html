008package org.jsimpledb.cli;
009
010import java.io.IOException;
011import java.io.PrintWriter;
012import java.util.ArrayDeque;
013
014import org.jsimpledb.core.Database;
015import org.jsimpledb.core.SchemaVersion;
016import org.jsimpledb.core.Transaction;
017import org.jsimpledb.schema.NameIndex;
018import org.jsimpledb.schema.SchemaModel;
019
020import jline.console.ConsoleReader;
021
022/**
023 * Represents one console session.
024 */
025public class Session {
026
027    private final Database db;
028    private final ConsoleReader console;
029    private final PrintWriter writer;
030    private final ArrayDeque<Channel<?>> stack = new ArrayDeque<>();
031
032    private Transaction tx;
033    private SchemaModel schemaModel;
034    private NameIndex nameIndex;
035    private int schemaVersion;
036    private boolean allowNewSchema;
037    private boolean done;
038    private boolean verbose;
039    private boolean readOnly;
040    private int lineLimit = 16;
041
042// Constructors
043
044    public Session(Database db, ConsoleReader console) {
045        this.db = db;
046        this.console = console;
047        this.writer = new PrintWriter(console.getOutput(), true);
048    }
049
050// Accessors
051
052    public Database getDatabase() {
053        return this.db;
054    }
055
056    public ConsoleReader getConsole() {
057        return this.console;
058    }
059
060    public PrintWriter getWriter() {
061        return this.writer;
062    }
063
064    public ArrayDeque<Channel<?>> getStack() {
065        return this.stack;
066    }
067
068    public Transaction getTransaction() {
069        if (this.tx == null)
070            throw new IllegalStateException("no transaction associated with session");
071        return this.tx;
072    }
073
074    public SchemaModel getSchemaModel() {
075        return this.schemaModel;
076    }
077    public void setSchemaModel(SchemaModel schemaModel) {
078        this.schemaModel = schemaModel;
079        this.nameIndex = this.schemaModel != null ? new NameIndex(this.schemaModel) : null;
080    }
081
082    public NameIndex getNameIndex() {
083        return this.nameIndex != null ? this.nameIndex : new NameIndex(new SchemaModel());
084    }
085
086    public int getSchemaVersion() {
087        return this.schemaVersion;
088    }
089    public void setSchemaVersion(int schemaVersion) {
090        this.schemaVersion = schemaVersion;
091    }
092
093    public int getLineLimit() {
094        return this.lineLimit;
095    }
096    public void setLineLimit(int lineLimit) {
097        this.lineLimit = lineLimit;
098    }
099
100    public boolean getAllowNewSchema() {
101        return this.allowNewSchema;
102    }
103    public void setAllowNewSchema(boolean allowNewSchema) {
104        this.allowNewSchema = allowNewSchema;
105    }
106
107    public boolean isVerbose() {
108        return this.verbose;
109    }
110    public void setVerbose(boolean verbose) {
111        this.verbose = verbose;
112    }
113
114    public boolean isReadOnly() {
115        return this.readOnly;
116    }
117    public void setReadOnly(boolean readOnly) {
118        this.readOnly = readOnly;
119    }
120
121    public boolean isDone() {
122        return this.done;
123    }
124    public void setDone(boolean done) {
125        this.done = done;
126    }
127
128// Errors
129
130    public void report(Exception e) {
131        final String message = e.getLocalizedMessage();
132        try {
133            if (e instanceof ParseException)
134                this.console.println("Error: " + message);
135            else
136                this.console.println("Error: " + e.getClass().getSimpleName() + (message != null ? ": " + message : ""));
137            if (this.verbose || this.showStackTrace(e))
138                e.printStackTrace(this.writer);
139        } catch (IOException ioe) {
140            this.setDone(true);
141        }
142    }
143
144    protected boolean showStackTrace(Exception e) {
145        return e instanceof NullPointerException;
146    }
147
148// Transactions
149
150    public boolean perform(Action action) {
151        try {
152            final boolean newTransaction = this.tx == null;
153            if (newTransaction) {
154                if (!this.openTransaction())
155                    return false;
156            }
157            boolean success = false;
158            try {
159                action.run(this);
160                success = true;
161            } finally {
162                if (newTransaction && this.tx != null) {
163                    if (success)
164                        this.commitTransaction();
165                    else
166                        this.rollbackTransaction();
167                }
168            }
169            return success;
170        } catch (Exception e) {
171            this.report(e);
172            return false;
173        }
174    }
175
176    public boolean openTransaction() {
177        try {
178            if (this.tx != null)
179                throw new IllegalStateException("a transaction is already open");
180            this.tx = this.db.createTransaction(this.schemaModel, this.schemaVersion, this.allowNewSchema);
181            final SchemaVersion version = this.tx.getSchemaVersion();
182            this.setSchemaModel(version.getSchemaModel());
183            this.setSchemaVersion(version.getVersionNumber());
184            this.tx.setReadOnly(this.readOnly);
185            return true;
186        } catch (Exception e) {
187            this.tx = null;
188            this.report(e);
189            return false;
190        }
191    }
192
193    public boolean commitTransaction() {
194        try {
195            if (this.tx == null)
196                throw new IllegalStateException("no transaction");
197            this.tx.commit();
198            return true;
199        } catch (Exception e) {
200            this.report(e);
201            return false;
202        } finally {
203            this.tx = null;
204        }
205    }
206
207    public boolean rollbackTransaction() {
208        try {
209            if (this.tx == null)
210                throw new IllegalStateException("no transaction");
211            this.tx.rollback();
212            return true;
213        } catch (Exception e) {
214            this.report(e);
215            return false;
216        } finally {
217            this.tx = null;
218        }
219    }
220}
