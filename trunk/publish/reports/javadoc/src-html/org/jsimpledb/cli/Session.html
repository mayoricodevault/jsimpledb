008package org.jsimpledb.cli;
009
010import java.io.IOException;
011import java.io.PrintWriter;
012
013import org.jsimpledb.core.Database;
014import org.jsimpledb.core.SchemaVersion;
015import org.jsimpledb.core.Transaction;
016import org.jsimpledb.schema.NameIndex;
017import org.jsimpledb.schema.SchemaModel;
018
019import jline.console.ConsoleReader;
020
021/**
022 * Represents one console session.
023 */
024public class Session {
025
026    private final Database db;
027    private final ConsoleReader console;
028
029    private Transaction tx;
030    private SchemaModel schemaModel;
031    private NameIndex nameIndex;
032    private int schemaVersion;
033    private boolean allowNewSchema;
034    private boolean done;
035    private boolean verbose;
036    private boolean readOnly;
037    private int lineLimit = 16;
038
039    public Session(Database db, ConsoleReader console) {
040        this.db = db;
041        this.console = console;
042    }
043
044    public ConsoleReader getConsole() {
045        return this.console;
046    }
047
048    public Transaction getTransaction() {
049        return this.tx;
050    }
051
052    public SchemaModel getSchemaModel() {
053        return this.schemaModel;
054    }
055    public void setSchemaModel(SchemaModel schemaModel) {
056        this.schemaModel = schemaModel;
057        this.nameIndex = this.schemaModel != null ? new NameIndex(this.schemaModel) : null;
058    }
059
060    public NameIndex getNameIndex() {
061        return this.nameIndex;
062    }
063
064    public int getSchemaVersion() {
065        return this.schemaVersion;
066    }
067    public void setSchemaVersion(int schemaVersion) {
068        this.schemaVersion = schemaVersion;
069    }
070
071    public int getLineLimit() {
072        return this.lineLimit;
073    }
074    public void setLineLimit(int lineLimit) {
075        this.lineLimit = lineLimit;
076    }
077
078    public boolean getAllowNewSchema() {
079        return this.allowNewSchema;
080    }
081    public void setAllowNewSchema(boolean allowNewSchema) {
082        this.allowNewSchema = allowNewSchema;
083    }
084
085    public boolean isVerbose() {
086        return this.verbose;
087    }
088    public void setVerbose(boolean verbose) {
089        this.verbose = verbose;
090    }
091
092    public boolean isReadOnly() {
093        return this.readOnly;
094    }
095    public void setReadOnly(boolean readOnly) {
096        this.readOnly = readOnly;
097    }
098
099    public boolean isDone() {
100        return this.done;
101    }
102    public void setDone(boolean done) {
103        this.done = done;
104    }
105
106    public void report(Exception e) {
107        final String message = e.getLocalizedMessage();
108        try {
109            this.console.println("ERROR: " + e.getClass().getSimpleName() + (message != null ? ": " + message : ""));
110            if (this.verbose)
111                e.printStackTrace(new PrintWriter(this.console.getOutput()));
112        } catch (IOException ioe) {
113            this.setDone(true);
114        }
115    }
116
117    public void perform(Action action) {
118        try {
119            final boolean temporaryTransaction = this.tx == null && action instanceof TransactionAction;
120            if (temporaryTransaction) {
121                if (!this.openTransaction())
122                    return;
123            }
124            boolean success = false;
125            try {
126                action.run(this);
127                success = true;
128            } finally {
129                if (temporaryTransaction && this.tx != null) {
130                    if (success)
131                        this.commitTransaction();
132                    else
133                        this.rollbackTransaction();
134                }
135            }
136        } catch (Exception e) {
137            this.report(e);
138        }
139    }
140
141    public boolean openTransaction() {
142        try {
143            if (this.tx != null)
144                throw new IllegalStateException("a transaction is already open");
145            this.tx = this.db.createTransaction(this.schemaModel, this.schemaVersion, this.allowNewSchema);
146            final SchemaVersion version = this.tx.getSchemaVersion();
147            this.setSchemaModel(version.getSchemaModel());
148            this.setSchemaVersion(version.getVersionNumber());
149            this.tx.setReadOnly(this.readOnly);
150            return true;
151        } catch (Exception e) {
152            this.tx = null;
153            this.report(e);
154            return false;
155        }
156    }
157
158    public boolean commitTransaction() {
159        try {
160            if (this.tx == null)
161                throw new IllegalStateException("no transaction");
162            this.tx.commit();
163            return true;
164        } catch (Exception e) {
165            this.report(e);
166            return false;
167        } finally {
168            this.tx = null;
169        }
170    }
171
172    public boolean rollbackTransaction() {
173        try {
174            if (this.tx == null)
175                throw new IllegalStateException("no transaction");
176            this.tx.rollback();
177            return true;
178        } catch (Exception e) {
179            this.report(e);
180            return false;
181        } finally {
182            this.tx = null;
183        }
184    }
185}
