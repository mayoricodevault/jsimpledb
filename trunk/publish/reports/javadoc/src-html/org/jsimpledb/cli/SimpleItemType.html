008package org.jsimpledb.cli;
009
010import com.google.common.reflect.TypeToken;
011
012public class SimpleItemType<T> implements ItemType<T> {
013
014    private final TypeToken<T> typeToken;
015
016    public SimpleItemType(Class<T> type) {
017        this(TypeToken.of(type));
018    }
019
020    public SimpleItemType(TypeToken<T> typeToken) {
021        if (typeToken == null)
022            throw new IllegalArgumentException("null typeToken");
023        this.typeToken = typeToken;
024    }
025
026    @Override
027    public TypeToken<T> getTypeToken() {
028        return this.typeToken;
029    }
030
031    @Override
032    public void print(Session session, T value, boolean verbose) {
033        final String prefix = verbose && value != null ?
034          "[" + (value.getClass().getName() + "@" + String.format("%08x", System.identityHashCode(value))) + "] " : "";
035        session.getWriter().println(prefix + value);
036    }
037
038    @Override
039    public boolean equals(Object obj) {
040        if (obj == null || obj.getClass() != this.getClass())
041            return false;
042        final SimpleItemType<?> that = (SimpleItemType<?>)obj;
043        return this.typeToken.equals(that.typeToken);
044    }
045
046    @Override
047    public int hashCode() {
048        return this.typeToken.hashCode();
049    }
050}
