008package org.jsimpledb.cli;
009
010import java.util.ArrayList;
011import java.util.Collections;
012
013import org.jsimpledb.util.ParseContext;
014
015public class DupCommand extends AbstractCommand {
016
017    public DupCommand() {
018        super("dup");
019    }
020
021    @Override
022    public String getUsage() {
023        return this.name + " [depth]";
024    }
025
026    @Override
027    public String getHelpSummary() {
028        return "duplicates channel(s) on top of the channel stack";
029    }
030
031    @Override
032    public String getHelpDetail() {
033        return "The `dup' command duplicates the top channel and pushes the copy onto the channel stack. If a `depth'"
034         + " other than `1' (the default) is given, that many channels are duplicated and pushed (the copies will appear"
035         + " in the same order as the originals).";
036    }
037
038    @Override
039    public Action parseParameters(Session session, ParseContext ctx) {
040
041        // Parse parameters
042        final ParamParser parser = new ParamParser(0, 1, this.getUsage()).parse(ctx);
043        final int depth;
044        if (parser.getParams().size() > 0) {
045            final String depthParam = parser.getParam(0);
046            try {
047                depth = Integer.parseInt(depthParam);
048                if (depth < 0)
049                    throw new IllegalArgumentException("depth is negative");
050            } catch (IllegalArgumentException e) {
051                throw new ParseException(ctx, "invalid depth `" + depthParam + "'");
052            }
053        } else
054            depth = 1;
055
056        // Return action
057        return new Action() {
058            @Override
059            public void run(Session session) throws Exception {
060                DupCommand.this.checkDepth(session, depth);
061                final ArrayList<Channel<?>> channels = new ArrayList<>(depth);
062                for (int i = 0; i < depth; i++)
063                    channels.add(DupCommand.this.get(session, i));
064                Collections.reverse(channels);
065                for (Channel<?> channel : channels)
066                    DupCommand.this.push(session, channel);
067            }
068        };
069    }
070}
