008package org.jsimpledb.cli;
009
010import com.google.common.collect.Iterables;
011import com.google.common.collect.Iterators;
012import com.google.common.reflect.TypeToken;
013
014import java.util.Collections;
015import java.util.Iterator;
016import java.util.Set;
017
018import org.jsimpledb.util.ParseContext;
019
020public abstract class AbstractCommand {
021
022    protected final String name;
023
024// Constructors
025
026    protected AbstractCommand(String name) {
027        if (name == null)
028            throw new IllegalArgumentException("null name");
029        this.name = name;
030    }
031
032// Command stuff
033
034    public String getName() {
035        return this.name;
036    }
037
038    public String getUsage() {
039        return this.getName();
040    }
041
042    public abstract String getHelpSummary();
043
044    public String getHelpDetail() {
045        return this.getHelpSummary();
046    }
047
048    /**
049     * Parse command parameters and return command action.
050     *
051     * @throws ParseException if command is invalid
052     */
053    public abstract Action parseParameters(Session session, ParseContext ctx);
054
055// Channel access
056
057    public Channel<?> get(Session session, int depth) {
058        this.checkDepth(session, depth + 1);
059        return Iterables.get(session.getStack(), depth);
060    }
061
062    public <T> Channel<? extends T> get(Session session, int depth, Class<T> type) {
063        return this.get(session, depth, TypeToken.of(type));
064    }
065
066    public <T> Channel<? extends T> get(Session session, int depth, TypeToken<T> typeToken) {
067        return this.cast(this.get(session, depth), typeToken);
068    }
069
070    public Channel<?> pop(Session session) {
071        this.checkDepth(session, 1);
072        return session.getStack().pop();
073    }
074
075    public <T> Channel<? extends T> pop(Session session, Class<T> type) {
076        return this.pop(session, TypeToken.of(type));
077    }
078
079    public <T> Channel<? extends T> pop(Session session, TypeToken<T> typeToken) {
080        return this.cast(this.pop(session), typeToken);
081    }
082
083    public Channel<?> remove(Session session, int depth) {
084        this.checkDepth(session, depth + 1);
085        final Iterator<Channel<?>> i = session.getStack().iterator();
086        Iterators.advance(i, depth);
087        Channel<?> channel = i.next();
088        i.remove();
089        return channel;
090    }
091
092    public <T> Channel<? extends T> remove(Session session, int depth, Class<T> type) {
093        return this.remove(session, depth, TypeToken.of(type));
094    }
095
096    public <T> Channel<? extends T> remove(Session session, int depth, TypeToken<T> typeToken) {
097        return this.cast(this.remove(session, depth), typeToken);
098    }
099
100    public <T> Channel<? extends T> cast(Channel<?> channel, Class<T> type) {
101        return this.cast(channel, TypeToken.of(type));
102    }
103
104    public void push(Session session, Channel<?> channel) {
105        session.getStack().push(channel);
106    }
107
108    public <T> void push(Session session, Class<T> type, final T value) {
109        this.push(session, new AbstractChannel<T>(type) {
110            @Override
111            public Set<T> getItems(Session session) {
112                return Collections.<T>singleton(value);
113            }
114        });
115    }
116
117    @SuppressWarnings("unchecked")
118    public <T> Channel<? extends T> cast(Channel<?> channel, TypeToken<T> typeToken) {
119        final TypeToken<?> itemType = channel.getItemType().getTypeToken();
120        if (!typeToken.isAssignableFrom(itemType)) {
121            throw new IllegalArgumentException("the `" + this.name + "' command expects a channel of type "
122              + typeToken + " but the channel found has type " + itemType);
123        }
124        return (Channel<? extends T>)channel;
125    }
126
127    protected void checkDepth(Session session, int min) {
128        final int depth = session.getStack().size();
129        if (depth < min) {
130            throw new IllegalArgumentException("the `" + this.name + "' command expects at least " + min
131              + " input channel" + (min != 1 ? "s" : "") + ", but there are only "
132              + depth + " channel" + (depth != 1 ? "s" : "") + " on the stack");
133        }
134    }
135}
