017import java.util.ArrayList;
018import java.util.List;

022import org.jsimpledb.core.Database;

034/**
035 * CLI console.
036 */
037public class Console {
038
039    private final Logger log = LoggerFactory.getLogger(this.getClass());
040    private final ConsoleReader console;
041    private final CliSession session;
042    private final CommandParser commandParser = new CommandParser();
043    private final CommandListParser commandListParser = new CommandListParser(this.commandParser);
044
045    private FileHistory history;
046
047    /**
048     * Constructor for core level access only.
049     */
050    public Console(Database db, InputStream in, OutputStream out) throws IOException {
051        this(null, db, in, out);
052    }
053
054    /**
055     * Constructor for {@link JSimpleDB} level access.
056     */
057    public Console(JSimpleDB jdb, InputStream in, OutputStream out) throws IOException {
058        this(jdb, null, in, out);
059    }
060
061    private Console(JSimpleDB jdb, Database db, InputStream in, OutputStream out) throws IOException {
062        this.console = new ConsoleReader(in, out);
063        this.console.setBellEnabled(true);
064        this.console.setHistoryEnabled(true);
065        this.console.setHandleUserInterrupt(true);
066        final PrintWriter writer = new PrintWriter(console.getOutput(), true);
067        this.session = jdb != null ? new CliSession(jdb, writer) : new CliSession(db, writer);
068    }
069
070    public CliSession getSession() {
071        return this.session;
072    }
073
074    public void setHistoryFile(File historyFile) {
075        if (this.history != null)
076            throw new IllegalStateException("history file already configured");
077        try {
078            this.history = new FileHistory(historyFile);
079        } catch (IOException e) {
080            // ignore
081        }
082        this.console.setHistory(this.history);
083    }
084
085    /**
086     * Run this instance.
087     */
088    public void run() throws IOException {
089
090        // Input buffer
091        final StringBuilder lineBuffer = new StringBuilder();
092
093        // Set up tab completion
094        console.addCompleter(new ConsoleCompleter(lineBuffer));
095
096        // Open a transaction to load and verify database schema
097        this.session.perform(new Session.Action() {
098            @Override
099            public void run(Session session) { }
100        });
101
102        // Main command loop
103        try {
104
105            this.console.println("Welcome to JSimpleDB. You are in "
106              + (this.session.hasJSimpleDB() ? "JSimpleDB" : "Core API") + " Mode. Type `help' for help.");
107            while (!session.isDone()) {
108
109                // Read command line
110                String line;
111                try {
112                    line = this.console.readLine(lineBuffer.length() == 0 ? "JSimpleDB> " : "        -> ");
113                } catch (UserInterruptException e) {
114                    this.console.print("^C");
115                    line = null;
116                }
117                if (line == null) {
118                    this.console.println();
119                    break;
120                }
121
122                // Detect backslash continuations
123                boolean continuation = false;
124                if (line.length() > 0 && line.charAt(line.length() - 1) == '\\') {
125                    line = line.substring(0, line.length() - 1) + "\n";
126                    continuation = true;
127                }
128
129                // Append line to buffer
130                lineBuffer.append(line);
131
132                // Handle backslash continuations
133                if (continuation)
134                    continue;
135                final ParseContext ctx = new ParseContext(lineBuffer.toString());
136                lineBuffer.setLength(0);
137
138                // Skip initial whitespace
139                ctx.skipWhitespace();
140
141                // Ignore blank input
142                if (ctx.getInput().length() == 0)
143                    continue;
144
145                // Parse command(s)
146                final ArrayList<CliSession.Action> actions = new ArrayList<>();
147                if (!this.session.perform(new CliSession.Action() {
148                    @Override
149                    public void run(CliSession session) {
150                        actions.addAll(commandListParser.parse(session, ctx, false));
151                    }
152                }))
153                    continue;
154
155                // Execute commands
156                for (CliSession.Action action : actions) {
157                    if (!this.session.perform(action))
158                        break;
159                }
160
161                // Proceed
162                this.console.flush();
163            }
164        } finally {
165            if (this.history != null)
166                this.history.flush();
167            this.console.flush();
168            this.console.shutdown();
169        }
170    }
171
172// ConsoleCompleter
173
174    private class ConsoleCompleter implements Completer {
175
176        private final StringBuilder lineBuffer;
177
178        ConsoleCompleter(StringBuilder lineBuffer) {
179            this.lineBuffer = lineBuffer;
180        }
181
182        @Override
183        public int complete(final String buffer, final int cursor, final List<CharSequence> candidates) {
184            final int[] result = new int[1];
185            Console.this.session.perform(new CliSession.Action() {
186                @Override
187                public void run(CliSession session) {
188                    result[0] = ConsoleCompleter.this.completeInTransaction(session, buffer, cursor, candidates);
189                }
190            });
191            return result[0];
192        }
193
194        private int completeInTransaction(CliSession session, String buffer, int cursor, List<CharSequence> candidates) {
195            final ParseContext ctx = new ParseContext(this.lineBuffer + buffer.substring(0, cursor));
196            try {
197                Console.this.commandListParser.parse(session, ctx, true);
198            } catch (ParseException e) {
199                String prefix = "";
200                int index = ctx.getIndex();
201                while (index > 0 && Character.isJavaIdentifierPart(ctx.getOriginalInput().charAt(index - 1)))
202                    prefix = ctx.getOriginalInput().charAt(--index) + prefix;
203                candidates.addAll(Lists.transform(e.getCompletions(), new AddPrefixFunction(prefix)));
204                return index;
205            } catch (Exception e) {
206                try {
207                    Console.this.console.println();
208                    Console.this.console.println("Error: got exception calculating command line completions");
209                    e.printStackTrace(session.getWriter());
210                } catch (IOException e2) {
211                    // ignore
212                }
213            }
214            return cursor;
215        }
216    }
