008package org.jsimpledb.cli;
009
010import com.google.common.collect.Lists;
011
012import java.io.File;
013import java.io.IOException;
014import java.io.InputStream;
015import java.io.OutputStream;
016import java.util.List;
017
018import org.jsimpledb.core.Database;
019import org.jsimpledb.util.ParseContext;
020
021import jline.console.ConsoleReader;
022import jline.console.UserInterruptException;
023import jline.console.completer.Completer;
024import jline.console.history.FileHistory;
025
026/**
027 * CLI console.
028 */
029public class Console {
030
031    private final Database db;
032    private final ConsoleReader console;
033    private final Session session;
034
035    private FileHistory history;
036
037    /**
038     * Constrcutor.
039     */
040    public Console(Database db, InputStream in, OutputStream out) throws IOException {
041        this.db = db;
042        this.console = new ConsoleReader(in, out);
043        this.console.setBellEnabled(true);
044        this.console.setHistoryEnabled(true);
045        this.console.setHandleUserInterrupt(true);
046        this.session = new Session(this.db, this.console);
047    }
048
049    public Database getDatabase() {
050        return this.db;
051    }
052
053    public ConsoleReader getConsole() {
054        return this.console;
055    }
056
057    public Session getSession() {
058        return this.session;
059    }
060
061    public void setHistoryFile(File historyFile) {
062        if (this.history != null)
063            throw new IllegalStateException("history file already configured");
064        try {
065            this.history = new FileHistory(historyFile);
066        } catch (IOException e) {
067            // ignore
068        }
069        this.console.setHistory(this.history);
070    }
071
072    /**
073     * Run this instance.
074     */
075    public void run() throws IOException {
076
077        // Input buffer
078        final StringBuilder lineBuffer = new StringBuilder();
079
080        // Set up commands
081        console.addCompleter(new Completer() {
082            @Override
083            public int complete(String buffer, int cursor, List<CharSequence> candidates) {
084                final ParseContext ctx = new ParseContext(lineBuffer + buffer.substring(0, cursor));
085                try {
086                    new CommandListParser().parse(session, ctx);
087                } catch (ParseException e) {
088                    String prefix = "";
089                    int index = ctx.getIndex();
090                    while (index > 0 && !Character.isWhitespace(ctx.getOriginalInput().charAt(index - 1)))
091                        prefix = ctx.getOriginalInput().charAt(--index) + prefix;
092                    candidates.addAll(Lists.transform(e.getCompletions(), new AddPrefixFunction(prefix)));
093                    return index;
094                } catch (Exception e) {
095                    try {
096                        Console.this.console.println();
097                        Console.this.console.println("ERROR: error calculating command line completions");
098                        e.printStackTrace(session.getWriter());
099                    } catch (IOException e2) {
100                        // ignore
101                    }
102                }
103                return cursor;
104            }
105        });
106
107        // Open a transaction to load and verify database schema
108        this.session.perform(new Action() {
109            @Override
110            public void run(Session session) { }
111        });
112
113        // Main command loop
114        try {
115
116            this.console.println("Welcome to JSimpleDB. Type `help' for help.");
117            while (!session.isDone()) {
118
119                // Read command line
120                String line;
121                try {
122                    line = this.console.readLine(lineBuffer.length() == 0 ? "JSimpleDB> " : "        -> ");
123                } catch (UserInterruptException e) {
124                    this.console.print("^C");
125                    line = null;
126                }
127                if (line == null) {
128                    this.console.println();
129                    break;
130                }
131
132                // Detect backslash continuations
133                boolean continuation = false;
134                if (line.length() > 0 && line.charAt(line.length() - 1) == '\\') {
135                    line = line.substring(0, line.length() - 1) + "\n";
136                    continuation = true;
137                }
138
139                // Append line to buffer
140                lineBuffer.append(line);
141
142                // Handle backslash continuations
143                if (continuation)
144                    continue;
145                final ParseContext ctx = new ParseContext(lineBuffer.toString());
146                lineBuffer.setLength(0);
147
148                // Skip initial whitespace
149                ctx.skipWhitespace();
150
151                // Ignore blank input
152                if (ctx.getInput().length() == 0)
153                    continue;
154
155                // Parse command(s)
156                final List<Action> actions;
157                try {
158                    actions = new CommandListParser().parse(this.session, ctx);
159                } catch (Exception e) {
160                    this.session.report(e);             // "should never happen"
161                    continue;
162                }
163
164                // Execute commands
165                for (Action action : actions) {
166                    if (!this.session.perform(action))
167                        break;
168                }
169
170                // Proceed
171                this.console.flush();
172            }
173        } finally {
174            if (this.history != null)
175                this.history.flush();
176            this.console.flush();
177            this.console.shutdown();
178        }
179    }
180}
