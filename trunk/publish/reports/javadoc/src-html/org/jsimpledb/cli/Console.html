008package org.jsimpledb.cli;
009
010import java.io.File;
011import java.io.IOException;
012import java.io.InputStream;
013import java.io.OutputStream;
014import java.util.List;
015
016import org.dellroad.stuff.string.ParseContext;
017import org.jsimpledb.core.Database;
018
019import jline.console.ConsoleReader;
020import jline.console.UserInterruptException;
021import jline.console.completer.Completer;
022import jline.console.history.FileHistory;
023
024/**
025 * CLI console.
026 */
027public class Console {
028
029    private final Database db;
030    private final ConsoleReader console;
031    private final Session session;
032
033    private FileHistory history;
034
035    /**
036     * Constrcutor.
037     */
038    public Console(Database db, InputStream in, OutputStream out) throws IOException {
039        this.db = db;
040        this.console = new ConsoleReader(in, out);
041        this.console.setBellEnabled(true);
042        this.console.setHistoryEnabled(true);
043        this.console.setHandleUserInterrupt(true);
044        this.session = new Session(this.db, this.console);
045    }
046
047    public Database getDatabase() {
048        return this.db;
049    }
050
051    public ConsoleReader getConsole() {
052        return this.console;
053    }
054
055    public Session getSession() {
056        return this.session;
057    }
058
059    public void setHistoryFile(File historyFile) {
060        if (this.history != null)
061            throw new IllegalStateException("history file already configured");
062        try {
063            this.history = new FileHistory(historyFile);
064        } catch (IOException e) {
065            // ignore
066        }
067        this.console.setHistory(this.history);
068    }
069
070    /**
071     * Run this instance.
072     */
073    public void run() throws IOException {
074
075        // Input buffer
076        final StringBuilder lineBuffer = new StringBuilder();
077
078        // Set up commands
079        final RootCommand rootCommand = new RootCommand();
080        console.addCompleter(new Completer() {
081            @Override
082            public int complete(String buffer, int cursor, List<CharSequence> candidates) {
083                final ParseContext ctx = new ParseContext(lineBuffer + buffer.substring(0, cursor));
084                try {
085                    rootCommand.parse(session, ctx);
086                } catch (ParseException e) {
087                    candidates.addAll(e.getCompletions());
088                    return e.getParseContext().getIndex();
089                } catch (Exception e) {
090                    try {
091                        Console.this.console.println();
092                        Console.this.console.println("ERROR: " + e.getMessage());
093                    } catch (IOException e2) {
094                        // ignore
095                    }
096                }
097                return cursor;
098            }
099        });
100
101        // Open a transaction to verify database schema
102        this.session.perform(new TransactionAction() {
103            @Override
104            public void run(Session session) throws Exception { }
105        });
106
107        // Main command loop
108        try {
109
110            this.console.println("Welcome to JSimpleDB. Type `help' for help.");
111            while (!session.isDone()) {
112
113                // Read command line
114                String line;
115                try {
116                    line = this.console.readLine(lineBuffer.length() == 0 ? "JSimpleDB> " : "        -> ");
117                } catch (UserInterruptException e) {
118                    this.console.print("^C");
119                    line = null;
120                }
121                if (line == null) {
122                    this.console.println();
123                    break;
124                }
125
126                // Detect backslash continuations
127                boolean continuation = false;
128                if (line.length() > 0 && line.charAt(line.length() - 1) == '\\') {
129                    line = line.substring(0, line.length() - 1) + "\n";
130                    continuation = true;
131                }
132
133                // Append line to buffer
134                lineBuffer.append(line);
135
136                // Handle backslash continuations
137                if (continuation)
138                    continue;
139                final ParseContext ctx = new ParseContext(lineBuffer.toString());
140                lineBuffer.setLength(0);
141
142                // Skip initial whitespace
143                ctx.skipWhitespace();
144
145                // Ignore blank input
146                if (ctx.getInput().length() == 0)
147                    continue;
148
149                // Parse command line
150                Action action;
151                try {
152                    action = rootCommand.parse(session, ctx);
153                } catch (Exception e) {
154                    this.console.println("ERROR: " + e.getMessage());
155                    continue;
156                }
157
158                // Execute command
159                session.perform(action);
160                this.console.flush();
161            }
162        } finally {
163            if (this.history != null)
164                this.history.flush();
165            this.console.flush();
166            this.console.shutdown();
167        }
168    }
169}
