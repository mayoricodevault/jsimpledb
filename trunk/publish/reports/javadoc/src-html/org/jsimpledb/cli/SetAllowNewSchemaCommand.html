008package org.jsimpledb.cli;
009
010import com.google.common.collect.Iterables;
011
012import java.util.Arrays;
013
014import org.dellroad.stuff.string.ParseContext;
015
016public class SetAllowNewSchemaCommand extends Command {
017
018    public SetAllowNewSchemaCommand(AggregateCommand parent) {
019        super(parent, "allow-new-schema");
020    }
021
022    @Override
023    public Action parse(Session session, ParseContext ctx) throws ParseException {
024        final String usage = "Usage: " + this.getFullName() + " [ true | false ]";
025        final boolean value;
026        try {
027            value = Boolean.valueOf(ctx.matchPrefix("(true|false)\\s*$").group(1));
028        } catch (IllegalArgumentException e) {
029            throw new ParseException(ctx, usage).addCompletions(
030              Iterables.filter(Arrays.asList("true", "false"), new PrefixPredicate(ctx.getInput())));
031        }
032        return new Action() {
033            @Override
034            public void run(Session session) throws Exception {
035                session.setAllowNewSchema(value);
036            }
037        };
038    }
039
040    @Override
041    public String getHelpSummary() {
042        return "Sets whether recording a new schema version into the database is allowed";
043    }
044}
