008package org.jsimpledb.cli;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseSession;
012import org.jsimpledb.parse.Parser;
013import org.jsimpledb.parse.WordParser;
014
015public class CommandParser implements Parser<CliSession.Action> {
016
017    /**
018     * Parse a command and return the corresponding {@link CliSession.Action}.
019     */
020    @Override
021    public CliSession.Action parse(ParseSession parseSession, ParseContext ctx, boolean complete) {
022        final CliSession session = (CliSession)parseSession;
023        final String commandName = new WordParser(session.getCommands().keySet(), "command").parse(session, ctx, complete);
024        return session.getCommands().get(commandName).parse(session, ctx, complete);
025    }
026}
