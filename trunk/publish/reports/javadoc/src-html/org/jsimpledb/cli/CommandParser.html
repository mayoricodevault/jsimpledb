008package org.jsimpledb.cli;
009
010import java.util.SortedMap;
011import java.util.TreeMap;
012
013import org.jsimpledb.util.ParseContext;
014
015public final class CommandParser {
016
017    private static final TreeMap<String, AbstractCommand> COMMANDS = new TreeMap<>();
018
019    static {
020        CommandParser.registerCommand(new CountCommand());
021        CommandParser.registerCommand(new CreateCommand());
022        CommandParser.registerCommand(new DeleteCommand());
023        CommandParser.registerCommand(new DupCommand());
024        CommandParser.registerCommand(new ExportCommand());
025        CommandParser.registerCommand(new GetCommand());
026        CommandParser.registerCommand(new HelpCommand());
027        CommandParser.registerCommand(new ImportCommand());
028        CommandParser.registerCommand(new IterateCommand());
029        CommandParser.registerCommand(new LimitCommand());
030        CommandParser.registerCommand(new ObjectCommand());
031        CommandParser.registerCommand(new PrintCommand());
032        CommandParser.registerCommand(new PopCommand());
033        CommandParser.registerCommand(new SetAllowNewSchemaCommand());
034        CommandParser.registerCommand(new SetSchemaVersionCommand());
035        CommandParser.registerCommand(new ShowAllSchemasCommand());
036        CommandParser.registerCommand(new ShowSchemaCommand());
037        CommandParser.registerCommand(new StackCommand());
038        CommandParser.registerCommand(new SwapCommand());
039    }
040
041    private CommandParser() {
042    }
043
044    public static SortedMap<String, AbstractCommand> getCommands() {
045        return COMMANDS;
046    }
047
048    public static void registerCommand(AbstractCommand command) {
049        COMMANDS.put(command.getName(), command);
050    }
051
052    public static Action parse(Session session, ParseContext ctx) {
053        while (true) {
054
055            // Skip whitespace and empty commands
056            ctx.skipWhitespace();
057            if (ctx.tryLiteral(";"))
058                continue;
059            if (ctx.isEOF())
060                return null;
061
062            // Get command name
063            final String commandName = ctx.matchPrefix("[^\\s;]+").group();
064
065            // Get matching command
066            final AbstractCommand command = COMMANDS.get(commandName);
067            if (command == null) {
068                throw new ParseException(ctx, "unknown command `" + commandName + "'")
069                  .addCompletions(Util.complete(COMMANDS.keySet(), commandName, false));
070            }
071
072            // Try to parse command parameters
073            return command.parseParameters(session, ctx);
074        }
075    }
076}
