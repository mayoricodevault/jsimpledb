008package org.jsimpledb.cli.cmd;
009
010import java.util.Map;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014import org.jsimpledb.cli.CliSession;
015import org.jsimpledb.cli.ParamParser;
016import org.jsimpledb.parse.FieldTypeParser;
017import org.jsimpledb.parse.ObjIdParser;
018import org.jsimpledb.parse.ObjTypeParser;
019import org.jsimpledb.parse.ParseContext;
020import org.jsimpledb.parse.ParseSession;
021import org.jsimpledb.parse.Parser;
022import org.jsimpledb.parse.expr.ExprParser;
023import org.slf4j.Logger;
024import org.slf4j.LoggerFactory;
025
026/**
027 * Superclass of all CLI commands.
028 *
029 * @see Command
030 */
031public abstract class AbstractCommand implements Parser<CliSession.Action> {
032
033    protected final Logger log = LoggerFactory.getLogger(this.getClass());
034    protected final String name;
035    protected final ParamParser paramParser;
036
037// Constructors
038
039    /**
040     * Constructor.
041     *
042     * @param spec {@link ParamParser} spec string, possibly containing custom type names (see {@link #getParser getParser()})
043     */
044    protected AbstractCommand(String spec) {
045        if (spec == null)
046            throw new IllegalArgumentException("null spec");
047        final Matcher matcher = Pattern.compile("([^\\s]+)(\\s+(.*))?").matcher(spec);
048        if (!matcher.matches())
049            throw new IllegalArgumentException("invalid command specification `" + spec + "'");
050        this.name = matcher.group(1);
051        final String paramSpec = matcher.group(3);
052        this.paramParser = new ParamParser(paramSpec != null ? paramSpec : "") {
053            @Override
054            protected Parser<?> getParser(String typeName) {
055                try {
056                    return super.getParser(typeName);
057                } catch (IllegalArgumentException e) {
058                    return AbstractCommand.this.getParser(typeName);
059                }
060            }
061        };
062    }
063
064// Command stuff
065
066    /**
067     * Get the name of this command.
068     *
069     * @return command name
070     */
071    public String getName() {
072        return this.name;
073    }
074
075    /**
076     * Get command usage string.
077     *
078     * <p>
079     * The implementation in {@link Command} delegates to {@link ParamParser#getUsage ParamParser.getUsage()}.
080     * </p>
081     *
082     * @return command usage string
083     */
084    public String getUsage() {
085        return this.paramParser.getUsage(this.name);
086    }
087
088    /**
089     * Get summarized help (typically a single line).
090     *
091     * @return one line command summary
092     */
093    public abstract String getHelpSummary();
094
095    /**
096     * Get expanded help (typically multiple lines).
097     *
098     * <p>
099     * The implementation in {@link Command} delegates to {@link #getHelpSummary getHelpSummary()}.
100     * </p>
101     *
102     * @return detailed command description
103     */
104    public String getHelpDetail() {
105        return this.getHelpSummary();
106    }
107
108    /**
109     * Parse command line and return command action.
110     *
111     * <p>
112     * The implementation in {@link ParamParser} parses the parameters and delegates to {@link #getAction} with the result.
113     * </p>
114     *
115     * @param session CLI session
116     * @param ctx input to parse
117     * @param complete false if parse is "for real", true if only for tab completion calculation
118     * @return action to perform for the parsed command
119     * @throws org.jsimpledb.parse.ParseException if parse fails, or if {@code complete} is true and there are valid completions
120     * @throws org.jsimpledb.parse.ParseException if parameters are invalid
121     */
122    @Override
123    public CliSession.Action parse(ParseSession session, ParseContext ctx, boolean complete) {
124        return this.getAction((CliSession)session, ctx, complete, this.paramParser.parse(session, ctx, complete));
125    }
126
127    /**
128     * Process command line parameters and return action.
129     *
130     * @param session CLI session
131     * @param ctx input to parse
132     * @param complete false if parse is "for real", true if only for tab completion calculation
133     * @param params parsed parameters indexed by name
134     * @return action to perform for the parsed command
135     * @throws org.jsimpledb.parse.ParseException if parse fails, or if {@code complete} is true and there are valid completions
136     * @throws org.jsimpledb.parse.ParseException if parameters are invalid
137     */
138    public abstract CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params);
139
140    /**
141     * Convert parameter spec type name into a {@link Parser}. Used for custom type names not supported by {@link ParamParser}.
142     *
143     * <p>
144     * The implementation in {@link ParamParser} supports all {@link org.jsimpledb.core.FieldType}s registered with the database,
145     * {@code type} for an object type name (returns {@link Integer}), and {@code objid} for an object ID
146     * (returns {@link org.jsimpledb.core.ObjId}).
147     * </p>
148     *
149     * @param typeName parameter type name
150     * @return parser for parameters of the specified type
151     */
152    protected Parser<?> getParser(String typeName) {
153        if (typeName == null)
154            throw new IllegalArgumentException("null typeName");
155        if (typeName.equals("type"))
156            return new ObjTypeParser();
157        if (typeName.equals("objid"))
158            return new ObjIdParser();
159        if (typeName.equals("expr"))
160            return new ExprParser();
161        return FieldTypeParser.getFieldTypeParser(typeName);
162    }
163}
