008package org.jsimpledb.cli;
009
010import java.io.PrintWriter;
011import java.lang.reflect.InvocationTargetException;
012import java.util.SortedMap;
013import java.util.TreeMap;
014
015import org.jsimpledb.JSimpleDB;
016import org.jsimpledb.Session;
017import org.jsimpledb.cli.cmd.AbstractCommand;
018import org.jsimpledb.core.Database;
019import org.jsimpledb.core.Transaction;
020import org.jsimpledb.parse.ParseException;
021import org.jsimpledb.parse.ParseSession;
022
023/**
024 * Represents one CLI console session.
025 */
026public class CliSession extends ParseSession {
027
028    private final PrintWriter writer;
029    private final TreeMap<String, AbstractCommand> commands = new TreeMap<>();
030
031    private boolean done;
032    private boolean verbose;
033    private int lineLimit = 16;
034
035// Constructors
036
037    /**
038     * Constructor for core level access only.
039     *
040     * @param db core database
041     * @param writer console output
042     * @throws IllegalArgumentException if either parameter is null
043     */
044    public CliSession(Database db, PrintWriter writer) {
045        super(db);
046        if (writer == null)
047            throw new IllegalArgumentException("null writer");
048        this.writer = writer;
049    }
050
051    /**
052     * Constructor for {@link JSimpleDB} level access.
053     *
054     * @param jdb database
055     * @param writer console output
056     * @throws IllegalArgumentException if either parameter is null
057     */
058    public CliSession(JSimpleDB jdb, PrintWriter writer) {
059        super(jdb);
060        if (writer == null)
061            throw new IllegalArgumentException("null writer");
062        this.writer = writer;
063    }
064
065// Accessors
066
067    /**
068     * Get the output {@link PrintWriter} for this CLI session.
069     */
070    public PrintWriter getWriter() {
071        return this.writer;
072    }
073
074    /**
075     * Get the {@link AbstractCommand}s registered with this instance.
076     */
077    public SortedMap<String, AbstractCommand> getCommands() {
078        return this.commands;
079    }
080
081    public int getLineLimit() {
082        return this.lineLimit;
083    }
084    public void setLineLimit(int lineLimit) {
085        this.lineLimit = lineLimit;
086    }
087
088    public boolean isVerbose() {
089        return this.verbose;
090    }
091    public void setVerbose(boolean verbose) {
092        this.verbose = verbose;
093    }
094
095    public boolean isDone() {
096        return this.done;
097    }
098    public void setDone(boolean done) {
099        this.done = done;
100    }
101
102// Command registration
103
104    /**
105     * Create an instance of the specified class and register it as a {@link AbstractCommand}.
106     * The class must have a public constructor taking either a single {@link CliSession} parameter
107     * or no parameters; they will be tried in that order.
108     *
109     * @param cl command class
110     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
111     * @throws IllegalArgumentException if {@code cl} instantiation fails
112     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractCommand}
113     */
114    public void registerCommand(Class<?> cl) {
115        if (!AbstractCommand.class.isAssignableFrom(cl))
116            throw new IllegalArgumentException(cl + " does not subclass " + AbstractCommand.class.getName());
117        final AbstractCommand command = this.instantiate(cl.asSubclass(AbstractCommand.class));
118        this.commands.put(command.getName(), command);
119    }
120
121    private <T> T instantiate(Class<T> cl) {
122        Throwable failure;
123        try {
124            return cl.getConstructor(CliSession.class).newInstance(this);
125        } catch (NoSuchMethodException e) {
126            try {
127                return cl.getConstructor().newInstance();
128            } catch (NoSuchMethodException e2) {
129                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
130            } catch (Exception e2) {
131                failure = e2;
132            }
133        } catch (Exception e) {
134            failure = e;
135        }
136        if (failure instanceof InvocationTargetException)
137            failure = failure.getCause();
138        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
139    }
140
141// Errors
142
143    @Override
144    protected void reportException(Exception e) {
145        final String message = e.getLocalizedMessage();
146        if (e instanceof ParseException && message != null)
147            this.writer.println("Error: " + message);
148        else
149            this.writer.println("Error: " + e.getClass().getSimpleName() + (message != null ? ": " + message : ""));
150        if (this.verbose || this.showStackTrace(e))
151            e.printStackTrace(this.writer);
152    }
153
154    protected boolean showStackTrace(Exception e) {
155        return e instanceof NullPointerException || (e instanceof ParseException && e.getLocalizedMessage() == null);
156    }
157
158// Action
159
160    /**
161     * Perform the given action. This is a convenience method, equivalent to: {@code perform(null, action)}
162     *
163     * @param action action to perform
164     * @throws IllegalArgumentException if {@code action} is null
165     */
166    public boolean perform(Action action) {
167        return this.perform(null, action);
168    }
169
170    /**
171     * Perform the given action within the given existing transaction, if any, otherwise within a new transaction.
172     * If {@code tx} is not null, it will used and left open when this method returns. Otherwise,
173     * if there is already an open transaction associated with this instance, it will be used;
174     * otherwise, a new transaction is created for the duration of {@code action} and then committed.
175     *
176     * <p>
177     * If {@code tx} is not null and there is already an open transaction associated with this instance and they
178     * are not the same transaction, an {@link IllegalStateException} is thrown.
179     * </p>
180     *
181     * @param tx transaction in which to perform the action, or null to create a new one (if necessary)
182     * @param action action to perform
183     * @throws IllegalStateException if {@code tx} conflict with the already an open transaction associated with this instance
184     * @throws IllegalArgumentException if {@code action} is null
185     */
186    public boolean perform(Transaction tx, final Action action) {
187        return this.perform(tx, new Session.Action() {
188            @Override
189            public void run(Session session) throws Exception {
190                action.run((CliSession)session);
191            }
192        });
193    }
194
195    /**
196     * Callback interface used by {@link CliSession#perform CliSession.perform()}.
197     */
198    public interface Action {
199
200        /**
201         * Perform some action using the given {@link CliSession} while a transaction is open.
202         */
203        void run(CliSession session) throws Exception;
204    }
205}
