008package org.jsimpledb.cli.cmd;
009
010import java.io.PrintWriter;
011import java.util.HashSet;
012import java.util.Map;
013
014import org.jsimpledb.cli.CliSession;
015import org.jsimpledb.parse.ParseContext;
016import org.jsimpledb.parse.Parser;
017import org.jsimpledb.parse.WordParser;
018import org.jsimpledb.parse.func.AbstractFunction;
019
020@Command
021public class HelpCommand extends AbstractCommand {
022
023    private final CliSession session;
024
025    public HelpCommand(CliSession session) {
026        super("help command-or-function:cmdfunc?");
027        this.session = session;
028    }
029
030    @Override
031    public String getHelpSummary() {
032        return "display help information";
033    }
034
035    @Override
036    public String getHelpDetail() {
037        return "Displays the list of known commands and functions, or help information about a specific command or function.";
038    }
039
040    @Override
041    protected Parser<?> getParser(String typeName) {
042        if ("cmdfunc".equals(typeName)) {
043            return new WordParser("command/function") {
044                @Override
045                protected HashSet<String> getWords() {
046                    final HashSet<String> names = new HashSet<>();
047                    names.addAll(HelpCommand.this.session.getCommands().keySet());
048                    names.addAll(HelpCommand.this.session.getFunctions().keySet());
049                    return names;
050                }
051            };
052        }
053        return super.getParser(typeName);
054    }
055
056    @Override
057    public CliSession.Action getAction(CliSession session, ParseContext ctx, boolean complete, Map<String, Object> params) {
058        final String name = (String)params.get("command-or-function");
059        return new CliSession.Action() {
060            @Override
061            public void run(CliSession session) throws Exception {
062                final PrintWriter writer = session.getWriter();
063                if (name == null) {
064                    writer.println("Available commands:");
065                    for (AbstractCommand availableCommand : session.getCommands().values())
066                        writer.println(String.format("%24s - %s", availableCommand.getName(), availableCommand.getHelpSummary()));
067                    writer.println("Available functions:");
068                    for (AbstractFunction availableFunction : session.getFunctions().values())
069                        writer.println(String.format("%24s - %s", availableFunction.getName(), availableFunction.getHelpSummary()));
070                } else {
071                    final AbstractCommand command = session.getCommands().get(name);
072                    if (command != null) {
073                        writer.println("Usage: " + command.getUsage());
074                        writer.println(command.getHelpDetail());
075                    }
076                    final AbstractFunction function = session.getFunctions().get(name);
077                    if (function != null) {
078                        writer.println("Usage: " + function.getUsage());
079                        writer.println(function.getHelpDetail());
080                    }
081                    if (command == null && function == null)
082                        writer.println("No command or function named `" + name + "' exists.");
083                }
084            }
085        };
086    }
087}
