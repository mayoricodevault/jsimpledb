008package org.jsimpledb.cli;
009
010import com.google.common.base.Function;
011
012/**
013 * Adds a suffix.
014 */
015public class AddSuffixFunction implements Function<String, String> {
016
017    private final String suffix;
018
019    public AddSuffixFunction(String suffix) {
020        if (suffix == null)
021            throw new IllegalArgumentException("null suffix");
022        this.suffix = suffix;
023    }
024
025    @Override
026    public String apply(String string) {
027        if (string == null)
028            throw new IllegalArgumentException("null string");
029        return string + this.suffix;
030    }
031}
