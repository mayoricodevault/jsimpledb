008package org.jsimpledb.demo;
009
010import com.vaadin.server.StreamResource;
011import com.vaadin.shared.ui.label.ContentMode;
012import com.vaadin.ui.Component;
013import com.vaadin.ui.Field;
014import com.vaadin.ui.Image;
015
016import java.io.ByteArrayInputStream;
017
018import org.dellroad.stuff.vaadin7.BlobField;
019import org.dellroad.stuff.vaadin7.FieldBuilder;
020import org.dellroad.stuff.vaadin7.ProvidesProperty;
021import org.jsimpledb.gui.NullableField;
022import org.jsimpledb.gui.SizedLabel;
023
024/**
025 * Support superclass for {@code Body} implementations.
026 */
027public abstract class AbstractBody implements Body {
028
029// Object
030
031    // Show the database object ID instead of system hash code
032    @Override
033    public String toString() {
034        return this.getClass().getName() + "@" + this.getObjId();
035    }
036
037// Vaadin GUI customizations
038
039    // The @ProvidesProperty annotation here indicates that this method should be invoked
040    // to build the Vaadin property named "image" instead of doing the default thing (using toString()).
041    // We are converting the byte[] value into a Vaadin Image object.
042    @ProvidesProperty("image")
043    @SuppressWarnings("serial")
044    private Component propertyImage() {
045        final byte[] imageData = this.getImage();
046        if (imageData == null)
047            return new SizedLabel("<i>Null</i>", ContentMode.HTML);
048        return new Image(null, new StreamResource(new StreamResource.StreamSource() {
049            @Override
050            public ByteArrayInputStream getStream() {
051                return new ByteArrayInputStream(imageData);
052            }
053        }, "image.png"));
054    }
055
056    // The @FieldBuilder.ProvidesField annotation indicates that if anyone wants to edit
057    // the "image" field, they should use this method to build an appropriate Vaaddin Field instead
058    // of doing the default thing (using a TextField).
059    @FieldBuilder.ProvidesField("image")
060    private Field<byte[]> buildImagePropertyField() {
061        return new NullableField<byte[]>(new BlobField());
062    }
063}
