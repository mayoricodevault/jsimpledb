008package org.jsimpledb.util;
009
010import java.util.Collections;
011import java.util.Comparator;
012import java.util.Iterator;
013import java.util.NavigableSet;
014import java.util.NoSuchElementException;
015
016/**
017 * Support superclass for {@link NavigableSet} implementations for which calculating {@link #size size()} requires
018 * an iteration through all of the set's elements to count them.
019 *
020 * <p>
021 * For a read-only implementation, subclasses should implement {@link #comparator comparator()}, {@link #contains contains()},
022 * {@link #iterator iterator()}, and {@link #createSubSet createSubSet()} to handle reversed and restricted range sub-sets.
023 * </p>
024 *
025 * <p>
026 * For a mutable implementation, subclasses should also implement {@link #add add()}, {@link #remove remove()},
027 * {@link #clear clear()}, and make the {@link #iterator iterator()} mutable.
028 * </p>
029 *
030 * <p>
031 * All overridden methods must be aware of the {@linkplain #bounds range restriction bounds}, if any.
032 * </p>
033 *
034 * @param <E> element type
035 */
036public abstract class AbstractNavigableSet<E> extends AbstractIterationSet<E> implements NavigableSet<E> {
037
038    /**
039     * Element range bounds associated with this instance.
040     */
041    protected final Bounds<E> bounds;
042
043    /**
044     * Convenience constructor for the case where there are no lower or upper bounds.
045     */
046    protected AbstractNavigableSet() {
047        this(new Bounds<E>());
048    }
049
050    /**
051     * Primary constructor.
052     *
053     * @param bounds range restriction
054     * @throws IllegalArgumentException if {@code bounds} is null
055     */
056    protected AbstractNavigableSet(Bounds<E> bounds) {
057        if (bounds == null)
058            throw new IllegalArgumentException("null bounds");
059        this.bounds = bounds;
060    }
061
062    @Override
063    public E first() {
064        return this.iterator().next();
065    }
066
067    @Override
068    public E last() {
069        return this.descendingIterator().next();
070    }
071
072    @Override
073    public E pollFirst() {
074        final Iterator<E> i = this.iterator();
075        if (!i.hasNext())
076            return null;
077        final E value = i.next();
078        i.remove();
079        return value;
080    }
081
082    @Override
083    public E pollLast() {
084        return this.descendingSet().pollFirst();
085    }
086
087    @Override
088    public Iterator<E> descendingIterator() {
089        return this.descendingSet().iterator();
090    }
091
092    @Override
093    public E lower(E elem) {
094        if (!this.bounds.isWithinLowerBound(this.comparator(), elem))
095            return null;
096        final NavigableSet<E> subSet = this.bounds.isWithinUpperBound(this.comparator(), elem) ? this.headSet(elem, false) : this;
097        try {
098            return subSet.last();
099        } catch (NoSuchElementException e) {
100            return null;
101        }
102    }
103
104    @Override
105    public E floor(E elem) {
106        if (!this.bounds.isWithinLowerBound(this.comparator(), elem))
107            return null;
108        final NavigableSet<E> subSet = this.bounds.isWithinUpperBound(this.comparator(), elem) ? this.headSet(elem, true) : this;
109        try {
110            return subSet.last();
111        } catch (NoSuchElementException e) {
112            return null;
113        }
114    }
115
116    @Override
117    public E ceiling(E elem) {
118        if (!this.bounds.isWithinUpperBound(this.comparator(), elem))
119            return null;
120        final NavigableSet<E> subSet = this.bounds.isWithinLowerBound(this.comparator(), elem) ? this.tailSet(elem, true) : this;
121        try {
122            return subSet.first();
123        } catch (NoSuchElementException e) {
124            return null;
125        }
126    }
127
128    @Override
129    public E higher(E elem) {
130        if (!this.bounds.isWithinUpperBound(this.comparator(), elem))
131            return null;
132        final NavigableSet<E> subSet = this.bounds.isWithinLowerBound(this.comparator(), elem) ? this.tailSet(elem, false) : this;
133        try {
134            return subSet.first();
135        } catch (NoSuchElementException e) {
136            return null;
137        }
138    }
139
140    @Override
141    public NavigableSet<E> headSet(E newMaxElement) {
142        return this.headSet(newMaxElement, false);
143    }
144
145    @Override
146    public NavigableSet<E> tailSet(E newMinElement) {
147        return this.tailSet(newMinElement, true);
148    }
149
150    @Override
151    public NavigableSet<E> subSet(E newMinElement, E newMaxElement) {
152        return this.subSet(newMinElement, true, newMaxElement, false);
153    }
154
155    @Override
156    public NavigableSet<E> descendingSet() {
157        return this.createSubSet(true, this.bounds.reverse());
158    }
159
160    @Override
161    public NavigableSet<E> headSet(E newMaxElement, boolean inclusive) {
162        final Bounds<E> newBounds = this.bounds.withUpperBound(newMaxElement, BoundType.of(inclusive));
163        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
164            throw new IllegalArgumentException("upper bound is out of range");
165        return this.createSubSet(false, newBounds);
166    }
167
168    @Override
169    public NavigableSet<E> tailSet(E newMinElement, boolean inclusive) {
170        final Bounds<E> newBounds = this.bounds.withLowerBound(newMinElement, BoundType.of(inclusive));
171        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
172            throw new IllegalArgumentException("lower bound is out of range");
173        return this.createSubSet(false, newBounds);
174    }
175
176    @Override
177    public NavigableSet<E> subSet(E newMinElement, boolean minInclusive, E newMaxElement, boolean maxInclusive) {
178        final Bounds<E> newBounds = new Bounds<>(newMinElement,
179          BoundType.of(minInclusive), newMaxElement, BoundType.of(maxInclusive));
180        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
181            throw new IllegalArgumentException("bound(s) are out of range");
182        return this.createSubSet(false, newBounds);
183    }
184
185    /**
186     * Get an {@link Comparator} that sorts consistently with (and optionally reversed from) this instance.
187     *
188     * @param reversed whether to return a reversed {@link Comparator}
189     * @return a non-null {@link Comparator}
190     */
191    protected Comparator<? super E> getComparator(boolean reversed) {
192        final Comparator<? super E> comparator = this.comparator();
193        if (comparator != null)
194            return AbstractNavigableSet.possiblyReverse(comparator, reversed);
195        return reversed ? Collections.<E>reverseOrder() : new NaturalComparator();
196    }
197
198    // This method exists solely to bind the generic type parameters
199    private static <T> Comparator<T> possiblyReverse(Comparator<T> comparator, boolean reverse) {
200        return reverse ? Collections.reverseOrder(comparator) : comparator;
201    }
202
203    /**
204     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds.
205     * The {@code newBounds} are consistent with the new ordering (i.e., reversed relative to this instance's ordering if
206     * {@code reverse} is true) and have already been range-checked against {@linkplain #bounds this instance's current bounds}.
207     *
208     * @param reverse whether the new set's ordering should be reversed relative to this instance's ordering
209     * @param newBounds new bounds
210     * @throws IllegalArgumentException if {@code newBounds} is null
211     * @throws IllegalArgumentException if a bound in {@code newBounds} is null and this set does not permit null elements
212     */
213    protected abstract NavigableSet<E> createSubSet(boolean reverse, Bounds<E> newBounds);
214
215// NaturalComparator
216
217    private class NaturalComparator implements Comparator<E> {
218
219        @Override
220        @SuppressWarnings("unchecked")
221        public int compare(E e1, E e2) {
222            return ((Comparable<E>)e1).compareTo(e2);
223        }
224    }
225}
