008package org.jsimpledb.util;
009
010import java.util.Arrays;
011
012import javax.xml.namespace.QName;
013import javax.xml.stream.XMLStreamConstants;
014import javax.xml.stream.XMLStreamException;
015import javax.xml.stream.XMLStreamReader;
016import javax.xml.stream.XMLStreamWriter;
017
018/**
019 * Support superclass for classes that serialize and deserialize via XML.
020 */
021public abstract class AbstractXMLStreaming {
022
023    protected AbstractXMLStreaming() {
024    }
025
026    /**
027     * Scan forward until we see an opening or closing tag.
028     * If opening tag is seen, it must match one of {@code names} and then we return true, if not or if {@code names}
029     * is empty throw an exception; if a closing tag is seen, return false if {@code closingOK}, else throw exception.
030     *
031     * @param reader XML input
032     * @param closingOK true if a closing tag is OK, otherwise false
033     * @param names expected opening tag, or null if we expected a closing tag
034     * @return true if matching opening tag seen, false otherwise
035     * @throws XMLStreamException if something unexpected is encountered
036     */
037    protected boolean expect(XMLStreamReader reader, boolean closingOK, QName... names) throws XMLStreamException {
038        while (true) {
039            if (!reader.hasNext())
040                throw new XMLStreamException("unexpected end of input", reader.getLocation());
041            final int eventType = reader.next();
042            if (eventType == XMLStreamConstants.START_ELEMENT)
043                break;
044            if (eventType == XMLStreamConstants.END_ELEMENT) {
045                if (!closingOK) {
046                    throw new XMLStreamException("expected " + this.description(names) + " but found closing <"
047                      + reader.getName() + "> tag instead", reader.getLocation());
048                }
049                return false;
050            }
051        }
052        if (names.length == 0) {
053            throw new XMLStreamException("expected closing tag but found opening <"
054              + reader.getName() + "> tag instead", reader.getLocation());
055        }
056        if (!Arrays.asList(names).contains(reader.getName())) {
057            throw new XMLStreamException("expected " + this.description(names)
058              + " but found <" + reader.getName() + "> instead", reader.getLocation());
059        }
060        return true;
061    }
062
063    /**
064     * Scan forward expecting to see a closing tag.
065     *
066     * @param reader XML input
067     * @throws XMLStreamException if something other than a closing tag is encountered
068     */
069    protected boolean expectClose(XMLStreamReader reader) throws XMLStreamException {
070        return this.expect(reader, true);
071    }
072
073    private String description(QName[] names) {
074        switch (names.length) {
075        case 0:
076            return "closing tag";
077        case 1:
078            return "opening <" + names[0] + "> tag";
079        default:
080            final StringBuilder buf = new StringBuilder();
081            for (QName name : names) {
082                if (buf.length() == 0)
083                    buf.append("one of ");
084                else
085                    buf.append(", ");
086                buf.append('<').append(name).append('>');
087            }
088            return buf.toString();
089        }
090    }
091
092    /**
093     * Write out a simple XML element containing the given content.
094     *
095     * @param writer XML output
096     * @param element element name
097     * @param content simple content
098     */
099    protected void writeElement(XMLStreamWriter writer, QName element, String content) throws XMLStreamException {
100        writer.writeStartElement(element.getNamespaceURI(), element.getLocalPart());
101        writer.writeCharacters(content);
102        writer.writeEndElement();
103    }
104
105    /**
106     * Get an attribute from the current element.
107     *
108     * @param reader XML input
109     * @param name attribute name
110     * @param required whether attribute must be present
111     * @return attribute value, or null if not {@code required} and no attribute is present
112     * @throws IllegalStateException if the current event is not a start element event
113     * @throws XMLStreamException if {@code required} is true and no such attribute is found
114     */
115    protected String getAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
116        final String value = reader.getAttributeValue(name.getNamespaceURI(), name.getLocalPart());
117        if (value == null && required) {
118            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element is missing required \""
119              + name + "\" attribute", reader.getLocation());
120        }
121        return value;
122    }
123
124    /**
125     * Get a requried attribute from the current element. Equivalent to: {@code getAttr(reader, name, true)}.
126     *
127     * @param reader XML input
128     * @param name attribute name
129     * @return attribute value
130     * @throws IllegalStateException if the current event is not a start element event
131     * @throws XMLStreamException if no such attribute is found
132     */
133    protected String getAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
134        return this.getAttr(reader, name, true);
135    }
136
137    /**
138     * Get an attribute from the current element and parse as an integer value.
139     *
140     * @param reader XML input
141     * @param name attribute name
142     * @param required whether attribute must be present
143     * @return attribute value, or null if not {@code required} and no attribute is present
144     * @throws IllegalStateException if the current event is not a start element event
145     * @throws XMLStreamException if {@code required} is true and no such attribute is found
146     * @throws XMLStreamException if attribute is not an integer value
147     */
148    protected Integer getIntAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
149        final String text = this.getAttr(reader, name, required);
150        if (text == null)
151            return null;
152        try {
153            return Integer.parseInt(text, 10);
154        } catch (NumberFormatException e) {
155            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element attribute \""
156              + name + "\" value `" + text + "' is invalid: not an integer value", reader.getLocation(), e);
157        }
158    }
159
160    /**
161     * Get a requried integer attribute from the current element. Equivalent to: {@code getIntAttr(reader, name, true)}.
162     *
163     * @param reader XML input
164     * @param name attribute name
165     * @return attribute value
166     * @throws IllegalStateException if the current event is not a start element event
167     * @throws XMLStreamException if no such attribute is found
168     * @throws XMLStreamException if attribute is not an integer value
169     */
170    protected int getIntAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
171        return this.getIntAttr(reader, name, true);
172    }
173
174    /**
175     * Get an attribute from the current element and parse as a boolean value.
176     *
177     * @param reader XML input
178     * @param name attribute name
179     * @param required whether attribute must be present
180     * @return attribute value, or null if not {@code required} and no attribute is present
181     * @throws IllegalStateException if the current event is not a start element event
182     * @throws XMLStreamException if {@code required} is true and no such attribute is found
183     * @throws XMLStreamException if attribute is not {@code "true"} or {@code "false"}
184     */
185    protected Boolean getBooleanAttr(XMLStreamReader reader, QName name, boolean required) throws XMLStreamException {
186        final String text = this.getAttr(reader, name, required);
187        if (text == null)
188            return null;
189        switch (text) {
190        case "true":
191            return true;
192        case "false":
193            return false;
194        default:
195            throw new XMLStreamException("<" + reader.getName().getLocalPart() + "> element attribute \""
196              + name + "\" value `" + text + "' is invalid: expected either \"true\" or \"false\"", reader.getLocation());
197        }
198    }
199
200    /**
201     * Get a requried boolean attribute from the current element. Equivalent to: {@code getBooleanAttr(reader, name, true)}.
202     *
203     * @param reader XML input
204     * @param name attribute name
205     * @return attribute value
206     * @throws IllegalStateException if the current event is not a start element event
207     * @throws XMLStreamException if no such attribute is found
208     * @throws XMLStreamException if attribute is not {@code "true"} or {@code "false"}
209     */
210    protected boolean getBooleanAttr(XMLStreamReader reader, QName name) throws XMLStreamException {
211        return this.getBooleanAttr(reader, name, true);
212    }
213}
