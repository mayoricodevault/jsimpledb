058/**
059 * Utility methods for serializing and deserializing objects in a {@link Transaction} to/from XML.
060 *
061 * <p>
062 * There are two supported formats. The "Storage ID Format" specifies objects and fields using their storage IDs
063 * and supports all possible database contents.
064 *  <pre>
065 *  &lt;objects&gt;
066 *      &lt;object storageId="100" id="64a971e1aef01cc8" version="1"&gt;
067 *          &lt;field storageId="101"&gt;George Washington&lt;/field&gt;
068 *          &lt;field storageId="102"&gt;true&lt;/field&gt;
069 *          &lt;field storageId="103"&gt;
070 *              &lt;entry&gt;
071 *                  &lt;key&gt;teeth&lt;/key&gt;
072 *                  &lt;value&gt;wooden&lt;/value&gt;
073 *              &lt;/entry&gt;
074 *          &lt;/field&gt;
075 *          &lt;field storageId="104"&gt;c8b84a08e5c2b1a2&lt;/field&gt;
076 *      &lt;/object&gt;
077 *      ...
078 *  &lt;/objects&gt;
079 *  </pre>
080 *
081 * <p>
082 * "Name Format" differs from "Storage ID Format" in that object types and fields are specified by name instead of storage ID.
083 * However, it does not support schemas that use names that are not valid XML tags.
084 * <pre>
085 *  &lt;objects&gt;
086 *      &lt;Person id="64a971e1aef01cc8" version="1"&gt;
087 *          &lt;name&gt;George Washington&lt;/name&gt;
088 *          &lt;wasPresident&gt;true&lt;/wasPresident&gt;
089 *          &lt;attributes&gt;
090 *              &lt;entry&gt;
091 *                  &lt;key&gt;teeth&lt;/key&gt;
092 *                  &lt;value&gt;wooden&lt;/value&gt;
093 *              &lt;/entry&gt;
094 *          &lt;/attributes&gt;
095 *          &lt;spouse&gt;c8b84a08e5c2b1a2&lt;/spouse&gt;
096 *      &lt;/Person&gt;
097 *      ...
098 *  &lt;/objects&gt;
099 *  </pre>
100 *
101 * <p>
102 * Some details on the input logic:
103 * <ul>
104 *  <li>Simple fields that are equal to their default values, and complex fields that are empty, may be omitted</li>
105 *  <li>The {@code "version"} attribute may be omitted, in which case the default schema version associated with
106 *      the {@link Transaction} being written to is assumed.</li>
107 *  <li>The {@code "id"} attribute may be omitted, in which case a random unassigned ID is generated</li>
108 *  <li>Any object ID (including the {@code "id"} attribute) may have the special form
109 *      <code>generated:<i>TYPE</i>:<i>SUFFIX</i></code>, where <code><i>TYPE</i></code> is the object type
110 *      name or storage ID and <code><i>SUFFIX</i></code> is an arbitrary string.
111 *      In this case, a random, unassigned object ID for type <code><i>TYPE</i></code> is generated
112 *      on first occurrence, and on subsequent occurences the previously generated ID is recalled. This facilitates
113 *      input generated via XSL and the {@code generate-id()} function.
114 *      The {@linkplain #getGeneratedIdCache configured} {@link GeneratedIdCache} keeps track of generated IDs.</li>
115 * </ul>
116 */

194    /**
195     * Import objects into the {@link Transaction} associated with this instance from the given XML input.
196     * This method expects to see an opening {@code <objects>} as the next event (not counting whitespace, comments, etc.),
197     * which is then consumed up through the closing {@code </objects>} event. Therefore this tag could be part of a
198     * larger XML document.
199     *
200     * <p>
201     * The input format is auto-detected for each {@code <object>} based on the presense of the {@code "type"} attribute.
202     * </p>
203     *
204     * @param reader XML reader
205     * @return the number of objects read
206     * @throws XMLStreamException if an error occurs
207     * @throws IllegalArgumentException if {@code reader} is null
208     */
209    @SuppressWarnings("unchecked")
210    public int read(XMLStreamReader reader) throws XMLStreamException {
211        if (reader == null)
212            throw new IllegalArgumentException("null reader");
213        this.expect(reader, false, OBJECTS_TAG);
214
215        // Create a snapshot transaction so we can replace objects without triggering DeleteAction's
216        final SnapshotTransaction snapshot = this.tx.createSnapshotTransaction();
217
218        // Iterate over objects
219        QName name;
220        int count;
221        for (count = 0; (name = this.next(reader)) != null; count++) {
222
223            // Determine schema version
224            final String versionAttr = this.getAttr(reader, VERSION_ATTR, false);
225            final Schema schema;
226            if (versionAttr != null) {
227                try {
228                    schema = this.tx.getSchemas().getVersion(Integer.parseInt(versionAttr));
229                } catch (IllegalArgumentException e) {
230                    throw new XMLStreamException("invalid object schema version `" + versionAttr + "': " + e,
231                      reader.getLocation(), e);
232                }
233            } else
234                schema = this.tx.getSchema();
235            final NameIndex nameIndex = this.nameIndexMap.get(schema.getVersionNumber());
236            final SchemaModel schemaModel = schema.getSchemaModel();
237
238            // Determine object type
239            String storageIdAttr = this.getAttr(reader, STORAGE_ID_ATTR, false);
240            ObjType objType = null;
241            if (storageIdAttr != null) {
242                try {
243                    objType = schema.getObjType(Integer.parseInt(storageIdAttr));
244                } catch (UnknownTypeException e) {
245                    throw new XMLStreamException("invalid object type storage ID `" + storageIdAttr + "': " + e,
246                      reader.getLocation(), e);
247                }
248                if (!name.equals(OBJECT_TAG)) {
249                    if (!XMLConstants.NULL_NS_URI.equals(name.getNamespaceURI())) {
250                        throw new XMLStreamException("unexpected element <" + name.getPrefix() + ":"
251                          + name.getLocalPart() + ">; expected <" + OBJECT_TAG.getLocalPart() + ">", reader.getLocation());
252                    }
253                    if (!objType.getName().equals(name.getLocalPart())) {
254                        throw new XMLStreamException("element <" + name.getLocalPart() + "> does not match storage ID "
255                          + objType.getStorageId() + "; should be <" + objType.getName() + ">", reader.getLocation());
256                    }
257                }
258            } else {
259                if (!XMLConstants.NULL_NS_URI.equals(name.getNamespaceURI())) {
260                    throw new XMLStreamException("unexpected element <" + name.getPrefix() + ":"
261                      + name.getLocalPart() + ">; expected <object> or object type name", reader.getLocation());
262                }
263                if (!name.equals(OBJECT_TAG)) {
264                    final SchemaObjectType schemaObjectType = nameIndex.getSchemaObjectType(name.getLocalPart());
265                    if (schemaObjectType == null) {
266                        throw new XMLStreamException("unexpected element <" + name.getLocalPart()
267                          + ">; no object type named `" + name.getLocalPart() + "' found in schema version "
268                          + schema.getVersionNumber(), reader.getLocation());
269                    }
270                    objType = schema.getObjType(schemaObjectType.getStorageId());
271                }
272            }
273
274            // Reset snapshot
275            snapshot.reset();
276
277            // Determine object ID and create object in snapshot
278            final String idAttr = this.getAttr(reader, ID_ATTR, true);
279            ObjId id;
280            if (idAttr == null) {
281
282                // Verify we know object type
283                if (objType == null) {
284                    throw new XMLStreamException("invalid <" + OBJECT_TAG.getLocalPart() + "> element: either \""
285                      + STORAGE_ID_ATTR.getLocalPart() + "\" or \"" + ID_ATTR.getLocalPart() + "\" attribute is required",
286                      reader.getLocation());
287                }
288
289                // Create object
290                id = snapshot.create(objType.getStorageId(), schema.getVersionNumber());
291            } else {
292
293                // Parse id
294                try {
295                    id = new ObjId(idAttr);
296                } catch (IllegalArgumentException e) {
297                    final Matcher matcher = GENERATED_ID_PATTERN.matcher(idAttr);
298                    if (!matcher.matches())
299                        throw new XMLStreamException("invalid object ID `" + idAttr + "'", reader.getLocation());
300                    final String typeAttr = matcher.group(1);
301                    final ObjType genType = this.parseGeneratedType(reader, idAttr, typeAttr, schema);
302                    if (objType == null)
303                        objType = genType;
304                    else if (!genType.equals(objType)) {
305                        throw new XMLStreamException("type `" + typeAttr + "' in generated object ID `" + idAttr
306                          + "' does not match type `" + objType.getName() + "' in schema version "
307                          + schema.getVersionNumber(), reader.getLocation());
308                    }
309                    id = this.generatedIdCache.getGeneratedId(this.tx, objType.getStorageId(), matcher.group(2));
310                }
311
312                // Create object
313                snapshot.create(id, schema.getVersionNumber());
314            }
315
316            // Iterate over fields
317            final SchemaObjectType schemaObjectType = schemaModel.getSchemaObjectTypes().get(objType.getStorageId());
318            while ((name = this.next(reader)) != null) {
319                final QName fieldTagName = name;
320
321                // Determine the field
322                storageIdAttr = this.getAttr(reader, STORAGE_ID_ATTR, false);
323                final Field<?> field;
324                if (storageIdAttr != null) {
325                    try {
326                        field = objType.getFields().get(Integer.parseInt(storageIdAttr));
327                    } catch (IllegalArgumentException e) {
328                        throw new XMLStreamException("invalid field storage ID `" + storageIdAttr
329                          + "': " + e, reader.getLocation(), e);
330                    }
331                    if (field == null) {
332                        throw new XMLStreamException("unknown field storage ID `" + storageIdAttr + "' for object type `"
333                          + objType.getName() + "' #" + objType.getStorageId() + " in schema version "
334                          + schema.getVersionNumber(), reader.getLocation());
335                    }
336                } else {
337                    if (!XMLConstants.NULL_NS_URI.equals(name.getNamespaceURI())) {
338                        throw new XMLStreamException("unexpected element <" + name.getPrefix() + ":"
339                          + name.getLocalPart() + ">; expected field name", reader.getLocation());
340                    }
341                    final SchemaField schemaField = nameIndex.getSchemaField(schemaObjectType, name.getLocalPart());
342                    if (schemaField == null) {
343                        throw new XMLStreamException("unexpected element <" + name.getLocalPart() + ">; unknown field `"
344                          + name.getLocalPart() + "' in object type `" + objType.getName() + "' #"
345                          + objType.getStorageId() + " in schema version " + schema.getVersionNumber(),
346                          reader.getLocation());
347                    }
348                    field = objType.getFields().get(schemaField.getStorageId());
349                    assert field != null : "field=" + schemaField + " fields=" + objType.getFields();
350                }
351
352                // Parse the field
353                if (field instanceof SimpleField)
354                    snapshot.writeSimpleField(id, field.getStorageId(), this.readSimpleField(reader, (SimpleField<?>)field), false);
355                else if (field instanceof CounterField) {
356                    final long value;
357                    try {
358                        value = Long.parseLong(reader.getElementText());
359                    } catch (Exception e) {
360                        throw new XMLStreamException("invalid counter value for field `" + field.getName() + "': " + e,
361                          reader.getLocation(), e);
362                    }
363                    snapshot.writeCounterField(id, field.getStorageId(), value, false);
364                } else if (field instanceof CollectionField) {
365                    final SimpleField<?> elementField = ((CollectionField<?, ?>)field).getElementField();
366                    final Collection<?> collection;
367                    if (field instanceof SetField)
368                        collection = snapshot.readSetField(id, field.getStorageId(), false);
369                    else if (field instanceof ListField)
370                        collection = snapshot.readListField(id, field.getStorageId(), false);
371                    else
372                        throw new RuntimeException("internal error: " + field);
373                    while ((name = this.next(reader)) != null) {
374                        if (!ELEMENT_TAG.equals(name)) {
375                            throw new XMLStreamException("invalid field element; expected <" + ELEMENT_TAG.getLocalPart()
376                              + "> but found opening <" + name.getLocalPart() + ">", reader.getLocation());
377                        }
378                        ((Collection<Object>)collection).add(this.readSimpleField(reader, elementField));
379                    }
380                } else if (field instanceof MapField) {
381                    final SimpleField<?> keyField = ((MapField<?, ?>)field).getKeyField();
382                    final SimpleField<?> valueField = ((MapField<?, ?>)field).getValueField();
383                    final Map<?, ?> map = snapshot.readMapField(id, field.getStorageId(), false);
384                    while ((name = this.next(reader)) != null) {
385                        if (!ENTRY_TAG.equals(name)) {
386                            throw new XMLStreamException("invalid map field entry; expected <" + ENTRY_TAG.getLocalPart()
387                              + "> but found opening <" + name.getLocalPart() + ">", reader.getLocation());
388                        }
389                        if (!KEY_TAG.equals(this.next(reader))) {
390                            throw new XMLStreamException("invalid map entry key; expected <" + KEY_TAG.getLocalPart()
391                              + ">", reader.getLocation());
392                        }
393                        final Object key = this.readSimpleField(reader, keyField);
394                        if (!VALUE_TAG.equals(this.next(reader))) {
395                            throw new XMLStreamException("invalid map entry value; expected <" + VALUE_TAG.getLocalPart()
396                              + ">", reader.getLocation());
397                        }
398                        final Object value = this.readSimpleField(reader, valueField);
399                        ((Map<Object, Object>)map).put(key, value);
400                        if ((name = this.next(reader)) != null) {
401                            throw new XMLStreamException("invalid map field entry; expected closing <" + ENTRY_TAG.getLocalPart()
402                              + "> but found opening <" + name.getLocalPart() + ">", reader.getLocation());
403                        }
404                    }
405                } else
406                    throw new RuntimeException("internal error: " + field);
407            }
408
409            // Copy over object, replacing any previous
410            snapshot.copy(id, id, this.tx);
411        }
412
413        // Done
414        return count;
415    }

417    /**
418     * Export all objects from the {@link Transaction} associated with this instance to the given output.
419     *
420     * @param output XML output; will not be closed by this method
421     * @param nameFormat true for Name Format, false for Storage ID Format
422     * @param indent true to indent output, false for all on one line
423     * @return the number of objects written
424     * @throws XMLStreamException if an error occurs
425     * @throws IllegalArgumentException if {@code output} is null
426     */
427    public int write(OutputStream output, boolean nameFormat, boolean indent) throws XMLStreamException {
428        if (output == null)
429            throw new IllegalArgumentException("null output");
430        XMLStreamWriter xmlWriter = XMLOutputFactory.newInstance().createXMLStreamWriter(output, "UTF-8");
431        if (indent)
432            xmlWriter = new IndentXMLStreamWriter(xmlWriter);
433        xmlWriter.writeStartDocument("UTF-8", "1.0");
434        return this.write(xmlWriter, nameFormat);
435    }

457    private int write(XMLStreamWriter writer, boolean nameFormat) throws XMLStreamException {
458
459        // Gather all known object storage IDs
460        final TreeSet<Integer> storageIds = new TreeSet<>();
461        for (Schema schema : this.tx.getSchemas().getVersions().values())
462            storageIds.addAll(schema.getSchemaModel().getSchemaObjectTypes().keySet());
463
464        // Get corresponding iterators
465        final ArrayList<NavigableSet<ObjId>> sets = new ArrayList<>(storageIds.size());
466        for (int storageId : storageIds)
467            sets.add(this.tx.getAll(storageId));
468
469        // Output all objects
470        return this.write(writer, nameFormat, Iterables.concat(sets));
471    }

473    /**
474     * Export the given objects from the {@link Transaction} associated with this instance to the given XML output.
475     *
476     * <p>
477     * This method writes a start element as its first action, allowing the output to be embedded into a larger XML document.
478     * Callers not embedding the output may with to precede invocation of this method with a call to
479     * {@link XMLStreamWriter#writeStartDocument writer.writeStartDocument()}.
480     * </p>
481     *
482     * @param writer XML writer; will not be closed by this method
483     * @param nameFormat true for Name Format, false for Storage ID Format
484     * @param objIds object IDs
485     * @return the number of objects written
486     * @throws XMLStreamException if an error occurs
487     * @throws IllegalArgumentException if {@code writer} is null
488     */
489    public int write(XMLStreamWriter writer, boolean nameFormat, Iterable<? extends ObjId> objIds) throws XMLStreamException {
490        if (writer == null)
491            throw new IllegalArgumentException("null writer");
492        if (objIds == null)
493            throw new IllegalArgumentException("null objIds");
494        final NameComparator nameComparator = new NameComparator();
495        writer.setDefaultNamespace(OBJECTS_TAG.getNamespaceURI());
496        writer.writeStartElement(OBJECTS_TAG.getNamespaceURI(), OBJECTS_TAG.getLocalPart());
497        int count = 0;
498        for (ObjId id : objIds) {
499
500            // Get object info
501            final int typeStorageId = id.getStorageId();
502            final int version = this.tx.getSchemaVersion(id);
503            final Schema schema = this.tx.getSchemas().getVersion(version);
504            final ObjType objType = schema.getObjType(typeStorageId);
505
506            // Get format info
507            final QName objectElement = nameFormat ? new QName(objType.getName()) : OBJECT_TAG;
508            final int storageIdAttr = nameFormat ? -1 : typeStorageId;
509
510            // Output fields; if all are default, output empty tag
511            boolean tagOutput = false;
512            ArrayList<Field<?>> fieldList = new ArrayList<>(objType.getFields().values());
513            if (nameFormat)
514                Collections.sort(fieldList, nameComparator);
515            for (Field<?> field : fieldList) {
516
517                // Determine if field equals its default value; if so, skip it
518                if (field.hasDefaultValue(this.tx, id))
519                    continue;
520
521                // Output <object> opening tag if not output yet
522                if (!tagOutput) {
523                    this.writeOpenTag(writer, false, objectElement, storageIdAttr, id, version);
524                    tagOutput = true;
525                }
526
527                // Get tag name
528                final QName fieldTag = nameFormat ? new QName(field.getName()) : FIELD_TAG;
529
530                // Special case for simple fields, which use empty tags when null
531                if (field instanceof SimpleField) {
532                    final Object value = this.tx.readSimpleField(id, field.getStorageId(), false);
533                    if (value != null)
534                        writer.writeStartElement(fieldTag.getNamespaceURI(), fieldTag.getLocalPart());
535                    else
536                        writer.writeEmptyElement(fieldTag.getNamespaceURI(), fieldTag.getLocalPart());
537                    if (!nameFormat)
538                        this.writeAttribute(writer, STORAGE_ID_ATTR, field.getStorageId());
539                    if (value != null) {
540                        this.writeSimpleField(writer, (SimpleField<?>)field, value);
541                        writer.writeEndElement();
542                    } else
543                        this.writeAttribute(writer, NULL_ATTR, "true");
544                    continue;
545                }
546
547                // Output field opening tag
548                writer.writeStartElement(fieldTag.getNamespaceURI(), fieldTag.getLocalPart());
549                if (!nameFormat)
550                    this.writeAttribute(writer, STORAGE_ID_ATTR, field.getStorageId());
551
552                // Output field value
553                if (field instanceof CounterField)
554                    writer.writeCharacters("" + this.tx.readCounterField(id, field.getStorageId(), false));
555                else if (field instanceof CollectionField) {
556                    final SimpleField<?> elementField = ((CollectionField<?, ?>)field).getElementField();
557                    final Iterable<?> collection = field instanceof SetField ?
558                      this.tx.readSetField(id, field.getStorageId(), false) :
559                      this.tx.readListField(id, field.getStorageId(), false);
560                    for (Object element : collection)
561                        this.writeSimpleTag(writer, ELEMENT_TAG, elementField, element);
562                } else if (field instanceof MapField) {
563                    final SimpleField<?> keyField = ((MapField<?, ?>)field).getKeyField();
564                    final SimpleField<?> valueField = ((MapField<?, ?>)field).getValueField();
565                    for (Map.Entry<?, ?> entry : this.tx.readMapField(id, field.getStorageId(), false).entrySet()) {
566                        writer.writeStartElement(ENTRY_TAG.getNamespaceURI(), ENTRY_TAG.getLocalPart());
567                        this.writeSimpleTag(writer, KEY_TAG, keyField, entry.getKey());
568                        this.writeSimpleTag(writer, VALUE_TAG, valueField, entry.getValue());
569                        writer.writeEndElement();
570                    }
571                } else
572                    throw new RuntimeException("internal error: " + field);
573
574                // Output field closing tag
575                writer.writeEndElement();
576            }
577
578            // Output empty opening tag if not output yet, otherwise closing tag
579            if (!tagOutput)
580                this.writeOpenTag(writer, true, objectElement, storageIdAttr, id, version);
581            else
582                writer.writeEndElement();
583            count++;
584        }
585
586        // Done
587        writer.writeEndElement();
588        writer.flush();
589        return count;
590    }

594    private <T> boolean isDefaultValue(SimpleField<T> field, Object value) throws XMLStreamException {
595        final ByteWriter writer = new ByteWriter();
596        final FieldType<T> fieldType = field.getFieldType();
597        fieldType.write(writer, fieldType.validate(value));
598        return Arrays.equals(writer.getBytes(), field.getFieldType().getDefaultValue());
599    }

601    private void writeSimpleTag(XMLStreamWriter writer, QName tag, SimpleField<?> field, Object value)
602      throws XMLStreamException {
603        if (value != null) {
604            writer.writeStartElement(tag.getNamespaceURI(), tag.getLocalPart());
605            this.writeSimpleField(writer, field, value);
606            writer.writeEndElement();
607        } else {
608            writer.writeEmptyElement(tag.getNamespaceURI(), tag.getLocalPart());
609            this.writeAttribute(writer, NULL_ATTR, "true");
610        }
611    }

613    private <T> void writeSimpleField(XMLStreamWriter writer, SimpleField<T> field, Object value) throws XMLStreamException {
614        final FieldType<T> fieldType = field.getFieldType();
615        writer.writeCharacters(fieldType.toString(fieldType.validate(value)));
616    }

618    private <T> T readSimpleField(XMLStreamReader reader, SimpleField<T> field) throws XMLStreamException {
619
620        // Get field type
621        final FieldType<T> fieldType = field.getFieldType();
622
623        // Check for null
624        final String nullAttr = this.getAttr(reader, NULL_ATTR, false);
625        boolean isNull = false;
626        if (nullAttr != null) {
627            switch (nullAttr) {
628            case "true":
629            case "false":
630                isNull = Boolean.valueOf(nullAttr);
631                break;
632            default:
633                throw new XMLStreamException("invalid value `" + nullAttr + "' for `" + NULL_ATTR.getLocalPart()
634                 + "' attribute: value must be \"true\" or \"false\"", reader.getLocation());
635            }
636        }
637
638        // Get text content
639        final String text;
640        try {
641            text = reader.getElementText();
642        } catch (Exception e) {
643            throw new XMLStreamException("invalid value for field `" + field.getName() + "': " + e, reader.getLocation(), e);
644        }
645
646        // If null, verify there is no text content
647        if (isNull) {
648            if (text.length() != 0)
649                throw new XMLStreamException("text content not allowed for values with null=\"true\"", reader.getLocation());
650            return null;
651        }
652
653        // Handle generated ID's for reference fields
654        if (field instanceof ReferenceField) {
655            final Matcher matcher = GENERATED_ID_PATTERN.matcher(text);
656            if (matcher.matches()) {
657                final int storageId = this.parseGeneratedType(reader, text, matcher.group(1));
658                return fieldType.validate(this.generatedIdCache.getGeneratedId(this.tx, storageId, matcher.group(2)));
659            }
660        }
661
662        // Parse field value
663        try {
664            return fieldType.fromString(text);
665        } catch (Exception e) {
666            throw new XMLStreamException("invalid value `" + text + "' for field `" + field.getName() + "': " + e,
667              reader.getLocation(), e);
668        }
669    }
670
671    // Parse a generated ID type found in a reference
672    private int parseGeneratedType(XMLStreamReader reader, String text, String attr) throws XMLStreamException {
673        int storageId = -1;
674        for (Schema schema : this.tx.getSchemas().getVersions().values()) {
675            final ObjType objType;
676            try {
677                objType = this.parseGeneratedType(reader, text, attr, schema);
678            } catch (XMLStreamException e) {
679                continue;
680            }
681            if (storageId != -1 && storageId != objType.getStorageId()) {
682                throw new XMLStreamException("invalid object type `" + attr + "' in generated object ID `"
683                  + text + "': two or more incompatible object types named `" + attr
684                  + "' exist (in different schema versions)", reader.getLocation());
685            }
686            storageId = objType.getStorageId();
687        }
688        if (storageId == -1) {
689            throw new XMLStreamException("invalid object type `" + attr + "' in generated object ID `"
690              + text + "': no such object type found in any schema version", reader.getLocation());
691        }
692        return storageId;
693    }
694
695    // Parse a generated ID type found in an object definition
696    private ObjType parseGeneratedType(XMLStreamReader reader, String text, String attr, Schema schema)
697      throws XMLStreamException {
698
699        // Try object type name
700        final NameIndex nameIndex = this.nameIndexMap.get(schema.getVersionNumber());
701        final SchemaObjectType schemaObjectType = nameIndex.getSchemaObjectType(attr);
702        if (schemaObjectType != null)
703            return schema.getObjType(schemaObjectType.getStorageId());
704
705        // Try storage ID
706        final int storageId;
707        try {
708            storageId = Integer.parseInt(attr);
709        } catch (IllegalArgumentException e) {
710            throw new XMLStreamException("invalid object type `" + attr + "' in generated object ID `"
711              + text + "': no such object type found in schema version " + schema.getVersionNumber(), reader.getLocation());
712        }
713        try {
714            return schema.getObjType(storageId);
715        } catch (UnknownTypeException e) {
716            throw new XMLStreamException("invalid storage ID " + storageId + " in generated object ID `"
717              + text + "': no such object type found in schema version " + schema.getVersionNumber(), reader.getLocation());
718        }
719    }

721    private void writeOpenTag(XMLStreamWriter writer, boolean empty, QName element, int storageId, ObjId id, int version)
722      throws XMLStreamException {
723        if (empty)
724            writer.writeEmptyElement(element.getNamespaceURI(), element.getLocalPart());
725        else
726            writer.writeStartElement(element.getNamespaceURI(), element.getLocalPart());
727        if (storageId != -1)
728            this.writeAttribute(writer, STORAGE_ID_ATTR, storageId);
729        this.writeAttribute(writer, ID_ATTR, id);
730        this.writeAttribute(writer, VERSION_ATTR, version);
731    }
732
733    private void writeAttribute(XMLStreamWriter writer, QName attr, Object value) throws XMLStreamException {
734        writer.writeAttribute(attr.getNamespaceURI(), attr.getLocalPart(), "" + value);
735    }

737    private QName next(XMLStreamReader reader) throws XMLStreamException {
738        while (true) {
739            if (!reader.hasNext())
740                throw new XMLStreamException("unexpected end of input", reader.getLocation());
741            final int eventType = reader.next();
742            if (eventType == XMLStreamConstants.END_ELEMENT)
743                return null;
744            if (eventType == XMLStreamConstants.START_ELEMENT)
745                return reader.getName();
746        }
747    }
