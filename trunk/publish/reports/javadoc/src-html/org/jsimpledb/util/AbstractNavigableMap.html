008package org.jsimpledb.util;
009
010import java.util.AbstractMap;
011import java.util.Iterator;
012import java.util.Map;
013import java.util.NavigableMap;
014import java.util.NavigableSet;
015import java.util.NoSuchElementException;
016
017/**
018 * Support superclass for {@link NavigableMap} implementations.
019 *
020 * <p>
021 * For a read-only implementation, subclasses should implement {@link #comparator comparator()}, {@link #get get()},
022 * {@link #entrySet entrySet()}, {@link #navigableKeySet navigableKeySet()}, and {@link #createSubMap createSubMap()}
023 * to handle reversed and restricted range sub-maps.
024 * </p>
025 *
026 * <p>
027 * For a mutable implementation, subclasses should also implement {@link #put put()}, {@link #remove remove()},
028 * {@link #clear clear()}, and make the {@link #keySet keySet()} and {@link #entrySet entrySet()} iterators mutable.
029 * </p>
030 *
031 * <p>
032 * All overridden methods must be aware of the {@linkplain #bounds range restriction bounds}, if any.
033 * </p>
034 *
035 * @param <K> key type
036 * @param <V> value type
037 */
038public abstract class AbstractNavigableMap<K, V> extends AbstractMap<K, V> implements NavigableMap<K, V> {
039
040    /**
041     * Key range bounds associated with this instance.
042     */
043    protected final Bounds<K> bounds;
044
045    /**
046     * Convenience constructor for the case where there are no lower or upper key bounds.
047     */
048    protected AbstractNavigableMap() {
049        this(new Bounds<K>());
050    }
051
052    /**
053     * Primary constructor.
054     *
055     * @param bounds key range restriction
056     * @throws IllegalArgumentException if {@code bounds} is null
057     */
058    protected AbstractNavigableMap(Bounds<K> bounds) {
059        if (bounds == null)
060            throw new IllegalArgumentException("null bounds");
061        this.bounds = bounds;
062    }
063
064    @Override
065    public boolean isEmpty() {
066        return this.navigableKeySet().isEmpty();
067    }
068
069    @Override
070    public boolean containsKey(Object obj) {
071        return this.navigableKeySet().contains(obj);
072    }
073
074    @Override
075    public K firstKey() {
076        return this.navigableKeySet().iterator().next();
077    }
078
079    @Override
080    public K lastKey() {
081        return this.descendingMap().firstKey();
082    }
083
084    @Override
085    public NavigableSet<K> keySet() {
086        return this.navigableKeySet();
087    }
088
089    @Override
090    public Map.Entry<K, V> lowerEntry(K maxKey) {
091        if (!this.bounds.isWithinLowerBound(this.comparator(), maxKey))
092            return null;
093        final NavigableMap<K, V> subMap = this.bounds.isWithinUpperBound(this.comparator(), maxKey) ?
094          this.headMap(maxKey, false) : this;
095        try {
096            return subMap.lastEntry();
097        } catch (NoSuchElementException e) {
098            return null;
099        }
100    }
101
102    @Override
103    public Map.Entry<K, V> floorEntry(K maxKey) {
104        if (!this.bounds.isWithinLowerBound(this.comparator(), maxKey))
105            return null;
106        final NavigableMap<K, V> subMap = this.bounds.isWithinUpperBound(this.comparator(), maxKey) ?
107          this.headMap(maxKey, true) : this;
108        try {
109            return subMap.lastEntry();
110        } catch (NoSuchElementException e) {
111            return null;
112        }
113    }
114
115    @Override
116    public Map.Entry<K, V> ceilingEntry(K minKey) {
117        if (!this.bounds.isWithinUpperBound(this.comparator(), minKey))
118            return null;
119        final NavigableMap<K, V> subMap = this.bounds.isWithinLowerBound(this.comparator(), minKey) ?
120          this.tailMap(minKey, true) : this;
121        try {
122            return subMap.firstEntry();
123        } catch (NoSuchElementException e) {
124            return null;
125        }
126    }
127
128    @Override
129    public Map.Entry<K, V> higherEntry(K minKey) {
130        if (!this.bounds.isWithinUpperBound(this.comparator(), minKey))
131            return null;
132        final NavigableMap<K, V> subMap = this.bounds.isWithinLowerBound(this.comparator(), minKey) ?
133          this.tailMap(minKey, false) : this;
134        try {
135            return subMap.firstEntry();
136        } catch (NoSuchElementException e) {
137            return null;
138        }
139    }
140
141    @Override
142    public K lowerKey(K maxKey) {
143        final Map.Entry<K, V> entry = this.lowerEntry(maxKey);
144        return entry != null ? entry.getKey() : null;
145    }
146
147    @Override
148    public K floorKey(K maxKey) {
149        final Map.Entry<K, V> entry = this.floorEntry(maxKey);
150        return entry != null ? entry.getKey() : null;
151    }
152
153    @Override
154    public K ceilingKey(K minKey) {
155        final Map.Entry<K, V> entry = this.ceilingEntry(minKey);
156        return entry != null ? entry.getKey() : null;
157    }
158
159    @Override
160    public K higherKey(K minKey) {
161        final Map.Entry<K, V> entry = this.higherEntry(minKey);
162        return entry != null ? entry.getKey() : null;
163    }
164
165    @Override
166    public Map.Entry<K, V> firstEntry() {
167        try {
168            return this.entrySet().iterator().next();
169        } catch (NoSuchElementException e) {
170            return null;
171        }
172    }
173
174    @Override
175    public Map.Entry<K, V> lastEntry() {
176        return this.descendingMap().firstEntry();
177    }
178
179    @Override
180    public Map.Entry<K, V> pollFirstEntry() {
181        final Iterator<Map.Entry<K, V>> i = this.entrySet().iterator();
182        if (!i.hasNext())
183            return null;
184        final Map.Entry<K, V> entry = i.next();
185        i.remove();
186        return entry;
187    }
188
189    @Override
190    public Map.Entry<K, V> pollLastEntry() {
191        return this.descendingMap().pollFirstEntry();
192    }
193
194    @Override
195    public NavigableMap<K, V> descendingMap() {
196        return this.createSubMap(true, this.bounds.reverse());
197    }
198
199    @Override
200    public NavigableSet<K> descendingKeySet() {
201        return this.navigableKeySet().descendingSet();
202    }
203
204    @Override
205    public NavigableMap<K, V> subMap(K minKey, K maxKey) {
206        return this.subMap(minKey, true, maxKey, false);
207    }
208
209    @Override
210    public NavigableMap<K, V> headMap(K maxKey) {
211        return this.headMap(maxKey, false);
212    }
213
214    @Override
215    public NavigableMap<K, V> tailMap(K minKey) {
216        return this.tailMap(minKey, true);
217    }
218
219    @Override
220    public NavigableMap<K, V> headMap(K newMaxKey, boolean inclusive) {
221        final Bounds<K> newBounds = this.bounds.withUpperBound(newMaxKey, BoundType.of(inclusive));
222        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
223            throw new IllegalArgumentException("upper bound is out of range");
224        return this.createSubMap(false, newBounds);
225    }
226
227    @Override
228    public NavigableMap<K, V> tailMap(K newMinKey, boolean inclusive) {
229        final Bounds<K> newBounds = this.bounds.withLowerBound(newMinKey, BoundType.of(inclusive));
230        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
231            throw new IllegalArgumentException("lower bound is out of range");
232        return this.createSubMap(false, newBounds);
233    }
234
235    @Override
236    public NavigableMap<K, V> subMap(K newMinKey, boolean minInclusive, K newMaxKey, boolean maxInclusive) {
237        final Bounds<K> newBounds = new Bounds<>(newMinKey, BoundType.of(minInclusive), newMaxKey, BoundType.of(maxInclusive));
238        if (!this.bounds.isWithinBounds(this.comparator(), newBounds))
239            throw new IllegalArgumentException("bound(s) are out of range");
240        return this.createSubMap(false, newBounds);
241    }
242
243    /**
244     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds.
245     * The {@code newBounds} are consistent with the new ordering (i.e., reversed relative to this instance's ordering if
246     * {@code reverse} is true) and have already been range-checked against {@linkplain #bounds this instance's current bounds}.
247     *
248     * @param reverse whether the new map's ordering should be reversed relative to this instance's ordering
249     * @param newBounds new bounds
250     * @throws IllegalArgumentException if {@code newBounds} is null
251     * @throws IllegalArgumentException if a bound in {@code newBounds} is null and this set does not permit null elements
252     */
253    protected abstract NavigableMap<K, V> createSubMap(boolean reverse, Bounds<K> newBounds);
254}
