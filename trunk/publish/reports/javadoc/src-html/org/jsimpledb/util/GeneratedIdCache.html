008package org.jsimpledb.util;
009
010import java.util.HashMap;
011
012import org.jsimpledb.core.ObjId;
013import org.jsimpledb.core.Transaction;
014
015/**
016 * Cache for randomly generated object IDs based on unique strings.
017 *
018 * <p>
019 * Instances are thread safe.
020 * </p>
021 *
022 * @see XMLObjectSerializer
023 */
024public class GeneratedIdCache {
025
026    private final HashMap<String, ObjId> map = new HashMap<>();
027
028    /**
029     * Generate an object ID for the given object type storage ID and suffix.
030     *
031     * <p>
032     * If an {@link ObjId} has already been created for the specified {@code storageId} and {@code suffix} by this instance,
033     * it will be returned.
034     * Otherwise, a random {@link ObjId} for which no object exists in the specified {@link Transaction} is generated
035     * and returned.
036     * </p>
037     *
038     * @param storageId object type storage ID
039     * @throws IllegalArgumentException if {@code storageId} is invalid
040     * @throws IllegalArgumentException if {@code tx} or {@code suffix} is null
041     */
042    public synchronized ObjId getGeneratedId(Transaction tx, int storageId, String suffix) {
043        if (tx == null)
044            throw new IllegalArgumentException("null tx");
045        if (suffix == null)
046            throw new IllegalArgumentException("null suffix");
047        final String key = "" + storageId + ":" + suffix;
048        ObjId id = this.map.get(key);
049        if (id == null) {
050            id = tx.generateId(storageId);
051            this.map.put(key, id);
052        }
053        return id;
054    }
055
056    /**
057     * Clear this instance.
058     */
059    public synchronized void clear() {
060        this.map.clear();
061    }
062}
