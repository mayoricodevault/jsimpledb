008package org.jsimpledb.util;
009
010import com.google.common.base.Function;
011import com.google.common.collect.Lists;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.annotation.Annotation;
015import java.lang.reflect.Method;
016import java.lang.reflect.Modifier;
017import java.lang.reflect.Type;
018import java.util.Arrays;
019import java.util.List;
020import java.util.Set;
021
022import org.dellroad.stuff.java.MethodAnnotationScanner;
023import org.jsimpledb.JClass;
024
025/**
026 * Support superclass for Java model class annotation scanners.
027 */
028public abstract class AnnotationScanner<T, A extends Annotation> extends MethodAnnotationScanner<T, A> {
029
030    /**
031     * The associated Java model class, if any.
032     */
033    protected final JClass<T> jclass;
034
035    /**
036     * Constructor for when there is an associated {@link JClass}.
037     *
038     * @param jclass Java model class
039     * @param annotationType annotation to scan for
040     */
041    @SuppressWarnings("unchecked")
042    protected AnnotationScanner(JClass<T> jclass, Class<A> annotationType) {
043        super((Class<T>)jclass.getTypeToken().getRawType(), annotationType);
044        this.jclass = jclass;
045    }
046
047    /**
048     * Constructor for when there is no associated {@link JClass}.
049     *
050     * @param type Java type to scan
051     * @param annotationType annotation to scan for
052     */
053    @SuppressWarnings("unchecked")
054    protected AnnotationScanner(Class<T> type, Class<A> annotationType) {
055        super(type, annotationType);
056        this.jclass = null;
057    }
058
059// TODO: remove the next two methods after upgrading to newer dellroad-stuff version
060
061    @Override
062    protected void findMethodInfos(Class<?> klass, Set<MethodInfo> set) {
063
064        // Stop at null
065        if (klass == null)
066            return;
067
068        // Search methods
069        for (Method method : klass.getDeclaredMethods()) {
070            final A annotation = this.getAnnotation(method);
071            if (annotation == null || !this.includeMethod(method, annotation))
072                continue;
073            final MethodInfo methodInfo = this.createMethodInfo(method, annotation);
074            if (methodInfo != null)
075                set.add(methodInfo);
076        }
077
078        // Recurse on interfaces
079        for (Class<?> iface : klass.getInterfaces())
080            this.findMethodInfos(iface, set);
081
082        // Recurse on superclass
083        this.findMethodInfos(klass.getSuperclass(), set);
084    }
085
086    /**
087     * Get the annotation on the given method.
088     *
089     * <p>
090     * The implementation in {@link MethodAnnotationScanner} just invokes {@code method.getAnnotation()}.
091     * Subclasses may override to automatically synthesize annotations, etc.
092     * </p>
093     *
094     * @param method method in question
095     * @return annotation for {@code method}, or null if there is none
096     */
097    protected A getAnnotation(Method method) {
098        return method.getAnnotation(this.annotationType);
099    }
100
101    /**
102     * Get a simple description of the annotation being scanned for.
103     */
104    public String getAnnotationDescription() {
105        return "@" + this.annotationType.getSimpleName();
106    }
107
108    /**
109     * Verify method is not static.
110     *
111     * @throws IllegalArgumentException if method is static
112     */
113    protected void checkNotStatic(Method method) {
114        if ((method.getModifiers() & Modifier.STATIC) != 0)
115            throw new IllegalArgumentException(this.getErrorPrefix(method) + "annotation is not supported on static methods");
116    }
117
118    /**
119     * Verify method return type.
120     *
121     * @throws IllegalArgumentException if has an invalid return type
122     */
123    protected void checkReturnType(Method method, List<TypeToken<?>> expecteds) {
124        final TypeToken<?> actual = TypeToken.of(method.getGenericReturnType());
125        for (TypeToken<?> expected : expecteds) {
126            if (actual.equals(expected))
127                return;
128        }
129        throw new IllegalArgumentException(this.getErrorPrefix(method) + "method is required to return "
130          + (expecteds.size() != 1 ? "one of " + expecteds : expecteds.get(0)) + " but instead returns " + actual);
131    }
132
133    /**
134     * Verify method return type.
135     *
136     * @throws IllegalArgumentException if has an invalid return type
137     */
138    protected void checkReturnType(Method method, Class<?>... expecteds) {
139        final Class<?> actual = method.getReturnType();
140        for (Class<?> expected : expecteds) {
141            if (actual.equals(expected))
142                return;
143        }
144        throw new IllegalArgumentException(this.getErrorPrefix(method) + "method is required to return "
145          + (expecteds.length != 1 ? "one of " + Arrays.asList(expecteds) : expecteds[0]) + " but instead returns " + actual);
146    }
147
148    /**
149     * Verify method parameter type(s).
150     *
151     * @throws IllegalArgumentException if has an invalid parameter type(s)
152     */
153    protected void checkParameterTypes(Method method, List<TypeToken<?>> expected) {
154        final List<TypeToken<?>> actual = this.getParameterTypeTokens(method);
155        if (!actual.equals(expected)) {
156            throw new IllegalArgumentException(this.getErrorPrefix(method) + "method is required to take "
157              + (expected.isEmpty() ? "zero parameters" : expected.size() + " parameter(s) of type " + expected));
158        }
159    }
160
161    /**
162     * Verify method parameter type(s).
163     *
164     * @throws IllegalArgumentException if has an invalid parameter type(s)
165     */
166    protected void checkParameterTypes(Method method, TypeToken<?>... expected) {
167        this.checkParameterTypes(method, Arrays.asList(expected));
168    }
169
170    /**
171     * Verify a specific method parameter's type.
172     *
173     * @throws IllegalArgumentException if parameter type does not match
174     */
175    protected void checkParameterType(Method method, int index, List<TypeToken<?>> choices) {
176        final List<TypeToken<?>> actuals = this.getParameterTypeTokens(method);
177        if (actuals.size() <= index || !choices.contains(actuals.get(index))) {
178            throw new IllegalArgumentException(this.getErrorPrefix(method) + "method parameter #" + (index + 1)
179              + " is required to have type " + (choices.size() != 1 ? "one of " + choices : choices.get(0)));
180        }
181    }
182
183    /**
184     * Verify a specific method parameter's type.
185     *
186     * @throws IllegalArgumentException if parameter type does not match
187     */
188    protected void checkParameterType(Method method, int index, TypeToken<?>... choices) {
189        this.checkParameterType(method, index, Arrays.asList(choices));
190    }
191
192    /**
193     * Verify method takes a single parameter of the specified type.
194     *
195     * @throws IllegalArgumentException if parameter type does not match
196     */
197    protected void checkSingleParameterType(Method method, List<TypeToken<?>> choices) {
198        final List<TypeToken<?>> actuals = this.getParameterTypeTokens(method);
199        if (actuals.size() != 1 || !choices.contains(actuals.get(0))) {
200            throw new IllegalArgumentException(this.getErrorPrefix(method) + "method is required to take a single parameter"
201              + " with type " + (choices.size() != 1 ? "one of " + choices : choices.get(0)));
202        }
203    }
204
205    /**
206     * Get method parameter types as {@link TypeToken}s.
207     */
208    protected List<TypeToken<?>> getParameterTypeTokens(Method method) {
209        return Lists.transform(Arrays.asList(method.getGenericParameterTypes()), new Function<Type, TypeToken<?>>() {
210            @Override
211            public TypeToken<?> apply(Type type) {
212                return TypeToken.of(type);
213            }
214        });
215    }
216
217    /**
218     * Get "invalid annotation" error message prefix that describes the annotation on the specified method.
219     */
220    protected String getErrorPrefix(Method method) {
221        return "invalid " + this.getAnnotationDescription() + " annotation on method " + method
222          + " for type " + this.jclass.getTypeToken() + ": ";
223    }
224}
