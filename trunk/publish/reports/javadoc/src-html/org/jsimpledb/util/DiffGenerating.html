008package org.jsimpledb.util;
009
010/**
011 * Implemented by classes that can generate a set of differences between themselves and other instances.
012 *
013 * @param <T> the type being compared
014 */
015public interface DiffGenerating<T> {
016
017    /**
018     * Detect the differences of this instance when compared to the given instance.
019     *
020     * @param other other instance
021     * @return differences; will be empty if there are none detected
022     * @throws IllegalArgumentException if {@code other} is null
023     */
024    Diffs differencesFrom(T other);
025}
