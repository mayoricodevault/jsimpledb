008package org.jsimpledb.util;
009
010import com.google.common.collect.Iterators;
011import com.google.common.collect.PeekingIterator;
012
013import java.util.Comparator;
014import java.util.Iterator;
015
016/**
017 * {@link Iterator} implementation that provides a read-only view of an inner {@link Iterator}
018 * in which consecutive duplicate values are eliminated.
019 */
020public class UniqueIterator<E> implements Iterator<E> {
021
022    private final PeekingIterator<E> iterator;
023    private final Comparator<? super E> comparator;
024
025    /**
026     * Constructor. Object equality using {@link Object#equals} is used to detect duplicates.
027     *
028     * @param iterator wrapped {@link Iterator}
029     */
030    public UniqueIterator(Iterator<? extends E> iterator) {
031        this.iterator = Iterators.peekingIterator(iterator);
032        this.comparator = null;
033    }
034
035    /**
036     * Constructor. A zero result from the provided {@link Comparator} is used to detect duplicates.
037     *
038     * @param iterator wrapped {@link Iterator}
039     * @param comparator used to compare consecutive values
040     * @throws IllegalArgumentException if {@code comparator} is null
041     */
042    public UniqueIterator(Iterator<? extends E> iterator, Comparator<? super E> comparator) {
043        if (comparator == null)
044            throw new IllegalArgumentException("null comparator");
045        this.iterator = Iterators.peekingIterator(iterator);
046        this.comparator = comparator;
047    }
048
049    @Override
050    public boolean hasNext() {
051        return this.iterator.hasNext();
052    }
053
054    @Override
055    public E next() {
056        final E next = this.iterator.next();
057        while (this.iterator.hasNext()) {
058            final E peek = this.iterator.peek();
059            if (this.comparator != null ? this.comparator.compare(next, peek) != 0 :
060              next != null ? !next.equals(peek) : peek != null)
061                break;
062            this.iterator.next();
063        }
064        return next;
065    }
066
067    @Override
068    public void remove() {
069        throw new UnsupportedOperationException();
070    }
071}
