020/**
021 * A map with {@link ObjId} keys.
022 *
023 * <p>
024 * This implementation is space optimized for the 64-bits of information contained in an {@link ObjId}.
025 * Instances do not accept null keys and are not thread safe.
026 * </p>
027 */
028public class ObjIdMap<V> extends AbstractMap<ObjId, V> implements Cloneable {
029
030    // Algorithm described here: http://en.wikipedia.org/wiki/Open_addressing
031
032    private static final float EXPAND_THRESHOLD = 0.70f;    // expand array when > 70% full
033    private static final float SHRINK_THRESHOLD = 0.25f;    // shrink array when < 25% full
034
035    private static final int MIN_LOG2_LENGTH = 4;           // minimum array length = 16 slots
036    private static final int MAX_LOG2_LENGTH = 30;          // maximum array length = 1 billion slots
037
038    private long[] keys;                                    // has length always a power of 2
039    private V[] values;                                     // will be null if we are being used to implement ObjIdSet
040    private int size;                                       // the number of entries in the map
041    private int log2len;                                    // log2 of keys.length and values.length (if not null)
042    private int upperSizeLimit;                             // size threshold when to grow array
043    private int lowerSizeLimit;                             // size threshold when to shrink array
044    private int numHashShifts;                              // used by hash() function
045    private volatile int modcount;
046
047// Constructors
048
049    /**
050     * Constructs an empty instance.
051     */
052    public ObjIdMap() {
053        this(0, true);
054    }
055
056    /**
057     * Constructs an instance with the given initial capacity.
058     *
059     * @param capacity initial capacity
060     * @throws IllegalArgumentException if {@code capacity} is negative
061     */
062    public ObjIdMap(int capacity) {
063        this(capacity, true);
064    }
065
066    /**
067     * Constructs an instance initialized from the given map.
068     *
069     * @param map initial contents for this instance
070     * @throws NullPointerException if {@code map} is null
071     * @throws IllegalArgumentException if {@code map} contains a null key
072     */
073    public ObjIdMap(Map<ObjId, ? extends V> map) {
074        this(map.size(), true);
075        for (Map.Entry<ObjId, ? extends V> entry : map.entrySet())
076            this.put(entry.getKey(), entry.getValue());
077    }
078
079    // Internal constructor
080    ObjIdMap(int capacity, boolean withValues) {
081        if (capacity < 0)
082            throw new IllegalArgumentException("capacity < 0");
083        capacity &= 0x3fffffff;                                                 // avoid integer overflow from large values
084        capacity = (int)(capacity / EXPAND_THRESHOLD);                          // increase to account for overhead
085        capacity = Math.max(1, capacity);                                       // avoid zero, on which the next line fails
086        this.log2len = 32 - Integer.numberOfLeadingZeros(capacity - 1);         // round up to next power of 2
087        this.log2len = Math.max(MIN_LOG2_LENGTH, this.log2len);                 // clip to bounds
088        this.log2len = Math.min(MAX_LOG2_LENGTH, this.log2len);
089        this.createArrays(withValues);
090    }

101        return this.size == 0;

104    @Override
105    public boolean containsKey(Object obj) {
106
107        // Check type
108        if (!(obj instanceof ObjId))
109            return false;
110        final long value = ((ObjId)obj).asLong();
111        assert value != 0;
112
113        // Check slot for value
114        final int slot = this.findSlot(value);
115        if (this.keys[slot] == value)
116            return true;
117        assert this.keys[slot] == 0;
118        return false;
119    }
120
121    @Override
122    public V get(Object obj) {
123
124        // Check type
125        if (!(obj instanceof ObjId))
126            return null;
127        final long value = ((ObjId)obj).asLong();
128        assert value != 0;
129
130        // Check slot for value
131        final int slot = this.findSlot(value);
132        if (this.keys[slot] == value)
133            return this.values != null ? this.values[slot] : null;
134        assert this.keys[slot] == 0;
135        return null;
136    }
137
138    @Override
139    public V put(ObjId id, V value) {
140        if (id == null)
141            throw new IllegalArgumentException("illegal null key");
142        final long key = id.asLong();
143        assert key != 0;
144        return this.insert(key, value);
145    }
146
147    @Override
148    public V remove(Object obj) {
149        if (!(obj instanceof ObjId))
150            return null;
151        final long key = ((ObjId)obj).asLong();
152        assert key != 0;
153        return this.exsert(key);
154    }
155
156    @Override
157    public void clear() {
158        this.log2len = MIN_LOG2_LENGTH;
159        this.createArrays(this.values != null);
160        this.size = 0;
161        this.modcount++;
162    }

178        final StringBuilder buf = new StringBuilder();

196    public ObjIdMap<V> clone() {
197        final ObjIdMap<V> clone;
198        try {
199            clone = (ObjIdMap<V>)super.clone();
200        } catch (CloneNotSupportedException e) {
201            throw new RuntimeException(e);
202        }
203        clone.keys = clone.keys.clone();
204        if (clone.values != null)
205            clone.values = clone.values.clone();
206        return clone;
207    }

225    private V insert(long key, V value) {
226
227        // Find slot for key
228        assert key != 0;
229        final int slot = this.findSlot(key);
230
231        // Key already exists? Just replace value
232        if (this.keys[slot] == key) {
233            if (this.values == null)
234                return null;
235            final V prev = this.values[slot];
236            this.values[slot] = value;
237            return prev;
238        }
239
240        // Insert new key/value pair
241        assert this.keys[slot] == 0;
242        this.keys[slot] = key;
243        if (this.values != null) {
244            assert this.values[slot] == null;
245            this.values[slot] = value;
246        }
247
248        // Expand if necessary
249        if (++this.size > this.upperSizeLimit && this.log2len < MAX_LOG2_LENGTH) {
250            this.log2len++;
251            this.resize();
252        }
253        this.modcount++;
254        return null;
255    }
256
257    private V exsert(long key) {
258
259        // Find slot for key
260        final int slot = this.findSlot(key);
261        if (this.keys[slot] == 0) {
262            assert this.values == null || this.values[slot] == null;
263            return null;
264        }
265        assert this.keys[slot] == key;
266
267        // Remove key
268        return this.exsert(slot);
269    }
270
271    private V exsert(final int slot) {
272
273        // Sanity check
274        assert this.keys[slot] != 0;
275        final V ovalue = this.values != null ? this.values[slot] : null;
276
277        // Remove key/value pair and fixup subsequent entries
278        int i = slot;                                                   // i points to the new empty slot
279        int j = slot;                                                   // j points to the next slot to fixup
280loop:   while (true) {
281            this.keys[i] = 0;
282            if (this.values != null)
283                this.values[i] = null;
284            long jkey;
285            V jvalue;
286            while (true) {
287                j = (j + 1) & (this.keys.length - 1);
288                jkey = this.keys[j];
289                if (jkey == 0)                                          // end of hash chain, no more fixups required
290                    break loop;
291                jvalue = this.values != null ? this.values[j] : null;   // get corresponding value
292                final int k = this.hash(jkey);                          // find where jkey's hash chain started
293                if (i <= j ? (i < k && k <= j) : (i < k || k <= j))     // jkey is between i and j, so it's not cut off
294                    continue;
295                break;                                                  // jkey is cut off from its hash chain, need to fix
296            }
297            this.keys[i] = jkey;                                        // move jkey back into its hash chain
298            if (this.values != null)
299                this.values[i] = jvalue;
300            i = j;                                                      // restart fixups at jkey's old location
301        }
302
303        // Shrink if necessary
304        if (--this.size < this.lowerSizeLimit && this.log2len > MIN_LOG2_LENGTH) {
305            this.log2len--;
306            this.resize();
307        }
308        this.modcount++;
309        return ovalue;
310    }
311
312    private int findSlot(long value) {
313        assert value != 0;
314        int slot = this.hash(value);
315        while (true) {
316            final long existing = this.keys[slot];
317            if (existing == 0 || existing == value)
318                return slot;
319            slot = (slot + 1) & (this.keys.length - 1);
320        }
321    }
322
323    private int hash(long value) {
324        final int shift = this.log2len;
325        int hash = (int)value;
326        for (int i = 0; i < this.numHashShifts; i++) {
327            value >>>= shift;
328            hash ^= (int)value;
329        }
330        return hash & (this.keys.length - 1);
331    }
332
333    private void resize() {
334
335        // Grab a copy of old arrays and create new ones
336        final long[] oldKeys = this.keys;
337        final V[] oldValues = this.values;
338        assert oldValues == null || oldValues.length == oldKeys.length;
339        this.createArrays(oldValues != null);
340
341        // Rehash key/value pairs from old array into new array
342        for (int oldSlot = 0; oldSlot < oldKeys.length; oldSlot++) {
343            final long key = oldKeys[oldSlot];
344            if (key == 0) {
345                assert oldValues == null || oldValues[oldSlot] == null;
346                continue;
347            }
348            final int newSlot = this.findSlot(key);
349            assert this.keys[newSlot] == 0;
350            this.keys[newSlot] = key;
351            if (this.values != null) {
352                assert this.values[newSlot] == null;
353                this.values[newSlot] = oldValues[oldSlot];
354            }
355        }
356    }
357
358    @SuppressWarnings("unchecked")
359    private void createArrays(boolean withValues) {
360        assert this.log2len >= MIN_LOG2_LENGTH;
361        assert this.log2len <= MAX_LOG2_LENGTH;
362        final int arrayLength = 1 << this.log2len;
363        this.lowerSizeLimit = this.log2len > MIN_LOG2_LENGTH ? (int)(SHRINK_THRESHOLD * arrayLength) : 0;
364        this.upperSizeLimit = this.log2len < MAX_LOG2_LENGTH ? (int)(EXPAND_THRESHOLD * arrayLength) : arrayLength;
365        this.numHashShifts = (64 + (this.log2len - 1)) / this.log2len;
366        this.numHashShifts = Math.min(12, this.numHashShifts);
367        this.keys = new long[arrayLength];
368        if (withValues)
369            this.values = (V[])new Object[arrayLength];
370    }

374    class EntrySet extends AbstractSet<Map.Entry<ObjId, V>> {
375
376        @Override
377        public Iterator<Map.Entry<ObjId, V>> iterator() {
378            return new EntrySetIterator();
379        }
380
381        @Override
382        public int size() {
383            return ObjIdMap.this.size;
384        }
385
386        @Override
387        public boolean contains(Object obj) {
388            if (!(obj instanceof Map.Entry))
389                return false;
390            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
391            final Object key = entry.getKey();
392            final V actualValue = ObjIdMap.this.get(key);
393            if (actualValue == null && !ObjIdMap.this.containsKey(key))
394                return false;
395            return entry.equals(new AbstractMap.SimpleEntry<ObjId, V>((ObjId)key, actualValue));
396        }
397
398        @Override
399        public boolean remove(Object obj) {
400            if (!(obj instanceof Map.Entry))
401                return false;
402            final Map.Entry<?, ?> entry = (Map.Entry<?, ?>)obj;
403            final Object key = entry.getKey();
404            final V actualValue = ObjIdMap.this.get(key);
405            if (actualValue == null && !ObjIdMap.this.containsKey(key))
406                return false;
407            if (actualValue != null ? actualValue.equals(entry.getValue()) : entry.getValue() == null) {
408                ObjIdMap.this.remove(key);
409                return true;
410            }
411            return false;
412        }
413
414        @Override
415        public void clear() {
416            ObjIdMap.this.clear();
417        }
418
419        // This works because ObjId.hashCode() == ObjId.asLong().hashCode()
420        @Override
421        public int hashCode() {
422            final long[] keyArray = ObjIdMap.this.keys;
423            final V[] valueArray = ObjIdMap.this.values;
424            int hash = 0;
425            for (int i = 0; i < keyArray.length; i++) {
426                final long key = keyArray[i];
427                if (key != 0) {
428                    int entryHash = (int)(key >>> 32) ^ (int)key;
429                    if (valueArray != null) {
430                        final V value = valueArray[i];
431                        if (value != null)
432                            entryHash ^= value.hashCode();
433                    }
434                    hash += entryHash;
435                }
436            }
437            return hash;
438        }
439    }

443    class EntrySetIterator implements Iterator<Map.Entry<ObjId, V>> {
444
445        private int modcount = ObjIdMap.this.modcount;
446        private int removeSlot = -1;
447        private int nextSlot;
448
449        @Override
450        public boolean hasNext() {
451            return this.findNext(false) != -1;
452        }
453
454        @Override
455        public Entry next() {
456            final int slot = this.findNext(true);
457            if (slot == -1)
458                throw new NoSuchElementException();
459            final long key = ObjIdMap.this.keys[slot];
460            assert key != 0;
461            this.removeSlot = slot;
462            return new Entry(slot);
463        }
464
465        @Override
466        public void remove() {
467            if (this.removeSlot == -1)
468                throw new IllegalStateException();
469            if (this.modcount != ObjIdMap.this.modcount)
470                throw new ConcurrentModificationException();
471            ObjIdMap.this.exsert(this.removeSlot);
472            this.removeSlot = -1;
473            this.modcount++;                            // keep synchronized with ObjIdMap.this.modcount
474        }
475
476        private int findNext(boolean advance) {
477            if (this.modcount != ObjIdMap.this.modcount)
478                throw new ConcurrentModificationException();
479            for (int slot = this.nextSlot; slot < ObjIdMap.this.keys.length; slot++) {
480                if (ObjIdMap.this.keys[slot] == 0)
481                    continue;
482                this.nextSlot = advance ? slot + 1 : slot;
483                return slot;
484            }
485            return -1;
486        }
487    }

491    @SuppressWarnings("serial")
492    class Entry extends AbstractMap.SimpleEntry<ObjId, V> {
493
494        private final int modcount = ObjIdMap.this.modcount;
495        private final int slot;
496
497        public Entry(int slot) {
498            super(new ObjId(ObjIdMap.this.keys[slot]), ObjIdMap.this.values != null ? ObjIdMap.this.values[slot] : null);
499            this.slot = slot;
500        }
501
502        @Override
503        public V setValue(V value) {
504            if (ObjIdMap.this.modcount != this.modcount)
505                throw new ConcurrentModificationException();
506            if (ObjIdMap.this.values != null)
507                ObjIdMap.this.values[this.slot] = value;
508            return super.setValue(value);
509        }
510    }
