008package org.jsimpledb.util;
009
010import java.util.Arrays;
011
012import org.dellroad.stuff.string.ByteArrayEncoder;
013
014/**
015 * Encodes {@code long} values to/from binary, preserving sort order, and such that the length of the
016 * encoding is optimized for values near zero.
017 *
018 * <p>
019 * Encoded values are guaranteed to not start with {@code 0x00} or {@code 0xff}.
020 * </p>
021 */
022public final class LongEncoder {
023
024    /**
025     * Maximum possible length of an encoded value.
026     */
027    public static final int MAX_ENCODED_LENGTH = 9;
028
029    /**
030     * Minimum value for the first byte of a single byte encoded value.
031     * Lower values indicate a multiple byte encoded negative value.
032     */
033    public static final int MIN_SINGLE_BYTE_ENCODED = 0x09;        // values 0x01 ... 0x08 prefix negative values
034
035    /**
036     * Maximum value for the first byte of a single byte encoded value.
037     * Higher values indicate a multiple byte encoded positive value.
038     */
039    public static final int MAX_SINGLE_BYTE_ENCODED = 0xf6;        // values 0xf7 ... 0xfe prefix positive values
040
041    /**
042     * Adjustment applied to single byte encoded values before encoding.
043     */
044    public static final int ZERO_ADJUST = 127;                     // single byte value that represents zero
045
046    /**
047     * Minimum value that can be encoded as a single byte.
048     */
049    public static final int MIN_SINGLE_BYTE_VALUE = MIN_SINGLE_BYTE_ENCODED - ZERO_ADJUST;          // -118
050
051    /**
052     * Maximum value that can be encoded as a single byte.
053     */
054    public static final int MAX_SINGLE_BYTE_VALUE = MAX_SINGLE_BYTE_ENCODED - ZERO_ADJUST;          // 119
055
056    /**
057     * Adjustment applied to multi-byte encoded negative values before encoding.
058     */
059    public static final int NEGATIVE_ADJUST = -MIN_SINGLE_BYTE_VALUE;                               // 118
060
061    /**
062     * Adjustment applied to multi-byte encoded positive values before encoding.
063     */
064    public static final int POSITIVE_ADJUST = -(MAX_SINGLE_BYTE_VALUE + 1);                         // -120
065
066    // Cutoff values at which the encoded length changes (this field is package private for testing purposes)
067    static final long[] CUTOFF_VALUES = new long[] {
068        0xff00000000000000L - NEGATIVE_ADJUST,      // [ 0] requires 8 bytes
069        0xffff000000000000L - NEGATIVE_ADJUST,      // [ 1] requires 7 bytes
070        0xffffff0000000000L - NEGATIVE_ADJUST,      // [ 2] requires 6 bytes
071        0xffffffff00000000L - NEGATIVE_ADJUST,      // [ 3] requires 5 bytes
072        0xffffffffff000000L - NEGATIVE_ADJUST,      // [ 4] requires 4 bytes
073        0xffffffffffff0000L - NEGATIVE_ADJUST,      // [ 5] requires 3 bytes
074        0xffffffffffffff00L - NEGATIVE_ADJUST,      // [ 6] requires 2 bytes
075        MIN_SINGLE_BYTE_VALUE,                      // [ 7] requires 1 byte
076        MAX_SINGLE_BYTE_VALUE + 1,                  // [ 8] requires 2 bytes
077        0x0000000000000100L - POSITIVE_ADJUST,      // [ 9] requires 3 bytes
078        0x0000000000010000L - POSITIVE_ADJUST,      // [10] requires 4 bytes
079        0x0000000001000000L - POSITIVE_ADJUST,      // [11] requires 5 bytes
080        0x0000000100000000L - POSITIVE_ADJUST,      // [12] requires 6 bytes
081        0x0000010000000000L - POSITIVE_ADJUST,      // [13] requires 7 bytes
082        0x0001000000000000L - POSITIVE_ADJUST,      // [14] requires 8 bytes
083        0x0100000000000000L - POSITIVE_ADJUST,      // [15] requires 9 bytes
084    };
085
086    private LongEncoder() {
087    }
088
089    /**
090     * Encode the given value to the output.
091     *
092     * @param writer destination for the encoded value
093     * @param value value to encode
094     */
095    public static void write(ByteWriter writer, long value) {
096        writer.makeRoom(MAX_ENCODED_LENGTH);
097        writer.len += LongEncoder.encode(value, writer.buf, writer.len);
098    }
099
100    /**
101     * Read and decode a value from the input.
102     *
103     * @param reader input holding an encoded value
104     * @return the decoded value
105     * @throws IllegalArgumentException if the first byte is {@code 0xff}
106     */
107    public static long read(ByteReader reader) {
108        int first = reader.readByte();
109        if (first < MIN_SINGLE_BYTE_ENCODED) {
110            if (first == 0x00)
111                throw new IllegalArgumentException("invalid encoded value starting with 0x00");
112            long value = ~0L;
113            while (first++ < MIN_SINGLE_BYTE_ENCODED)
114                value = (value << 8) | reader.readByte();
115            return value - NEGATIVE_ADJUST;
116        }
117        if (first > MAX_SINGLE_BYTE_ENCODED) {
118            if (first == 0xff)
119                throw new IllegalArgumentException("invalid encoded value starting with 0xff");
120            long value = 0L;
121            while (first-- > MAX_SINGLE_BYTE_ENCODED)
122                value = (value << 8) | reader.readByte();
123            return value - POSITIVE_ADJUST;
124        }
125        return (byte)(first - ZERO_ADJUST);
126    }
127
128    /**
129     * Skip a value from the input.
130     *
131     * @param reader input holding an encoded value
132     * @throws IllegalArgumentException if the first byte is {@code 0xff}
133     */
134    public static void skip(ByteReader reader) {
135        final int first = reader.readByte();
136        if (first == 0x00 || first == 0xff)
137            throw new IllegalArgumentException("invalid encoded value starting with 0x" + Integer.toHexString(first));
138        reader.skip(LongEncoder.decodeLength(first) - 1);
139    }
140
141    /**
142     * Determine the length (in bytes) of an encoded value based on the first byte.
143     *
144     * @param first first byte of encoded value (in lower eight bits; other bits are ignored)
145     * @return the length of the encoded value (including {@code first})
146     * @throws IllegalArgumentException if the first byte is {@code 0xff}
147     */
148    public static int decodeLength(int first) {
149        first &= 0xff;
150        if (first == 0x00 || first == 0xff)
151            throw new IllegalArgumentException("invalid encoded value starting with 0x" + Integer.toHexString(first));
152        if (first < MIN_SINGLE_BYTE_ENCODED)
153            return 1 + MIN_SINGLE_BYTE_ENCODED - first;
154        if (first > MAX_SINGLE_BYTE_ENCODED)
155            return 1 + first - MAX_SINGLE_BYTE_ENCODED;
156        return 1;
157    }
158
159    /**
160     * Determine the length (in bytes) of the encoded value.
161     *
162     * @param value value to encode
163     * @return the length of the encoded value, a value between one and {@link #MAX_ENCODED_LENGTH}
164     */
165    public static int encodeLength(long value) {
166        int index = Arrays.binarySearch(CUTOFF_VALUES, value);
167        if (index < 0)
168            index = ~index - 1;
169        return index < 8 ? 8 - index : index - 6;
170    }
171
172    /**
173     * Encode the given value and write the encoded bytes into the given buffer.
174     *
175     * @param value value to encode
176     * @param buf output buffer
177     * @param off starting offset into output buffer
178     * @return the number of encoded bytes written
179     * @throws ArrayIndexOutOfBoundsException if {@code off} is negative or the encoded value exceeds the given buffer
180     */
181    private static int encode(long value, byte[] buf, int off) {
182        int len = 1;
183        if (value < MIN_SINGLE_BYTE_VALUE) {
184            value += NEGATIVE_ADJUST;
185            long mask = 0x00ffffffffffffffL;
186            for (int shift = 56; shift != 0; shift -= 8, mask >>= 8) {
187                if ((value | mask) != ~0L)
188                    buf[off + len++] = (byte)(value >> shift);
189            }
190            buf[off] = (byte)(MIN_SINGLE_BYTE_ENCODED - len);
191        } else if (value > MAX_SINGLE_BYTE_VALUE) {
192            value += POSITIVE_ADJUST;
193            long mask = 0xff00000000000000L;
194            for (int shift = 56; shift != 0; shift -= 8, mask >>= 8) {
195                if ((value & mask) != 0L)
196                    buf[off + len++] = (byte)(value >> shift);
197            }
198            buf[off] = (byte)(MAX_SINGLE_BYTE_ENCODED + len);
199        } else {
200            buf[off] = (byte)(value + ZERO_ADJUST);
201            return 1;
202        }
203        buf[off + len++] = (byte)value;
204        return len;
205    }
206
207    /**
208     * Test routine.
209     */
210    static void main(String[] args) throws Exception {
211        for (String arg : args) {
212            byte[] bytes = null;
213            try {
214                bytes = ByteArrayEncoder.decode(arg);
215            } catch (IllegalArgumentException e) {
216                if (arg.startsWith("0x"))
217                    bytes = ByteArrayEncoder.decode(arg.substring(2));
218            }
219            if (bytes != null) {
220                final long value = LongEncoder.read(new ByteReader(bytes));
221                System.out.println("0x" + ByteArrayEncoder.encode(bytes)
222                  + " decodes to " + value + " (" + String.format("0x%016x", value) + ")");
223            }
224            Long value = null;
225            try {
226                value = Long.parseLong(arg);
227            } catch (IllegalArgumentException e) {
228                // ignore
229            }
230            if (value != null) {
231                final ByteWriter writer = new ByteWriter();
232                LongEncoder.write(writer, value);
233                System.out.println(value + " (" + String.format("0x%016x", value)
234                  + ") encodes to " + ByteArrayEncoder.encode(writer.getBytes()));
235            }
236        }
237    }
238}
