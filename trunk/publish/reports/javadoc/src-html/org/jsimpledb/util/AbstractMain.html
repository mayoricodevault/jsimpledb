061    /**
062     * Parse command line options.
063     *
064     * @return -1 to proceed, otherwise process exit value
065     */
066    public int parseOptions(ArrayDeque<String> params) {
067
068        // Parse options
069        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
070            final String option = params.removeFirst();
071            if (option.equals("-h") || option.equals("--help")) {
072                this.usageMessage();
073                return 0;
074            } else if (option.equals("-ro") || option.equals("--read-only"))
075                this.readOnly = true;
076            else if (option.equals("-cp") || option.equals("--classpath")) {
077                if (params.isEmpty())
078                    this.usageError();
079                this.appendClasspath(params.removeFirst());
080            } else if (option.equals("-v") || option.equals("--verbose"))
081                this.verbose = true;
082            else if (option.equals("--version")) {
083                if (params.isEmpty())
084                    this.usageError();
085                final String vstring = params.removeFirst();
086                try {
087                    this.schemaVersion = Integer.parseInt(vstring);
088                    if (this.schemaVersion < 0)
089                        throw new IllegalArgumentException("schema version is negative");
090                } catch (Exception e) {
091                    System.err.println(this.getName() + ": invalid schema version `" + vstring + "': " + e.getMessage());
092                    this.usageError();
093                }
094            } else if (option.equals("--schema-pkg")) {
095                if (params.isEmpty())
096                    this.usageError();
097                this.scanSchemaClasses(params.removeFirst());
098                this.allowAutoDemo = false;
099            } else if (option.equals("--types-pkg")) {
100                if (params.isEmpty())
101                    this.usageError();
102                this.scanTypeClasses(params.removeFirst());
103            } else if (option.equals("-p") || option.equals("--scan-pkg")) {
104                if (params.isEmpty())
105                    this.usageError();
106                final String packageName = params.removeFirst();
107                this.scanSchemaClasses(packageName);
108                this.scanTypeClasses(packageName);
109                this.allowAutoDemo = false;
110            } else if (option.equals("--new-schema"))
111                this.allowNewSchema = true;
112            else if (option.equals("--mem")) {
113                this.kvType = KV_MEM;
114                this.allowAutoDemo = false;
115            } else if (option.equals("--prefix")) {
116                if (params.isEmpty())
117                    this.usageError();
118                final String value = params.removeFirst();
119                try {
120                    this.keyPrefix = ByteUtil.parse(value);
121                } catch (IllegalArgumentException e) {
122                    this.keyPrefix = value.getBytes(Charset.forName("UTF-8"));
123                }
124                if (this.keyPrefix.length > 0)
125                    this.allowAutoDemo = false;
126            } else if (option.equals("--fdb")) {
127                if (params.isEmpty())
128                    this.usageError();
129                this.kvType = KV_FDB;
130                this.fdbClusterFile = params.removeFirst();
131                if (!new File(this.fdbClusterFile).exists())
132                    System.err.println(this.getName() + ": file `" + this.fdbClusterFile + "' does not exist");
133                this.allowAutoDemo = false;
134            } else if (option.equals("--xml")) {
135                if (params.isEmpty())
136                    this.usageError();
137                this.kvType = KV_XML;
138                this.xmlFile = new File(params.removeFirst());
139                this.allowAutoDemo = false;
140            } else if (option.equals("--"))
141                break;
142            else if (!this.parseOption(option, params)) {
143                System.err.println(this.getName() + ": unknown option `" + option + "'");
144                this.usageError();
145            }
146        }
147        if (this.kvType != KV_FDB && this.keyPrefix != null) {
148            System.err.println(this.getName() + ": option `--prefix' is only valid in combination with `--fdb'");
149            this.usageError();
150        }
151
152        // Automatically go into demo mode if appropriate
153        if (this.allowAutoDemo && DEMO_XML_FILE.exists() && DEMO_SUBDIR.exists())
154            this.configureDemoMode();
155
156        // Done
157        return -1;
158    }

192    protected void configureDemoMode() {
193
194        // Configure database
195        System.err.println(this.getName() + ": auto-configuring use of demo database `" + DEMO_XML_FILE + "'");
196        this.kvType = KV_XML;
197        this.xmlFile = DEMO_XML_FILE;
198
199        // Add demo subdirectory to class path
200        this.appendClasspath(DEMO_SUBDIR.toString());
201
202        // Scan classes
203        this.scanSchemaClasses("org.jsimpledb.demo");
204    }

206    /**
207     * Append a path to the classpath.
208     */
209    protected boolean appendClasspath(String path) {
210        try {
211
212            // Get URLClassLoader.addURL() method and make accessible
213            final Method addURLMethod = URLClassLoader.class.getDeclaredMethod("addURL", URL.class);
214            addURLMethod.setAccessible(true);
215
216            // Split path and add components
217            for (String file : path.split(System.getProperty("file.separator", ":"))) {
218                if (file.length() == 0)
219                    continue;
220                addURLMethod.invoke(ClassLoader.getSystemClassLoader(), new Object[] { new File(file).toURI().toURL() });
221            }
222            return true;
223        } catch (Exception e) {
224            System.err.println(this.getName() + ": can't append `" + path + " to classpath: " + e);
225            return false;
226        }
227    }

229    private void scanSchemaClasses(String pkgname) {
230        if (this.schemaClasses == null)
231            this.schemaClasses = new HashSet<>();
232        for (String className : new JSimpleDBClassScanner().scanForClasses(pkgname.split("[\\s,]")))
233            this.schemaClasses.add(this.loadClass(className));
234    }
235
236    private void scanTypeClasses(String pkgname) {
237
238        // Check types of annotated classes as we scan them
239        final Function<Class<?>, Class<? extends FieldType<?>>> checkFunction
240          = new Function<Class<?>, Class<? extends FieldType<?>>>() {
241            @Override
242            @SuppressWarnings("unchecked")
243            public Class<? extends FieldType<?>> apply(Class<?> type) {
244                try {
245                    return (Class<? extends FieldType<?>>)type.asSubclass(FieldType.class);
246                } catch (ClassCastException e) {
247                    throw new IllegalArgumentException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
248                      + type + ": type is not a subclass of " + FieldType.class);
249                }
250            }
251        };
252
253        // Scan classes
254        if (this.fieldTypeClasses == null)
255            this.fieldTypeClasses = new HashSet<>();
256        for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(pkgname.split("[\\s,]")))
257            this.fieldTypeClasses.add(checkFunction.apply(this.loadClass(className)));
258    }
259
260    /**
261     * Load a class.
262     *
263     * @throws RuntimeException if load fails
264     */
265    protected Class<?> loadClass(String className) {
266        try {
267            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
268        } catch (ClassNotFoundException e) {
269            throw new RuntimeException("failed to load class `" + className + "'", e);
270        }
271    }
272
273    protected void startupKVDatabase() {
274        if (this.kvdb != null)
275            this.shutdownKVDatabase();
276        switch (this.kvType) {
277        case KV_MEM:
278            this.kvdb = new SimpleKVDatabase();
279            this.databaseDescription = "In-Memory Database";
280            break;
281        case KV_FDB:
282        {
283            final FoundationKVDatabase fdb = new FoundationKVDatabase();
284            fdb.setClusterFilePath(this.fdbClusterFile);
285            fdb.setKeyPrefix(this.keyPrefix);
286            fdb.start();
287            this.kvdb = fdb;
288            this.databaseDescription = "FoundationDB";
289            if (this.keyPrefix != null)
290                this.databaseDescription += " [0x" + ByteUtil.toString(this.keyPrefix) + "]";
291            break;
292        }
293        case KV_XML:
294            this.kvdb = new XMLKVDatabase(this.xmlFile);
295            this.databaseDescription = this.xmlFile.getName();
296            break;
297        default:
298            throw new RuntimeException("internal error");
299        }
300    }
301
302    protected void shutdownKVDatabase() {
303        if (this.kvdb != null) {
304            switch (this.kvType) {
305            case KV_FDB:
306                ((FoundationKVDatabase)this.kvdb).stop();
307                break;
308            default:
309                break;
310            }
311            this.kvdb = null;
312        }
313    }

317    /**
318     * Output usage message flag listing.
319     */
320    protected void outputFlags(String[][] subclassOpts) {
321        final String[][] baseOpts = new String[][] {
322            { "-cp, --classpath path",      "Append to the classpath (useful with `java -jar ...')" },
323            { "--fdb file",                 "Use FoundationDB with specified cluster file" },
324            { "--mem",                      "Use an empty in-memory database (default)" },
325            { "--prefix prefix",            "FoundationDB key prefix (hex or string)" },
326            { "-ro, --read-only",           "Disallow database modifications" },
327            { "--new-schema",               "Allow recording of a new database schema version" },
328            { "--xml file",                 "Use the specified XML flat file database" },
329            { "--version num",              "Specify database schema version (default highest recorded)" },
330            { "--schema-pkg package",       "Scan for @JSimpleClass types under Java package to build schema (=> JSimpleDB mode)" },
331            { "--types-pkg package",        "Scan for @JFieldType types under Java package to register custom types" },
332            { "-p, --scan-pkg package",     "Equivalent to `--schema-pkg package --types-pkg package'" },
333            { "-h, --help, -h",                 "Show this help message" },
334            { "--verbose, -v",              "Show verbose error messages" },
335        };
336        final String[][] combinedOpts = new String[baseOpts.length + subclassOpts.length][];
337        System.arraycopy(baseOpts, 0, combinedOpts, 0, baseOpts.length);
338        System.arraycopy(subclassOpts, 0, combinedOpts, baseOpts.length, subclassOpts.length);
339        Arrays.sort(combinedOpts, new Comparator<String[]>() {
340            @Override
341            public int compare(String[] opt1, String[] opt2) {
342                return opt1[0].compareTo(opt2[0]);
343            }
344        });
345        int width = 0;
346        for (String[] opt : combinedOpts)
347            width = Math.max(width, opt[0].length());
348        for (String[] opt : combinedOpts)
349            System.err.println(String.format("  %-" + width + "s  %s", opt[0], opt[1]));
350    }
