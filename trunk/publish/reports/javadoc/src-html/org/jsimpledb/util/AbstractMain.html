031/**
032 * Support superclass for main entry point classes.
033 */
034public abstract class AbstractMain extends MainClass {
035
036    protected static final int KV_MEM = 0;
037    protected static final int KV_FDB = 1;
038    protected static final int KV_XML = 2;
039
040    protected int kvType = KV_MEM;
041    protected String fdbClusterFile;
042    protected File xmlFile;
043    protected byte[] keyPrefix;
044    protected int schemaVersion;
045    protected HashSet<Class<?>> schemaClasses;
046    protected HashSet<Class<? extends FieldType<?>>> fieldTypeClasses;
047    protected boolean allowNewSchema;
048    protected boolean verbose;
049    protected boolean readOnly;
050
051    protected KVDatabase kvdb;
052    protected String databaseDescription;

054    /**
055     * Parse command line options.
056     *
057     * @return -1 to proceed, otherwise process exit value
058     */
059    public int parseOptions(ArrayDeque<String> params) {
060        while (!params.isEmpty() && params.peekFirst().startsWith("-")) {
061            final String option = params.removeFirst();
062            if (option.equals("-h") || option.equals("--help")) {
063                this.usageMessage();
064                return 0;
065            } else if (option.equals("-ro") || option.equals("--read-only"))
066                this.readOnly = true;
067            else if (option.equals("-v") || option.equals("--verbose"))
068                this.verbose = true;
069            else if (option.equals("--version")) {
070                if (params.isEmpty())
071                    this.usageError();
072                final String vstring = params.removeFirst();
073                try {
074                    this.schemaVersion = Integer.parseInt(vstring);
075                    if (this.schemaVersion < 0)
076                        throw new IllegalArgumentException("schema version is negative");
077                } catch (Exception e) {
078                    System.err.println(this.getName() + ": invalid schema version `" + vstring + "': " + e.getMessage());
079                    this.usageError();
080                }
081            } else if (option.equals("--schema-pkg")) {
082                if (params.isEmpty())
083                    this.usageError();
084                this.scanSchemaClasses(params.removeFirst());
085            } else if (option.equals("--types-pkg")) {
086                if (params.isEmpty())
087                    this.usageError();
088                this.scanTypeClasses(params.removeFirst());
089            } else if (option.equals("--scan-pkg")) {
090                if (params.isEmpty())
091                    this.usageError();
092                final String packageName = params.removeFirst();
093                this.scanSchemaClasses(packageName);
094                this.scanTypeClasses(packageName);
095            } else if (option.equals("--new-schema"))
096                this.allowNewSchema = true;
097            else if (option.equals("--mem"))
098                this.kvType = KV_MEM;
099            else if (option.equals("--prefix")) {
100                if (params.isEmpty())
101                    this.usageError();
102                final String value = params.removeFirst();
103                try {
104                    this.keyPrefix = ByteUtil.parse(value);
105                } catch (IllegalArgumentException e) {
106                    this.keyPrefix = value.getBytes(Charset.forName("UTF-8"));
107                }
108            } else if (option.equals("--fdb")) {
109                if (params.isEmpty())
110                    this.usageError();
111                this.kvType = KV_FDB;
112                this.fdbClusterFile = params.removeFirst();
113                if (!new File(this.fdbClusterFile).exists())
114                    System.err.println(this.getName() + ": file `" + this.fdbClusterFile + "' does not exist");
115            } else if (option.equals("--xml")) {
116                if (params.isEmpty())
117                    this.usageError();
118                this.kvType = KV_XML;
119                this.xmlFile = new File(params.removeFirst());
120            } else if (option.equals("--"))
121                break;
122            else if (!this.parseOption(option, params)) {
123                System.err.println(this.getName() + ": unknown option `" + option + "'");
124                this.usageError();
125            }
126        }
127        if (this.kvType != KV_FDB && this.keyPrefix != null) {
128            System.err.println(this.getName() + ": option `--prefix' is only valid in combination with `--fdb'");
129            this.usageError();
130        }
131        return -1;
132    }

150    public JSimpleDBFactory getJSimpleDBFactory(Database db) {
151        return new JSimpleDBFactory()

153          .setSchemaVersion(this.schemaVersion)
154          .setDatabase(db);

163        return false;

166    private void scanSchemaClasses(String pkgname) {
167        if (this.schemaClasses == null)
168            this.schemaClasses = new HashSet<>();
169        for (String className : new JSimpleDBClassScanner().scanForClasses(pkgname.split("[\\s,]")))
170            this.schemaClasses.add(this.loadClass(className));
171    }
172
173    private void scanTypeClasses(String pkgname) {
174
175        // Check types of annotated classes as we scan them
176        final Function<Class<?>, Class<? extends FieldType<?>>> checkFunction
177          = new Function<Class<?>, Class<? extends FieldType<?>>>() {
178            @Override
179            @SuppressWarnings("unchecked")
180            public Class<? extends FieldType<?>> apply(Class<?> type) {
181                try {
182                    return (Class<? extends FieldType<?>>)type.asSubclass(FieldType.class);
183                } catch (ClassCastException e) {
184                    throw new IllegalArgumentException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
185                      + type + ": type is not a subclass of " + FieldType.class);
186                }
187            }
188        };
189
190        // Scan classes
191        if (this.fieldTypeClasses == null)
192            this.fieldTypeClasses = new HashSet<>();
193        for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(pkgname.split("[\\s,]")))
194            this.fieldTypeClasses.add(checkFunction.apply(this.loadClass(className)));
195    }
196
197    /**
198     * Load a class.
199     *
200     * @throws RuntimeException if load fails
201     */
202    protected Class<?> loadClass(String className) {
203        try {
204            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
205        } catch (ClassNotFoundException e) {
206            throw new RuntimeException("failed to load class `" + className + "'", e);
207        }
208    }
209
210    protected void startupKVDatabase() {
211        if (this.kvdb != null)
212            this.shutdownKVDatabase();
213        switch (this.kvType) {
214        case KV_MEM:
215            this.kvdb = new SimpleKVDatabase();
216            this.databaseDescription = "In-Memory Database";
217            break;
218        case KV_FDB:
219        {
220            final FoundationKVDatabase fdb = new FoundationKVDatabase();
221            fdb.setClusterFilePath(this.fdbClusterFile);
222            fdb.setKeyPrefix(this.keyPrefix);
223            fdb.start();
224            this.kvdb = fdb;
225            this.databaseDescription = "FoundationDB";
226            if (this.keyPrefix != null)
227                this.databaseDescription += " [0x" + ByteUtil.toString(this.keyPrefix) + "]";
228            break;
229        }
230        case KV_XML:
231            this.kvdb = new XMLKVDatabase(this.xmlFile);
232            this.databaseDescription = this.xmlFile.getName();
233            break;
234        default:
235            throw new RuntimeException("internal error");
236        }
237    }
238
239    protected void shutdownKVDatabase() {
240        if (this.kvdb != null) {
241            switch (this.kvType) {
242            case KV_FDB:
243                ((FoundationKVDatabase)this.kvdb).stop();
244                break;
245            default:
246                break;
247            }
248            this.kvdb = null;
249        }
250    }

254    /**
255     * Output usage message flag listing.
256     */
257    protected void outputFlags(String[][] subclassOpts) {
258        final String[][] baseOpts = new String[][] {
259            { "--fdb file",             "Use FoundationDB with specified cluster file" },
260            { "--mem",                  "Use an empty in-memory database (default)" },
261            { "--prefix prefix",        "FoundationDB key prefix (hex or string)" },
262            { "--read-only",            "Disallow database modifications" },
263            { "--new-schema",           "Allow recording of a new database schema version" },
264            { "--xml file",             "Use the specified XML flat file database" },
265            { "--version num",          "Specify database schema version (default highest recorded)" },
266            { "--schema-pkg package",   "Scan for @JSimpleClass types under Java package to build schema (=> JSimpleDB mode)" },
267            { "--types-pkg package",    "Scan for @JFieldType types under Java package to register custom types" },
268            { "--scan-pkg package",     "Equivalent to `--schema-pkg package --types-pkg package'" },
269            { "--help, -h",             "Show this help message" },
270            { "--verbose, -v",          "Show verbose error messages" },
271        };
272        final String[][] combinedOpts = new String[baseOpts.length + subclassOpts.length][];
273        System.arraycopy(baseOpts, 0, combinedOpts, 0, baseOpts.length);
274        System.arraycopy(subclassOpts, 0, combinedOpts, baseOpts.length, subclassOpts.length);
275        Arrays.sort(combinedOpts, new Comparator<String[]>() {
276            @Override
277            public int compare(String[] opt1, String[] opt2) {
278                return opt1[0].compareTo(opt2[0]);
279            }
280        });
281        int width = 0;
282        for (String[] opt : combinedOpts)
283            width = Math.max(width, opt[0].length());
284        for (String[] opt : combinedOpts)
285            System.err.println(String.format("  %-" + width + "s  %s", opt[0], opt[1]));
286    }
