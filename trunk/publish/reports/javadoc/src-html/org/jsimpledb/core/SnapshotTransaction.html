008package org.jsimpledb.core;
009
010/**
011 * An initially empty, in-memory {@link Transaction} that persists indefinitely.
012 *
013 * <p>
014 * {@link SnapshotTransaction}s hold a "snapshot" some portion of the state of a {@link Transaction} for later use.
015 * Each {@link SnapshotTransaction}s contains its own set of "snapshot" objects that are copies of
016 * corresponding "real" database objects (as of the time they were copied).
017 * </p>
018 *
019 * <p>
020 * {@link SnapshotTransaction}s can never be closed (i.e., committed or rolled-back); they persist in memory until
021 * no longer referenced. {@link Transaction.Callback}s may be registered but they will never be invoked.
022 * </p>
023 *
024 * @see Transaction#createSnapshotTransaction
025 */
026public class SnapshotTransaction extends Transaction {
027
028    SnapshotTransaction(Transaction parent) {
029        super(parent.db, new SnapshotKVTransaction(parent), parent.schemas, parent.schema);
030    }
031
032    /**
033     * Delete all objects contained in this snapshot transaction and reset it back to its initial empty state.
034     *
035     * <p>
036     * It will contain schema meta-data but no objects.
037     * </p>
038     */
039    public void reset() {
040
041        // Sanity check
042        if (this.stale)
043            throw new StaleTransactionException(this);
044
045        // Delete all object and index keys
046        this.db.reset(this);
047    }
048
049    /**
050     * Commit this transaction.
051     *
052     * <p>
053     * {@link SnapshotTransaction}s do not support this method and will always throw {@link UnsupportedOperationException}.
054     * </p>
055     *
056     * @throws UnsupportedOperationException always
057     */
058    @Override
059    public void commit() {
060        throw new UnsupportedOperationException("snapshot transaction");
061    }
062
063    /**
064     * Roll back this transaction.
065     *
066     * <p>
067     * {@link SnapshotTransaction}s do not support this method and will always throw {@link UnsupportedOperationException}.
068     * </p>
069     *
070     * @throws UnsupportedOperationException always
071     */
072    @Override
073    public void rollback() {
074        throw new UnsupportedOperationException("snapshot transaction");
075    }
076
077    /**
078     * Mark this transaction for rollback only.
079     *
080     * <p>
081     * {@link SnapshotTransaction}s do not support this method and will always throw {@link UnsupportedOperationException}.
082     * </p>
083     *
084     * @throws UnsupportedOperationException always
085     */
086    @Override
087    public void setRollbackOnly() {
088        throw new UnsupportedOperationException("snapshot transaction");
089    }
090
091    /**
092     * Register a transaction callback to be invoked when this transaction completes.
093     *
094     * <p>
095     * {@link Transaction.Callback}s registered with a {@link SnapshotTransaction} will by definition never be invoked.
096     * Therefore, this method simply discards {@code callback}.
097     * </p>
098     */
099    @Override
100    public void addCallback(Callback callback) {
101    }
102
103    /**
104     * Determine whether this transaction is still valid.
105     *
106     * <p>
107     * {@link SnapshotTransaction}s are always valid.
108     * </p>
109     *
110     * @return true always
111     */
112    @Override
113    public boolean isValid() {
114        return super.isValid();
115    }
116}
