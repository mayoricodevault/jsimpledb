008package org.jsimpledb.core;
009
010import java.util.Set;
011import java.util.SortedSet;
012
013/**
014 * A field that references another {@link Database} object.
015 *
016 * <p>
017 * Null values sort last.
018 * </p>
019 *
020 * <p>
021 * Reference fields are always indexed.
022 * </p>
023 */
024public class ReferenceField extends SimpleField<ObjId> {
025
026    final DeleteAction onDelete;
027    final boolean cascadeDelete;
028
029    /**
030     * Constructor.
031     *
032     * @param name the name of the field
033     * @param storageId field storage ID
034     * @param version schema version
035     * @param onDelete deletion behavior
036     * @param cascadeDelete whether to cascade deletes
037     * @param objectTypes allowed object type storage IDs, or null for no restriction
038     * @throws IllegalArgumentException if any parameter is null
039     * @throws IllegalArgumentException if {@code name} is invalid
040     * @throws IllegalArgumentException if {@code storageId} is invalid
041     */
042    ReferenceField(String name, int storageId, SchemaVersion version,
043      DeleteAction onDelete, boolean cascadeDelete, Set<Integer> objectTypes) {
044        super(name, storageId, version, new ReferenceFieldType(objectTypes), true);
045        if (onDelete == null)
046            throw new IllegalArgumentException("null onDelete");
047        this.onDelete = onDelete;
048        this.cascadeDelete = cascadeDelete;
049    }
050
051// Public methods
052
053    /**
054     * Get the desired behavior when an object referred to by this field is deleted.
055     */
056    public DeleteAction getOnDelete() {
057        return this.onDelete;
058    }
059
060    /**
061     * Determine whether the referred-to object should be deleted when an object containing this field is deleted.
062     */
063    public boolean isCascadeDelete() {
064        return this.cascadeDelete;
065    }
066
067    /**
068     * Get the object types this field is allowed to reference, if so restricted.
069     *
070     * @return storage IDs of allowed object types, or null if there is no restriction
071     */
072    public SortedSet<Integer> getObjectTypes() {
073        return ((ReferenceFieldType)this.getFieldType()).getObjectTypes();
074    }
075
076    @Override
077    public <R> R visit(FieldSwitch<R> target) {
078        return target.caseReferenceField(this);
079    }
080
081    @Override
082    public String toString() {
083        return "reference field `" + this.name + "'";
084    }
085
086// Non-public methods
087
088    @Override
089    ReferenceFieldStorageInfo toStorageInfo() {
090        return new ReferenceFieldStorageInfo(this, this.parent != null ? this.parent.storageId : 0);
091    }
092}
