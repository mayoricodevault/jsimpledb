008package org.jsimpledb.core;
009
010import java.util.ArrayList;
011import java.util.Collections;
012import java.util.SortedMap;
013import java.util.TreeMap;
014
015import org.jsimpledb.schema.SchemaModel;
016import org.jsimpledb.schema.SchemaObject;
017
018/**
019 * Contains information describing one {@link Schema} version.
020 */
021public class SchemaVersion {
022
023    final int versionNumber;
024    final byte[] encodedXML;
025    final SchemaModel schemaModel;
026    final TreeMap<Integer, SchemaItem> schemaItemMap = new TreeMap<>();
027    final TreeMap<Integer, StorageInfo> storageInfoMap = new TreeMap<>();
028
029    /**
030     * Constructor.
031     *
032     * @throws IllegalArgumentException if a storage ID is used twice
033     */
034    SchemaVersion(int versionNumber, byte[] encodedXML, SchemaModel schemaModel, FieldTypeRegistry fieldTypeRegistry) {
035
036        // Sanity check
037        if (versionNumber <= 0)
038            throw new IllegalArgumentException("versionNumber <= 0");
039        if (schemaModel == null)
040            throw new IllegalArgumentException("null schemaModel");
041        this.versionNumber = versionNumber;
042        this.encodedXML = encodedXML;
043        this.schemaModel = schemaModel.clone();
044
045        // Build object types and fields, and associated StorageInfo objects
046        final ArrayList<ObjType> objTypes = new ArrayList<>();
047        for (SchemaObject schemaObject : this.schemaModel.getSchemaObjects().values()) {
048            final ObjType objType = new ObjType(schemaObject, this, fieldTypeRegistry);
049            this.addSchemaItem(objType);
050            for (Field<?> field : objType.getFieldsAndSubFields())
051                this.addSchemaItem(field);
052            this.addStorageInfo(objType);
053        }
054    }
055
056    /**
057     * Get the version number associated with this instance.
058     *
059     * @return version number, always greater than zero
060     */
061    public int getVersionNumber() {
062        return this.versionNumber;
063    }
064
065    /**
066     * Get the {@link SchemaModel} on which this schema version is based.
067     */
068    public SchemaModel getSchemaModel() {
069        return this.schemaModel;
070    }
071
072    /**
073     * Get all of the {@link SchemaItem}s that constitute this schema version, indexed by storage ID.
074     *
075     * @return unmodifiable mapping from {@linkplain ObjType#getStorageId storage ID} to schema object
076     */
077    public SortedMap<Integer, SchemaItem> getSchemaItemMap() {
078        return Collections.unmodifiableSortedMap(this.schemaItemMap);
079    }
080
081    /**
082     * Get the {@link SchemaItem} with the given storage ID and having the specified type.
083     *
084     * @param storageId storage ID
085     * @param type expected type
086     * @return the {@link SchemaItem} with storage ID {@code storageID} and having type {@code type}
087     * @throws IllegalArgumentException if none such exists or it is not of type {@code type}
088     */
089    public <T extends SchemaItem> T getSchemaItem(int storageId, Class<T> type) {
090        final SchemaItem schemaItem = this.schemaItemMap.get(storageId);
091        if (schemaItem == null)
092            throw new IllegalArgumentException("storage ID " + storageId + " does not exist in " + this);
093        try {
094            return type.cast(schemaItem);
095        } catch (ClassCastException e) {
096            throw new IllegalArgumentException("storage ID " + storageId + " does not correspond to any "
097              + this.getDescription(type) + " in " + this + " (found " + this.getDescription(schemaItem.getClass())
098              + " instead)");
099        }
100    }
101
102    @Override
103    public String toString() {
104        return "schema version " + this.versionNumber;
105    }
106
107    private void addSchemaItem(SchemaItem schemaItem) {
108        final int storageId = schemaItem.getStorageId();
109        final SchemaItem previous = this.schemaItemMap.put(storageId, schemaItem);
110        if (previous != null) {
111
112            // Allow the field with the same storage ID to appear in mulitple object types if equivalent
113            if (schemaItem instanceof Field && previous instanceof Field && ((Field<?>)schemaItem).isEquivalent((Field<?>)previous))
114                return;
115
116            // Not the same
117            throw new IllegalArgumentException("incompatible use of storage ID " + storageId
118              + " by both " + previous + " and " + schemaItem + " in " + this);
119        }
120    }
121
122    private void addStorageInfo(ObjType objType) {
123        final ObjTypeStorageInfo objInfo = objType.toStorageInfo();
124        this.addStorageInfo(objInfo);
125        for (SimpleField<?> field : objType.simpleFields.values()) {
126            final SimpleFieldStorageInfo fieldInfo = field.toStorageInfo(0);
127            this.addStorageInfo(fieldInfo);
128            objInfo.getFields().put(field.storageId, fieldInfo);
129        }
130        for (ComplexField<?> field : objType.complexFields.values()) {
131            final ComplexFieldStorageInfo fieldInfo = field.toStorageInfo();
132            objInfo.getFields().put(field.storageId, fieldInfo);
133            final ArrayList<SimpleFieldStorageInfo> subFieldInfos = new ArrayList<>();
134            for (SimpleField<?> subField : field.getSubFields()) {
135                final SimpleFieldStorageInfo subFieldInfo = subField.toStorageInfo(field.storageId);
136                this.addStorageInfo(subFieldInfo);
137                subFieldInfos.add(subFieldInfo);
138            }
139            fieldInfo.setSubFields(subFieldInfos);
140            this.addStorageInfo(fieldInfo);
141        }
142        for (CounterField field : objType.counterFields.values()) {
143            final CounterFieldStorageInfo fieldInfo = field.toStorageInfo();
144            this.addStorageInfo(fieldInfo);
145            objInfo.getFields().put(field.storageId, fieldInfo);
146        }
147    }
148
149    private void addStorageInfo(StorageInfo storageInfo) {
150        final int storageId = storageInfo.getStorageId();
151        final StorageInfo previous = this.storageInfoMap.put(storageId, storageInfo);
152        if (previous != null && !previous.canShareStorageId(storageInfo))
153            throw new IllegalArgumentException("duplicate use of storage ID " + storageId);     // should never happen
154    }
155
156    private String getDescription(Class<? extends SchemaItem> type) {
157        if (type == ObjType.class)
158            return "object";
159        if (type == CounterField.class)
160            return "counter field";
161        if (type == SimpleField.class)
162            return "simple field";
163        if (type == ReferenceField.class)
164            return "reference field";
165        if (type == SetField.class)
166            return "set field";
167        if (type == ListField.class)
168            return "list field";
169        if (type == MapField.class)
170            return "map field";
171        return type.getSimpleName();
172    }
173}
