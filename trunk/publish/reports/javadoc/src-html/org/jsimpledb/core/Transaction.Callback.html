041/**
042 * A JSimpleDB {@link Database} transaction.
043 *
044 * <p>
045 * Note: this is the lower level, core API for {@link org.jsimpledb.JSimpleDB}. In most cases this API
046 * will only be used indirectly through the higher level {@link org.jsimpledb.JSimpleDB}, {@link org.jsimpledb.JTransaction},
047 * and {@link org.jsimpledb.JObject} APIs.
048 * </p>
049 *
050 * <p>
051 * Methods in this class can be divided into the following categories:
052 * </p>
053 *
054 * <p>

056 * <ul>
057 *  <li>{@link #getDatabase getDatabase()} - Get the associated {@link Database}</li>
058 *  <li>{@link #getSchema getSchema()} - Get the database {@link Schema}, as seen by this transaction</li>

061 * </ul>
062 * </p>
063 *
064 * <p>
065 * <b>Transaction Lifecycle</b>
066 * <ul>
067 *  <li>{@link #commit commit()} - Commit transaction</li>
068 *  <li>{@link #rollback rollback()} - Roll back transaction</li>

071 *  <li>{@link #setReadOnly setReadOnly()} - Set transaction to read-only</li>
072 *  <li>{@link #setRollbackOnly setRollbackOnly()} - Set transaction for rollack only</li>

074 *  <li>{@link #createSnapshotTransaction createSnapshotTransaction()} - Create a empty, in-memory copy of this transaction</li>
075 * </ul>
076 * </p>
077 *
078 * <p>
079 * <b>Object Lifecycle</b>
080 * <ul>

236     * Delete a schema version that is no longer being used. There must be no objects with the given version
237     * in the database, and {@code version} must not be the version being used by this transaction.
238     *
239     * @param version schema version to remove
240     * @return true if schema version was found and removed, false if schema version does not exist in database
241     * @throws InvalidSchemaException if one or more objects with schema version {@code version} still exist
242     * @throws InvalidSchemaException if {@code version} is equal to this transaction's version
243     * @throws IllegalArgumentException if {@code version} is zero or negative
244     * @throws StaleTransactionException if this transaction is no longer usable
245     */
246    public synchronized boolean deleteSchemaVersion(int version) {
247
248        // Sanity check
249        if (version <= 0)
250            throw new IllegalArgumentException("invalid schema version " + version);
251        if (version == this.version.getVersionNumber())
252            throw new InvalidSchemaException("version " + version + " is this transaction's version");
253        if (this.stale)
254            throw new StaleTransactionException(this);
255        if (this.queryVersion().asMap().containsKey(version))
256            throw new InvalidSchemaException("one or more version " + version + " objects still exist in database");
257
258        // Delete schema version
259        if (!this.schema.deleteVersion(version))
260            return false;
261        this.db.deleteSchema(this.kvt, version);
262        return true;
263    }

279    /**
280     * Commit this transaction.
281     *
282     * @throws StaleTransactionException if this transaction is no longer usable
283     * @throws RetryTransactionException from {@link KVTransaction#commit KVTransaction.commit()}
284     * @throws RollbackOnlyTransactionException if this instance has been {@linkplain #setRollbackOnly marked} rollback only;
285     *  this instance will be automatically rolled back
286     */
287    public synchronized void commit() {
288
289        // Sanity check
290        if (this.stale)
291            throw new StaleTransactionException(this);
292
293        // Rollback only?
294        if (this.rollbackOnly) {
295            this.log.info("commit() invoked on transaction " + this + " marked rollback-only, rolling back");
296            this.rollback();
297            throw new RollbackOnlyTransactionException(this);
298        }
299        this.stale = true;
300
301        // Do before completion callbacks
302        this.log.debug("commit() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
303        Callback failedCallback = null;
304        try {
305            for (Callback callback : this.callbacks) {
306                failedCallback = callback;
307                callback.beforeCommit(this.readOnly);
308            }
309            failedCallback = null;
310        } finally {
311            if (failedCallback != null) {
312                this.log.warn("error invoking beforeCommit() method on transaction "
313                  + this + " callback " + failedCallback + ", rolling back");
314            }
315            this.triggerBeforeCompletion();
316            if (failedCallback != null) {
317                try {
318                    try {
319                        this.kvt.rollback();
320                    } catch (KVTransactionException e) {
321                        // ignore
322                    }
323                } finally {
324                    this.triggerAfterCompletion(false);
325                }
326            }
327        }
328
329        // Commit KVTransaction and trigger after completion callbacks
330        failedCallback = null;
331        try {
332            this.kvt.commit();
333            for (Callback callback : this.callbacks) {
334                failedCallback = callback;
335                callback.afterCommit();
336            }
337            failedCallback = null;
338        } finally {
339            if (failedCallback != null)
340                this.log.warn("error invoking afterCommit() method on transaction " + this + " callback " + failedCallback);
341            this.triggerAfterCompletion(true);
342        }
343    }

345    /**
346     * Roll back this transaction.
347     *
348     * @throws StaleTransactionException if this transaction is no longer usable
349     */
350    public synchronized void rollback() {
351
352        // Sanity check
353        if (this.stale)
354            throw new StaleTransactionException(this);
355        this.stale = true;
356        this.log.debug("rollback() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
357
358        // Do before completion callbacks
359        this.triggerBeforeCompletion();
360
361        // Roll back KVTransaction and trigger after completion callbacks
362        try {
363            this.kvt.rollback();
364        } finally {
365            this.triggerAfterCompletion(false);
366        }
367    }

369    private /*synchronized*/ void triggerBeforeCompletion() {
370        for (Callback callback : this.callbacks) {
371            try {
372                callback.beforeCompletion();
373            } catch (Throwable t) {
374                this.log.error("error invoking beforeCompletion() method on transaction callback " + callback, t);
375            }
376        }
377    }
378
379    private /*synchronized*/ void triggerAfterCompletion(boolean committed) {
380        for (Callback callback : this.callbacks) {
381            try {
382                callback.afterCompletion(committed);
383            } catch (Throwable t) {
384                this.log.error("error invoking afterCompletion() method on transaction callback " + callback, t);
385            }
386        }
387    }

613    private /*synchronized*/ ObjId generateIdValidated(int storageId) {
614
615        // Create a new, unique key
616        final ByteWriter keyWriter = new ByteWriter();
617        for (int attempts = 0; attempts < MAX_GENERATED_KEY_ATTEMPTS; attempts++) {
618            final ObjId id = new ObjId(storageId);
619            id.writeTo(keyWriter);
620            if (this.kvt.get(keyWriter.getBytes()) == null)
621                return id;
622            keyWriter.reset(0);
623        }
624
625        // Give up
626        throw new DatabaseException("could not find a new, unused object ID after "
627          + MAX_GENERATED_KEY_ATTEMPTS + " attempts; is our source of randomness truly random?");
628    }

630    private synchronized void initialize(ObjId id, ObjType objType) {
631
632        // Sanity check
633        if (this.stale)
634            throw new StaleTransactionException(this);
635        if (this.readOnly)
636            throw new ReadOnlyTransactionException(this);
637
638        // Write object meta-data
639        ObjInfo.write(this, id, objType.version.versionNumber, false);
640
641        // Write object version index entry
642        this.kvt.put(Database.buildVersionIndexKey(id, objType.version.versionNumber), ByteUtil.EMPTY);
643
644        // Initialize counters to zero
645        if (!objType.counterFields.isEmpty()) {
646            for (CounterField field : objType.counterFields.values())
647                this.kvt.put(field.buildKey(id), this.kvt.encodeCounter(0));
648        }
649
650        // Write simple field index entries
651        for (SimpleField<?> field : objType.simpleFields.values()) {
652            if (field.indexed)
653                this.kvt.put(Transaction.buildSimpleIndexEntry(field, id, null), ByteUtil.EMPTY);
654        }
655
656        // Write composite index entries
657        for (CompositeIndex index : objType.compositeIndexes.values())
658            this.kvt.put(Transaction.buildDefaultCompositeIndexEntry(id, index), ByteUtil.EMPTY);
659
660        // Notify listeners
661        for (CreateListener listener : this.createListeners.toArray(new CreateListener[this.createListeners.size()]))
662            listener.onCreate(this, id);
663    }

665    /**
666     * Delete an object. Does nothing if object does not exist (e.g., has already been deleted).
667     *
668     * <p>
669     * This method does <i>not</i> change the object's schema version if it has a different version that this transaction.
670     * </p>
671     *
672     * <p><b>Secondary Deletions</b></p>
673     *
674     * <p>
675     * Deleting an object can trigger additional secondary deletions. Specifically,
676     * if the object contains reference fields with {@linkplain ReferenceField#cascadeDelete delete cascade} enabled,
677     * any objects referred to through those fields will also be deleted, and if the object is referred to by any other objects
678     * through fields configured for {@link DeleteAction#DELETE}, those referring objects will be deleted.
679     * </p>
680     *
681     * <p>
682     * In any case, deletions occur one at a time, and only when an object is actually deleted are any associated secondary
683     * deletions added to an internal deletion queue. However, the order in which objects on this deletion queue are
684     * processed is unspecified.
685     * </p>
686     *
687     * @param id object ID of the object to delete
688     * @return true if object was found and deleted, false if object was not found,
689     *  or if {@code id} specifies an unknown object type
690     * @throws ReferencedObjectException if the object is referenced by some other object
691     *  through a reference field configured for {@link DeleteAction#EXCEPTION}
692     * @throws IllegalArgumentException if {@code id} is null
693     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
694     * @throws StaleTransactionException if this transaction is no longer usable
695     */
696    public synchronized boolean delete(ObjId id) {
697
698        // Sanity check
699        if (id == null)
700            throw new IllegalArgumentException("null id");
701        if (this.stale)
702            throw new StaleTransactionException(this);
703        if (this.readOnly)
704            throw new ReadOnlyTransactionException(this);
705
706        // Get object info
707        try {
708            this.getObjectInfo(id, false);
709        } catch (DeletedObjectException | UnknownTypeException e) {
710            return false;
711        }
712
713        // Handle delete cascade and recurive DeleteAction.DELETE without hogging Java stack
714        final ObjIdSet deletables = new ObjIdSet();
715        deletables.add(id);
716        boolean found = false;
717        while (!deletables.isEmpty())
718            found |= this.doDelete(deletables.iterator().next(), deletables);
719
720        // Done
721        return found;
722    }

724    private synchronized boolean doDelete(final ObjId id, ObjIdSet deletables) {
725
726        // Loop here to handle any mutations within delete notification listener callbacks
727        ObjInfo info;
728        while (true) {
729
730            // Get object info
731            try {
732                info = this.getObjectInfo(id, false);
733            } catch (DeletedObjectException e) {                    // possibly due to a cycle of DeleteAction.DELETE references
734                deletables.remove(id);
735                return false;
736            } catch (UnknownTypeException e) {
737                throw new InconsistentDatabaseException("encountered reference with unknown type during delete cascade: " + id, e);
738            }
739
740            // Determine if any EXCEPTION reference fields refer to the object (from some other object); if so, throw exception
741            for (ReferenceFieldStorageInfo fieldInfo :
742              Iterables.filter(this.schema.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
743                for (ObjId referrer : this.findReferrers(id, DeleteAction.EXCEPTION, fieldInfo.storageId)) {
744                    if (!referrer.equals(id))
745                        throw new ReferencedObjectException(id, referrer, fieldInfo.storageId);
746                }
747            }
748
749            // Do we need to issue delete notifications for the object being deleted?
750            if (info.isDeleteNotified() || this.deleteListeners.isEmpty())
751                break;
752
753            // Issue delete notifications and retry
754            ObjInfo.write(this, id, info.getVersionNumber(), true);
755            for (DeleteListener listener : this.deleteListeners.toArray(new DeleteListener[this.deleteListeners.size()]))
756                listener.onDelete(this, id);
757        }
758
759        // Find all objects referred to by a reference field with cascadeDelete = true and add them to deletables
760        for (ReferenceField field : Iterables.filter(
761          Iterables.filter(info.getObjType().referenceFields.values(), ReferenceField.class), new HasCascadeDelete())) {
762            final Iterable<ObjId> refs = field.parent != null ?
763              field.parent.iterateSubField(this, id, field) : Collections.singleton(field.getValue(this, id));
764            for (ObjId ref : refs) {
765                if (ref != null)
766                    deletables.add(ref);
767            }
768        }
769
770        // Actually delete the object
771        this.deleteObjectData(info);
772        deletables.remove(id);
773
774        // Find all UNREFERENCE references and unreference them
775        for (ReferenceFieldStorageInfo fieldInfo :
776          Iterables.filter(this.schema.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
777            final NavigableSet<ObjId> referrers = this.findReferrers(id, DeleteAction.UNREFERENCE, fieldInfo.storageId);
778            if (fieldInfo.isSubField()) {
779                final ComplexFieldStorageInfo<?> superFieldInfo = this.schema.verifyStorageInfo(
780                  fieldInfo.superFieldStorageId, ComplexFieldStorageInfo.class);
781                superFieldInfo.unreferenceAll(this, fieldInfo.storageId, id, referrers);
782            } else {
783                final int storageId = fieldInfo.storageId;
784                for (ObjId referrer : referrers)
785                    this.writeSimpleField(referrer, storageId, null, false);
786            }
787        }
788
789        // Find all DELETE references and mark the containing object for deletion (caller will call us back to actually delete)
790        for (ObjId referrer : this.findReferrers(id, DeleteAction.DELETE, -1))
791            deletables.add(referrer);
792
793        // Done
794        return true;
795    }

797    // Delete all of an object's data
798    private void deleteObjectData(ObjInfo info) {
799
800        // Delete object's simple field index entries
801        final ObjId id = info.getId();
802        final ObjType type = info.getObjType();
803        for (SimpleField<?> field : type.simpleFields.values()) {
804            if (field.indexed)
805                this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, this.kvt.get(field.buildKey(id))));
806        }
807
808        // Delete object's composite index entries
809        for (CompositeIndex index : type.compositeIndexes.values())
810            this.kvt.remove(this.buildCompositeIndexEntry(id, index));
811
812        // Delete object's complex field index entries
813        for (ComplexField<?> field : type.complexFields.values())
814            field.removeIndexEntries(this, id);
815
816        // Delete object meta-data and all field content
817        final byte[] minKey = info.getId().getBytes();
818        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
819        this.kvt.removeRange(minKey, maxKey);
820
821        // Delete object schema version entry
822        this.kvt.remove(Database.buildVersionIndexKey(id, info.getVersionNumber()));
823    }

847    /**
848     * Copy all of an object's fields onto a target object in a (possibly) different transaction, replacing any previous values.
849     *
850     * <p>
851     * The schema version associated with the {@code source} object must be identical in this transaction and {@code dest}.
852     * Only the object's fields are copied; any other objects they reference are not copied. If the {@code target} object
853     * does not exist in {@code dest}, it will be created first (and {@link CreateListener}s notified); if {@code target}
854     * does exist in {@code dest}, it's schema version will be upgraded if necessary to match {@code source} (and any registered
855     * {@link VersionChangeListener}s notified).
856     * (Meaningful) changes to {@code target}'s fields generate change listener notifications.
857     * </p>
858     *
859     * <p>
860     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
861     * deadlock could result.
862     * </p>
863     *
864     * <p>
865     * If {@code dest} is this instance, and {@code source} equals {@code target}, no changes are made and false is returned.
866     * </p>
867     *
868     * @param source object ID of the source object in this transaction
869     * @param target object ID of the target object in {@code dest}
870     * @param dest destination transaction containing {@code target} (possibly same as this transaction)
871     * @return false if object already existed in {@code dest}, true if {@code target} did not exist in {@code dest}
872     * @throws DeletedObjectException if no object with ID equal to {@code id} is found in this transaction
873     * @throws UnknownTypeException if {@code source} or {@code target} specifies an unknown object type
874     * @throws IllegalArgumentException if {@code source} and {@code target} specify different object types
875     * @throws IllegalArgumentException if any parameter is null
876     * @throws IllegalArgumentException if any {@code source} and {@code target} have different object types
877     * @throws ReadOnlyTransactionException if {@code dest} has been {@linkplain #setReadOnly set read-only}
878     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
879     * @throws SchemaMismatchException if the schema version associated with {@code source} differs between
880     *  this transaction and {@code dest}
881     */
882    public synchronized boolean copy(ObjId source, final ObjId target, final Transaction dest) {
883
884        // Sanity check
885        if (source == null)
886            throw new IllegalArgumentException("null source");
887        if (target == null)
888            throw new IllegalArgumentException("null target");
889        if (dest == null)
890            throw new IllegalArgumentException("null dest");
891        if (this.stale)
892            throw new StaleTransactionException(this);
893
894        // Do nothing if nothing to do
895        if (this == dest && source.equals(target))
896            return false;
897
898        // Get object info
899        final ObjInfo srcInfo = this.getObjectInfo(source, false);
900
901        // Do the copy while both transactions are locked
902        synchronized (dest) {
903
904            // Sanity check
905            if (dest.stale)
906                throw new StaleTransactionException(dest);
907            if (dest.readOnly)
908                throw new ReadOnlyTransactionException(dest);
909
910            // Copy fields
911            return dest.mutateAndNotify(new Mutation<Boolean>() {
912                @Override
913                public Boolean mutate() {
914                    return Transaction.doCopyFields(srcInfo, target, Transaction.this, dest);
915                }
916            });
917        }
918    }

920    // This method assumes both transactions are locked
921    private static boolean doCopyFields(ObjInfo srcInfo, ObjId dstId, Transaction srcTx, Transaction dstTx) {
922
923        // Sanity check
924        final ObjId srcId = srcInfo.getId();
925        if (srcId.equals(dstId) && srcTx == dstTx)
926            throw new RuntimeException("internal error");
927
928        // Verify objects have the same type
929        if (srcId.getStorageId() != dstId.getStorageId()) {
930            throw new IllegalArgumentException("can't copy " + srcId + " to " + dstId
931              + " due to non-equal object types (" + srcId.getStorageId() + " != " + dstId.getStorageId() + ")");
932        }
933
934        // Find and verify source object's schema version in destination transaction
935        final int objectVersion = srcInfo.getVersionNumber();
936        final SchemaVersion dstSchema;
937        try {
938            dstSchema = dstTx.schema.getVersion(objectVersion);
939        } catch (IllegalArgumentException e) {
940            throw new SchemaMismatchException("destination transaction has no schema version " + objectVersion);
941        }
942        if (!Arrays.equals(srcTx.version.encodedXML, dstTx.version.encodedXML))
943            throw new SchemaMismatchException("destination transaction schema version " + objectVersion + " does not match");
944
945        // Create destination object if it doesn't already exist
946        final boolean existed = !dstTx.create(dstId, objectVersion);
947        final ObjInfo dstInfo = dstTx.getObjectInfo(dstId, false);
948        final boolean needUpgrade = dstInfo.getVersionNumber() != objectVersion;   // can only happen if object already existed
949
950        // Do field-by-field copy if there are change or version listeners, otherwise do fast copy of key/value pairs
951        final ObjType type = srcInfo.getObjType();
952        if (dstTx.hasFieldMonitor(type) || (needUpgrade && !dstTx.versionChangeListeners.isEmpty())) {
953
954            // Upgrade object first
955            if (needUpgrade)
956                dstTx.updateVersion(dstInfo, dstSchema);
957
958            // Copy fields
959            for (Field<?> field : type.fields.values())
960                field.copy(srcId, dstId, srcTx, dstTx);
961        } else {
962
963            // Delete pre-existing object's field content, if any
964            dstTx.deleteObjectData(dstInfo);
965
966            // Add schema version index entry
967            dstTx.kvt.put(Database.buildVersionIndexKey(dstId, objectVersion), ByteUtil.EMPTY);
968
969            // Copy object meta-data and all field content in one key range sweep
970            final byte[] srcMinKey = srcId.getBytes();
971            final byte[] srcMaxKey = ByteUtil.getKeyAfterPrefix(srcMinKey);
972            final ByteWriter dstWriter = new ByteWriter();
973            dstWriter.write(dstId.getBytes());
974            final int dstMark = dstWriter.mark();
975            for (Iterator<KVPair> i = srcTx.kvt.getRange(srcMinKey, srcMaxKey, false); i.hasNext(); ) {
976                final KVPair kv = i.next();
977                final ByteReader srcReader = new ByteReader(kv.getKey());
978                srcReader.skip(srcMinKey.length);
979                dstWriter.reset(dstMark);
980                dstWriter.write(srcReader);
981                dstTx.kvt.put(dstWriter.getBytes(), kv.getValue());
982            }
983
984            // Create object's simple field index entries
985            for (SimpleField<?> field : type.simpleFields.values()) {
986                if (field.indexed) {
987                    final byte[] fieldValue = dstTx.kvt.get(field.buildKey(dstId));     // can be null (if field has default value)
988                    final byte[] indexKey = Transaction.buildSimpleIndexEntry(field, dstId, fieldValue);
989                    dstTx.kvt.put(indexKey, ByteUtil.EMPTY);
990                }
991            }
992
993            // Create object's composite index entries
994            for (CompositeIndex index : type.compositeIndexes.values())
995                dstTx.kvt.put(Transaction.buildCompositeIndexEntry(dstTx, dstId, index), ByteUtil.EMPTY);
996
997            // Create object's complex field index entries
998            for (ComplexField<?> field : type.complexFields.values()) {
999                for (SimpleField<?> subField : field.getSubFields()) {
1000                    if (subField.indexed)
1001                        field.addIndexEntries(dstTx, dstId, subField);
1002                }
1003            }
1004        }
1005
1006        // Done
1007        return !existed;
1008    }

1140    /**
1141     * Update object to the current schema version.
1142     *
1143     * @param info original object info
1144     * @param targetVersion version to change to
1145     */
1146    private synchronized void updateVersion(final ObjInfo info, final SchemaVersion targetVersion) {
1147
1148        // Get version numbers
1149        final ObjId id = info.getId();
1150        final int oldVersion = info.getVersionNumber();
1151        final int newVersion = targetVersion.versionNumber;
1152
1153        // Sanity check
1154        if (newVersion == oldVersion)
1155            throw new IllegalArgumentException("object already at version " + newVersion);
1156        if (this.readOnly)
1157            throw new ReadOnlyTransactionException(this);
1158
1159        // Get old and new types
1160        final ObjType oldType = info.getObjType();
1161        final ObjType newType;
1162        try {
1163            newType = targetVersion.getObjType(id.getStorageId());
1164        } catch (UnknownTypeException e) {
1165            throw (TypeNotInSchemaVersionException)new TypeNotInSchemaVersionException(id, newVersion).initCause(e);
1166        }
1167
1168        // Gather removed fields' values here for user migration if any VersionChangeListeners are registered
1169        final TreeMap<Integer, Object> oldValueMap = !this.versionChangeListeners.isEmpty() ?
1170          new TreeMap<Integer, Object>() : null;
1171
1172    //////// Remove the index entries corresponding to removed composite indexes
1173
1174        // Get composite index storage IDs (old or new)
1175        final TreeSet<Integer> compositeIndexStorageIds = new TreeSet<>();
1176        compositeIndexStorageIds.addAll(oldType.compositeIndexes.keySet());
1177        compositeIndexStorageIds.addAll(newType.compositeIndexes.keySet());
1178
1179        // Remove index entries for composite indexes that are going away
1180        for (CompositeIndex index : oldType.compositeIndexes.values()) {
1181            if (!newType.compositeIndexes.containsKey(index.storageId))
1182                this.kvt.remove(this.buildCompositeIndexEntry(id, index));
1183        }
1184
1185    //////// Update counter fields
1186
1187        // Get counter field storage IDs (old or new)
1188        final TreeSet<Integer> counterFieldStorageIds = new TreeSet<>();
1189        counterFieldStorageIds.addAll(oldType.counterFields.keySet());
1190        counterFieldStorageIds.addAll(newType.counterFields.keySet());
1191
1192        // Update counter fields
1193        for (int storageId : counterFieldStorageIds) {
1194
1195            // Get old and new counter fields having this storage ID
1196            final CounterField oldField = oldType.counterFields.get(storageId);
1197            final CounterField newField = newType.counterFields.get(storageId);
1198
1199            // Save old field values for version change notification
1200            final byte[] key = Field.buildKey(id, storageId);
1201            if (oldField != null && oldValueMap != null) {
1202                final byte[] oldValue = this.kvt.get(key);
1203                final long value = oldValue != null ? this.kvt.decodeCounter(oldValue) : 0;
1204                oldValueMap.put(storageId, value);
1205            }
1206
1207            // Remove old value if field has disappeared, otherwise leave alone
1208            if (newField == null)
1209                this.kvt.remove(key);
1210        }
1211
1212    //////// Update simple fields and corresponding index entries
1213
1214        // Get simple field storage IDs (old or new)
1215        final TreeSet<Integer> simpleFieldStorageIds = new TreeSet<>();
1216        simpleFieldStorageIds.addAll(oldType.simpleFields.keySet());
1217        simpleFieldStorageIds.addAll(newType.simpleFields.keySet());
1218
1219        // Update simple fields
1220        for (int storageId : simpleFieldStorageIds) {
1221
1222            // Get old and new simple fields having this storage ID
1223            final SimpleField<?> oldField = oldType.simpleFields.get(storageId);
1224            final SimpleField<?> newField = newType.simpleFields.get(storageId);
1225
1226            // Save old field values for version change notification
1227            final byte[] key = Field.buildKey(id, storageId);
1228            final byte[] oldValue = oldField != null ? this.kvt.get(key) : null;
1229            if (oldField != null && oldValueMap != null) {
1230                final byte[] bytes = oldValue != null ? oldValue : oldField.fieldType.getDefaultValue();
1231                final Object value = oldField.fieldType.read(new ByteReader(bytes));
1232                oldValueMap.put(storageId, value);
1233            }
1234
1235            // If field is being removed, discard the old value
1236            if (newField == null && oldValue != null)
1237                this.kvt.remove(key);
1238
1239            // Remove old index entry if index removed in new version
1240            if (oldField != null && oldField.indexed && (newField == null || !newField.indexed))
1241                this.kvt.remove(Transaction.buildSimpleIndexEntry(oldField, id, oldValue));
1242
1243            // Add new index entry if index added in new version
1244            if (newField != null && newField.indexed && (oldField == null || !oldField.indexed))
1245                this.kvt.put(Transaction.buildSimpleIndexEntry(newField, id, oldValue), ByteUtil.EMPTY);
1246        }
1247
1248    //////// Add composite index entries for newly added composite indexes
1249
1250        // Add index entries for composite indexes that are newly added
1251        for (CompositeIndex index : newType.compositeIndexes.values()) {
1252            if (!oldType.compositeIndexes.containsKey(index.storageId))
1253                this.kvt.put(this.buildCompositeIndexEntry(id, index), ByteUtil.EMPTY);
1254        }
1255
1256    //////// Update complex fields and corresponding index entries
1257
1258        // Get complex field storage IDs (old or new)
1259        final TreeSet<Integer> complexFieldStorageIds = new TreeSet<>();
1260        complexFieldStorageIds.addAll(oldType.complexFields.keySet());
1261        complexFieldStorageIds.addAll(newType.complexFields.keySet());
1262
1263        // Notes:
1264        //
1265        // - The only changes we support are sub-field changes that don't affect the corresponding StorageInfo's
1266        // - New complex fields do not need to be explicitly initialized because their initial state is to have zero KV pairs
1267        //
1268        for (int storageId : complexFieldStorageIds) {
1269
1270            // Get old and new complex fields having this storage ID
1271            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1272            final ComplexField<?> newField = newType.complexFields.get(storageId);
1273
1274            // If there is no old field, new field and any associated indexes are already initialized (i.e., they're empty)
1275            if (oldField == null)
1276                continue;
1277
1278            // Save old field's value
1279            if (oldField != null && oldValueMap != null)
1280                oldValueMap.put(storageId, oldField.getValueReadOnlyCopy(this, id));
1281
1282            // If field is being removed, delete old field content, otherwise check if index entries should be added/removed
1283            if (newField == null) {
1284                oldField.removeIndexEntries(this, id);
1285                oldField.deleteContent(this, id);
1286            } else
1287                newField.updateSubFieldIndexes(this, oldField, id);
1288        }
1289
1290    //////// Remove references that are no longer valid
1291
1292        // Unreference any reference fields that refer to no-longer-valid object types
1293        for (int storageId : simpleFieldStorageIds) {
1294
1295            // Get storage IDs that are no longer allowed, if any
1296            final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, storageId);
1297            if (removedObjectTypes == null)
1298                continue;
1299
1300            // Unreference field if needed
1301            final ReferenceField oldField = (ReferenceField)oldType.simpleFields.get(storageId);
1302            final ObjId ref = oldField.getValue(this, id);
1303            if (ref != null && removedObjectTypes.contains(ref.getStorageId()))
1304                this.writeSimpleField(id, oldField.storageId, null, false);
1305        }
1306
1307        // Unreference any reference sub-fields of complex fields that refer to no-longer-valid object types
1308        for (int storageId : complexFieldStorageIds) {
1309
1310            // Get old and new complex fields having this storage ID
1311            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1312            final ComplexField<?> newField = newType.complexFields.get(storageId);
1313            if (oldField == null || newField == null)
1314                continue;
1315            final List<? extends SimpleField<?>> oldSubFields = oldField.getSubFields();
1316            final List<? extends SimpleField<?>> newSubFields = newField.getSubFields();
1317            final int numSubFields = oldSubFields.size();
1318            assert numSubFields == newSubFields.size();
1319
1320            // Iterate over subfields
1321            for (int i = 0; i < numSubFields; i++) {
1322                final SimpleField<?> oldSubField = oldSubFields.get(i);
1323                final SimpleField<?> newSubField = newSubFields.get(i);
1324                assert oldSubField.storageId == newSubField.storageId;
1325                final int subStorageId = oldSubField.storageId;
1326
1327                // Get storage IDs that are no longer allowed, if any
1328                final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, subStorageId);
1329                if (removedObjectTypes == null)
1330                    continue;
1331
1332                // Unreference sub-fields as needed
1333                oldField.unreferenceRemovedObjectTypes(this, id, (ReferenceField)oldSubField, removedObjectTypes);
1334            }
1335        }
1336
1337    //////// Update object version and corresponding index entry
1338
1339        // Update object version
1340        ObjInfo.write(this, id, newVersion, info.isDeleteNotified());
1341
1342        // Update object version index entry
1343        this.kvt.remove(Database.buildVersionIndexKey(id, oldVersion));
1344        this.kvt.put(Database.buildVersionIndexKey(id, newVersion), ByteUtil.EMPTY);
1345
1346    //////// Notify listeners
1347
1348        // Lock down old field values map
1349        final NavigableMap<Integer, Object> readOnlyOldValuesMap = oldValueMap != null ?
1350          Maps.unmodifiableNavigableMap(oldValueMap) : null;
1351
1352        // Notify about version update
1353        for (VersionChangeListener listener : this.versionChangeListeners)
1354            listener.onVersionChange(this, id, oldVersion, newVersion, readOnlyOldValuesMap);
1355    }

1357    /**
1358     * Find storage ID's which are no longer allowed by a reference field when upgrading to the specified
1359     * schema version and therefore need to be scrubbed during the upgrade.
1360     */
1361    private TreeSet<Integer> findRemovedObjectTypes(ObjInfo info, SchemaVersion newVersion, int storageId) {
1362
1363        // Get old schema version
1364        final SchemaVersion oldVersion = info.getSchemaVersion();
1365
1366        // Get old and new object types
1367        final ObjType oldType = info.getObjType();
1368        final ObjType newType = newVersion.getObjType(info.getId().getStorageId());
1369
1370        // Get old and new reference fields having specified storage ID
1371        final ReferenceField oldField = oldType.referenceFields.get(storageId);
1372        final ReferenceField newField = newType.referenceFields.get(storageId);
1373        if (oldField == null || newField == null)
1374            return null;
1375
1376        // Check allowed storage IDs
1377        final SortedSet<Integer> newObjectTypes = newField.getObjectTypes();
1378        if (newObjectTypes == null)
1379            return null;                                                // new field can refer to any type in any schema version
1380        SortedSet<Integer> oldObjectTypes = oldField.getObjectTypes();
1381        if (oldObjectTypes == null)
1382            oldObjectTypes = this.schema.objTypeStorageIds;             // old field can refer to any type in any schema version
1383
1384        // Identify storage IDs which are were allowed by old field but are no longer allowed by new field
1385        final TreeSet<Integer> removedObjectTypes = new TreeSet<>(oldObjectTypes);
1386        removedObjectTypes.removeAll(newObjectTypes);
1387        return !removedObjectTypes.isEmpty() ? removedObjectTypes : null;
1388    }

1564    private synchronized void doWriteSimpleField(ObjId id, int storageId, final Object newObj, boolean updateVersion) {
1565
1566        // Get object info
1567        final ObjInfo info = this.getObjectInfo(id, updateVersion);
1568
1569        // Find field
1570        final SimpleField<?> field = info.getObjType().simpleFields.get(storageId);
1571        if (field == null)
1572            throw new UnknownFieldException(info.getObjType(), storageId, "simple field");
1573
1574        // Get new value
1575        final byte[] key = field.buildKey(id);
1576        final byte[] newValue = field.encode(newObj);
1577
1578        // Before setting the new value, read the old value if one of the following is true:
1579        //  - The field is being monitored -> we need to filter out "changes" that don't actuallly change anything
1580        //  - The field is indexed -> we need the old value so we can remove the old index entry
1581        // If neither of the above is true, then there's no need to read the old value.
1582        byte[] oldValue = null;
1583        if (field.indexed || field.compositeIndexMap != null || this.hasFieldMonitor(id, field)) {
1584
1585            // Get old value
1586            oldValue = this.kvt.get(key);
1587
1588            // Compare new to old value
1589            if (oldValue != null ? newValue != null && Arrays.equals(oldValue, newValue) : newValue == null)
1590                return;
1591        }
1592
1593        // Update value
1594        if (newValue != null)
1595            this.kvt.put(key, newValue);
1596        else
1597            this.kvt.remove(key);
1598
1599        // Update simple index, if any
1600        if (field.indexed) {
1601            this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, oldValue));
1602            this.kvt.put(Transaction.buildSimpleIndexEntry(field, id, newValue), ByteUtil.EMPTY);
1603        }
1604
1605        // Update affected composite indexes, if any
1606        if (field.compositeIndexMap != null) {
1607            for (Map.Entry<CompositeIndex, Integer> entry : field.compositeIndexMap.entrySet()) {
1608                final CompositeIndex index = entry.getKey();
1609                final int fieldIndexOffset = entry.getValue();
1610
1611                // Build old composite index entry
1612                final ByteWriter oldWriter = new ByteWriter();
1613                UnsignedIntEncoder.write(oldWriter, index.storageId);
1614                int fieldStart = -1;
1615                int fieldEnd = -1;
1616                for (SimpleField<?> otherField : index.fields) {
1617                    final byte[] otherValue;
1618                    if (otherField == field) {
1619                        fieldStart = oldWriter.getLength();
1620                        otherValue = oldValue;
1621                    } else
1622                        otherValue = this.kvt.get(otherField.buildKey(id));         // can be null (if field has default value)
1623                    oldWriter.write(otherValue != null ? otherValue : otherField.fieldType.getDefaultValue());
1624                    if (otherField == field)
1625                        fieldEnd = oldWriter.getLength();
1626                }
1627                assert fieldStart != -1;
1628                assert fieldEnd != -1;
1629                id.writeTo(oldWriter);
1630
1631                // Remove old composite index entry
1632                final byte[] oldIndexEntry = oldWriter.getBytes();
1633                this.kvt.remove(oldIndexEntry);
1634
1635                // Patch in new field value to create new composite index entry
1636                final ByteWriter newWriter = new ByteWriter(oldIndexEntry.length);
1637                newWriter.write(oldIndexEntry, 0, fieldStart);
1638                newWriter.write(newValue != null ? newValue : field.fieldType.getDefaultValue());
1639                newWriter.write(oldIndexEntry, fieldEnd, oldIndexEntry.length - fieldEnd);
1640
1641                // Add new composite index entry
1642                this.kvt.put(newWriter.getBytes(), ByteUtil.EMPTY);
1643            }
1644        }
1645
1646        // Notify monitors
1647        final Object oldObj = field.fieldType.read(new ByteReader(oldValue != null ? oldValue : field.fieldType.getDefaultValue()));
1648        this.addFieldChangeNotification(new SimpleFieldChangeNotifier(field, id) {
1649            @Override
1650            @SuppressWarnings("unchecked")
1651            public void notify(Transaction tx, SimpleFieldChangeListener listener, int[] path, NavigableSet<ObjId> referrers) {
1652                listener.onSimpleFieldChange(tx, this.id, (SimpleField<Object>)field, path, referrers, oldObj, newObj);
1653            }
1654        });
1655    }

2012    /**
2013     * Monitor for changes within this transaction of the value of the given field, as seen through a path of references.
2014     *
2015     * <p>
2016     * When the specified field is changed, a listener notification will be delivered for each referring object that
2017     * refers to the object containing the changed field through the specified path of references. Notifications are
2018     * delivered at the end of the mutation operation just prior to returning to the caller. If the listener method
2019     * performs additional mutation(s) which are themselves being listened for, those notifications will also be delivered
2020     * prior to the returning to the original caller. Therefore, care must be taken to avoid change notification dependency
2021     * loops when listeners can themselves mutate fields, to avoid infinite loops.
2022     * </p>
2023     *
2024     * <p>
2025     * If a non-null {@code types} is provided, then only objects whose types have one of the specified storage ID's
2026     * will trigger notifications to {@code listener}.
2027     * </p>
2028     *
2029     * <p>
2030     * A referring object may refer to the changed object through more than one actual path of references matching {@code path};
2031     * if so, it will still appear only once in the {@link NavigableSet} provided to the listener (this is of course required
2032     * by set semantics).
2033     * </p>
2034     *
2035     * <p>
2036     * Although the reference back-tracking algorithm does consolidate multiple paths between the same two objects,
2037     * be careful to avoid an explosion of notifications when objects in the {@code path} have a high degree of fan-in.
2038     * </p>
2039     *
2040     * <p>
2041     * When a {@link ReferenceField} in {@code path} also happens to be the field being changed, then there is ambiguity
2042     * about how the set of referring objects is calculated: does it use the field's value before or after the change?
2043     * This API guarantees that the answer is "after the change"; however, if another listener on the same field is
2044     * invoked before {@code listener} and mutates any reference field(s) in {@code path}, then whether that additional
2045     * change is also be included in the calculation is undefined.
2046     * </p>
2047     *
2048     * <p>
2049     * Therefore, for consistency, avoid changing any {@link ReferenceField} from within a listener callback when that
2050     * field is also in some other listener's reference path, and both listeners are watching the same field.
2051     * </p>
2052     *
2053     * @param storageId storage ID of the field to monitor
2054     * @param path path of reference fields (represented by storage IDs) through which to monitor field
2055     * @param types set of allowed storage IDs for the changed object, or null for no restriction
2056     * @param listener callback for notifications on changes in value
2057     * @throws UnknownFieldException if no {@link SimpleField} corresponding to {@code storageId} exists
2058     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2059     * @throws IllegalArgumentException if {@code storageId} refers to a sub-field of a complex field
2060     * @throws IllegalArgumentException if {@code path} or {@code listener} is null
2061     * @throws StaleTransactionException if this transaction is no longer usable
2062     */

2064      SimpleFieldChangeListener listener) {
2065        this.validateChangeListener(SimpleField.class, storageId, path, listener);
2066        this.getMonitorsForField(storageId, true).add(new FieldMonitor(storageId, path, types, listener));
2067    }

2314    private synchronized <V> V mutateAndNotify(Mutation<V> mutation) {
2315
2316        // Validate transaction
2317        if (this.stale)
2318            throw new StaleTransactionException(this);
2319        if (this.readOnly)
2320            throw new ReadOnlyTransactionException(this);
2321
2322        // If re-entrant invocation, we're already set up
2323        if (this.pendingNotifications.get() != null)
2324            return mutation.mutate();
2325
2326        // Set up pending report list, perform mutation, and then issue reports
2327        this.pendingNotifications.set(new TreeMap<Integer, ArrayList<FieldChangeNotifier>>());
2328        try {
2329            return mutation.mutate();
2330        } finally {
2331            try {
2332                final TreeMap<Integer, ArrayList<FieldChangeNotifier>> pendingNotificationMap = this.pendingNotifications.get();
2333                while (!pendingNotificationMap.isEmpty()) {
2334
2335                    // Get the next field with pending notifications
2336                    final Map.Entry<Integer, ArrayList<FieldChangeNotifier>> entry = pendingNotificationMap.pollFirstEntry();
2337                    final int storageId = entry.getKey();
2338
2339                    // For all pending notifications, back-track references and notify all field monitors for the field
2340                    for (FieldChangeNotifier notifier : entry.getValue()) {
2341                        assert notifier.getStorageId() == storageId;
2342                        final ArrayList<FieldMonitor> monitorList = new ArrayList<>(this.getMonitorsForField(storageId, false));
2343                        if (!monitorList.isEmpty())
2344                            this.notifyFieldMonitors(notifier, NavigableSets.singleton(notifier.getId()), monitorList, 0);
2345                    }
2346                }
2347            } finally {
2348                this.pendingNotifications.remove();
2349            }
2350        }
2351    }

2353    // Recursively back-track references along monitor paths and notify monitors when we reach the end (i.e., beginning)
2354    private void notifyFieldMonitors(FieldChangeNotifier notifier,
2355      NavigableSet<ObjId> objects, ArrayList<FieldMonitor> monitorList, int step) {
2356
2357        // Find the monitors for whom we have completed all the steps in their (inverse) path,
2358        // and group the remaining monitors by their next inverted reference path step.
2359        final HashMap<Integer, ArrayList<FieldMonitor>> remainingMonitorsMap = new HashMap<>();
2360        for (FieldMonitor monitor : monitorList) {
2361
2362            // On the first step, apply the monitor's type filter, if any
2363            if (step == 0 && monitor.types != null && !monitor.types.contains(notifier.getId().getBytes()))
2364                continue;
2365
2366            // Issue notification callback if we have back-tracked through the whole path
2367            if (monitor.path.length == step) {
2368                notifier.notify(this, monitor.listener, monitor.path, objects);
2369                continue;
2370            }
2371
2372            // Sort the unfinished monitors by their next (i.e., previous) reference field
2373            final int storageId = monitor.path[monitor.path.length - step - 1];
2374            ArrayList<FieldMonitor> remainingMonitors = remainingMonitorsMap.get(storageId);
2375            if (remainingMonitors == null) {
2376                remainingMonitors = new ArrayList<FieldMonitor>();
2377                remainingMonitorsMap.put(storageId, remainingMonitors);
2378            }
2379            remainingMonitors.add(monitor);
2380        }
2381
2382        // Invert references for each group of remaining monitors and recurse
2383        for (Map.Entry<Integer, ArrayList<FieldMonitor>> entry : remainingMonitorsMap.entrySet()) {
2384            final int storageId = entry.getKey();
2385
2386            // Gather all objects that refer to any object in our current "objects" set
2387            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2388            for (ObjId object : objects) {
2389                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(object);
2390                if (refs != null)
2391                    refsList.add(refs);
2392            }
2393
2394            // Recurse on the union of those objects
2395            if (!refsList.isEmpty())
2396                this.notifyFieldMonitors(notifier, NavigableSets.union(refsList), entry.getValue(), step + 1);
2397        }
2398    }

2402    /**
2403     * Find all objects that refer to any object in the given target set through the specified path of references.
2404     *
2405     * @param path path of one or more reference fields (represented by storage IDs) through which to reach the target objects
2406     * @param targetObjects target objects
2407     * @return set of objects that refer to the {@code targetObjects} via {@code path}
2408     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2409     * @throws IllegalArgumentException if {@code targetObjects} or {@code path} is null
2410     * @throws IllegalArgumentException if {@code path} is empty
2411     * @throws StaleTransactionException if this transaction is no longer usable
2412     */
2413    public NavigableSet<ObjId> invertReferencePath(int[] path, Iterable<ObjId> targetObjects) {
2414
2415        // Sanity check
2416        if (targetObjects == null)
2417            throw new IllegalArgumentException("null targetObjects");
2418        if (path == null)
2419            throw new IllegalArgumentException("null path");
2420        if (path.length == 0)
2421            throw new IllegalArgumentException("empty path");
2422
2423        // Verify all fields in the path are reference fields
2424        for (int storageId : path)
2425            this.schema.verifyStorageInfo(storageId, ReferenceFieldStorageInfo.class);
2426
2427        // Invert references in reverse order
2428        NavigableSet<ObjId> result = null;
2429        for (int i = path.length - 1; i >= 0; i--) {
2430            final int storageId = path[i];
2431
2432            // Gather all objects that refer to any object in our current target objects set
2433            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2434            for (ObjId id : targetObjects) {
2435                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(id);
2436                if (refs != null)
2437                    refsList.add(refs);
2438            }
2439            if (refsList.isEmpty())
2440                return NavigableSets.empty();
2441
2442            // Recurse on the union of those objects
2443            targetObjects = result = NavigableSets.union(refsList);
2444        }
2445
2446        // Done
2447        return result;
2448    }

2598    /**
2599     * Find all objects that refer to the given target object through the/any reference field with the specified
2600     * {@link DeleteAction}.
2601     *
2602     * <p>
2603     * Because different schema versions can have different {@link DeleteAction}'s configured for the
2604     * same field, we have to iterate through each schema version separately.
2605     * </p>
2606     *
2607     * @param target referred-to object
2608     * @param onDelete {@link DeleteAction} to match
2609     * @param fieldStorageId reference field storage ID, or -1 to match any reference field
2610     */
2611    private NavigableSet<ObjId> findReferrers(ObjId target, DeleteAction onDelete, int fieldStorageId) {
2612        final ArrayList<NavigableSet<ObjId>> refSets = new ArrayList<>();
2613        for (SchemaVersion schemaVersion : this.schema.versions.values()) {
2614
2615            // Check whether any object of this version exist; if not, skip
2616            final NavigableSet<ObjId> versionObjects = this.queryVersion().asMap().get(schemaVersion.versionNumber);
2617            if (versionObjects == null)
2618                continue;
2619
2620            // Find all reference fields with storage ID matching fieldStorageId (if not -1) and check them.
2621            // Do this separately for each such field in each object type because the fields may have different DeleteAction's.
2622            for (ObjType objType : schemaVersion.objTypeMap.values()) {
2623                for (ReferenceField field : Iterables.filter(objType.getFieldsAndSubFields(), ReferenceField.class)) {
2624
2625                    // Check delete action and field
2626                    if (field.onDelete != onDelete || (fieldStorageId != -1 && field.storageId != fieldStorageId))
2627                        continue;
2628
2629                    // Query index on this field, restricting to those only references coming from objType objects
2630                    final NavigableSet<ObjId> refs = this.queryIndex(field.storageId)
2631                      .filter(1, new KeyRanges(ObjId.getKeyRange(objType.storageId))).asMap().get(target);
2632                    if (refs == null)
2633                        continue;
2634
2635                    // Restrict further to the specific schema version
2636                    refSets.add(NavigableSets.intersection(versionObjects, refs));
2637                }
2638            }
2639        }
2640        return !refSets.isEmpty() ? NavigableSets.union(refSets) : NavigableSets.empty(FieldTypeRegistry.OBJ_ID);
2641    }
