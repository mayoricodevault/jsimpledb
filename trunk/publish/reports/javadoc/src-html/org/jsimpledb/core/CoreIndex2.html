008package org.jsimpledb.core;
009
010import java.util.NavigableSet;
011
012import org.jsimpledb.index.Index;
013import org.jsimpledb.index.Index2;
014import org.jsimpledb.kv.KeyFilter;
015import org.jsimpledb.tuple.Tuple2;
016import org.jsimpledb.tuple.Tuple3;
017
018/**
019 * Core API {@link Index} implementation representing a composite index on two fields.
020 *
021 * <p>
022 * Instances are immutable.
023 * </p>
024 *
025 * @param <V1> first index value type
026 * @param <V2> second index value type
027 * @param <T> index target type
028 */
029public class CoreIndex2<V1, V2, T> extends AbstractCoreIndex implements Index2<V1, V2, T> {
030
031// Constructors
032
033    CoreIndex2(Transaction tx, Index2View<V1, V2, T> indexView) {
034        super(tx, 3, indexView);
035    }
036
037// Methods
038
039    @Override
040    public CoreIndex2<V1, V2, T> filter(int index, KeyFilter filter) {
041        return new CoreIndex2<V1, V2, T>(this.tx, this.getIndex2View().filter(index, filter));
042    }
043
044    @SuppressWarnings("unchecked")
045    Index2View<V1, V2, T> getIndex2View() {
046        return (Index2View<V1, V2, T>)this.indexView;
047    }
048
049// Index2
050
051    @Override
052    public IndexSet<Tuple3<V1, V2, T>> asSet() {
053
054        // Get index view
055        final Index2View<V1, V2, T> iv = this.getIndex2View();
056
057        // Create field type for Tuple3<V1, V2, T>
058        final Tuple3FieldType<V1, V2, T> fieldType = new Tuple3FieldType<V1, V2, T>(
059          iv.getValue1Type(), iv.getValue2Type(), iv.getTargetType());
060
061        // Build set and apply filtering
062        IndexSet<Tuple3<V1, V2, T>> indexSet = new IndexSet<Tuple3<V1, V2, T>>(this.tx, fieldType, iv.prefixMode, iv.prefix);
063        if (iv.hasFilters())
064            indexSet = indexSet.filterKeys(new IndexKeyFilter(this.tx, iv, 3));
065
066        // Done
067        return indexSet;
068    }
069
070    @Override
071    public IndexMap<Tuple2<V1, V2>, NavigableSet<T>> asMap() {
072
073        // Get index view
074        final Index2View<V1, V2, T> iv = this.getIndex2View();
075
076        // Create new IndexView
077        final IndexView<Tuple2<V1, V2>, T> tupleIV = iv.asTuple2IndexView();
078
079        // Build map and apply filtering
080        IndexMap<Tuple2<V1, V2>, NavigableSet<T>> indexMap = new IndexMap.OfValues<Tuple2<V1, V2>, T>(this.tx, tupleIV);
081        if (tupleIV.hasFilters())
082            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, tupleIV, 1));
083
084        // Done
085        return indexMap;
086    }
087
088    @Override
089    public IndexMap<V1, Index<V2, T>> asMapOfIndex() {
090
091        // Get index view
092        final Index2View<V1, V2, T> iv = this.getIndex2View();
093
094        // Build map and apply filtering
095        IndexMap<V1, Index<V2, T>> indexMap = new IndexMap.OfIndex<V1, V2, T>(this.tx, iv);
096        if (iv.hasFilters())
097            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, iv, 1));
098
099        // Done
100        return indexMap;
101    }
102
103    @Override
104    public CoreIndex<V1, V2> asIndex() {
105        return new CoreIndex<V1, V2>(this.tx, this.getIndex2View().asIndexView());
106    }
107}
