008package org.jsimpledb.core;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeToken;
012
013/**
014 * A {@link FieldType} implementation for any Java type that can be encoded uniquely as a {@link String}.
015 * A {@link Converter} is used to convert between native and {@link String} forms.
016 *
017 * <p>
018 * This class provides a convenient way to implement custom {@link FieldType}s.
019 * Null values are supported and null is the default value. This type will sort instances according to
020 * the lexicographical sort order of their {@link String} encodings; null will sort last.
021 * </p>
022 *
023 * @param <T> The associated Java type
024 */
025public class StringEncodedType<T> extends NullSafeType<T> {
026
027    /**
028     * Convenience constructor. Uses the name of the {@code type} as this {@link FieldType}'s type name.
029     *
030     * @param type represented Java type
031     * @param signature binary encoding signature (in this case, {@link String} encoding signature)
032     * @param converter converts between native form and {@link String} form
033     * @throws IllegalArgumentException if {@code converter} does not convert null to null
034     * @throws IllegalArgumentException if any parameter is null
035     */
036    public StringEncodedType(Class<T> type, long signature, Converter<T, String> converter) {
037        this(type.getName(), type, signature, converter);
038    }
039
040    /**
041     * Primary constructor.
042     *
043     * @param name the name for this {@link FieldType}
044     * @param type represented Java type
045     * @param signature binary encoding signature (in this case, {@link String} encoding signature)
046     * @param converter converts between native form and {@link String} form
047     * @throws IllegalArgumentException if {@code converter} does not convert null to null
048     * @throws IllegalArgumentException if any parameter is null
049     */
050    public StringEncodedType(String name, Class<T> type, long signature, Converter<T, String> converter) {
051        super(new StringConvertedType<T>(name, TypeToken.of(type), signature, converter));
052    }
053}
