008package org.jsimpledb.core;
009
010import com.google.common.base.Converter;
011
012/**
013 * A {@link FieldType} implementation for any Java type that can be encoded uniquely as a {@link String}.
014 * A {@link Converter} is used to convert between native and {@link String} forms.
015 *
016 * <p>
017 * This class provides a convenient way to implement custom {@link FieldType}s.
018 * Null values are supported and null is the default value.
019 * </p>
020 *
021 * @param <T> The associated Java type
022 */
023public class StringEncodedType<T> extends NullSafeType<T> {
024
025    /**
026     * Convenience constructor. Uses the simple name of the {@code type} as this {@link FieldType}'s type name.
027     *
028     * @param type represented Java type
029     * @param converter converts between native form and {@link String} form
030     * @throws IllegalArgumentException if any parameter is null
031     */
032    public StringEncodedType(Class<T> type, Converter<T, String> converter) {
033        this(type, type.getSimpleName(), converter);
034    }
035
036    /**
037     * Primary constructor.
038     *
039     * @param type represented Java type
040     * @param name the name for this {@link FieldType}
041     * @param converter converts between native form and {@link String} form
042     * @throws IllegalArgumentException if any parameter is null
043     */
044    public StringEncodedType(Class<T> type, String name, Converter<T, String> converter) {
045        super(new StringConvertedType<T>(type, name, converter));
046    }
047}
