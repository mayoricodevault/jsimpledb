008package org.jsimpledb.core;
009
010import java.security.SecureRandom;
011import java.util.regex.Pattern;
012
013import org.jsimpledb.kv.KeyRange;
014import org.jsimpledb.util.ByteReader;
015import org.jsimpledb.util.ByteUtil;
016import org.jsimpledb.util.ByteWriter;
017import org.jsimpledb.util.UnsignedIntEncoder;
018
019/**
020 * Object IDs. Instances identify individual {@link Database} objects.
021 */
022public class ObjId implements Comparable<ObjId> {
023
024    /**
025     * The number of bytes in the binary encoding of an {@link ObjId}.
026     */
027    public static final int NUM_BYTES = 8;
028
029    /**
030     * Regular expression that matches the string encoding of an {@link ObjId}.
031     */
032    public static final Pattern PATTERN = Pattern.compile("\\p{XDigit}{" + (NUM_BYTES * 2) + "}");
033
034    private static final ThreadLocal<SecureRandom> RANDOM = new ThreadLocal<SecureRandom>() {
035        @Override
036        protected SecureRandom initialValue() {
037            return new SecureRandom();
038        }
039    };
040
041    private final long value;
042
043// Constructors
044
045    /**
046     * Create a new, random instance with the given storage ID.
047     *
048     * @param storageId storage ID, must be greater than zero
049     * @throws IllegalArgumentException if {@code storageId} is zero or negative
050     */
051    public ObjId(int storageId) {
052        this(ObjId.buildRandom(storageId));
053    }
054
055    /**
056     * Constructor that parses a string previously returned by {@link #toString}.
057     *
058     * @param string string encoding of an instance
059     * @throws IllegalArgumentException if {@code string} is invalid
060     */
061    public ObjId(String string) {
062        this(ObjId.parseString(string));
063    }
064
065    /**
066     * Constructor that reads an encoded instance from the given {@link ByteReader}.
067     *
068     * @param reader input for binary encoding of an instance
069     * @throws IllegalArgumentException if {@code reader} contains invalid data
070     */
071    public ObjId(ByteReader reader) {
072        if (reader == null)
073            throw new IllegalArgumentException("null reader");
074        this.value = ByteUtil.readLong(reader);
075        this.validateStorageId();
076    }
077
078    /**
079     * Constructor using a long value previously returned by {@link #asLong}.
080     *
081     * @param value long encoding of an instance
082     * @throws IllegalArgumentException if {@code value} is invalid
083     */
084    public ObjId(long value) {
085        this.value = value;
086        this.validateStorageId();
087    }
088
089    private void validateStorageId() {
090        final int storageId;
091        try {
092            storageId = this.getStorageId();
093        } catch (IllegalArgumentException e) {
094            throw new IllegalArgumentException("invalid object ID", e);
095        }
096        if (storageId <= 0)
097            throw new IllegalArgumentException("invalid object ID containing storage ID " + storageId);
098    }
099
100// Methods
101
102    /**
103     * Get the storage ID associated with this instance.
104     */
105    public int getStorageId() {
106        return UnsignedIntEncoder.read(new ByteReader(this.getBytes()));
107    }
108
109    /**
110     * Get the binary encoding of this instance.
111     */
112    public byte[] getBytes() {
113        final ByteWriter writer = new ByteWriter(NUM_BYTES);
114        this.writeTo(writer);
115        return writer.getBytes();
116    }
117
118    /**
119     * Get this instance encoded as a {@code long} value.
120     */
121    public long asLong() {
122        return this.value;
123    }
124
125    /**
126     * Write the binary encoding of this instance to the given output.
127     */
128    public void writeTo(ByteWriter writer) {
129        ByteUtil.writeLong(writer, this.value);
130    }
131
132    /**
133     * Get the smallest (i.e., first) instance having the given storage ID.
134     *
135     * @param storageId storage ID, must be greater than zero
136     * @return smallest instance with storage ID {@code storageId} (inclusive)
137     * @throws IllegalArgumentException if {@code storageId} is zero or negative
138     */
139    public static ObjId getMin(int storageId) {
140        return ObjId.getFill(storageId, 0x00);
141    }
142
143    /**
144     * Get the largest (i.e., last) instance having the given storage ID.
145     *
146     * @param storageId storage ID, must be greater than zero
147     * @return largest instance with storage ID {@code storageId} (inclusive)
148     * @throws IllegalArgumentException if {@code storageId} is zero or negative
149     */
150    public static ObjId getMax(int storageId) {
151        return ObjId.getFill(storageId, 0xff);
152    }
153
154    /**
155     * Get the {@link KeyRange} containing all object IDs with the given storage ID.
156     *
157     * @param storageId storage ID, must be greater than zero
158     * @return {@link KeyRange} containing all object IDs having the specified type
159     * @throws IllegalArgumentException if {@code storageId} is zero or negative
160     */
161    public static KeyRange getKeyRange(int storageId) {
162        if (storageId <= 0)
163            throw new IllegalArgumentException("invalid storageId " + storageId);
164        final ByteWriter writer = new ByteWriter(NUM_BYTES);
165        UnsignedIntEncoder.write(writer, storageId);
166        return KeyRange.forPrefix(writer.getBytes());
167    }
168
169    private static ObjId getFill(int storageId, int value) {
170        if (storageId <= 0)
171            throw new IllegalArgumentException("invalid storage ID " + storageId);
172        final ByteWriter writer = new ByteWriter(NUM_BYTES);
173        UnsignedIntEncoder.write(writer, storageId);
174        for (int remain = NUM_BYTES - writer.getLength(); remain > 0; remain--)
175            writer.writeByte(value);
176        return new ObjId(new ByteReader(writer));
177    }
178
179// Object
180
181    /**
182     * Encode this instance as a string.
183     */
184    @Override
185    public String toString() {
186        final ByteWriter writer = new ByteWriter(NUM_BYTES);
187        this.writeTo(writer);
188        final byte[] buf = writer.getBytes();
189        final char[] result = new char[NUM_BYTES * 2];
190        int off = 0;
191        for (int i = 0; i < buf.length; i++) {
192            int b = buf[i];
193            result[off++] = Character.forDigit((b >> 4) & 0x0f, 16);
194            result[off++] = Character.forDigit(b & 0x0f, 16);
195        }
196        return new String(result);
197    }
198
199    @Override
200    public boolean equals(Object obj) {
201        if (obj == this)
202            return true;
203        if (obj == null || obj.getClass() != this.getClass())
204            return false;
205        final ObjId that = (ObjId)obj;
206        return this.value == that.value;
207    }
208
209    @Override
210    public int hashCode() {
211        return (int)(this.value >> 32) ^ (int)this.value;
212    }
213
214// Comparable
215
216    @Override
217    public int compareTo(ObjId that) {
218        return ByteUtil.compare(this.getBytes(), that.getBytes());
219    }
220
221// Internal methods
222
223    private static ByteReader buildRandom(int storageId) {
224        if (storageId <= 0)
225            throw new IllegalArgumentException("invalid storage ID " + storageId);
226        final ByteWriter writer = new ByteWriter(NUM_BYTES);
227        UnsignedIntEncoder.write(writer, storageId);
228        final byte[] randomPart = new byte[NUM_BYTES - writer.getLength()];
229        ObjId.RANDOM.get().nextBytes(randomPart);
230        writer.write(randomPart);
231        return new ByteReader(writer);
232    }
233
234    private static ByteReader parseString(String string) {
235        if (string == null)
236            throw new IllegalArgumentException("null string");
237        if (string.length() != NUM_BYTES * 2)
238            throw new IllegalArgumentException("invalid object ID `" + string + "'");
239        final byte[] buf = new byte[NUM_BYTES];
240        int off = 0;
241        for (int i = 0; i < buf.length; i++) {
242            final int digit1 = Character.digit(string.charAt(off++), 16);
243            final int digit2 = Character.digit(string.charAt(off++), 16);
244            if (digit1 == -1 || digit2 == -1)
245                throw new IllegalArgumentException("invalid object ID `" + string + "'");
246            buf[i] = (byte)((digit1 << 4) | digit2);
247        }
248        return new ByteReader(buf);
249    }
250}
