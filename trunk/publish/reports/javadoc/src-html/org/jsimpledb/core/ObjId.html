008package org.jsimpledb.core;
009
010import java.security.SecureRandom;
011import java.util.regex.Pattern;
012
013import org.jsimpledb.util.ByteReader;
014import org.jsimpledb.util.ByteUtil;
015import org.jsimpledb.util.ByteWriter;
016import org.jsimpledb.util.UnsignedIntEncoder;
017
018/**
019 * Object IDs. Instances identify individual {@link Database} objects.
020 */
021public class ObjId implements Comparable<ObjId> {
022
023    /**
024     * The number of bytes in the binary encoding of an {@link ObjId}.
025     */
026    public static final int NUM_BYTES = 8;
027
028    /**
029     * Regular expression that matches the string encoding of an {@link ObjId}.
030     */
031    public static final Pattern PATTERN = Pattern.compile("\\p{XDigit}{" + (NUM_BYTES * 2) + "}");
032
033    private static final ThreadLocal<SecureRandom> RANDOM = new ThreadLocal<SecureRandom>() {
034        @Override
035        protected SecureRandom initialValue() {
036            return new SecureRandom();
037        }
038    };
039
040    private final long value;
041
042    /**
043     * Create a new, random instance with the given storage ID.
044     *
045     * @param storageId storage ID, must be greater than zero
046     * @throws IllegalArgumentException if {@code storageId} is zero or negative
047     */
048    public ObjId(int storageId) {
049        this(ObjId.buildRandom(storageId));
050    }
051
052    /**
053     * Constructor that parses a string previously returned by {@link #toString}.
054     *
055     * @param string string encoding of an instance
056     * @throws IllegalArgumentException if {@code string} is invalid
057     */
058    public ObjId(String string) {
059        this(ObjId.parseString(string));
060    }
061
062    /**
063     * Constructor that reads an encoded instance from the given {@link ByteReader}.
064     *
065     * @param reader input for binary encoding of an instance
066     * @throws IllegalArgumentException if {@code reader} contains invalid data
067     */
068    public ObjId(ByteReader reader) {
069        if (reader == null)
070            throw new IllegalArgumentException("null reader");
071        this.value = ByteUtil.readLong(reader);
072        final int storageId;
073        try {
074            storageId = this.getStorageId();
075        } catch (IllegalArgumentException e) {
076            throw new IllegalArgumentException("invalid object ID", e);
077        }
078        if (storageId <= 0)
079            throw new IllegalArgumentException("invalid object ID containing storage ID " + storageId);
080    }
081
082    /**
083     * Get the storage ID associated with this instance.
084     */
085    public int getStorageId() {
086        return UnsignedIntEncoder.read(new ByteReader(this.getBytes()));
087    }
088
089    /**
090     * Get the binary encoding of this instance.
091     */
092    public byte[] getBytes() {
093        final ByteWriter writer = new ByteWriter(NUM_BYTES);
094        this.writeTo(writer);
095        return writer.getBytes();
096    }
097
098    /**
099     * Get this instance encoded as a {@code long} value.
100     */
101    public long asLong() {
102        return this.value;
103    }
104
105    /**
106     * Write the binary encoding of this instance to the given output.
107     */
108    public void writeTo(ByteWriter writer) {
109        ByteUtil.writeLong(writer, this.value);
110    }
111
112    static ObjId getMin(int storageId) {
113        if (storageId <= 0)
114            throw new IllegalArgumentException("invalid storage ID " + storageId);
115        final ByteWriter writer = new ByteWriter(NUM_BYTES);
116        UnsignedIntEncoder.write(writer, storageId);
117        for (int remain = NUM_BYTES - writer.getLength(); remain > 0; remain--)
118            writer.writeByte(0);
119        return new ObjId(new ByteReader(writer));
120    }
121
122// Object
123
124    /**
125     * Encode this instance as a string.
126     */
127    @Override
128    public String toString() {
129        final ByteWriter writer = new ByteWriter(NUM_BYTES);
130        this.writeTo(writer);
131        final byte[] buf = writer.getBytes();
132        final char[] result = new char[NUM_BYTES * 2];
133        int off = 0;
134        for (int i = 0; i < buf.length; i++) {
135            int b = buf[i];
136            result[off++] = Character.forDigit((b >> 4) & 0x0f, 16);
137            result[off++] = Character.forDigit(b & 0x0f, 16);
138        }
139        return new String(result);
140    }
141
142    @Override
143    public boolean equals(Object obj) {
144        if (obj == this)
145            return true;
146        if (obj == null || obj.getClass() != this.getClass())
147            return false;
148        final ObjId that = (ObjId)obj;
149        return this.value == that.value;
150    }
151
152    @Override
153    public int hashCode() {
154        return (int)(this.value >> 32) ^ (int)this.value;
155    }
156
157// Comparable
158
159    @Override
160    public int compareTo(ObjId that) {
161        return ByteUtil.compare(this.getBytes(), that.getBytes());
162    }
163
164// Internal methods
165
166    private static ByteReader buildRandom(int storageId) {
167        if (storageId <= 0)
168            throw new IllegalArgumentException("invalid storage ID " + storageId);
169        final ByteWriter writer = new ByteWriter(NUM_BYTES);
170        UnsignedIntEncoder.write(writer, storageId);
171        final byte[] randomPart = new byte[NUM_BYTES - writer.getLength()];
172        ObjId.RANDOM.get().nextBytes(randomPart);
173        writer.write(randomPart);
174        return new ByteReader(writer);
175    }
176
177    private static ByteReader parseString(String string) {
178        if (string == null)
179            throw new IllegalArgumentException("null string");
180        if (string.length() != NUM_BYTES * 2)
181            throw new IllegalArgumentException("invalid object ID `" + string + "'");
182        final byte[] buf = new byte[NUM_BYTES];
183        int off = 0;
184        for (int i = 0; i < buf.length; i++) {
185            final int digit1 = Character.digit(string.charAt(off++), 16);
186            final int digit2 = Character.digit(string.charAt(off++), 16);
187            if (digit1 == -1 || digit2 == -1)
188                throw new IllegalArgumentException("invalid object ID `" + string + "'");
189            buf[i] = (byte)((digit1 << 4) | digit2);
190        }
191        return new ByteReader(buf);
192    }
193}
