041/**
042 * A JSimpleDB {@link Database} transaction.
043 *
044 * <p>
045 * Note: this is the lower level, core API for {@link org.jsimpledb.JSimpleDB}. In most cases this API
046 * will only be used indirectly through the higher level {@link org.jsimpledb.JSimpleDB}, {@link org.jsimpledb.JTransaction},
047 * and {@link org.jsimpledb.JObject} APIs.
048 * </p>
049 *
050 * <p>
051 * Methods in this class can be divided into the following categories:
052 * </p>
053 *
054 * <p>
055 * <b>Transaction Meta-Data</b>
056 * <ul>
057 *  <li>{@link #getDatabase getDatabase()} - Get the associated {@link Database}</li>
058 *  <li>{@link #getSchema getSchema()} - Get the database {@link Schema}, as seen by this transaction</li>
059 *  <li>{@link #getSchemaVersion() getSchemaVersion()} - Get the {@link SchemaVersion} that will be used by this transaction</li>
060 *  <li>{@link #deleteSchemaVersion deleteSchemaVersion()} - Delete a schema version that is no longer being used</li>
061 * </ul>
062 * </p>
063 *
064 * <p>
065 * <b>Transaction Lifecycle</b>
066 * <ul>
067 *  <li>{@link #commit commit()} - Commit transaction</li>
068 *  <li>{@link #rollback rollback()} - Roll back transaction</li>
069 *  <li>{@link #isValid isValid()} - Test transaction validity</li>
070 *  <li>{@link #setTimeout setTimeout()} - Set transaction timeout</li>
071 *  <li>{@link #setReadOnly setReadOnly()} - Set transaction to read-only</li>
072 *  <li>{@link #setRollbackOnly setRollbackOnly()} - Set transaction for rollack only</li>
073 *  <li>{@link #addCallback addCallback()} - Register a {@link Callback} on transaction completion</li>
074 *  <li>{@link #createSnapshotTransaction createSnapshotTransaction()} - Create a empty, in-memory copy of this transaction</li>
075 * </ul>
076 * </p>
077 *
078 * <p>
079 * <b>Object Lifecycle</b>
080 * <ul>
081 *  <li>{@link #create(int) create()} - Create a database object</li>
082 *  <li>{@link #delete delete()} - Delete a database object</li>

084 *  <li>{@link #copy copy()} - Copy an object's fields onto another object in a (possibly) different transaction</li>
085 *  <li>{@link #addCreateListener addCreateListener()} - Register a {@link CreateListener} for notifications about new objects</li>
086 *  <li>{@link #removeCreateListener removeCreateListener()} - Unregister a {@link CreateListener}</li>

091 * </p>
092 *

234     * Delete a schema version that is no longer being used. There must be no objects with the given version
235     * in the database, and {@code version} must not be the version being used by this transaction.
236     *
237     * @param version schema version to remove
238     * @return true if schema version was found and removed, false if schema version does not exist in database
239     * @throws InvalidSchemaException if one or more objects with schema version {@code version} still exist
240     * @throws InvalidSchemaException if {@code version} is equal to this transaction's version
241     * @throws IllegalArgumentException if {@code version} is zero or negative
242     * @throws StaleTransactionException if this transaction is no longer usable
243     */
244    public synchronized boolean deleteSchemaVersion(int version) {
245
246        // Sanity check
247        if (version <= 0)
248            throw new IllegalArgumentException("invalid schema version " + version);
249        if (version == this.version.getVersionNumber())
250            throw new InvalidSchemaException("version " + version + " is this transaction's version");
251        if (this.stale)
252            throw new StaleTransactionException(this);
253        if (this.queryVersion().asMap().containsKey(version))
254            throw new InvalidSchemaException("one or more version " + version + " objects still exist in database");
255
256        // Delete schema version
257        if (!this.schema.deleteVersion(version))
258            return false;
259        this.db.deleteSchema(this.kvt, version);
260        return true;
261    }

285    /**
286     * Commit this transaction.
287     *
288     * @throws StaleTransactionException if this transaction is no longer usable
289     * @throws RetryTransactionException from {@link KVTransaction#commit KVTransaction.commit()}
290     * @throws RollbackOnlyTransactionException if this instance has been {@linkplain #setRollbackOnly marked} rollback only;
291     *  this instance will be automatically rolled back
292     */
293    public synchronized void commit() {
294
295        // Sanity check
296        if (this.stale)
297            throw new StaleTransactionException(this);
298
299        // Rollback only?
300        if (this.rollbackOnly) {
301            this.log.info("commit() invoked on transaction " + this + " marked rollback-only, rolling back");
302            this.rollback();
303            throw new RollbackOnlyTransactionException(this);
304        }
305        this.stale = true;
306
307        // Do before completion callbacks
308        this.log.debug("commit() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
309        Callback failedCallback = null;
310        try {
311            for (Callback callback : this.callbacks) {
312                failedCallback = callback;
313                callback.beforeCommit(this.readOnly);
314            }
315            failedCallback = null;
316        } finally {
317            if (failedCallback != null) {
318                this.log.warn("error invoking beforeCommit() method on transaction "
319                  + this + " callback " + failedCallback + ", rolling back");
320            }
321            this.triggerBeforeCompletion();
322            if (failedCallback != null) {
323                try {
324                    try {
325                        this.kvt.rollback();
326                    } catch (KVTransactionException e) {
327                        // ignore
328                    }
329                } finally {
330                    this.triggerAfterCompletion(false);
331                }
332            }
333        }
334
335        // Commit KVTransaction and trigger after completion callbacks
336        failedCallback = null;
337        try {
338            this.kvt.commit();
339            for (Callback callback : this.callbacks) {
340                failedCallback = callback;
341                callback.afterCommit();
342            }
343            failedCallback = null;
344        } finally {
345            if (failedCallback != null)
346                this.log.warn("error invoking afterCommit() method on transaction " + this + " callback " + failedCallback);
347            this.triggerAfterCompletion(true);
348        }
349    }

351    /**
352     * Roll back this transaction.
353     *
354     * @throws StaleTransactionException if this transaction is no longer usable
355     */
356    public synchronized void rollback() {
357
358        // Sanity check
359        if (this.stale)
360            throw new StaleTransactionException(this);
361        this.stale = true;
362        this.log.debug("rollback() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
363
364        // Do before completion callbacks
365        this.triggerBeforeCompletion();
366
367        // Roll back KVTransaction and trigger after completion callbacks
368        try {
369            this.kvt.rollback();
370        } finally {
371            this.triggerAfterCompletion(false);
372        }
373    }

375    private /*synchronized*/ void triggerBeforeCompletion() {
376        for (Callback callback : this.callbacks) {
377            try {
378                callback.beforeCompletion();
379            } catch (Throwable t) {
380                this.log.error("error invoking beforeCompletion() method on transaction callback " + callback, t);
381            }
382        }
383    }
384
385    private /*synchronized*/ void triggerAfterCompletion(boolean committed) {
386        for (Callback callback : this.callbacks) {
387            try {
388                callback.afterCompletion(committed);
389            } catch (Throwable t) {
390                this.log.error("error invoking afterCompletion() method on transaction callback " + callback, t);
391            }
392        }
393    }

619    private /*synchronized*/ ObjId generateIdValidated(int storageId) {
620
621        // Create a new, unique key
622        final ByteWriter keyWriter = new ByteWriter();
623        for (int attempts = 0; attempts < MAX_GENERATED_KEY_ATTEMPTS; attempts++) {
624            final ObjId id = new ObjId(storageId);
625            id.writeTo(keyWriter);
626            if (this.kvt.get(keyWriter.getBytes()) == null)
627                return id;
628            keyWriter.reset(0);
629        }
630
631        // Give up
632        throw new DatabaseException("could not find a new, unused object ID after "
633          + MAX_GENERATED_KEY_ATTEMPTS + " attempts; is our source of randomness truly random?");
634    }

636    private synchronized void initialize(ObjId id, ObjType objType) {
637
638        // Sanity check
639        if (this.stale)
640            throw new StaleTransactionException(this);
641        if (this.readOnly)
642            throw new ReadOnlyTransactionException(this);
643
644        // Write object meta-data
645        ObjInfo.write(this, id, objType.version.versionNumber, false);
646
647        // Write object version index entry
648        this.kvt.put(Database.buildVersionIndexKey(id, objType.version.versionNumber), ByteUtil.EMPTY);
649
650        // Initialize counters to zero
651        if (!objType.counterFields.isEmpty()) {
652            final CountingKVStore ckv = this.getCountingKVTransaction();
653            for (CounterField field : objType.counterFields.values())
654                ckv.put(field.buildKey(id), ckv.encodeCounter(0));
655        }
656
657        // Write simple field index entries
658        for (SimpleField<?> field : objType.simpleFields.values()) {
659            if (field.indexed)
660                this.kvt.put(Transaction.buildSimpleIndexEntry(field, id, null), ByteUtil.EMPTY);
661        }
662
663        // Write composite index entries
664        for (CompositeIndex index : objType.compositeIndexes.values())
665            this.kvt.put(Transaction.buildDefaultCompositeIndexEntry(id, index), ByteUtil.EMPTY);
666
667        // Notify listeners
668        for (CreateListener listener : this.createListeners.toArray(new CreateListener[this.createListeners.size()]))
669            listener.onCreate(this, id);
670    }

672    /**
673     * Delete an object. Does nothing if object does not exist (e.g., has already been deleted).
674     *
675     * <p>
676     * This method does <i>not</i> change the object's schema version if it has a different version that this transaction.
677     * </p>
678     *
679     * @param id object ID of the object to delete
680     * @return true if object was found and deleted, false if object was not found,
681     *  or if {@code id} specifies an unknown object type
682     * @throws ReferencedObjectException if the object is referenced by some other object
683     *  through a reference field configured for {@link DeleteAction#EXCEPTION}
684     * @throws IllegalArgumentException if {@code id} is null
685     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
686     * @throws StaleTransactionException if this transaction is no longer usable
687     */
688    public synchronized boolean delete(ObjId id) {
689
690        // Sanity check
691        if (id == null)
692            throw new IllegalArgumentException("null id");
693        if (this.stale)
694            throw new StaleTransactionException(this);
695        if (this.readOnly)
696            throw new ReadOnlyTransactionException(this);
697
698        // Get object info
699        try {
700            this.getObjectInfo(id, false);
701        } catch (DeletedObjectException | UnknownTypeException e) {
702            return false;
703        }
704
705        // Handle recurive DeleteAction.DELETE without hogging Java stack
706        final ArrayList<ObjId> deletables = new ArrayList<>(1);
707        deletables.add(id);
708        boolean found = false;
709        while (true) {
710            final int size = deletables.size();
711            if (size == 0)
712                break;
713            id = deletables.remove(size - 1);
714            found |= this.doDelete(id, deletables);
715        }
716
717        // Done
718        return found;
719    }
720
721    private synchronized boolean doDelete(final ObjId id, List<ObjId> deletables) {
722
723        // Loop here to handle any mutations within delete notification listener callbacks
724        ObjInfo info;
725        while (true) {
726
727            // Get object info
728            try {
729                info = this.getObjectInfo(id, false);
730            } catch (DeletedObjectException e) {                    // possibly due to a cycle of DeleteAction.DELETE references
731                return false;
732            } catch (UnknownTypeException e) {
733                throw new InconsistentDatabaseException("encountered reference with unknown type during delete cascade: " + id, e);
734            }
735
736            // Determine if any EXCEPTION reference fields refer to the object (from some other object); if so, throw exception
737            for (ReferenceFieldStorageInfo fieldInfo :
738              Iterables.filter(this.schema.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
739                for (ObjId referrer : this.findReferrers(id, DeleteAction.EXCEPTION, fieldInfo.storageId)) {
740                    if (!referrer.equals(id))
741                        throw new ReferencedObjectException(id, referrer, fieldInfo.storageId);
742                }
743            }
744
745            // Do we need to issue delete notifications for the object being deleted?
746            if (info.isDeleteNotified() || this.deleteListeners.isEmpty())
747                break;
748
749            // Issue delete notifications and retry
750            ObjInfo.write(this, id, info.getVersionNumber(), true);
751            for (DeleteListener listener : this.deleteListeners.toArray(new DeleteListener[this.deleteListeners.size()]))
752                listener.onDelete(this, id);
753        }
754
755        // Actually delete the object
756        this.deleteObjectData(info);
757
758        // Find all UNREFERENCE references and unreference them
759        for (ReferenceFieldStorageInfo fieldInfo :
760          Iterables.filter(this.schema.storageInfos.values(), ReferenceFieldStorageInfo.class)) {
761            final NavigableSet<ObjId> referrers = this.findReferrers(id, DeleteAction.UNREFERENCE, fieldInfo.storageId);
762            if (fieldInfo.isSubField()) {
763                final ComplexFieldStorageInfo<?> superFieldInfo = this.schema.verifyStorageInfo(
764                  fieldInfo.superFieldStorageId, ComplexFieldStorageInfo.class);
765                superFieldInfo.unreferenceAll(this, fieldInfo.storageId, id, referrers);
766            } else {
767                final int storageId = fieldInfo.storageId;
768                for (ObjId referrer : referrers)
769                    this.writeSimpleField(referrer, storageId, null, false);
770            }
771        }
772
773        // Find all DELETE references and mark the containing object for deletion (caller will call us back to actually delete)
774        for (ObjId referrer : this.findReferrers(id, DeleteAction.DELETE, -1))
775            deletables.add(referrer);
776
777        // Done
778        return true;
779    }

781    // Delete all of an object's data
782    private void deleteObjectData(ObjInfo info) {
783
784        // Delete object's simple field index entries
785        final ObjId id = info.getId();
786        final ObjType type = info.getObjType();
787        for (SimpleField<?> field : type.simpleFields.values()) {
788            if (field.indexed)
789                this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, this.kvt.get(field.buildKey(id))));
790        }
791
792        // Delete object's composite index entries
793        for (CompositeIndex index : type.compositeIndexes.values())
794            this.kvt.remove(this.buildCompositeIndexEntry(id, index));
795
796        // Delete object's complex field index entries
797        for (ComplexField<?> field : type.complexFields.values())
798            field.removeIndexEntries(this, id);
799
800        // Delete object meta-data and all field content
801        final byte[] minKey = info.getId().getBytes();
802        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
803        this.kvt.removeRange(minKey, maxKey);
804
805        // Delete object schema version entry
806        this.kvt.remove(Database.buildVersionIndexKey(id, info.getVersionNumber()));
807    }

831    /**
832     * Copy all of an object's fields onto a target object in a (possibly) different transaction, replacing any previous values.
833     *
834     * <p>
835     * The schema version associated with the {@code source} object must be identical in this transaction and {@code dest}.
836     * Only the object's fields are copied; any other objects they reference are not copied. If the {@code target} object
837     * does not exist in {@code dest}, it will be created first (and {@link CreateListener}s notified); if {@code target}
838     * does exist in {@code dest}, it's schema version will be upgraded if necessary to match {@code source} (and any registered
839     * {@link VersionChangeListener}s notified).
840     * (Meaningful) changes to {@code target}'s fields generate change listener notifications.
841     * </p>
842     *
843     * <p>
844     * Note: if two threads attempt to copy objects between the same two transactions at the same time but in opposite directions,
845     * deadlock could result.
846     * </p>
847     *
848     * <p>
849     * If {@code dest} is this instance, and {@code source} equals {@code target}, no changes are made and false is returned.
850     * </p>
851     *
852     * @param source object ID of the source object in this transaction
853     * @param target object ID of the target object in {@code dest}
854     * @param dest destination transaction containing {@code target} (possibly same as this transaction)
855     * @return false if object already existed in {@code dest}, true if {@code target} did not exist in {@code dest}
856     * @throws DeletedObjectException if no object with ID equal to {@code id} is found in this transaction
857     * @throws UnknownTypeException if {@code source} or {@code target} specifies an unknown object type
858     * @throws IllegalArgumentException if {@code source} and {@code target} specify different object types
859     * @throws IllegalArgumentException if any parameter is null
860     * @throws IllegalArgumentException if any {@code source} and {@code target} have different object types
861     * @throws ReadOnlyTransactionException if {@code dest} has been {@linkplain #setReadOnly set read-only}
862     * @throws StaleTransactionException if this transaction or {@code dest} is no longer usable
863     * @throws SchemaMismatchException if the schema version associated with {@code source} differs between
864     *  this transaction and {@code dest}
865     */
866    public synchronized boolean copy(ObjId source, final ObjId target, final Transaction dest) {
867
868        // Sanity check
869        if (source == null)
870            throw new IllegalArgumentException("null source");
871        if (target == null)
872            throw new IllegalArgumentException("null target");
873        if (dest == null)
874            throw new IllegalArgumentException("null dest");
875        if (this.stale)
876            throw new StaleTransactionException(this);
877
878        // Do nothing if nothing to do
879        if (this == dest && source.equals(target))
880            return false;
881
882        // Get object info
883        final ObjInfo srcInfo = this.getObjectInfo(source, false);
884
885        // Do the copy while both transactions are locked
886        synchronized (dest) {
887
888            // Sanity check
889            if (dest.stale)
890                throw new StaleTransactionException(dest);
891            if (dest.readOnly)
892                throw new ReadOnlyTransactionException(dest);
893
894            // Copy fields
895            return dest.mutateAndNotify(new Mutation<Boolean>() {
896                @Override
897                public Boolean mutate() {
898                    return Transaction.doCopyFields(srcInfo, target, Transaction.this, dest);
899                }
900            });
901        }
902    }

904    // This method assumes both transactions are locked
905    private static boolean doCopyFields(ObjInfo srcInfo, ObjId dstId, Transaction srcTx, Transaction dstTx) {
906
907        // Sanity check
908        final ObjId srcId = srcInfo.getId();
909        if (srcId.equals(dstId) && srcTx == dstTx)
910            throw new RuntimeException("internal error");
911
912        // Verify objects have the same type
913        if (srcId.getStorageId() != dstId.getStorageId()) {
914            throw new IllegalArgumentException("can't copy " + srcId + " to " + dstId
915              + " due to non-equal object types (" + srcId.getStorageId() + " != " + dstId.getStorageId() + ")");
916        }
917
918        // Find and verify source object's schema version in destination transaction
919        final int objectVersion = srcInfo.getVersionNumber();
920        final SchemaVersion dstSchema;
921        try {
922            dstSchema = dstTx.schema.getVersion(objectVersion);
923        } catch (IllegalArgumentException e) {
924            throw new SchemaMismatchException("destination transaction has no schema version " + objectVersion);
925        }
926        if (!Arrays.equals(srcTx.version.encodedXML, dstTx.version.encodedXML))
927            throw new SchemaMismatchException("destination transaction schema version " + objectVersion + " does not match");
928
929        // Create destination object if it doesn't already exist
930        final boolean existed = !dstTx.create(dstId, objectVersion);
931        final ObjInfo dstInfo = dstTx.getObjectInfo(dstId, false);
932        final boolean needUpgrade = dstInfo.getVersionNumber() != objectVersion;   // can only happen if object already existed
933
934        // Do field-by-field copy if there are change or version listeners, otherwise do fast copy of key/value pairs
935        final ObjType type = srcInfo.getObjType();
936        if (dstTx.hasFieldMonitor(type) || (needUpgrade && !dstTx.versionChangeListeners.isEmpty())) {
937
938            // Upgrade object first
939            if (needUpgrade)
940                dstTx.updateVersion(dstInfo, dstSchema);
941
942            // Copy fields
943            for (Field<?> field : type.fields.values())
944                field.copy(srcId, dstId, srcTx, dstTx);
945        } else {
946
947            // Delete pre-existing object's field content, if any
948            dstTx.deleteObjectData(dstInfo);
949
950            // Add schema version index entry
951            dstTx.kvt.put(Database.buildVersionIndexKey(dstId, objectVersion), ByteUtil.EMPTY);
952
953            // Copy object meta-data and all field content in one key range sweep
954            final byte[] srcMinKey = srcId.getBytes();
955            final byte[] srcMaxKey = ByteUtil.getKeyAfterPrefix(srcMinKey);
956            final ByteWriter dstWriter = new ByteWriter();
957            dstWriter.write(dstId.getBytes());
958            final int dstMark = dstWriter.mark();
959            for (Iterator<KVPair> i = srcTx.kvt.getRange(srcMinKey, srcMaxKey, false); i.hasNext(); ) {
960                final KVPair kv = i.next();
961                final ByteReader srcReader = new ByteReader(kv.getKey());
962                srcReader.skip(srcMinKey.length);
963                dstWriter.reset(dstMark);
964                dstWriter.write(srcReader);
965                dstTx.kvt.put(dstWriter.getBytes(), kv.getValue());
966            }
967
968            // Create object's simple field index entries
969            for (SimpleField<?> field : type.simpleFields.values()) {
970                if (field.indexed) {
971                    final byte[] fieldValue = dstTx.kvt.get(field.buildKey(dstId));     // can be null (if field has default value)
972                    final byte[] indexKey = Transaction.buildSimpleIndexEntry(field, dstId, fieldValue);
973                    dstTx.kvt.put(indexKey, ByteUtil.EMPTY);
974                }
975            }
976
977            // Create object's composite index entries
978            for (CompositeIndex index : type.compositeIndexes.values())
979                dstTx.kvt.put(Transaction.buildCompositeIndexEntry(dstTx, dstId, index), ByteUtil.EMPTY);
980
981            // Create object's complex field index entries
982            for (ComplexField<?> field : type.complexFields.values()) {
983                for (SimpleField<?> subField : field.getSubFields()) {
984                    if (subField.indexed)
985                        field.addIndexEntries(dstTx, dstId, subField);
986                }
987            }
988        }
989
990        // Done
991        return !existed;
992    }

1124    /**
1125     * Update object to the current schema version.
1126     *
1127     * @param info original object info
1128     * @param targetVersion version to change to
1129     */
1130    private synchronized void updateVersion(final ObjInfo info, final SchemaVersion targetVersion) {
1131
1132        // Get version numbers
1133        final ObjId id = info.getId();
1134        final int oldVersion = info.getVersionNumber();
1135        final int newVersion = targetVersion.versionNumber;
1136
1137        // Sanity check
1138        if (newVersion == oldVersion)
1139            throw new IllegalArgumentException("object already at version " + newVersion);
1140        if (this.readOnly)
1141            throw new ReadOnlyTransactionException(this);
1142
1143        // Get old and new types
1144        final ObjType oldType = info.getObjType();
1145        final ObjType newType;
1146        try {
1147            newType = targetVersion.getObjType(id.getStorageId());
1148        } catch (UnknownTypeException e) {
1149            throw (TypeNotInSchemaVersionException)new TypeNotInSchemaVersionException(id, newVersion).initCause(e);
1150        }
1151
1152        // Gather removed fields' values here for user migration if any VersionChangeListeners are registered
1153        final TreeMap<Integer, Object> oldValueMap = !this.versionChangeListeners.isEmpty() ?
1154          new TreeMap<Integer, Object>() : null;
1155
1156    //////// Remove the index entries corresponding to removed composite indexes
1157
1158        // Get composite index storage IDs (old or new)
1159        final TreeSet<Integer> compositeIndexStorageIds = new TreeSet<>();
1160        compositeIndexStorageIds.addAll(oldType.compositeIndexes.keySet());
1161        compositeIndexStorageIds.addAll(newType.compositeIndexes.keySet());
1162
1163        // Remove index entries for composite indexes that are going away
1164        for (CompositeIndex index : oldType.compositeIndexes.values()) {
1165            if (!newType.compositeIndexes.containsKey(index.storageId))
1166                this.kvt.remove(this.buildCompositeIndexEntry(id, index));
1167        }
1168
1169    //////// Update counter fields
1170
1171        // Get counter field storage IDs (old or new)
1172        final TreeSet<Integer> counterFieldStorageIds = new TreeSet<>();
1173        counterFieldStorageIds.addAll(oldType.counterFields.keySet());
1174        counterFieldStorageIds.addAll(newType.counterFields.keySet());
1175
1176        // Update counter fields
1177        for (int storageId : counterFieldStorageIds) {
1178
1179            // Get old and new counter fields having this storage ID
1180            final CounterField oldField = oldType.counterFields.get(storageId);
1181            final CounterField newField = newType.counterFields.get(storageId);
1182
1183            // Save old field values for version change notification
1184            final byte[] key = Field.buildKey(id, storageId);
1185            if (oldField != null && oldValueMap != null) {
1186                final byte[] oldValue = this.kvt.get(key);
1187                final long value = oldValue != null ? this.getCountingKVTransaction().decodeCounter(oldValue) : 0;
1188                oldValueMap.put(storageId, value);
1189            }
1190
1191            // Remove old value if field has disappeared, otherwise leave alone
1192            if (newField == null)
1193                this.kvt.remove(key);
1194        }
1195
1196    //////// Update simple fields and corresponding index entries
1197
1198        // Get simple field storage IDs (old or new)
1199        final TreeSet<Integer> simpleFieldStorageIds = new TreeSet<>();
1200        simpleFieldStorageIds.addAll(oldType.simpleFields.keySet());
1201        simpleFieldStorageIds.addAll(newType.simpleFields.keySet());
1202
1203        // Update simple fields
1204        for (int storageId : simpleFieldStorageIds) {
1205
1206            // Get old and new simple fields having this storage ID
1207            final SimpleField<?> oldField = oldType.simpleFields.get(storageId);
1208            final SimpleField<?> newField = newType.simpleFields.get(storageId);
1209
1210            // Save old field values for version change notification
1211            final byte[] key = Field.buildKey(id, storageId);
1212            final byte[] oldValue = oldField != null ? this.kvt.get(key) : null;
1213            if (oldField != null && oldValueMap != null) {
1214                final byte[] bytes = oldValue != null ? oldValue : oldField.fieldType.getDefaultValue();
1215                final Object value = oldField.fieldType.read(new ByteReader(bytes));
1216                oldValueMap.put(storageId, value);
1217            }
1218
1219            // If field is being removed, discard the old value
1220            if (newField == null && oldValue != null)
1221                this.kvt.remove(key);
1222
1223            // Remove old index entry if index removed in new version
1224            if (oldField != null && oldField.indexed && (newField == null || !newField.indexed))
1225                this.kvt.remove(Transaction.buildSimpleIndexEntry(oldField, id, oldValue));
1226
1227            // Add new index entry if index added in new version
1228            if (newField != null && newField.indexed && (oldField == null || !oldField.indexed))
1229                this.kvt.put(Transaction.buildSimpleIndexEntry(newField, id, oldValue), ByteUtil.EMPTY);
1230        }
1231
1232    //////// Add composite index entries for newly added composite indexes
1233
1234        // Add index entries for composite indexes that are newly added
1235        for (CompositeIndex index : newType.compositeIndexes.values()) {
1236            if (!oldType.compositeIndexes.containsKey(index.storageId))
1237                this.kvt.put(this.buildCompositeIndexEntry(id, index), ByteUtil.EMPTY);
1238        }
1239
1240    //////// Update complex fields and corresponding index entries
1241
1242        // Get complex field storage IDs (old or new)
1243        final TreeSet<Integer> complexFieldStorageIds = new TreeSet<>();
1244        complexFieldStorageIds.addAll(oldType.complexFields.keySet());
1245        complexFieldStorageIds.addAll(newType.complexFields.keySet());
1246
1247        // Notes:
1248        //
1249        // - The only changes we support are sub-field changes that don't affect the corresponding StorageInfo's
1250        // - New complex fields do not need to be explicitly initialized because their initial state is to have zero KV pairs
1251        //
1252        for (int storageId : complexFieldStorageIds) {
1253
1254            // Get old and new complex fields having this storage ID
1255            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1256            final ComplexField<?> newField = newType.complexFields.get(storageId);
1257
1258            // If there is no old field, new field and any associated indexes are already initialized (i.e., they're empty)
1259            if (oldField == null)
1260                continue;
1261
1262            // Save old field's value
1263            if (oldField != null && oldValueMap != null)
1264                oldValueMap.put(storageId, oldField.getValueReadOnlyCopy(this, id));
1265
1266            // If field is being removed, delete old field content, otherwise check if index entries should be added/removed
1267            if (newField == null) {
1268                oldField.removeIndexEntries(this, id);
1269                oldField.deleteContent(this, id);
1270            } else
1271                newField.updateSubFieldIndexes(this, oldField, id);
1272        }
1273
1274    //////// Remove references that are no longer valid
1275
1276        // Unreference any reference fields that refer to no-longer-valid object types
1277        for (int storageId : simpleFieldStorageIds) {
1278
1279            // Get storage IDs that are no longer allowed, if any
1280            final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, storageId);
1281            if (removedObjectTypes == null)
1282                continue;
1283
1284            // Unreference field if needed
1285            final ReferenceField oldField = (ReferenceField)oldType.simpleFields.get(storageId);
1286            final ObjId ref = oldField.getValue(this, id);
1287            if (ref != null && removedObjectTypes.contains(ref.getStorageId()))
1288                this.writeSimpleField(id, oldField.storageId, null, false);
1289        }
1290
1291        // Unreference any reference sub-fields of complex fields that refer to no-longer-valid object types
1292        for (int storageId : complexFieldStorageIds) {
1293
1294            // Get old and new complex fields having this storage ID
1295            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
1296            final ComplexField<?> newField = newType.complexFields.get(storageId);
1297            if (oldField == null || newField == null)
1298                continue;
1299            final List<? extends SimpleField<?>> oldSubFields = oldField.getSubFields();
1300            final List<? extends SimpleField<?>> newSubFields = newField.getSubFields();
1301            final int numSubFields = oldSubFields.size();
1302            assert numSubFields == newSubFields.size();
1303
1304            // Iterate over subfields
1305            for (int i = 0; i < numSubFields; i++) {
1306                final SimpleField<?> oldSubField = oldSubFields.get(i);
1307                final SimpleField<?> newSubField = newSubFields.get(i);
1308                assert oldSubField.storageId == newSubField.storageId;
1309                final int subStorageId = oldSubField.storageId;
1310
1311                // Get storage IDs that are no longer allowed, if any
1312                final TreeSet<Integer> removedObjectTypes = this.findRemovedObjectTypes(info, targetVersion, subStorageId);
1313                if (removedObjectTypes == null)
1314                    continue;
1315
1316                // Unreference sub-fields as needed
1317                oldField.unreferenceRemovedObjectTypes(this, id, (ReferenceField)oldSubField, removedObjectTypes);
1318            }
1319        }
1320
1321    //////// Update object version and corresponding index entry
1322
1323        // Update object version
1324        ObjInfo.write(this, id, newVersion, info.isDeleteNotified());
1325
1326        // Update object version index entry
1327        this.kvt.remove(Database.buildVersionIndexKey(id, oldVersion));
1328        this.kvt.put(Database.buildVersionIndexKey(id, newVersion), ByteUtil.EMPTY);
1329
1330    //////// Notify listeners
1331
1332        // Lock down old field values map
1333        final NavigableMap<Integer, Object> readOnlyOldValuesMap = oldValueMap != null ?
1334          Maps.unmodifiableNavigableMap(oldValueMap) : null;
1335
1336        // Notify about version update
1337        for (VersionChangeListener listener : this.versionChangeListeners)
1338            listener.onVersionChange(this, id, oldVersion, newVersion, readOnlyOldValuesMap);
1339    }

1341    /**
1342     * Find storage ID's which are no longer allowed by a reference field when upgrading to the specified
1343     * schema version and therefore need to be scrubbed during the upgrade.
1344     */
1345    private TreeSet<Integer> findRemovedObjectTypes(ObjInfo info, SchemaVersion newVersion, int storageId) {
1346
1347        // Get old schema version
1348        final SchemaVersion oldVersion = info.getSchemaVersion();
1349
1350        // Get old and new object types
1351        final ObjType oldType = info.getObjType();
1352        final ObjType newType = newVersion.getObjType(info.getId().getStorageId());
1353
1354        // Get old and new reference fields having specified storage ID
1355        final ReferenceField oldField = oldType.referenceFields.get(storageId);
1356        final ReferenceField newField = newType.referenceFields.get(storageId);
1357        if (oldField == null || newField == null)
1358            return null;
1359
1360        // Check allowed storage IDs
1361        final SortedSet<Integer> newObjectTypes = newField.getObjectTypes();
1362        if (newObjectTypes == null)
1363            return null;                                                // new field can refer to any type in any schema version
1364        SortedSet<Integer> oldObjectTypes = oldField.getObjectTypes();
1365        if (oldObjectTypes == null)
1366            oldObjectTypes = this.schema.objTypeStorageIds;             // old field can refer to any type in any schema version
1367
1368        // Identify storage IDs which are were allowed by old field but are no longer allowed by new field
1369        final TreeSet<Integer> removedObjectTypes = new TreeSet<>(oldObjectTypes);
1370        removedObjectTypes.removeAll(newObjectTypes);
1371        return !removedObjectTypes.isEmpty() ? removedObjectTypes : null;
1372    }

1548    private synchronized void doWriteSimpleField(ObjId id, int storageId, final Object newObj, boolean updateVersion) {
1549
1550        // Get object info
1551        final ObjInfo info = this.getObjectInfo(id, updateVersion);
1552
1553        // Find field
1554        final SimpleField<?> field = info.getObjType().simpleFields.get(storageId);
1555        if (field == null)
1556            throw new UnknownFieldException(info.getObjType(), storageId, "simple field");
1557
1558        // Get new value
1559        final byte[] key = field.buildKey(id);
1560        final byte[] newValue = field.encode(newObj);
1561
1562        // Before setting the new value, read the old value if one of the following is true:
1563        //  - The field is being monitored -> we need to filter out "changes" that don't actuallly change anything
1564        //  - The field is indexed -> we need the old value so we can remove the old index entry
1565        // If neither of the above is true, then there's no need to read the old value.
1566        byte[] oldValue = null;
1567        if (field.indexed || field.compositeIndexMap != null || this.hasFieldMonitor(id, field)) {
1568
1569            // Get old value
1570            oldValue = this.kvt.get(key);
1571
1572            // Compare new to old value
1573            if (oldValue != null ? newValue != null && Arrays.equals(oldValue, newValue) : newValue == null)
1574                return;
1575        }
1576
1577        // Update value
1578        if (newValue != null)
1579            this.kvt.put(key, newValue);
1580        else
1581            this.kvt.remove(key);
1582
1583        // Update simple index, if any
1584        if (field.indexed) {
1585            this.kvt.remove(Transaction.buildSimpleIndexEntry(field, id, oldValue));
1586            this.kvt.put(Transaction.buildSimpleIndexEntry(field, id, newValue), ByteUtil.EMPTY);
1587        }
1588
1589        // Update affected composite indexes, if any
1590        if (field.compositeIndexMap != null) {
1591            for (Map.Entry<CompositeIndex, Integer> entry : field.compositeIndexMap.entrySet()) {
1592                final CompositeIndex index = entry.getKey();
1593                final int fieldIndexOffset = entry.getValue();
1594
1595                // Build old composite index entry
1596                final ByteWriter oldWriter = new ByteWriter();
1597                UnsignedIntEncoder.write(oldWriter, index.storageId);
1598                int fieldStart = -1;
1599                int fieldEnd = -1;
1600                for (SimpleField<?> otherField : index.fields) {
1601                    final byte[] otherValue;
1602                    if (otherField == field) {
1603                        fieldStart = oldWriter.getLength();
1604                        otherValue = oldValue;
1605                    } else
1606                        otherValue = this.kvt.get(otherField.buildKey(id));         // can be null (if field has default value)
1607                    oldWriter.write(otherValue != null ? otherValue : otherField.fieldType.getDefaultValue());
1608                    if (otherField == field)
1609                        fieldEnd = oldWriter.getLength();
1610                }
1611                assert fieldStart != -1;
1612                assert fieldEnd != -1;
1613                id.writeTo(oldWriter);
1614
1615                // Remove old composite index entry
1616                final byte[] oldIndexEntry = oldWriter.getBytes();
1617                this.kvt.remove(oldIndexEntry);
1618
1619                // Patch in new field value to create new composite index entry
1620                final ByteWriter newWriter = new ByteWriter(oldIndexEntry.length);
1621                newWriter.write(oldIndexEntry, 0, fieldStart);
1622                newWriter.write(newValue != null ? newValue : field.fieldType.getDefaultValue());
1623                newWriter.write(oldIndexEntry, fieldEnd, oldIndexEntry.length - fieldEnd);
1624
1625                // Add new composite index entry
1626                this.kvt.put(newWriter.getBytes(), ByteUtil.EMPTY);
1627            }
1628        }
1629
1630        // Notify monitors
1631        final Object oldObj = field.fieldType.read(new ByteReader(oldValue != null ? oldValue : field.fieldType.getDefaultValue()));
1632        this.addFieldChangeNotification(new SimpleFieldChangeNotifier(field, id) {
1633            @Override
1634            @SuppressWarnings("unchecked")
1635            public void notify(Transaction tx, SimpleFieldChangeListener listener, int[] path, NavigableSet<ObjId> referrers) {
1636                listener.onSimpleFieldChange(tx, this.id, (SimpleField<Object>)field, path, referrers, oldObj, newObj);
1637            }
1638        });
1639    }

1998     * Monitor for changes within this transaction of the value of the given field, as seen through a path of references.
1999     *
2000     * <p>
2001     * When the specified field is changed, a listener notification will be delivered for each referring object that
2002     * refers to the object containing the changed field through the specified path of references. Notifications are
2003     * delivered at the end of the mutation operation just prior to returning to the caller. If the listener method
2004     * performs additional mutation(s) which are themselves being listened for, those notifications will also be delivered
2005     * prior to the returning to the original caller. Therefore, care must be taken to avoid change notification dependency
2006     * loops when listeners can themselves mutate fields, to avoid infinite loops.
2007     * </p>
2008     *
2009     * <p>
2010     * If a non-null {@code types} is provided, then only objects whose types have one of the specified storage ID's
2011     * will trigger notifications to {@code listener}.
2012     * </p>
2013     *
2014     * <p>
2015     * A referring object may refer to the changed object through more than one actual path of references matching {@code path};
2016     * if so, it will still appear only once in the {@link NavigableSet} provided to the listener (this is of course required
2017     * by set semantics).
2018     * </p>
2019     *
2020     * <p>
2021     * Although the reference back-tracking algorithm does consolidate multiple paths between the same two objects,
2022     * be careful to avoid an explosion of notifications when objects in the {@code path} have a high degree of fan-in.
2023     * </p>
2024     *
2025     * <p>
2026     * When a {@link ReferenceField} in {@code path} also happens to be the field being changed, then there is ambiguity
2027     * about how the set of referring objects is calculated: does it use the field's value before or after the change?
2028     * This API guarantees that the answer is "after the change"; however, if another listener on the same field is
2029     * invoked before {@code listener} and mutates any reference field(s) in {@code path}, then whether that additional
2030     * change is also be included in the calculation is undefined.
2031     * </p>
2032     *
2033     * <p>
2034     * Therefore, for consistency, avoid changing any {@link ReferenceField} from within a listener callback when that
2035     * field is also in some other listener's reference path, and both listeners are watching the same field.
2036     * </p>
2037     *
2038     * @param storageId storage ID of the field to monitor
2039     * @param path path of reference fields (represented by storage IDs) through which to monitor field
2040     * @param types set of allowed storage IDs for the changed object, or null for no restriction
2041     * @param listener callback for notifications on changes in value
2042     * @throws UnknownFieldException if no {@link SimpleField} corresponding to {@code storageId} exists
2043     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2044     * @throws IllegalArgumentException if {@code storageId} refers to a sub-field of a complex field
2045     * @throws IllegalArgumentException if {@code path} or {@code listener} is null
2046     * @throws StaleTransactionException if this transaction is no longer usable

2299    private synchronized <V> V mutateAndNotify(Mutation<V> mutation) {
2300
2301        // Validate transaction
2302        if (this.stale)
2303            throw new StaleTransactionException(this);
2304        if (this.readOnly)
2305            throw new ReadOnlyTransactionException(this);
2306
2307        // If re-entrant invocation, we're already set up
2308        if (this.pendingNotifications.get() != null)
2309            return mutation.mutate();
2310
2311        // Set up pending report list, perform mutation, and then issue reports
2312        this.pendingNotifications.set(new TreeMap<Integer, ArrayList<FieldChangeNotifier>>());
2313        try {
2314            return mutation.mutate();
2315        } finally {
2316            try {
2317                final TreeMap<Integer, ArrayList<FieldChangeNotifier>> pendingNotificationMap = this.pendingNotifications.get();
2318                while (!pendingNotificationMap.isEmpty()) {
2319
2320                    // Get the next field with pending notifications
2321                    final Map.Entry<Integer, ArrayList<FieldChangeNotifier>> entry = pendingNotificationMap.pollFirstEntry();
2322                    final int storageId = entry.getKey();
2323
2324                    // For all pending notifications, back-track references and notify all field monitors for the field
2325                    for (FieldChangeNotifier notifier : entry.getValue()) {
2326                        assert notifier.getStorageId() == storageId;
2327                        final ArrayList<FieldMonitor> monitorList = new ArrayList<>(this.getMonitorsForField(storageId, false));
2328                        if (!monitorList.isEmpty())
2329                            this.notifyFieldMonitors(notifier, NavigableSets.singleton(notifier.getId()), monitorList, 0);
2330                    }
2331                }
2332            } finally {
2333                this.pendingNotifications.remove();
2334            }
2335        }
2336    }

2338    // Recursively back-track references along monitor paths and notify monitors when we reach the end (i.e., beginning)
2339    private void notifyFieldMonitors(FieldChangeNotifier notifier,
2340      NavigableSet<ObjId> objects, ArrayList<FieldMonitor> monitorList, int step) {
2341
2342        // Find the monitors for whom we have completed all the steps in their (inverse) path,
2343        // and group the remaining monitors by their next inverted reference path step.
2344        final HashMap<Integer, ArrayList<FieldMonitor>> remainingMonitorsMap = new HashMap<>();
2345        for (FieldMonitor monitor : monitorList) {
2346
2347            // On the first step, apply the monitor's type filter, if any
2348            if (step == 0 && monitor.types != null && !monitor.types.contains(notifier.getId().getBytes()))
2349                continue;
2350
2351            // Issue notification callback if we have back-tracked through the whole path
2352            if (monitor.path.length == step) {
2353                notifier.notify(this, monitor.listener, monitor.path, objects);
2354                continue;
2355            }
2356
2357            // Sort the unfinished monitors by their next (i.e., previous) reference field
2358            final int storageId = monitor.path[monitor.path.length - step - 1];
2359            ArrayList<FieldMonitor> remainingMonitors = remainingMonitorsMap.get(storageId);
2360            if (remainingMonitors == null) {
2361                remainingMonitors = new ArrayList<FieldMonitor>();
2362                remainingMonitorsMap.put(storageId, remainingMonitors);
2363            }
2364            remainingMonitors.add(monitor);
2365        }
2366
2367        // Invert references for each group of remaining monitors and recurse
2368        for (Map.Entry<Integer, ArrayList<FieldMonitor>> entry : remainingMonitorsMap.entrySet()) {
2369            final int storageId = entry.getKey();
2370
2371            // Gather all objects that refer to any object in our current "objects" set
2372            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2373            for (ObjId object : objects) {
2374                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(object);
2375                if (refs != null)
2376                    refsList.add(refs);
2377            }
2378
2379            // Recurse on the union of those objects
2380            if (!refsList.isEmpty())
2381                this.notifyFieldMonitors(notifier, NavigableSets.union(refsList), entry.getValue(), step + 1);
2382        }
2383    }

2387    /**
2388     * Find all objects that refer to any object in the given target set through the specified path of references.
2389     *
2390     * @param path path of one or more reference fields (represented by storage IDs) through which to reach the target objects
2391     * @param targetObjects target objects
2392     * @return set of objects that refer to the {@code targetObjects} via {@code path}
2393     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
2394     * @throws IllegalArgumentException if {@code targetObjects} or {@code path} is null
2395     * @throws IllegalArgumentException if {@code path} is empty
2396     * @throws StaleTransactionException if this transaction is no longer usable
2397     */
2398    public NavigableSet<ObjId> invertReferencePath(int[] path, Iterable<ObjId> targetObjects) {
2399
2400        // Sanity check
2401        if (targetObjects == null)
2402            throw new IllegalArgumentException("null targetObjects");
2403        if (path == null)
2404            throw new IllegalArgumentException("null path");
2405        if (path.length == 0)
2406            throw new IllegalArgumentException("empty path");
2407
2408        // Verify all fields in the path are reference fields
2409        for (int storageId : path)
2410            this.schema.verifyStorageInfo(storageId, ReferenceFieldStorageInfo.class);
2411
2412        // Invert references in reverse order
2413        NavigableSet<ObjId> result = null;
2414        for (int i = path.length - 1; i >= 0; i--) {
2415            final int storageId = path[i];
2416
2417            // Gather all objects that refer to any object in our current target objects set
2418            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
2419            for (ObjId id : targetObjects) {
2420                final NavigableSet<ObjId> refs = this.queryReferences(storageId).get(id);
2421                if (refs != null)
2422                    refsList.add(refs);
2423            }
2424            if (refsList.isEmpty())
2425                return NavigableSets.empty();
2426
2427            // Recurse on the union of those objects
2428            targetObjects = result = NavigableSets.union(refsList);
2429        }
2430
2431        // Done
2432        return result;
2433    }

2560    /**
2561     * Find all objects that refer to the given target object through the/any reference field with the specified
2562     * {@link DeleteAction}.
2563     *
2564     * <p>
2565     * Because different schema versions can have different {@link DeleteAction}'s configured for the
2566     * same field, we have to iterate through each schema version separately.
2567     * </p>
2568     *
2569     * @param target referred-to object
2570     * @param onDelete {@link DeleteAction} to match
2571     * @param fieldStorageId reference field storage ID, or -1 to match any reference field
2572     */
2573    private NavigableSet<ObjId> findReferrers(ObjId target, DeleteAction onDelete, int fieldStorageId) {
2574        final ArrayList<NavigableSet<ObjId>> refSets = new ArrayList<>();
2575        for (SchemaVersion schemaVersion : this.schema.versions.values()) {
2576
2577            // Check whether any object of this version exist; if not, skip
2578            final NavigableSet<ObjId> versionObjects = this.queryVersion().asMap().get(schemaVersion.versionNumber);
2579            if (versionObjects == null)
2580                continue;
2581
2582            // Find all reference fields with storage ID matching fieldStorageId (if not -1) and check them.
2583            // Do this separately for each such field in each object type because the fields may have different DeleteAction's.
2584            for (ObjType objType : schemaVersion.objTypeMap.values()) {
2585                for (ReferenceField field : Iterables.filter(objType.getFieldsAndSubFields(), ReferenceField.class)) {
2586
2587                    // Check delete action and field
2588                    if (field.onDelete != onDelete || (fieldStorageId != -1 && field.storageId != fieldStorageId))
2589                        continue;
2590
2591                    // Query index on this field, restricting to those only references coming from objType objects
2592                    final NavigableSet<ObjId> refs = this.querySimpleField(field.storageId)
2593                      .filter(1, new KeyRanges(ObjId.getKeyRange(objType.storageId))).asMap().get(target);
2594                    if (refs == null)
2595                        continue;
2596
2597                    // Restrict further to the specific schema version
2598                    refSets.add(NavigableSets.intersection(versionObjects, refs));
2599                }
2600            }
2601        }
2602        return !refSets.isEmpty() ? NavigableSets.union(refSets) : NavigableSets.empty(FieldTypeRegistry.OBJ_ID);
2603    }
