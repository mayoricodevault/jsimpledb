008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeToken;
011
012import java.util.Comparator;
013
014import org.jsimpledb.parse.ParseContext;
015import org.jsimpledb.parse.ParseUtil;
016import org.jsimpledb.util.ByteReader;
017import org.jsimpledb.util.ByteWriter;
018
019/**
020 * Defines the encoding, ordering, and range of possible values for a {@link SimpleField}.
021 *
022 * <p>
023 * A {@link FieldType} maps between instances of its supported Java type and the self-delimited {@code byte[]} encoding of
024 * those instances used in the database. The {@code byte[]} encoding also implicitly defines database the sort order
025 * (from unsigned lexicographical ordering) which is also reflected via {@link #compare Comparator.compare()}.
026 * </p>
027 *
028 * <p>
029 * A {@link FieldType} also defines a mapping between Java instances and {@link String} values;
030 * there are actually two separate {@link String} forms, one of which is self-delimiting.
031 * </p>
032 *
033 * <p>
034 * {@link FieldType}s have these requirements and properties:
035 * <ul>
036 *  <li>They have a unique {@linkplain #getName name}; typically the same as their {@linkplain #getTypeToken supported type}.</li>
037 *  <li>All possible values can be represented in Java as an instance of the associated Java type (possibly including null).</li>
038 *  <li>Instances {@linkplain #compare totally order} the Java values. If the associated Java type implements {@link Comparable},
039 *      then the two orderings do not necessarily have to match, but they should if possible.</li>
040 *  <li>All possible values can be encoded/decoded into a self-delimiting binary string (i.e., {@code byte[]} array)
041 *      without losing information, and these binary strings, when sorted lexicographically using unsigned comparison,
042 *      sort consistently with the total ordering of the corresponding Java values defined by {@link #compare compare()}.</li>
043 *  <li>All possible values can be encoded/decoded to/from {@link String}s without losing information,
044 *      with both a {@linkplain #toString(Object) regular string form} for non-null values and a
045 *      {@linkplain #toParseableString self-delimiting string form} for any value (these may be the same).</li>
046 *  <li>{@code null} may or may not be a supported value; if so, it must be handled by {@link #compare} and
047 *      have binary and string encodings just like any other value. Typically, null sorts last.</li>
048 *  <li>There is a {@linkplain #getDefaultValue default value}; it must be null for types that support null.</li>
049 *  <li>An optional {@linkplain #getEncodingSignature encoding signature} protects against incompatible encodings
050 *      when a {@link FieldType}'s binary or string encoding changes without changing the {@linkplain #getName name}</li>
051 * </ul>
052 * </p>
053 *
054 * <p>
055 * Two {@link FieldType} instances should be equal according to {@link #equals equals()} if only if they behave identically
056 * with respect to all of the above.
057 * </p>
058 *
059 * <p>
060 * A {@link FieldTypeRegistry} contains a registry of {@link FieldType}s indexed by name.
061 * </p>
062 *
063 * @param <T> The associated Java type
064 * @see FieldTypeRegistry
065 */
066public abstract class FieldType<T> implements Comparator<T> {
067
068    /**
069     * The regular expression that {@link FieldType} names must match. This pattern is the same as is required
070     * for Java identifiers, except that the following additional characters are allowed after the first character:
071     * dot (`.') and dash (`-'), and lastly up to 255 pairs of square brackets (`[]') to indicate an array type.
072     */
073    public static final String NAME_PATTERN = "\\p{javaJavaIdentifierStart}[-.\\p{javaJavaIdentifierPart}]*(\\[\\]){0,255}";
074
075    /**
076     * Type name for reference types.
077     */
078    public static final String REFERENCE_TYPE_NAME = "reference";
079
080    final String name;
081    final TypeToken<T> typeToken;
082    final long signature;
083
084// Constructors
085
086    /**
087     * Constructor.
088     *
089     * @param name the name of this type
090     * @param typeToken Java type for the field's values
091     * @param signature binary encoding signature
092     * @throws IllegalArgumentException if any parameter is null
093     * @throws IllegalArgumentException if {@code name} is invalid
094     */
095    protected FieldType(String name, TypeToken<T> typeToken, long signature) {
096        if (name == null)
097            throw new IllegalArgumentException("null name");
098        if (!name.matches(FieldType.NAME_PATTERN))
099            throw new IllegalArgumentException("invalid type name `" + name + "'");
100        if (typeToken == null)
101            throw new IllegalArgumentException("null typeToken");
102        this.name = name;
103        this.typeToken = typeToken;
104        this.signature = signature;
105    }
106
107    /**
108     * Constructor taking a {@link Class} object.
109     * The {@linkplain #getName name} of this instance will be the {@linkplain Class#getName name} of the given class.
110     *
111     * @param type Java type for the field's values
112     * @param signature binary encoding signature
113     * @throws NullPointerException if {@code type} is null
114     */
115    protected FieldType(Class<T> type, long signature) {
116        this(type.getName(), TypeToken.of(type), signature);
117    }
118
119// Public methods
120
121    /**
122     * Get the name of this type. {@link FieldType} names must be unique in the registry.
123     */
124    public String getName() {
125        return this.name;
126    }
127
128    /**
129     * Get the Java type corresponding to this type's values.
130     */
131    public TypeToken<T> getTypeToken() {
132        return this.typeToken;
133    }
134
135    /**
136     * Get the binary encoding signature of this type.
137     *
138     * <p>
139     * The binary encoding signature is analogous to the {@code serialVersionUID} used by Java serialization.
140     * It represents a specific binary and/or {@link String} encoding for Java values. In the case that a {@link FieldType}
141     * implementation changes its binary encoding, but not it's name, it <b>must</b> use a new, different binary encoding
142     * signature to eliminate the possibility of mixing incompatible encodings in software vs. persistent storage.
143     * Typically a value of zero is used until if/when such a change occurs.
144     * </p>
145     *
146     * <p>
147     * Note that another option when encodings change is simply to change the {@linkplain #getName name} of the type
148     * (encoding signatures are scoped to a single {@link FieldType} name).
149     * </p>
150     */
151    public long getEncodingSignature() {
152        return this.signature;
153    }
154
155    /**
156     * Read a value from the given input.
157     *
158     * @param input byte input
159     * @return field value (possibly null)
160     * @throws IllegalArgumentException if invalid input is encountered
161     * @throws IndexOutOfBoundsException if input is truncated
162     */
163    public abstract T read(ByteReader input);
164
165    /**
166     * Write a value to the given output.
167     *
168     * @param writer byte output
169     * @param value value to write (possibly null)
170     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
171     */
172    public abstract void write(ByteWriter writer, T value);
173
174    /**
175     * Get the default value for this field type encoded as a {@code byte[]} array.
176     *
177     * @return encoded default value
178     */
179    public abstract byte[] getDefaultValue();
180
181    /**
182     * Read and discard a value from the given input.
183     *
184     * @param reader byte input
185     * @throws IllegalArgumentException if invalid input is encountered
186     * @throws IndexOutOfBoundsException if input is truncated
187     */
188    public abstract void skip(ByteReader reader);
189
190    /**
191     * Encode a non-null value as a {@link String} for later decoding by {@link #fromString fromString()}.
192     *
193     * <p>
194     * Each of the characters in the returned {@link String} must be one of the valid XML characters
195     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
196     * </p>
197     *
198     * <p>
199     * The implementation in {@link FieldType} checks that {@code value} is not null, then delegates to {@link #toParseableString}.
200     * Subclasses that override this method should also override {@link #fromString fromString()}.
201     * </p>
202     *
203     * @param value actual value, never null
204     * @return string encoding of {@code value}
205     * @throws IllegalArgumentException if {@code value} is null
206     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
207     */
208    public String toString(T value) {
209        if (value == null)
210            throw new IllegalArgumentException("null value");
211        return this.toParseableString(value);
212    }
213
214    /**
215     * Parse a non-null value previously encoded by {@link #toString(Object)}.
216     *
217     * <p>
218     * The implementation in {@link FieldType} creates a new {@link ParseContext} based on {@code string},
219     * delegates to {@link #toParseableString} to parse it, and verifies that all of {@code string} was consumed
220     * during the parse. Subclasses that override this method should also override {@link #toString(Object)}.
221     * </p>
222     *
223     * @param string non-null value previously encoded as a {@link String} by {@link #toString(Object)}
224     * @return actual value
225     * @throws IllegalArgumentException if the input is invalid
226     */
227    public T fromString(String string) {
228        final ParseContext ctx = new ParseContext(string);
229        final T value = this.fromParseableString(ctx);
230        if (!ctx.isEOF()) {
231            throw new IllegalArgumentException("found trailing garbage starting with `"
232              + ParseUtil.truncate(ctx.getInput(), 20) + "'");
233        }
234        return value;
235    }
236
237    /**
238     * Encode a possibly null value as a {@link String} for later decoding by {@link #fromParseableString fromParseableString()}.
239     * The string value must be <i>self-delimiting</i>, i.e., decodable even when followed by arbitrary additional characters,
240     * and must not start with whitespace or closing square bracket ({@code "]"}).
241     *
242     * <p>
243     * In addition, each of the characters in the returned {@link String} must be one of the valid XML characters
244     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
245     * </p>
246     *
247     * @param value actual value (possibly null)
248     * @return string encoding of {@code value}
249     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
250     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
251     */
252    public abstract String toParseableString(T value);
253
254    /**
255     * Parse a value previously encoded by {@link #toParseableString toParseableString()} as a self-delimited {@link String}
256     * and positioned at the start of the given parsing context.
257     *
258     * @param context parse context starting with a string previously encoded via {@link #toParseableString toParseableString()}
259     * @return actual value (possibly null)
260     * @throws IllegalArgumentException if the input is invalid
261     */
262    public abstract T fromParseableString(ParseContext context);
263
264    /**
265     * Verify the given object is a valid instance of this {@link FieldType}'s Java type and cast it to that type.
266     *
267     * <p>
268     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
269     * or {@link NullPointerException}, if {@code obj} does not have the correct type, or is an illegal null value.
270     * </p>
271     *
272     * <p>
273     * This method is allowed to perform widening conversions of the object that lose no information, e.g.,
274     * from {@link Integer} to {@link Long}.
275     * </p>
276     *
277     * <p>
278     * The implementation in {@link FieldType} simply casts the value using this instance's raw Java type.
279     * Subclasses should override this method to implement any other restrictions, e.g., disallowing null values.
280     * </p>
281     *
282     * @param obj object to validate
283     * @throws IllegalArgumentException if {@code obj} in not of type T
284     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
285     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
286     */
287    @SuppressWarnings("unchecked")
288    public T validate(Object obj) {
289        try {
290            return (T)this.typeToken.getRawType().cast(obj);
291        } catch (ClassCastException e) {
292            throw new IllegalArgumentException(this + " does not support values of type " + obj.getClass().getName());
293        }
294    }
295
296    /**
297     * Compare two values. This method must provide a total ordering of all supported Java values.
298     * If null is a supported Java value, then this method must accept it without throwing an exception
299     * (note, this is a stronger requirement than {@link Comparator} requires). By convention, null usually sorts last.
300     *
301     * @throws IllegalArgumentException if {@code value1} or {@code value2} is null and this type does not support null
302     */
303    @Override
304    public abstract int compare(T value1, T value2);
305
306    /**
307     * Determine whether any of this field type's encoded values start with a {@code 0x00} byte.
308     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
309     *
310     * <p>
311     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
312     * an incompatible type.
313     * </p>
314     *
315     * <p>
316     * The implementation in {@link FieldType} returns {@code true}.
317     * </p>
318     */
319    public boolean hasPrefix0x00() {
320        return true;
321    }
322
323    /**
324     * Determine whether any of this field type's encoded values start with a {@code 0xff} byte.
325     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
326     *
327     * <p>
328     * Note: changing the return value of this method usually means changing the binary encoding, resulting in
329     * an incompatible type.
330     * </p>
331     *
332     * <p>
333     * The implementation in {@link FieldType} returns {@code true}.
334     * </p>
335     */
336    public boolean hasPrefix0xff() {
337        return true;
338    }
339
340    /**
341     * Convenience method for generic type binding.
342     *
343     * @param obj object to validate
344     * @throws IllegalArgumentException if {@code obj} in not of type T
345     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
346     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
347     */
348    void validateAndWrite(ByteWriter writer, Object obj) {
349        this.write(writer, this.validate(obj));
350    }
351
352// Object
353
354    @Override
355    public String toString() {
356        return "field type `" + this.name + "'";
357    }
358
359    @Override
360    public int hashCode() {
361        return this.name.hashCode() ^ this.typeToken.hashCode();
362    }
363
364    @Override
365    public boolean equals(Object obj) {
366        if (obj == this)
367            return true;
368        if (obj == null || obj.getClass() != this.getClass())
369            return false;
370        final FieldType<?> that = (FieldType<?>)obj;
371        return this.name.equals(that.name) && this.typeToken.equals(that.typeToken);
372    }
373}
