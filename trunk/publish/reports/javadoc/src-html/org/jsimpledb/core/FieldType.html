024/**
025 * Defines the encoding, ordering, and range of possible values for a {@link SimpleField}.
026 *
027 * <p>
028 * {@link FieldType}s have these requirements and properties:
029 * <ul>
030 *  <li>They have a unique {@linkplain #getName name}; typically the same as their {@linkplain #getTypeToken supported type}.</li>
031 *  <li>All possible values can be represented in Java as an instance of the associated Java type (possibly including null).</li>
032 *  <li>Instances {@linkplain #compare totally order} the Java values. If the associated Java type implements {@link Comparable},
033 *      then the two orderings do not necessarily have to match, but they should if possible.</li>
034 *  <li>All possible values can be encoded/decoded into a self-delimiting binary string (i.e., {@code byte[]} array)
035 *      without losing information, and these binary strings, when sorted lexicographically using unsigned comparison,
036 *      sort consistently with the {@linkplain #compare total ordering} of the corresponding Java values.</li>
037 *  <li>All possible values can be encoded/decoded to/from {@link String}s without losing information,
038 *      with both a {@linkplain #toString(Object) regular string form} and a
039 *      {@linkplain #toParseableString self-delimiting string form} (these may be the same).</li>
040 *  <li>{@code null} may or may not be a supported value; if so, it must be handled by {@link #compare} and
041 *      have binary and string encodings just like any other value. Typically, null sorts last.</li>
042 *  <li>There is a {@linkplain #getDefaultValue default value}; it must be null for types that support null.</li>
043 * </ul>
044 * </p>
045 *
046 * <p>
047 * Two {@link FieldType} instances should be equal according to {@link #equals equals()} if only if they behave identically
048 * with respect to all of the above.
049 * </p>
050 *
051 * <p>
052 * A {@link FieldTypeRegistry} contains a registry of {@link FieldType}s indexed by name.
053 * </p>
054 *
055 * @param <T> The associated Java type
056 * @see FieldTypeRegistry
057 */
058public abstract class FieldType<T> implements Comparator<T> {
059

062     */

200    final String name;
201    final TypeToken<T> typeToken;
202
203    /**
204     * Constructor.
205     *
206     * @param name the name of this type
207     * @param typeToken Java type for the field's values
208     * @throws IllegalArgumentException if any parameter is null
209     * @throws IllegalArgumentException if {@code name} is invalid
210     */
211    FieldType(String name, TypeToken<T> typeToken) {
212        if (name == null)
213            throw new IllegalArgumentException("null name");
214        if (typeToken == null)
215            throw new IllegalArgumentException("null typeToken");
216        this.name = name;
217        this.typeToken = typeToken;
218    }
219
220    FieldType(Class<T> type) {
221        this(type.getName(), TypeToken.of(type));
222    }
223
224    /**
225     * Get the name of this type. {@link FieldType} names must be unique in the registry.
226     */
227    public String getName() {
228        return this.name;
229    }
230
231    /**
232     * Get the Java type corresponding to this type's values.
233     */
234    public TypeToken<T> getTypeToken() {
235        return this.typeToken;
236    }
237
238    /**
239     * Read a value from the given input.
240     *
241     * @param input byte input
242     * @return field value (possibly null)
243     * @throws IllegalArgumentException if invalid input is encountered
244     * @throws IndexOutOfBoundsException if input is truncated
245     */
246    public abstract T read(ByteReader input);
247
248    /**
249     * Write a value to the given output.
250     *
251     * @param writer byte output
252     * @param value value to write (possibly null)
253     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
254     */
255    public abstract void write(ByteWriter writer, T value);
256
257    /**
258     * Get the default value for this field encoded as a {@code byte[]} array.
259     *
260     * @return encoded default value
261     */
262    public abstract byte[] getDefaultValue();
263
264    /**
265     * Read and discard a value from the given input.
266     *
267     * @param reader byte input
268     * @throws IllegalArgumentException if invalid input is encountered
269     * @throws IndexOutOfBoundsException if input is truncated
270     */
271    public abstract void skip(ByteReader reader);
272
273    /**
274     * Encode a value as a {@link String} for later decoding by {@link #fromString fromString()}.
275     *
276     * <p>
277     * Each of the characters in the returned {@link String} must be one of the valid XML characters
278     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
279     * </p>
280     *
281     * <p>
282     * The implementation in {@link FieldType} delegates to {@link #toParseableString}.
283     * Subclasses that override this method should also override {@link #fromString fromString()}.
284     * </p>
285     *
286     * @param value actual value, never null
287     * @return string encoding of {@code value}
288     * @throws IllegalArgumentException if {@code value} is null
289     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
290     */
291    public String toString(T value) {
292        if (value == null)
293            throw new IllegalArgumentException("null value");
294        return this.toParseableString(value);
295    }
296
297    /**
298     * Parse a value previously enoded by {@link #toString(Object)}.
299     *
300     * <p>
301     * The implementation in {@link FieldType} creates a new {@link ParseContext} based on {@code string},
302     * delegates to {@link #toParseableString} to parse it, and verifies that all of {@code string} was consumed
303     * during the parse. Subclasses that override this method should also override {@link #fromString toString()}.
304     * </p>
305     *
306     * @param string string previously encoded by {@link #toString(Object)}
307     * @return actual value
308     * @throws IllegalArgumentException if the input is invalid
309     */
310    public T fromString(String string) {
311        final ParseContext ctx = new ParseContext(string);
312        final T value = this.fromParseableString(ctx);
313        if (!ctx.isEOF()) {
314            throw new IllegalArgumentException("found trailing garbage starting with `"
315              + ParseUtil.truncate(ctx.getInput(), 20) + "'");
316        }
317        return value;
318    }
319
320    /**
321     * Encode a possibly null value as a {@link String} for later decoding by {@link #fromParseableString fromParseableString()}.
322     * The string value must be <i>self-delimiting</i>, i.e., decodable even when followed by arbitrary additional characters,
323     * and must not start with whitespace or closing square bracket ({@code "]"}).
324     *
325     * <p>
326     * In addition, each of the characters in the returned {@link String} must be one of the valid XML characters
327     * (tab, newline, carriage return, <code>&#92;u0020 - &#92;ud7ff</code>, and <code>&#92;ue000 - &#92;fffdf</code>).
328     * </p>
329     *
330     * @param value actual value (possibly null)
331     * @return string encoding of {@code value}
332     * @throws IllegalArgumentException if {@code value} is null and this type does not support null
333     * @see <a href="http://www.w3.org/TR/REC-xml/#charsets">The XML 1.0 Specification</a>
334     */
335    public abstract String toParseableString(T value);
336
337    /**
338     * Parse a value previously encoded by {@link #toParseableString toParseableString()} as a self-delimited {@link String}
339     * and positioned at the start of the given parsing context.
340     *
341     * @param context parse context starting with a string previously encoded via {@link #toParseableString toParseableString()}
342     * @return actual value (possibly null)
343     * @throws IllegalArgumentException if the input is invalid
344     */
345    public abstract T fromParseableString(ParseContext context);
346
347    /**
348     * Verify the given object is a valid instance of this {@link FieldType}'s Java type and cast it to that type.
349     *
350     * <p>
351     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
352     * or {@code NullPointerException}, if {@code obj} does not have the correct type, or is an illegal null value.
353     * </p>
354     *
355     * <p>
356     * The implementation in {@link FieldType} simply casts the value using this instance's raw Java type.
357     * Subclasses should override this method to implement any other restrictions, e.g., disallowing null values.
358     * </p>
359     *
360     * @param obj object to validate
361     * @throws IllegalArgumentException if {@code obj} in not of type T
362     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
363     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
364     */
365    @SuppressWarnings("unchecked")
366    public T validate(Object obj) {
367        try {
368            return (T)this.typeToken.getRawType().cast(obj);
369        } catch (ClassCastException e) {
370            throw new IllegalArgumentException(this + " does not support values of type " + obj.getClass().getName());
371        }
372    }
373
374    /**
375     * Compare two values. This method must provide a total ordering of all supported Java values.
376     * If null is a supported Java value, then this method must accept it without throwing an exception
377     * (note, this is a stronger requirement than {@link Comparator} requires).
378     *
379     * @throws IllegalArgumentException if {@code value1} or {@code value2} is null and this type does not support null
380     */
381    @Override
382    public abstract int compare(T value1, T value2);
383
384    /**
385     * Determine whether any of this field type's encoded values start with a {@code 0x00} byte.
386     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
387     *
388     * <p>
389     * Note: changing the return value of this method usually means changing the binary encoding.
390     * </p>
391     *
392     * <p>
393     * The implementation in {@link FieldType} returns {@code true}.
394     * </p>
395     */
396    public boolean hasPrefix0x00() {
397        return true;
398    }
399
400    /**
401     * Determine whether any of this field type's encoded values start with a {@code 0xff} byte.
402     * Certain optimizations are possible when this is not the case. It is safe for this method to always return true.
403     *
404     * <p>
405     * Note: changing the return value of this method usually means changing the binary encoding.
406     * </p>
407     *
408     * <p>
409     * The implementation in {@link FieldType} returns {@code true}.
410     * </p>
411     */
412    public boolean hasPrefix0xff() {
413        return true;
414    }
415
416    /**
417     * Returns this instance's {@linkplain #getName name}.
418     */
419    @Override
420    public String toString() {
421        return this.name;
422    }
423
424    /**
425     * Convenience method for generic type binding.
426     *
427     * @param obj object to validate
428     * @throws IllegalArgumentException if {@code obj} in not of type T
429     * @throws IllegalArgumentException if {@code obj} is null and this type does not support null values
430     * @throws IllegalArgumentException if {@code obj} is in any other way not supported by this {@link FieldType}
431     */
432    void validateAndWrite(ByteWriter writer, Object obj) {
433        this.write(writer, this.validate(obj));
434    }
435
436    @Override
437    public int hashCode() {
438        return this.name.hashCode() ^ this.typeToken.hashCode();
439    }
440
441    @Override
442    public boolean equals(Object obj) {
443        if (obj == this)
444            return true;
445        if (obj == null || obj.getClass() != this.getClass())
446            return false;
447        final FieldType<?> that = (FieldType<?>)obj;
448        return this.name.equals(that.name) && this.typeToken.equals(that.typeToken);
449    }
