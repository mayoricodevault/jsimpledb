008package org.jsimpledb.core;
009
010import java.util.NavigableSet;
011
012/**
013 * Listener interface for notifications of a change in value of a {@link SimpleField}.
014 *
015 * @see Transaction#addSimpleFieldChangeListener Transaction.addSimpleFieldChangeListener()
016 */
017public interface SimpleFieldChangeListener {
018
019    /**
020     * Receive notification of a change in the value of a {@link SimpleField} in an object.
021     *
022     * <p>
023     * Notifications are only delivered when the set of referring objects is non-empty.
024     * </p>
025     *
026     * <p>
027     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
028     * </p>
029     *
030     * @param tx associated transaction
031     * @param id the ID of the affected object (i.e., the object containing the field that changed)
032     * @param storageId the storage ID of the field that changed
033     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
034     * @param referrers all objects that (indirectly) refer to the affected object via the {@code path}
035     * @param oldValue the field's previous value
036     * @param newValue the field's new value
037     * @param <V> Java type for {@code field}'s values
038     */
039    <V> void onSimpleFieldChange(Transaction tx, ObjId id, int storageId,
040      int[] path, NavigableSet<ObjId> referrers, V oldValue, V newValue);
041}
