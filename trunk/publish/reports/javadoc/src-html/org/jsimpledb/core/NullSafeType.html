008package org.jsimpledb.core;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.util.ByteReader;
012import org.jsimpledb.util.ByteWriter;
013
014/**
015 * A {@link FieldType} that wraps any other {@link FieldType} not supporting null values and adds support for null values
016 * by pre-pending to the binary encoding a {@code 0x00} for non-null values or a {@code 0xff} byte for null values. Therefore,
017 * null values sort last.
018 *
019 * <p>
020 * The default value becomes null, for which {@code null} is the parseable {@link String} encoding.
021 * Therefore, {@code null} must not be a valid parseable {@link String} encoding for the wrapped type.
022 * </p>
023 *
024 * <p>
025 * This class will automatically "inline" the {@code 0xff} for null values if the wrapped {@link FieldType}'s
026 * {@link FieldType#hasPrefix0xff} method returns false.
027 * </p>
028 */
029public class NullSafeType<T> extends FieldType<T> {
030
031    /**
032     * Not null sentinel byte value. Used only when inlining is not in effect.
033     */
034    public static final int NOT_NULL_SENTINEL = 0x01;
035
036    /**
037     * Null sentinel byte value.
038     */
039    public static final int NULL_SENTINEL = 0xff;
040
041    private static final byte[] DEFAULT_VALUE = new byte[] { (byte)NULL_SENTINEL };
042
043    /**
044     * The inner {@link FieldType} that this instance wraps.
045     */
046    protected final FieldType<T> inner;
047
048    private final boolean inline;
049
050    /**
051     * Constructor.
052     *
053     * @param name type name
054     * @param inner inner type that is not null safe
055     */
056    public NullSafeType(String name, FieldType<T> inner) {
057        super(name, inner.getTypeToken().wrap());
058        if (inner instanceof NullSafeType)
059            throw new IllegalArgumentException("inner type is already null-safe");
060        this.inner = inner;
061        this.inline = !inner.hasPrefix0xff();
062    }
063
064    /**
065     * Constructor. Takes type name from {@code inner}; therefore, this instance and {@code inner}
066     * cannot be both registered in a {@link FieldTypeRegistry}.
067     *
068     * @param inner inner type that is not null safe
069     */
070    public NullSafeType(FieldType<T> inner) {
071       this(inner.name, inner);
072    }
073
074    @Override
075    public T read(ByteReader reader) {
076        if (this.inline) {
077            if (reader.peek() == NULL_SENTINEL) {
078                reader.skip(1);
079                return null;
080            }
081            return this.inner.read(reader);
082        } else {
083            switch (reader.readByte()) {
084            case NULL_SENTINEL:
085                return null;
086            case NOT_NULL_SENTINEL:
087                return this.inner.read(reader);
088            default:
089                throw new IllegalArgumentException("invalid encoding of " + this);
090            }
091        }
092    }
093
094    @Override
095    public void write(ByteWriter writer, T value) {
096        if (value == null) {
097            writer.writeByte(NULL_SENTINEL);
098            return;
099        }
100        if (!this.inline)
101            writer.writeByte(NOT_NULL_SENTINEL);
102        this.inner.write(writer, value);
103    }
104
105    @Override
106    public byte[] getDefaultValue() {
107        return DEFAULT_VALUE;
108    }
109
110    @Override
111    public void skip(ByteReader reader) {
112        if (this.inline) {
113            final int prefix = reader.peek();
114            switch (prefix) {
115            case NULL_SENTINEL:
116                reader.skip(1);
117                break;
118            default:
119                this.inner.skip(reader);
120                break;
121            }
122        } else {
123            final int prefix = reader.readByte();
124            switch (prefix) {
125            case NULL_SENTINEL:
126                break;
127            case NOT_NULL_SENTINEL:
128                this.inner.skip(reader);
129                break;
130            default:
131                throw new IllegalArgumentException("invalid encoding of " + this);
132            }
133        }
134    }
135
136    @Override
137    public T fromString(String string) {
138        return this.inner.fromString(string);
139    }
140
141    @Override
142    public String toString(T value) {
143        if (value == null)
144            throw new IllegalArgumentException("null value");
145        return this.inner.toString(value);
146    }
147
148    @Override
149    public T fromParseableString(ParseContext context) {
150        return context.tryPattern("null\\b") != null ? null : this.inner.fromParseableString(context);
151    }
152
153    @Override
154    public String toParseableString(T value) {
155        return value == null ? "null" : this.inner.toParseableString(value);
156    }
157
158    @Override
159    public int compare(T value1, T value2) {
160        if (value1 == null)
161            return value2 == null ? 0 : 1;
162        if (value2 == null)
163            return value1 == null ? 0 : -1;
164        return this.inner.compare(value1, value2);
165    }
166
167    @Override
168    public boolean hasPrefix0xff() {
169        return true;
170    }
171
172    @Override
173    public boolean hasPrefix0x00() {
174        return this.inline && this.inner.hasPrefix0x00();
175    }
176
177// Object
178
179    @Override
180    public int hashCode() {
181        return super.hashCode() ^ this.inner.hashCode();
182    }
183
184    @Override
185    public boolean equals(Object obj) {
186        if (obj == this)
187            return true;
188        if (!super.equals(obj))
189            return false;
190        final NullSafeType<?> that = (NullSafeType<?>)obj;
191        return this.inner.equals(that.inner);
192    }
193}
