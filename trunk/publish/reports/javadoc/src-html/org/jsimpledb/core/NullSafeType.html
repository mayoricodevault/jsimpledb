008package org.jsimpledb.core;
009
010import org.dellroad.stuff.string.ParseContext;
011import org.jsimpledb.util.ByteReader;
012import org.jsimpledb.util.ByteWriter;
013
014/**
015 * A {@link FieldType} that wraps any other {@link FieldType} not supporting null values and adds support for null values
016 * by pre-pending to the binary encoding a {@code 0x00} for non-null values or a {@code 0xff} byte for null values. Therefore,
017 * null values sort last.
018 *
019 * <p>
020 * The default value becomes null and {@code "null"} is an accepted string value (therefore, {@code "null"} must not be
021 * a valid string value for the wrapped type).
022 * </p>
023 *
024 * <p>
025 * This class will automatically "inline" the {@code 0xff} for null values if the wrapped {@link FieldType}'s
026 * {@link FieldType#hasPrefix0xff} returns false.
027 * </p>
028 */
029public class NullSafeType<T> extends FieldType<T> {
030
031    /**
032     * Not null sentinel byte value. Used only when inlining is not in effect.
033     */
034    public static final int NOT_NULL_SENTINEL = 0x01;
035
036    /**
037     * Null sentinel byte value.
038     */
039    public static final int NULL_SENTINEL = 0xff;
040
041    private static final byte[] DEFAULT_VALUE = new byte[] { (byte)NULL_SENTINEL };
042
043    private final FieldType<T> inner;
044    private final boolean inline;
045
046    /**
047     * Constructor.
048     *
049     * @param name type name
050     * @param inner inner type that is not null safe
051     */
052    public NullSafeType(String name, FieldType<T> inner) {
053       super(name, inner.getTypeToken().wrap());
054       this.inner = inner;
055       this.inline = !inner.hasPrefix0xff();
056    }
057
058    /**
059     * Constructor. Takes type name from {@code inner}; therefore, this instance and {@code inner}
060     * cannot be both registered in a {@link FieldTypeRegistry}.
061     *
062     * @param inner inner type that is not null safe
063     */
064    public NullSafeType(FieldType<T> inner) {
065       this(inner.name, inner);
066    }
067
068    @Override
069    public T read(ByteReader reader) {
070        if (this.inline) {
071            if (reader.peek() == NULL_SENTINEL) {
072                reader.skip(1);
073                return null;
074            }
075            return this.inner.read(reader);
076        } else {
077            switch (reader.readByte()) {
078            case NULL_SENTINEL:
079                return null;
080            case NOT_NULL_SENTINEL:
081                return this.inner.read(reader);
082            default:
083                throw new IllegalArgumentException("invalid encoding of " + this);
084            }
085        }
086    }
087
088    @Override
089    public void write(ByteWriter writer, T value) {
090        if (value == null) {
091            writer.writeByte(NULL_SENTINEL);
092            return;
093        }
094        if (!this.inline)
095            writer.writeByte(NOT_NULL_SENTINEL);
096        this.inner.write(writer, value);
097    }
098
099    @Override
100    public void copy(ByteReader reader, ByteWriter writer) {
101        if (this.inline) {
102            final int prefix = reader.peek();
103            switch (prefix) {
104            case NULL_SENTINEL:
105                writer.writeByte(prefix);
106                break;
107            default:
108                this.inner.copy(reader, writer);
109                break;
110            }
111        } else {
112            final int prefix = reader.readByte();
113            writer.writeByte(prefix);
114            switch (prefix) {
115            case NULL_SENTINEL:
116                break;
117            case NOT_NULL_SENTINEL:
118                this.inner.copy(reader, writer);
119                break;
120            default:
121                throw new IllegalArgumentException("invalid encoding of " + this);
122            }
123        }
124    }
125
126    @Override
127    public byte[] getDefaultValue() {
128        return DEFAULT_VALUE;
129    }
130
131    @Override
132    public void skip(ByteReader reader) {
133        if (this.inline) {
134            final int prefix = reader.peek();
135            switch (prefix) {
136            case NULL_SENTINEL:
137                reader.skip(1);
138                break;
139            default:
140                this.inner.skip(reader);
141                break;
142            }
143        } else {
144            final int prefix = reader.readByte();
145            switch (prefix) {
146            case NULL_SENTINEL:
147                break;
148            case NOT_NULL_SENTINEL:
149                this.inner.skip(reader);
150                break;
151            default:
152                throw new IllegalArgumentException("invalid encoding of " + this);
153            }
154        }
155    }
156
157    @Override
158    public T fromString(ParseContext context) {
159        return context.tryLiteral("null") ? null : this.inner.fromString(context);
160    }
161
162    @Override
163    public String toString(T value) {
164        return value == null ? "null" : this.inner.toString(value);
165    }
166
167    @Override
168    public int compare(T value1, T value2) {
169        if (value1 == null)
170            return value2 == null ? 0 : 1;
171        if (value2 == null)
172            return value1 == null ? 0 : -1;
173        return this.inner.compare(value1, value2);
174    }
175
176    @Override
177    protected boolean hasPrefix0xff() {
178        return true;
179    }
180
181    @Override
182    protected boolean hasPrefix0x00() {
183        return this.inline && this.inner.hasPrefix0x00();
184    }
185
186// Object
187
188    @Override
189    public int hashCode() {
190        return super.hashCode() ^ this.inner.hashCode();
191    }
192
193    @Override
194    public boolean equals(Object obj) {
195        if (obj == this)
196            return true;
197        if (!super.equals(obj))
198            return false;
199        final NullSafeType<?> that = (NullSafeType<?>)obj;
200        return this.inner.equals(that.inner);
201    }
202}
