008package org.jsimpledb.core;
009
010import java.util.Collections;
011import java.util.HashMap;
012import java.util.Map;
013import java.util.SortedMap;
014import java.util.TreeMap;
015
016import org.jsimpledb.schema.SchemaModel;
017import org.jsimpledb.schema.SchemaObjectType;
018
019/**
020 * Contains information describing one schema version.
021 */
022public class Schema {
023
024    final int versionNumber;
025    final byte[] encodedXML;
026    final SchemaModel schemaModel;
027    final TreeMap<Integer, ObjType> objTypeMap = new TreeMap<>();
028    final TreeMap<Integer, StorageInfo> storageInfoMap = new TreeMap<>();
029
030    /**
031     * Constructor.
032     *
033     * @throws IllegalArgumentException if a storage ID is used twice
034     */
035    Schema(int versionNumber, byte[] encodedXML, SchemaModel schemaModel, FieldTypeRegistry fieldTypeRegistry) {
036
037        // Sanity check
038        if (versionNumber <= 0)
039            throw new IllegalArgumentException("versionNumber <= 0");
040        if (schemaModel == null)
041            throw new IllegalArgumentException("null schemaModel");
042        this.versionNumber = versionNumber;
043        this.encodedXML = encodedXML;
044        this.schemaModel = schemaModel.clone();
045
046        // Build object types
047        for (SchemaObjectType schemaObjectType : this.schemaModel.getSchemaObjectTypes().values()) {
048            final ObjType objType = new ObjType(schemaObjectType, this, fieldTypeRegistry);
049            final int storageId = objType.getStorageId();
050            final ObjType otherObjType = this.objTypeMap.put(storageId, objType);
051            if (otherObjType != null) {
052                throw new IllegalArgumentException("incompatible use of storage ID " + storageId
053                  + " by both " + otherObjType + " and " + objType + " in " + this);
054            }
055        }
056
057        // Create StorageInfo objects and check for conflicts
058        final HashMap<Integer, String> descriptionMap = new HashMap<>();
059        for (ObjType objType : this.objTypeMap.values()) {
060            this.addStorageInfo(objType, descriptionMap);
061            for (Field<?> field : objType.fields.values()) {
062                this.addStorageInfo(field, descriptionMap);
063                if (field instanceof ComplexField) {
064                    final ComplexField<?> complexField = (ComplexField<?>)field;
065                    for (SimpleField<?> subField : complexField.getSubFields())
066                        this.addStorageInfo(subField, descriptionMap);
067                }
068            }
069            for (CompositeIndex index : objType.compositeIndexes.values())
070                this.addStorageInfo(index, descriptionMap);
071        }
072    }
073
074    /**
075     * Get the version number associated with this instance.
076     *
077     * @return version number, always greater than zero
078     */
079    public int getVersionNumber() {
080        return this.versionNumber;
081    }
082
083    /**
084     * Get the {@link SchemaModel} on which this schema version is based.
085     *
086     * @return schema model
087     */
088    public SchemaModel getSchemaModel() {
089        return this.schemaModel;
090    }
091
092    /**
093     * Get all of the {@link ObjType}s that constitute this schema version, indexed by storage ID.
094     *
095     * @return unmodifiable mapping from {@linkplain ObjType#getStorageId storage ID} to {@link ObjType}
096     */
097    public SortedMap<Integer, ObjType> getObjTypes() {
098        return Collections.unmodifiableSortedMap(this.objTypeMap);
099    }
100
101    /**
102     * Get the {@link ObjType} in this schema with the given storage ID.
103     *
104     * @param storageId storage ID
105     * @return the {@link ObjType} with storage ID {@code storageID}
106     * @throws UnknownTypeException if no {@link ObjType} with storage ID {@code storageId} exists
107     */
108    public ObjType getObjType(int storageId) {
109        final ObjType objType = this.objTypeMap.get(storageId);
110        if (objType == null)
111            throw new UnknownTypeException(storageId, this.versionNumber);
112        return objType;
113    }
114
115    @Override
116    public String toString() {
117        return "schema version " + this.versionNumber;
118    }
119
120    // Add new StorageInfo, checking for conflicts
121    private void addStorageInfo(SchemaItem schemaItem, Map<Integer, String> descriptionMap) {
122        final StorageInfo storageInfo = schemaItem.toStorageInfo();
123        final StorageInfo previous = this.storageInfoMap.put(storageInfo.storageId, storageInfo);
124        if (previous != null && !previous.equals(storageInfo)) {
125            throw new IllegalArgumentException("invalid duplicate use of storage ID " + storageInfo.storageId
126              + " for both " + descriptionMap.get(storageInfo.storageId) + " and " + schemaItem);
127        }
128        descriptionMap.put(storageInfo.storageId, "" + schemaItem);
129    }
130}
