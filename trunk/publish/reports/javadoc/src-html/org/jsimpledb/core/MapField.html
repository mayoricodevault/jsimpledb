008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeParameter;
011import com.google.common.reflect.TypeToken;
012
013import java.util.ArrayList;
014import java.util.List;
015import java.util.NavigableMap;
016
017import org.jsimpledb.util.ByteReader;
018import org.jsimpledb.util.ByteWriter;
019
020/**
021 * Map field.
022 *
023 * @param <K> Map key type
024 * @param <V> Map value type
025 */
026public class MapField<K, V> extends ComplexField<NavigableMap<K, V>> {
027
028    public static final String KEY_FIELD_NAME = "key";
029    public static final String VALUE_FIELD_NAME = "value";
030
031    final SimpleField<K> keyField;
032    final SimpleField<V> valueField;
033
034    /**
035     * Constructor.
036     *
037     * @param objType the object type that contains this field
038     * @param name the name of the field
039     * @param storageId field content storage ID
040     * @param version schema version
041     * @param keyField this field's key sub-field
042     * @param valueField this field's value sub-field
043     * @throws IllegalArgumentException if any parameter is null
044     * @throws IllegalArgumentException if {@code storageId} is non-positive
045     */
046    @SuppressWarnings("serial")
047    MapField(String name, int storageId, SchemaVersion version, SimpleField<K> keyField, SimpleField<V> valueField) {
048        super(name, storageId, version, new TypeToken<NavigableMap<K, V>>() { }
049          .where(new TypeParameter<K>() { }, keyField.typeToken.wrap())
050          .where(new TypeParameter<V>() { }, valueField.typeToken.wrap()));
051        this.keyField = keyField;
052        this.valueField = valueField;
053    }
054
055    /**
056     * Get the key field.
057     */
058    public SimpleField<K> getKeyField() {
059        return this.keyField;
060    }
061
062    /**
063     * Get the value field.
064     */
065    public SimpleField<V> getValueField() {
066        return this.valueField;
067    }
068
069    @Override
070    public List<SimpleField<?>> getSubFields() {
071        final ArrayList<SimpleField<?>> list = new ArrayList<>(2);
072        list.add(this.keyField);
073        list.add(this.valueField);
074        return list;
075    }
076
077    @Override
078    MapFieldStorageInfo toStorageInfo() {
079        return new MapFieldStorageInfo(this);
080    }
081
082    @Override
083    public String toString() {
084        return "map field `" + this.name + "' of " + this.keyField.fieldType + ", " + this.valueField.fieldType;
085    }
086
087// Subclass methods
088
089    @Override
090    public NavigableMap<K, V> getValue(Transaction tx, ObjId id) {
091        return new JSMap<K, V>(tx, this, id);
092    }
093
094    @Override
095    void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer) {
096        if (subField == this.keyField) {
097            writer.write(reader);
098            id.writeTo(writer);
099            writer.write(value);
100        } else if (subField == this.valueField) {
101            writer.write(value);
102            id.writeTo(writer);
103            writer.write(reader);
104        } else
105            throw new RuntimeException("internal error");
106    }
107}
