008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeParameter;
011import com.google.common.reflect.TypeToken;
012
013import java.util.ArrayList;
014import java.util.List;
015import java.util.NavigableMap;
016
017import org.jsimpledb.util.ByteReader;
018import org.jsimpledb.util.ByteWriter;
019
020/**
021 * Map field.
022 *
023 * @param <K> Map key type
024 * @param <V> Map value type
025 */
026public class MapField<K, V> extends ComplexField<NavigableMap<K, V>> {
027
028    public static final String KEY_FIELD_NAME = "key";
029    public static final String VALUE_FIELD_NAME = "value";
030
031    final SimpleField<K> keyField;
032    final SimpleField<V> valueField;
033
034    /**
035     * Constructor.
036     *
037     * @param objType the object type that contains this field
038     * @param name the name of the field
039     * @param storageId field content storage ID
040     * @param version schema version
041     * @param keyField this field's key sub-field
042     * @param valueField this field's value sub-field
043     * @throws IllegalArgumentException if any parameter is null
044     * @throws IllegalArgumentException if {@code storageId} is non-positive
045     */
046    @SuppressWarnings("serial")
047    MapField(String name, int storageId, SchemaVersion version, SimpleField<K> keyField, SimpleField<V> valueField) {
048        super(name, storageId, version, new TypeToken<NavigableMap<K, V>>() { }
049          .where(new TypeParameter<K>() { }, keyField.typeToken.wrap())
050          .where(new TypeParameter<V>() { }, valueField.typeToken.wrap()));
051        this.keyField = keyField;
052        this.valueField = valueField;
053        assert this.keyField.parent == null;
054        assert this.valueField.parent == null;
055        this.keyField.parent = this;
056        this.valueField.parent = this;
057    }
058
059    /**
060     * Get the key field.
061     */
062    public SimpleField<K> getKeyField() {
063        return this.keyField;
064    }
065
066    /**
067     * Get the value field.
068     */
069    public SimpleField<V> getValueField() {
070        return this.valueField;
071    }
072
073    @Override
074    public List<SimpleField<?>> getSubFields() {
075        final ArrayList<SimpleField<?>> list = new ArrayList<>(2);
076        list.add(this.keyField);
077        list.add(this.valueField);
078        return list;
079    }
080
081    @Override
082    MapFieldStorageInfo toStorageInfo() {
083        return new MapFieldStorageInfo(this);
084    }
085
086    @Override
087    boolean hasComplexIndex(SimpleField<?> subField) {
088        return true;        // index value = object ID + (key or value)
089    }
090
091    @Override
092    public String toString() {
093        return "map field `" + this.name + "' of " + this.keyField.fieldType + ", " + this.valueField.fieldType;
094    }
095
096    @Override
097    public boolean hasDefaultValue(Transaction tx, ObjId id) {
098        if (tx == null)
099            throw new IllegalArgumentException("null tx");
100        return tx.readMapField(id, this.storageId, false).isEmpty();
101    }
102
103// Subclass methods
104
105    @Override
106    public NavigableMap<K, V> getValue(Transaction tx, ObjId id) {
107        return new JSMap<K, V>(tx, this, id);
108    }
109
110    @Override
111    void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer) {
112        if (subField == this.keyField) {
113            writer.write(reader);
114            id.writeTo(writer);
115            writer.write(value);
116        } else if (subField == this.valueField) {
117            writer.write(value);
118            id.writeTo(writer);
119            writer.write(reader);
120        } else
121            throw new RuntimeException("internal error");
122    }
123}
