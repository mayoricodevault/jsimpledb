008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeToken;
011
012import org.jsimpledb.util.ByteWriter;
013import org.jsimpledb.util.UnsignedIntEncoder;
014
015/**
016 * Represents a field in an {@link ObjType} or a ({@linkplain SimpleField simple}) sub-field of a {@link ComplexField}.
017 *
018 * @param <T> Java type for the field's values
019 */
020public abstract class Field<T> extends SchemaItem {
021
022    final TypeToken<T> typeToken;
023
024    /**
025     * Constructor for normal fields of an {@link ObjType}.
026     *
027     * @param name the name of the field
028     * @param storageId field storage ID
029     * @param version schema version
030     * @param typeToken Java type for the field's values
031     * @throws IllegalArgumentException if any parameter is null
032     * @throws IllegalArgumentException if {@code name} is invalid
033     * @throws IllegalArgumentException if {@code storageId} is non-positive
034     */
035    Field(String name, int storageId, SchemaVersion version, TypeToken<T> typeToken) {
036        super(name, storageId, version);
037        if (typeToken == null)
038            throw new IllegalArgumentException("null typeToken");
039        this.typeToken = typeToken;
040    }
041
042    /**
043     * Get the Java type corresponding to this field.
044     */
045    public TypeToken<T> getTypeToken() {
046        return this.typeToken;
047    }
048
049    /**
050     * Determine if this field in the specified object has its default value in the specified {@link Transaction}.
051     *
052     * @param tx {@link Transaction} containing field state
053     * @param id object ID
054     * @return true if this field is set to its initial default value in object {@code id}, otherwise false
055     * @throws IllegalArgumentException if either parameter is null
056     */
057    public abstract boolean hasDefaultValue(Transaction tx, ObjId id);
058
059    /**
060     * Build the key (or key prefix) for this field in the given object.
061     */
062    byte[] buildKey(ObjId id) {
063        return Field.buildKey(id, this.storageId);
064    }
065
066    /**
067     * Build the key (or key prefix) for this field in the given object.
068     */
069    static byte[] buildKey(ObjId id, int storageId) {
070        final ByteWriter writer = new ByteWriter();
071        id.writeTo(writer);
072        UnsignedIntEncoder.write(writer, storageId);
073        return writer.getBytes();
074    }
075
076    /**
077     * Determine if this field and the given field are exactly equivalent.
078     */
079    abstract boolean isEquivalent(Field<?> field);
080}
