008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeToken;
011
012import org.jsimpledb.util.ByteWriter;
013import org.jsimpledb.util.UnsignedIntEncoder;
014
015/**
016 * Represents a field in an {@link ObjType} or a ({@linkplain SimpleField simple}) sub-field of a {@link ComplexField}.
017 *
018 * @param <T> Java type for the field's values
019 */
020public abstract class Field<T> extends SchemaItem {
021
022    final TypeToken<T> typeToken;
023
024    /**
025     * Constructor for normal fields of an {@link ObjType}.
026     *
027     * @param name the name of the field
028     * @param storageId field storage ID
029     * @param version schema version
030     * @param typeToken Java type for the field's values
031     * @throws IllegalArgumentException if any parameter is null
032     * @throws IllegalArgumentException if {@code name} is invalid
033     * @throws IllegalArgumentException if {@code storageId} is non-positive
034     */
035    Field(String name, int storageId, SchemaVersion version, TypeToken<T> typeToken) {
036        super(name, storageId, version);
037        if (typeToken == null)
038            throw new IllegalArgumentException("null typeToken");
039        this.typeToken = typeToken;
040    }
041
042// Public methods
043
044    /**
045     * Get the Java type corresponding to this field.
046     */
047    public TypeToken<T> getTypeToken() {
048        return this.typeToken;
049    }
050
051    /**
052     * Get the value of this field in the given object.
053     * Does not alter the schema version of the object.
054     *
055     * @param tx transaction
056     * @param id object id
057     * @throws DeletedObjectException if no object with ID equal to {@code id} is found
058     * @throws UnknownTypeException if {@code id} specifies an unknown object type
059     * @throws UnknownFieldException if this field does not exist in the specified object
060     * @throws StaleTransactionException if this transaction is no longer usable
061     * @throws IllegalArgumentException if either parameter is null
062     */
063    public abstract T getValue(Transaction tx, ObjId id);
064
065    /**
066     * Determine if this field in the specified object has its default value in the specified {@link Transaction}.
067     *
068     * @param tx {@link Transaction} containing field state
069     * @param id object ID
070     * @return true if this field is set to its initial default value in object {@code id}, otherwise false
071     * @throws DeletedObjectException if no object with ID equal to {@code id} is found
072     * @throws UnknownTypeException if {@code id} specifies an unknown object type
073     * @throws StaleTransactionException if this transaction is no longer usable
074     * @throws IllegalArgumentException if either parameter is null
075     */
076    public abstract boolean hasDefaultValue(Transaction tx, ObjId id);
077
078    /**
079     * Apply visitor pattern.
080     *
081     * @param target target to invoke
082     * @return value from the method of {@code target} corresponding to this instance's type
083     * @throws NullPointerException if {@code target} is null
084     */
085    public abstract <R> R visit(FieldSwitch<R> target);
086
087// Non-public methods
088
089    @Override
090    abstract FieldStorageInfo toStorageInfo();
091
092    /**
093     * Copy field value between transactions.
094     *
095     * <p>
096     * This method assumes both objects exist and both transactions are locked.
097     * </p>
098     */
099    abstract void copy(ObjId srcId, ObjId dstId, Transaction srcTx, Transaction dstTx);
100
101    /**
102     * Build the key (or key prefix) for this field in the given object.
103     */
104    byte[] buildKey(ObjId id) {
105        return Field.buildKey(id, this.storageId);
106    }
107
108    /**
109     * Build the key (or key prefix) for this field in the given object.
110     */
111    static byte[] buildKey(ObjId id, int storageId) {
112        final ByteWriter writer = new ByteWriter();
113        id.writeTo(writer);
114        UnsignedIntEncoder.write(writer, storageId);
115        return writer.getBytes();
116    }
117}
