008package org.jsimpledb.core;
009
010import java.util.NavigableSet;
011
012/**
013 * Listener interface for notifications of a change in value of a {@link ListField}.
014 *
015 * @see Transaction#addListFieldChangeListener Transaction.addListFieldChangeListener()
016 */
017public interface ListFieldChangeListener {
018
019    /**
020     * Receive notification of the addition (or insertion) of an element to a {@link ListField}.
021     *
022     * <p>
023     * Notifications are only delivered when {@code referrers} is non-empty.
024     * </p>
025     *
026     * <p>
027     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
028     * </p>
029     *
030     * @param tx associated transaction
031     * @param id the ID of the affected object (i.e., the object containing the field that changed)
032     * @param storageId the storage ID of the field that changed
033     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
034     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
035     * @param index the position in the list at which {@code value} was added or inserted
036     * @param value the value added to the list
037     * @param <E> Java type for {@code field}'s elements
038     */
039    <E> void onListFieldAdd(Transaction tx, ObjId id, int storageId, int[] path, NavigableSet<ObjId> referrers, int index, E value);
040
041    /**
042     * Receive notification of a removal of an element from a {@link ListField}.
043     *
044     * <p>
045     * Notifications are only delivered when {@code referrers} is non-empty.
046     * </p>
047     *
048     * <p>
049     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
050     * </p>
051     *
052     * @param tx associated transaction
053     * @param id the ID of the affected object (i.e., the object containing the field that changed)
054     * @param storageId the storage ID of the field that changed
055     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
056     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
057     * @param index the position in the list at which {@code value} was removed
058     * @param value the value removed from the list
059     * @param <E> Java type for {@code field}'s elements
060     */
061    <E> void onListFieldRemove(Transaction tx, ObjId id, int storageId,
062      int[] path, NavigableSet<ObjId> referrers, int index, E value);
063
064    /**
065     * Receive notification of the change in element at a specific position in a {@link ListField}.
066     *
067     * <p>
068     * Notifications are only delivered when {@code referrers} is non-empty.
069     * </p>
070     *
071     * <p>
072     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
073     * </p>
074     *
075     * @param tx associated transaction
076     * @param id the ID of the affected object (i.e., the object containing the field that changed)
077     * @param storageId the storage ID of the field that changed
078     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
079     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
080     * @param index the position in the list at which the change occurred
081     * @param oldValue the old value at list position {@code index}
082     * @param newValue the new value at list position {@code index}
083     * @param <E> Java type for {@code field}'s elements
084     */
085    <E> void onListFieldReplace(Transaction tx, ObjId id, int storageId,
086      int[] path, NavigableSet<ObjId> referrers, int index, E oldValue, E newValue);
087
088    /**
089     * Receive notification of the clearing of a {@link ListField}.
090     *
091     * <p>
092     * This method is only used when the whole list is cleared; when a sub-range of a list is cleared, individual
093     * {@linkplain #onListFieldRemove onListFieldRemove()} invocations for each removed element are made instead.
094     * </p>
095     *
096     * <p>
097     * Notifications are only delivered when {@code referrers} is non-empty.
098     * </p>
099     *
100     * <p>
101     * Notifications are delivered in the same thread that made the change, before the outermost mutation operation returns.
102     * </p>
103     *
104     * @param tx associated transaction
105     * @param id the ID of the affected object (i.e., the object containing the field that was cleared)
106     * @param storageId the storage ID of the field that changed
107     * @param path path of reference fields (represented by storage IDs) that lead to {@code id}
108     * @param referrers all objects that (indirectly) refer to the affected object via {@code path}
109     */
110    void onListFieldClear(Transaction tx, ObjId id, int storageId, int[] path, NavigableSet<ObjId> referrers);
111}
