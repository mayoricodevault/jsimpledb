008package org.jsimpledb.core;
009
010/**
011 * Default {@link FieldType} for {@link Enum} types. Null values are supported.
012 *
013 * <p>
014 * Enums present a problem because they have both a {@link Enum#name name()} and an {@link Enum#ordinal ordinal()} value,
015 * and either or both of these might change over time (perhaps due to some code refactoring). Therefore,
016 * it's possible to encounter an unknown value when decoding an enum constant. To gracefully handle these situations,
017 * the binary encoding includes both the constant's {@link Enum#ordinal ordinal()}s and its {@link Enum#name name()},
018 * and when decoding an enum constant the following logic applies:
019 * <ol>
020 *  <li>If the name matches an enum constant, that constant is returned, regardless of whether the ordinal matches or not</li>
021 *  <li>If the name does not match but the ordinal matches an enum constant, that constant is returned</li>
022 *  <li>Otherwise, neither the name nor the ordinal matches, and null is returned</li>
023 * </ol>
024 * </p>
025 *
026 * <p>
027 * As with all {@link FieldType}s, a separate string encoding is also supported for enum values: this is just the enum name
028 * surrounded by square brackets. If an unrecognized name is encountered when {@linkplain #fromString decoding
029 * from a string}, null is returned.
030 * </p>
031 *
032 */
033public class EnumType<T extends Enum<T>> extends NullSafeType<T> {
034
035    private final IntegerType intType = new IntegerType();
036    private final StringType stringType = new StringType();
037
038    /**
039     * Constructor.
040     *
041     * @throws ClassCastException if {@code type} is not an {@link Enum}
042     */
043    EnumType(Class<T> type) {
044        super(new RawEnumType<T>(type));
045    }
046}
