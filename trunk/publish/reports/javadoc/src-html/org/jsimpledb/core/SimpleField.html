008package org.jsimpledb.core;
009
010import java.util.Arrays;
011
012import org.jsimpledb.util.ByteWriter;
013import org.jsimpledb.util.UnsignedIntEncoder;
014
015/**
016 * A simple {@link Field}.
017 *
018 * <p>
019 * {@link SimpleField}s have these requirements and properties:
020 *  <ul>
021 *  <li>They have an associated {@link FieldType} representing the domain of possible values</li>
022 *  <li>Theys can serve as the element sub-field for {@link SetField}s and {@link ListField}s,
023 *      and the key and value sub-fields for {@link MapField}s.</li>
024 *  <li>They can be indexed.</li>
025 * </p>
026 *
027 * @param <T> Java type for the field's values
028 */
029public class SimpleField<T> extends Field<T> {
030
031    final FieldType<T> fieldType;
032    final boolean indexed;
033
034    /**
035     * Constructor.
036     *
037     * @param name the name of the field
038     * @param storageId field storage ID
039     * @param version schema version
040     * @param fieldType field type
041     * @param indexed whether this field is indexed
042     * @throws IllegalArgumentException if any parameter is null
043     * @throws IllegalArgumentException if {@code name} is invalid
044     * @throws IllegalArgumentException if {@code storageId} is zero or less
045     */
046    SimpleField(String name, int storageId, SchemaVersion version, FieldType<T> fieldType, boolean indexed) {
047        super(name, storageId, version, fieldType.getTypeToken());
048        this.fieldType = fieldType;
049        this.indexed = indexed;
050    }
051
052    /**
053     * Get the {@link FieldType} associated with this field.
054     */
055    public FieldType<T> getFieldType() {
056        return this.fieldType;
057    }
058
059    /**
060     * Check compatibility with another {@link SimpleField} across a schema change.
061     * To be compatible, the two fields must be exactly the same in terms of binary encoding, Java representation,
062     * and default value.
063     * This is in effect an {@code #equals equals()} test with respect to those aspects. Note that compatibililty
064     * does not necessarily imply the fields are both indexed or both not indexed.
065     *
066     * <p>
067     * The implementation in {@link SimpleField} checks that the two fields have equivalent {@link FieldType}s.
068     * </p>
069     *
070     * @param that field to check for compatibility
071     * @throws NullPointerException if {@code that} is null
072     */
073    boolean isSchemaChangeCompatible(SimpleField<?> that) {
074        return this.fieldType.equals(that.fieldType);
075    }
076
077    SimpleFieldStorageInfo toStorageInfo(int superFieldStorageId) {
078        return new SimpleFieldStorageInfo(this, superFieldStorageId);
079    }
080
081    @Override
082    boolean isEquivalent(Field<?> field) {
083        if (field.getClass() != this.getClass())
084            return false;
085        final SimpleField<?> that = (SimpleField<?>)field;
086        return this.fieldType.equals(that.fieldType) && this.indexed == that.indexed;
087    }
088
089    @Override
090    public String toString() {
091        return "field `" + this.name + "' of type " + this.fieldType.typeToken;
092    }
093
094    /**
095     * Determine whether this field is indexed.
096     */
097    public boolean isIndexed() {
098        return this.indexed;
099    }
100
101    /**
102     * Encode the given value.
103     *
104     * @param obj value to encode
105     * @return encoded value, or null if value is the default value
106     * @throws IllegalArgumentException if {@code obj} cannot be encoded
107     */
108    byte[] encode(Object obj) {
109        T value;
110        try {
111            value = this.fieldType.validate(obj);
112        } catch (IllegalArgumentException e) {
113            throw new IllegalArgumentException("can't set " + this + " to value " + obj + ": " + e.getMessage(), e);
114        }
115        final ByteWriter writer = new ByteWriter();
116        this.fieldType.write(writer, value);
117        final byte[] result = writer.getBytes();
118        return Arrays.equals(result, this.fieldType.getDefaultValue()) ? null : result;
119    }
120
121    /**
122     * Bulid the index entry for a given object having the given value in this field.
123     * The encoded field value is read from the given {@link ByteReader}.
124     *
125     * @param id object containing the field
126     * @param value encoded field value, or null for default value
127     * @return index key
128     */
129    byte[] buildIndexKey(ObjId id, byte[] value) {
130
131        // Sanity check
132        if (!this.indexed)
133            throw new IllegalArgumentException(this + " is not indexed");
134
135        // Get default value if necessary
136        if (value == null)
137            value = this.fieldType.getDefaultValue();
138
139        // Build index entry
140        final ByteWriter writer = new ByteWriter();
141        UnsignedIntEncoder.write(writer, this.storageId);
142        writer.write(value);
143        id.writeTo(writer);
144
145        // Return index entry
146        return writer.getBytes();
147    }
148}
