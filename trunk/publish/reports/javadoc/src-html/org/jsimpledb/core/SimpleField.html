008package org.jsimpledb.core;
009
010import java.util.Arrays;
011
012import org.jsimpledb.util.ByteWriter;
013import org.jsimpledb.util.UnsignedIntEncoder;
014
015/**
016 * A simple {@link Field}.
017 *
018 * <p>
019 * {@link SimpleField}s have these requirements and properties:
020 *  <ul>
021 *  <li>They have an associated {@link FieldType} representing the domain of possible values</li>
022 *  <li>Theys can serve as the element sub-field for {@link SetField}s and {@link ListField}s,
023 *      and the key and value sub-fields for {@link MapField}s.</li>
024 *  <li>They can be indexed.</li>
025 * </p>
026 *
027 * @param <T> Java type for the field's values
028 */
029public class SimpleField<T> extends Field<T> {
030
031    final FieldType<T> fieldType;
032    final boolean indexed;
033
034    ComplexField<?> parent;
035
036    /**
037     * Constructor.
038     *
039     * @param name the name of the field
040     * @param storageId field storage ID
041     * @param version schema version
042     * @param fieldType field type
043     * @param indexed whether this field is indexed
044     * @throws IllegalArgumentException if any parameter is null
045     * @throws IllegalArgumentException if {@code name} is invalid
046     * @throws IllegalArgumentException if {@code storageId} is zero or less
047     */
048    SimpleField(String name, int storageId, SchemaVersion version, FieldType<T> fieldType, boolean indexed) {
049        super(name, storageId, version, fieldType.getTypeToken());
050        this.fieldType = fieldType;
051        this.indexed = indexed;
052    }
053
054// Public methods
055
056    /**
057     * Get the {@link FieldType} associated with this field.
058     */
059    public FieldType<T> getFieldType() {
060        return this.fieldType;
061    }
062
063    /**
064     * Determine whether this field is indexed.
065     */
066    public boolean isIndexed() {
067        return this.indexed;
068    }
069
070    /**
071     * Set the value of this field in the given object.
072     * Does not alter the schema version of the object.
073     *
074     * @param tx transaction
075     * @param id object id
076     * @param value new value
077     * @throws DeletedObjectException if no object with ID equal to {@code id} is found
078     * @throws StaleTransactionException if this transaction is no longer usable
079     * @throws IllegalArgumentException if {@code tx} or {@code id} is null
080     */
081    public void setValue(Transaction tx, ObjId id, T value) {
082        if (tx == null)
083            throw new IllegalArgumentException("null tx");
084        tx.writeSimpleField(id, this.storageId, value, false);
085    }
086
087    @Override
088    @SuppressWarnings("unchecked")
089    public T getValue(Transaction tx, ObjId id) {
090        if (tx == null)
091            throw new IllegalArgumentException("null tx");
092        return (T)tx.readSimpleField(id, this.storageId, false);
093    }
094
095    @Override
096    public boolean hasDefaultValue(Transaction tx, ObjId id) {
097        if (tx == null)
098            throw new IllegalArgumentException("null tx");
099        return tx.hasDefaultValue(id, this);
100    }
101
102    @Override
103    public <R> R visit(FieldSwitch<R> target) {
104        return target.caseSimpleField(this);
105    }
106
107    @Override
108    public String toString() {
109        return (this.indexed ? "indexed " : "") + "field `" + this.name + "' of type " + this.fieldType.typeToken;
110    }
111
112// Non-public methods
113
114    /**
115     * Check compatibility with another {@link SimpleField} across a schema change.
116     * To be compatible, the two fields must be exactly the same in terms of binary encoding, Java representation,
117     * and default value.
118     * This is in effect an {@code #equals equals()} test with respect to those aspects. Note that compatibililty
119     * does not necessarily imply the fields are both indexed or both not indexed.
120     *
121     * <p>
122     * The implementation in {@link SimpleField} checks that the two fields have equivalent {@link FieldType}s.
123     * </p>
124     *
125     * @param that field to check for compatibility
126     * @throws NullPointerException if {@code that} is null
127     */
128    boolean isSchemaChangeCompatible(SimpleField<?> that) {
129        return this.fieldType.equals(that.fieldType);
130    }
131
132    @Override
133    SimpleFieldStorageInfo toStorageInfo() {
134        return new SimpleFieldStorageInfo(this, this.parent != null ? this.parent.storageId : 0);
135    }
136
137    @Override
138    void copy(ObjId srcId, ObjId dstId, Transaction srcTx, Transaction dstTx) {
139        dstTx.writeSimpleField(dstId, this.storageId, srcTx.readSimpleField(srcId, this.storageId, false), false);
140    }
141
142    @Override
143    boolean isEquivalent(Field<?> field) {
144        if (field.getClass() != this.getClass())
145            return false;
146        final SimpleField<?> that = (SimpleField<?>)field;
147        return this.fieldType.equals(that.fieldType) && this.indexed == that.indexed;
148    }
149
150    /**
151     * Encode the given value.
152     *
153     * @param obj value to encode
154     * @return encoded value, or null if value is the default value
155     * @throws IllegalArgumentException if {@code obj} cannot be encoded
156     */
157    byte[] encode(Object obj) {
158        T value;
159        try {
160            value = this.fieldType.validate(obj);
161        } catch (IllegalArgumentException e) {
162            throw new IllegalArgumentException("can't set " + this + " to value " + obj + ": " + e.getMessage(), e);
163        }
164        final ByteWriter writer = new ByteWriter();
165        this.fieldType.write(writer, value);
166        final byte[] result = writer.getBytes();
167        return Arrays.equals(result, this.fieldType.getDefaultValue()) ? null : result;
168    }
169
170    /**
171     * Bulid the index entry for a given object having the given value in this field.
172     * The encoded field value is read from the given {@link ByteReader}.
173     *
174     * @param id object containing the field
175     * @param value encoded field value, or null for default value
176     * @return index key
177     */
178    byte[] buildIndexKey(ObjId id, byte[] value) {
179
180        // Sanity check
181        if (!this.indexed)
182            throw new IllegalArgumentException(this + " is not indexed");
183
184        // Get default value if necessary
185        if (value == null)
186            value = this.fieldType.getDefaultValue();
187
188        // Build index entry
189        final ByteWriter writer = new ByteWriter();
190        UnsignedIntEncoder.write(writer, this.storageId);
191        writer.write(value);
192        id.writeTo(writer);
193
194        // Return index entry
195        return writer.getBytes();
196    }
197}
