008package org.jsimpledb.core;
009
010/**
011 * Describes what action to take when an object that is still referred to by some other object is deleted.
012 *
013 * @see org.jsimpledb.core.Transaction#delete Transaction.delete()
014 */
015public enum DeleteAction {
016
017    /**
018     * Do nothing. The reference will still exist, but subsequent attempts to access the fields of the deleted object
019     * will result in {@link DeletedObjectException}.
020     */
021    NOTHING,
022
023    /**
024     * Disallow deleting the object, instead throwing {@link ReferencedObjectException}. This is the default if not specified.
025     *
026     * <p>
027     * Note: deleting an object that is only referred to by itself will not cause any exception to be thrown.
028     * </p>
029     */
030    EXCEPTION,
031
032    /**
033     * Remove the reference, either by setting it to null (in the case of a {@link SimpleField} in an object),
034     * or by removing the corresponding collection element or key/value pair (in the case of a {@link ComplexField} sub-field).
035     */
036    UNREFERENCE,
037
038    /**
039     * Also delete the object containing the reference. This action will be repeated recursively, if necessary.
040     */
041    DELETE;
042}
