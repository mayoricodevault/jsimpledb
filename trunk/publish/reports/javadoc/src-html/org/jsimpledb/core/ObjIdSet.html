008package org.jsimpledb.core;
009
010import java.util.AbstractSet;
011import java.util.ConcurrentModificationException;
012import java.util.Iterator;
013import java.util.NoSuchElementException;
014
015/**
016 * A set of {@link ObjId}s.
017 *
018 * <p>
019 * Typically used when performing a copy operation of a (possibly) cyclic graph of object references,
020 * to keep track of which objects have already been copied.
021 * </p>
022 *
023 * <p>
024 * This implementation is space optimized for the 64-bits of information contained in an {@link ObjId}.
025 * Instances do not accept null values and are not thread safe.
026 * </p>
027 *
028 * @see org.jsimpledb.JObject#copyTo
029 */
030public class ObjIdSet extends AbstractSet<ObjId> {
031
032    // Algorithm described here: http://en.wikipedia.org/wiki/Open_addressing
033
034    private static final int MIN_ARRAY_LENGTH = 0x00000008;
035    private static final int MAX_ARRAY_LENGTH = 0x40000000;
036
037    private long[] array;
038    private int size;
039    private volatile int modcount;
040
041    public ObjIdSet() {
042        this.array = new long[MIN_ARRAY_LENGTH];
043    }
044
045    @Override
046    public Iterator<ObjId> iterator() {
047        return new Iter();
048    }
049
050    @Override
051    public int size() {
052        return this.size;
053    }
054
055    @Override
056    public boolean contains(Object obj) {
057
058        // Check type
059        if (!(obj instanceof ObjId))
060            return false;
061        final long value = ((ObjId)obj).asLong();
062        assert value != 0;
063
064        // Check slot for value
065        if (this.array[this.findSlot(value)] == value)
066            return true;
067        assert this.array[this.findSlot(value)] == 0;
068        return false;
069    }
070
071    @Override
072    public boolean add(ObjId id) {
073        final long value = id.asLong();
074        if (!this.insert(value))
075            return false;
076        if (++this.size > (this.array.length >> 1))
077            this.expand();
078        this.modcount++;
079        return true;
080    }
081
082    @Override
083    public boolean remove(Object obj) {
084        if (!(obj instanceof ObjId))
085            return false;
086        final long value = ((ObjId)obj).asLong();
087        assert value != 0;
088        if (!this.exsert(value))
089            return false;
090        if (--this.size < (this.array.length >> 2))
091            this.shrink();
092        this.modcount++;
093        return true;
094    }
095
096    @Override
097    public void clear() {
098        this.array = new long[MIN_ARRAY_LENGTH];
099        this.size = 0;
100        this.modcount++;
101    }
102
103    /**
104     * Produce a debug dump of this instance.
105     */
106    String debugDump() {
107        final StringBuilder buf = new StringBuilder();
108        buf.append("OBJIDSET: size=" + this.size + " len=" + this.array.length + " modcount=" + this.modcount);
109        for (int i = 0; i < this.array.length; i++)
110            buf.append('\n').append(String.format(" [%2d] %016x (hash %d)", i, this.array[i], this.hash(this.array[i])));
111        return buf.toString();
112    }
113
114// Internal methods
115
116    private boolean insert(long value) {
117
118        // Find slot for value
119        assert value != 0;
120        final int index = this.findSlot(value);
121        if (this.array[index] == value)
122            return false;
123
124        // Add value
125        assert this.array[index] == 0;
126        this.array[index] = value;
127        return true;
128    }
129
130    private boolean exsert(long value) {
131
132        // Find slot for value
133        final int index = this.findSlot(value);
134        if (this.array[index] == 0)
135            return false;
136        assert this.array[index] == value;
137
138        // Remove value and fixup subsequent entries
139        int i = index;                                                  // i points to the new empty slot
140        int j = index;                                                  // j points to the next slot to fixup
141        while (true) {
142            this.array[i] = 0;
143            long jvalue;
144            while (true) {
145                j = (j + 1) & (this.array.length - 1);
146                jvalue = this.array[j];
147                if (jvalue == 0)                                        // end of hash chain, no more fixups required
148                    return true;
149                final int k = this.hash(jvalue);                        // find where jvalue's hash chain started
150                if (i <= j ? (i < k && k <= j) : (i < k || k <= j))     // jvalue is between i and j, so it's not cut off
151                    continue;
152                break;                                                  // jvalue is cut off from its hash chain
153            }
154            this.array[i] = jvalue;                                     // move jvalue back into its hash chain
155            i = j;                                                      // restart fixups at jvalue's old location
156        }
157    }
158
159    private int findSlot(long value) {
160        assert value != 0;
161        int index = this.hash(value) & (this.array.length - 1);
162        while (true) {
163            final long existing = this.array[index];
164            if (existing == 0 || existing == value)
165                return index;
166            index = (index + 1) & (this.array.length - 1);
167        }
168    }
169
170    private int hash(long value) {
171        return ((int)value ^ (int)(value >>> 32)) & (this.array.length - 1);
172    }
173
174    private void expand() {
175        final long[] oldarray = this.array;
176        if (oldarray.length == MAX_ARRAY_LENGTH)
177            return;
178        this.array = new long[oldarray.length << 1];
179        this.rehash(oldarray);
180    }
181
182    private void shrink() {
183        final long[] oldarray = this.array;
184        if (oldarray.length == MIN_ARRAY_LENGTH)
185            return;
186        this.array = new long[oldarray.length >> 1];
187        this.rehash(oldarray);
188    }
189
190    private void rehash(long[] oldarray) {
191        for (int i = 0; i < oldarray.length; i++) {
192            final long value = oldarray[i];
193            if (value != 0)
194                this.insert(value);
195        }
196    }
197
198// Iter
199
200    private class Iter implements Iterator<ObjId> {
201
202        private int modcount = ObjIdSet.this.modcount;
203        private long removee;
204        private int next;
205
206        @Override
207        public boolean hasNext() {
208            return this.findNext(false) != -1;
209        }
210
211        @Override
212        public ObjId next() {
213            final int index = this.findNext(true);
214            if (index == -1)
215                throw new NoSuchElementException();
216            final long value = ObjIdSet.this.array[index];
217            assert value != 0;
218            this.removee = value;
219            return new ObjId(value);
220        }
221
222        @Override
223        public void remove() {
224            if (this.removee == 0)
225                throw new IllegalStateException();
226            if (this.modcount != ObjIdSet.this.modcount)
227                throw new ConcurrentModificationException();
228            ObjIdSet.this.remove(new ObjId(this.removee));
229            this.modcount++;
230            this.removee = 0;
231        }
232
233        private int findNext(boolean advance) {
234            if (this.modcount != ObjIdSet.this.modcount)
235                throw new ConcurrentModificationException();
236            for (int index = this.next; index < ObjIdSet.this.array.length; index++) {
237                if (ObjIdSet.this.array[index] != 0) {
238                    this.next = advance ? index + 1 : index;
239                    return index;
240                }
241            }
242            return -1;
243        }
244    }
245}
