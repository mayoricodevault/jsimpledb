036/**
037 * Provides an object database abstraction on top of a key/value database.
038 *
039 * <p>
040 * Includes support for:
041 * <ul>
042 *  <li>Objects and fields defined by a {@link SchemaModel}</li>
043 *  <li>Simple values fields containing any atomic type</li>
044 *  <li>User-defined custom {@link FieldType}s</li>
045 *  <li>Invertable reference fields with strong referential integrity and configurable delete cascading</li>
046 *  <li>Complex fields of type {@link java.util.NavigableSet}, {@link java.util.List}, and {@link java.util.NavigableMap}</li>
047 *  <li>Configurable indexing of any simple or complex field</li>
048 *  <li>Notification of field changes as seen through an arbitrary path of references</li>
049 *  <li>Automatic schema tracking and object versioning with schema change notification support</li>
050 * </ul>
051 * </p>
052 *
053 * <p>
054 * See {@link Transaction} for further details on the above functionality.
055 * </p>
056 *
057 * <p>
058 * This class defines an abstraction layer that usually sits below a {@link org.jsimpledb.JSimpleDB} but is completely
059 * independent of {@link org.jsimpledb.JSimpleDB} and can be used on its own.
060 * Compared to {@link org.jsimpledb.JSimpleDB}, a {@link Database} has these differences:
061 * <ul>
062 *  <li>A {@link SchemaModel} must be explicitly provided to define the schema in use</li>
063 *  <li>Database objects are represented by {@link ObjId}s instead of Java objects, and there is no notion of object sub-type</li>
064 *  <li>All object types and fields must be referenced by storage ID using a {@link Transaction} object</li>
065 *  <li>There is no automatic validation support</li>
066 * </ul>
067 * </p>
068 *
069 * @see Transaction
070 * @see org.jsimpledb
071 */
072public class Database {
073

141    /**
142     * Create a new {@link Transaction} on this database and use the given schema version to access objects and fields.
143     *
144     * <p>
145     * <b>Schema Versions</b>
146     * </p>
147     *
148     * <p>
149     * Within each {@link Database} is stored a record of all schema versions previously used with the database.
150     * When creating a new transaction, the following checks are applied:
151     * <ul>
152     *  <li>If a schema with version number {@code version} is recorded in the database, and the given schema
153     *      matches it, then this method succeeds, and the {@link Transaction} will use schema version {@code version}.</li>
154     *  <li>If a schema with version number {@code version} is recorded in the database, and the given schema does not
155     *      match it, then this method fails and throws {@link InvalidSchemaException}.</li>
156     *  <li>If {@code allowNewSchema} is false, and no schema with version number {@code version} has yet been
157     *      recorded in the database, then this method fails and throws {@link InvalidSchemaException}.</li>
158     *  <li>If {@code allowNewSchema} is true, and no schema with version number {@code version} has yet been
159     *      recorded in the database, then {@code schemaModel} is checked for compabitility with all of the other
160     *      schemas recorded in the database; if so, this method succeeds, {@code schema} is recorded in the database
161     *      with version number {@code version}, and the {@link Transaction} will use schema version {@code version}.</li>
162     *  <li>If the schema compatibility check in the previous step fails, an {@link InvalidSchemaException} is thrown.
163     *      This can happen the same storage ID is used inconsistently in two schema versions.</li>
164     * </ul>
165     * </p>
166     *
167     * <p>
168     * Fields and objects have assigned storage IDs, and these identify the schema object across all schema versions.
169     * Once a storage ID is assigned, it cannot be re-assigned to a different schema object. Fields must have a consistent
170     * type and parent entity (object type or complex field) in all schema versions in which they appear.
171     * </p>
172     *
173     * <p>
174     * <b>Object Versions</b>
175     * </p>
176     *
177     * <p>
178     * Each object in a {@link Database} contains an internal version number that indicates its current schema version;
179     * this in turn dictates what fields that object contains.
180     * </p>
181     *
182     * <p>
183     * When an object is accessed during a {@link Transaction}, the object's version is compared to the {@code version} associated
184     * with that {@link Transaction}. If the versions are the same, no version change occurs and fields are accessed normally.
185     * </p>
186     *
187     * <p>
188     * If the object has a version {@code oldVersion} different from {@code version}, then the object version is changed
189     * to {@code version}. This may cause fields to be added or removed, as follows:
190     * <ul>
191     *  <li>Compatible fields that are common to both schema versions remain unchanged; fields are compatible
192     *      if they have the same storage ID and type (for complex fields, sub-fields must also be compatible).</li>
193     *  <li>Fields that exist in {@code oldVersion} but not in {@code version} are removed.</li>
194     *  <li>Fields that exist in {@code version} but not in {@code oldVersion} are initialized to their default values.</li>
195     *  <li>All {@link VersionChangeListener}s registered with the {@link Transaction} are notified.</li>
196     * </ul>
197     * </p>
198     *
199     * <p>
200     * Note that compatibility between schema versions does not depend on the field name, or whether the field is indexed.
201     * Fields are identified by storage ID, not name. A field's index may be added or removed between schema versions
202     * without losing information. Note, however, that querying a field's index will only return objects whose schema version
203     * corresponds to a schema in which the field is indexed.
204     * </p>
205     *
206     * <p>
207     * Note that an object's current schema version can go up as well as down, and may change by any amount.
208     * </p>
209     *
210     * @param schemaModel schema to use with the new transaction
211     * @param version the schema version number corresponding to {@code schemaModel}; must be greater than zero
212     * @param allowNewSchema whether creating a new schema version is allowed
213     * @throws IllegalArgumentException if {@code schemaModel} is null
214     * @throws IllegalArgumentException if {@code version} is not greater than zero
215     * @throws InvalidSchemaException if {@code schemaModel} is invalid (i.e., does not pass validation checks)
216     * @throws InvalidSchemaException if {@code schemaModel} does not match schema version {@code version}
217     *  as recorded in the database
218     * @throws InvalidSchemaException if schema version {@code version} is not recorded in the database
219     *  and {@code allowNewSchema} is false
220     * @throws InvalidSchemaException if schema version {@code version} is not recorded in the database,
221     *  {@code allowNewSchema} is true, but {@code schemaModel} is incompatible with one or more other schemas
222     *  alread recorded in the database (i.e., the same storage ID is used inconsistently between schema versions)
223     * @throws InconsistentDatabaseException if inconsistent or invalid schema information is detected in the database
224     * @throws InconsistentDatabaseException if an uninitialized database is encountered but the database is not empty
225     * @throws IllegalStateException if no underlying {@link KVDatabase} has been configured for this instance
226     */
227    public Transaction createTransaction(final SchemaModel schemaModel, final int version, final boolean allowNewSchema) {
228
229        // Sanity check
230        if (schemaModel == null)
231            throw new IllegalArgumentException("null schemaModel");
232        if (version <= 0)
233            throw new IllegalArgumentException("invalid schema version: " + version);
234
235        // Validate schema
236        final Set<ConstraintViolation<SchemaModel>> violations = ValidationUtil.validate(schemaModel);
237        if (!violations.isEmpty())
238            throw new InvalidSchemaException("invalid schema model:\n" + ValidationUtil.describe(violations));
239
240        // Open KV transaction
241        final KVTransaction kvt;
242        synchronized (this) {
243            if (this.kvdb == null)
244                throw new IllegalStateException("no KVDatabase configured");
245            kvt = this.kvdb.createTransaction();
246        }
247        boolean success = false;
248        this.log.debug("creating transaction using schema version " + version);
249        try {
250            Schema schema = null;
251            boolean firstAttempt = true;
252            while (true) {
253
254                // Check for uninitialized database
255                boolean uninitialized = false;
256                if (kvt.getAtLeast(null) == null) {
257                    this.log.info("detected an uninitialized database; initializing");
258                    final ByteWriter writer = new ByteWriter();
259                    UnsignedIntEncoder.write(writer, ENCODING_VERSION);
260                    kvt.put(ENCODING_KEY, writer.getBytes());
261                    uninitialized = true;
262                }
263
264                // Verify encoding version
265                final byte[] encodingBytes = kvt.get(ENCODING_KEY);
266                if (encodingBytes == null) {
267                    if (uninitialized)
268                        throw new InconsistentDatabaseException("database failed basic read/write test");
269                    else
270                        throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage");
271                }
272                final int encodingVersion;
273                try {
274                    encodingVersion = UnsignedIntEncoder.read(new ByteReader(encodingBytes));
275                } catch (IllegalArgumentException e) {
276                    throw new InconsistentDatabaseException("database contains invalid encoding version under key "
277                      + ByteUtil.toString(encodingBytes));
278                }
279                switch (encodingVersion) {
280                case ENCODING_VERSION:
281                    break;
282                default:
283                    throw new InconsistentDatabaseException("database contains unrecognized encoding version "
284                      + encodingVersion + " under key " + ByteUtil.toString(encodingBytes));
285                }
286
287                // Read existing database schema versions
288                final TreeMap<Integer, byte[]> bytesMap = new TreeMap<>();
289                for (KVPairIterator i = new KVPairIterator(kvt, SCHEMA_KEY_PREFIX); i.hasNext(); ) {
290                    final KVPair pair = i.next();
291                    final int vers = UnsignedIntEncoder.read(new ByteReader(pair.getKey(), SCHEMA_KEY_PREFIX.length));
292                    bytesMap.put(vers, pair.getValue());
293                }
294
295                // Read and decode database schemas, avoiding rebuild if possible
296                synchronized (this) {
297                    if (this.lastSchema != null && this.lastSchema.isSameVersions(bytesMap))
298                        schema = this.lastSchema;
299                }
300                if (schema == null) {
301                    try {
302                        schema = this.buildSchema(bytesMap);
303                    } catch (IllegalArgumentException e) {
304                        if (firstAttempt)
305                            throw new InconsistentDatabaseException("database contains invalid schema information", e);
306                        else
307                            throw new InvalidSchemaException("schema is not valid", e);
308                    }
309                }
310
311                // If transaction schema was not found in the database, add it and retry
312                if (!bytesMap.containsKey(version)) {
313
314                    // Log it
315                    if (bytesMap.isEmpty()) {
316                        if (!uninitialized)
317                            throw new InconsistentDatabaseException("database is initialized but contains zero schema versions");
318                    } else {
319                        this.log.info("schema version " + version
320                          + " not found in database; known versions are " + bytesMap.keySet());
321                    }
322
323                    // Check whether we can add a new schema version
324                    if (!allowNewSchema) {
325                        throw new IllegalArgumentException("schema version " + version
326                          + " not found in database and allowNewSchema is false");
327                    }
328
329                    // Record new schema in database
330                    this.log.info("recording new schema version " + version + " into database");
331                    this.writeSchema(kvt, version, schemaModel);
332
333                    // Try again
334                    schema = null;
335                    firstAttempt = false;
336                    continue;
337                }
338
339                // Compare transaction schema with the schema of the same version found in the database
340                this.log.debug("found schema version " + version + " in database; known versions are " + bytesMap.keySet());
341                final SchemaModel dbSchemaModel = schema.getVersion(version).getSchemaModel();
342                if (!schemaModel.equals(schema.getVersion(version).getSchemaModel())) {
343                    this.log.error("schema mismatch:\n  database schema:\n{}  provided schema:\n{}", dbSchemaModel, schemaModel);
344                    throw new IllegalArgumentException("the provided transaction schema does not match the schema with version "
345                      + version + " that is already recorded in the database");
346                }
347                break;
348            }
349
350            // Save schema for next time
351            synchronized (this) {
352                this.lastSchema = schema;
353            }
354
355            // Create transaction
356            final Transaction tx = new Transaction(this, kvt, schema, version);
357            success = true;
358            return tx;
359        } finally {
360            if (!success)
361                kvt.rollback();
362        }
363    }

365    /**
366     * Validate a {@link SchemaModel}.
367     *
368     * <p>
369     * This method only performs "static" checks; it does not access the database and therefore
370     * does not validate the schema against existing schema versions previously recorded.
371     * It does however use the {@link FieldTypeRegistry} associated with this instance to look up field types.
372     * </p>
373     *
374     * <p>
375     * To validate a schema against the database contents as well, simply attempt to create a transaction
376     * via {@link #createTransaction createTransaction()}.
377     * </p>
378     *
379     * @param schemaModel schema to validate
380     * @throws InvalidSchemaException if {@code schemaModel} is invalid
381     * @throws IllegalArgumentException if {@code schemaModel} is null
382     */
383    public void validateSchema(SchemaModel schemaModel) {
384
385        // Sanity check
386        if (schemaModel == null)
387            throw new IllegalArgumentException("null schemaModel");
388
389        // Validate
390        final Set<ConstraintViolation<SchemaModel>> violations = ValidationUtil.validate(schemaModel);
391        if (!violations.isEmpty())
392            throw new InvalidSchemaException("invalid schema:\n" + ValidationUtil.describe(violations));
393        try {
394            new SchemaVersion(1, new byte[0], schemaModel, this.fieldTypeRegistry);
395        } catch (IllegalArgumentException e) {
396            throw new InvalidSchemaException("invalid schema: " + e.getMessage(), e);
397        }
398    }
399
400    /**
401     * Build {@link Schema} object from a schema version XMLs.
402     */
403    private Schema buildSchema(SortedMap<Integer, byte[]> bytesMap) {
404        final TreeMap<Integer, SchemaVersion> versionMap = new TreeMap<>();
405        for (Map.Entry<Integer, byte[]> entry : bytesMap.entrySet()) {
406            final int version = entry.getKey();
407            final byte[] bytes = entry.getValue();
408            versionMap.put(version, new SchemaVersion(version, bytes, this.decodeSchema(version, bytes), this.fieldTypeRegistry));
409        }
410        return new Schema(versionMap);
411    }
412
413    /**
414     * Decode and validate schema XML.
415     */
416    private SchemaModel decodeSchema(int version, byte[] value) {
417
418        // Decode as XML
419        SchemaModel schemaModel;
420        try {
421            schemaModel = JiBXUtil.readObject(SchemaModel.class, new ByteArrayInputStream(value));
422        } catch (IOException e) {
423            throw new RuntimeException("unexpected exception", e);
424        } catch (JiBXException e) {
425            throw new IllegalArgumentException("can't parse schema version " + version, e);
426        }
427        if (this.log.isTraceEnabled())
428            this.log.trace("read schema version {} from database:\n{}", version, new String(value, Charset.forName("UTF-8")));
429
430        // Validate schema
431        final Set<ConstraintViolation<SchemaModel>> violations = ValidationUtil.validate(schemaModel);
432        if (!violations.isEmpty()) {
433            throw new IllegalArgumentException("found invalid schema version "
434              + version + ":\n" + ValidationUtil.describe(violations));
435        }
436
437        // Done
438        return schemaModel;
439    }

441    /**
442     * Record the given schema into the database.
443     */
444    private void writeSchema(KVTransaction kvt, int version, SchemaModel schema) {
445
446        // Encode as XML
447        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
448        try {
449            JiBXUtil.writeObject(schema, buf);
450        } catch (IOException e) {
451            throw new RuntimeException("unexpected exception", e);
452        } catch (JiBXException e) {
453            throw new RuntimeException("unexpected exception", e);
454        }
455
456        // Write XML
457        final ByteWriter writer = new ByteWriter();
458        writer.write(SCHEMA_KEY_PREFIX);
459        UnsignedIntEncoder.write(writer, version);
460        kvt.put(writer.getBytes(), buf.toByteArray());
461    }
