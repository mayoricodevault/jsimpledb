034/**
035 * Provides an object database abstraction on top of a key/value database.
036 *
037 * <p>
038 * Includes support for:
039 * <ul>
040 *  <li>Objects and fields defined by a {@link SchemaModel}</li>
041 *  <li>Simple values fields containing any atomic type</li>
042 *  <li>User-defined custom {@link FieldType}s</li>
043 *  <li>Invertable reference fields with strong referential integrity and configurable delete cascading</li>
044 *  <li>Complex fields of type {@link java.util.NavigableSet}, {@link java.util.List}, and {@link java.util.NavigableMap}</li>
045 *  <li>Configurable indexing of any simple or complex field</li>
046 *  <li>Notification of field changes as seen through an arbitrary path of references</li>
047 *  <li>Automatic schema tracking and object versioning with schema change notification support</li>
048 * </ul>
049 * </p>
050 *
051 * <p>
052 * See {@link Transaction} for further details on the above functionality.
053 * </p>
054 *
055 * <p>
056 * This class defines an abstraction layer that usually sits below a {@link org.jsimpledb.JSimpleDB} but is completely
057 * independent of {@link org.jsimpledb.JSimpleDB} and can be used on its own.
058 * Compared to {@link org.jsimpledb.JSimpleDB}, a {@link Database} has these differences:
059 * <ul>
060 *  <li>A {@link SchemaModel} must be explicitly provided to define the schema in use</li>
061 *  <li>Database objects are represented by {@link ObjId}s instead of Java objects, and there is no notion of object sub-type</li>
062 *  <li>All object types and fields must be referenced by storage ID using a {@link Transaction} object</li>
063 *  <li>There is no automatic validation support</li>
064 * </ul>
065 * </p>
066 *
067 * @see Transaction
068 * @see org.jsimpledb
069 */
070public class Database {

123    /**
124     * Create a new {@link Transaction} on this database and use the specified schema version to access objects and fields.
125     *
126     * <p>
127     * <b>Schema Versions</b>
128     * </p>
129     *
130     * <p>
131     * Within each {@link Database} is stored a record of all schema versions previously used with the database.
132     * When creating a new transaction, the caller provides an expected schema version and corresponding {@link SchemaModel}.
133     * Both of these are optional: a schema version of zero means "use the highest version recorded in the
134     * database", and a null {@link SchemaModel} measn "use the {@link SchemaModel} already recorded in the database under
135     * {@code version}". When this method is invoked, the following checks are applied:
136     * <ul>
137     *  <li>If a schema with version number {@code version != 0} is recorded in the database, and {@code schemaModel} is null or
138     *      matches it, then this method succeeds, and the {@link Transaction} will use that schema.</li>
139     *  <li>If a schema with version number {@code version} (or the highest numbered schema if {@code version == 0})
140     *      is recorded in the database, and {@code schemaModel} is not null and does not match it, then this method fails
141     *      and throws {@link SchemaMismatchException}.</li>
142     *  <li>If {@code allowNewSchema} is false, and no schema with version number {@code version != 0} has yet been
143     *      recorded in the database, then this method fails and throws {@link SchemaMismatchException}.</li>
144     *  <li>If {@code allowNewSchema} is true, and no schema with version number {@code version != 0} has yet been
145     *      recorded in the database, then if {@code schemaModel} is null an {@link SchemaMismatchException} is thrown;
146     *      otherwise {@code schemaModel} is checked for compabitility with all of the other schemas recorded in the database;
147     *      if compatible, this method succeeds, {@code schema} is recorded in the database with version number {@code version},
148     *      and the {@link Transaction} will use schema version {@code version}; otherwise an {@link SchemaMismatchException}
149     *      is thrown.</li>
150     *  <li>If the database is uninitialized and {@code version == 0} or {@code schemaModel} is null,
151     *      a {@link SchemaMismatchException} is thrown.</li>
152     * </ul>
153     * </p>
154     *
155     * <p>
156     * Fields and objects have assigned storage IDs, and these identify the schema object across all schema versions.
157     * Once a storage ID is assigned, it cannot be re-assigned to a different schema object. Fields must have a consistent
158     * type and parent entity (object type or complex field) in all schema versions in which they appear.
159     * </p>
160     *
161     * <p>
162     * <b>Object Versions</b>
163     * </p>
164     *
165     * <p>
166     * Each object in a {@link Database} contains an internal version number that indicates its current schema version;
167     * this in turn dictates what fields that object contains.
168     * </p>
169     *
170     * <p>
171     * When an object is accessed during a {@link Transaction}, the object's version is compared to the {@code version} associated
172     * with that {@link Transaction}. If the versions are the same, no version change occurs and fields are accessed normally.
173     * </p>
174     *
175     * <p>
176     * If the object has a version {@code oldVersion} different from {@code version}, then depending on which {@link Transaction}
177     * method is invoked, the object version may be automatically updated to {@code version}. This will cause fields to be added
178     * or removed, as follows:
179     * <ul>
180     *  <li>Compatible fields that are common to both schema versions remain unchanged; fields are compatible
181     *      if they have the same storage ID and type. For complex fields, sub-fields must also be compatible.</li>
182     *  <li>Fields that exist in {@code oldVersion} but not in {@code version} are removed.</li>
183     *  <li>Fields that exist in {@code version} but not in {@code oldVersion} are initialized to their default values.</li>
184     *  <li>All {@link VersionChangeListener}s registered with the {@link Transaction} are notified.</li>
185     * </ul>
186     * </p>
187     *
188     * <p>
189     * Note that compatibility between schema versions does not depend on the field name, or whether the field is indexed.
190     * Fields are identified by storage ID, not name. A field's index may be added or removed between schema versions
191     * without losing information. Note, however, that querying a field's index will only return objects whose schema version
192     * corresponds to a schema in which the field is indexed.
193     * </p>
194     *
195     * <p>
196     * Note that an object's current schema version can go up as well as down, and may change by any amount.
197     * Also, nothing requires schema version numbers to be consecutive.
198     * </p>
199     *
200     * @param schemaModel schema to use with the new transaction, or null to use the schema already recorded in the database
201     * @param version the schema version number corresponding to {@code schemaModel}, or zero to use the highest recorded version
202     * @param allowNewSchema whether creating a new schema version is allowed
203     * @throws IllegalArgumentException if {@code version} is less than zero
204     * @throws InvalidSchemaException if {@code schemaModel} is invalid (i.e., does not pass validation checks)
205     * @throws SchemaMismatchException if {@code schemaModel} does not match schema version {@code version}
206     *  as recorded in the database
207     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database
208     *  and {@code allowNewSchema} is false
209     * @throws SchemaMismatchException if schema version {@code version} is not recorded in the database,
210     *  {@code allowNewSchema} is true, but {@code schemaModel} is incompatible with one or more other schemas
211     *  already recorded in the database (i.e., the same storage ID is used inconsistently between schema versions)
212     * @throws SchemaMismatchException
213     *  if the database is uninitialized and {@code version == 0} or {@code schemaModel} is null
214     * @throws InconsistentDatabaseException if inconsistent or invalid schema information is detected in the database
215     * @throws InconsistentDatabaseException if an uninitialized database is encountered but the database is not empty
216     * @throws IllegalStateException if no underlying {@link KVDatabase} has been configured for this instance
217     */
218    public Transaction createTransaction(final SchemaModel schemaModel, int version, final boolean allowNewSchema) {
219
220        // Sanity check
221        if (version < 0)
222            throw new IllegalArgumentException("invalid schema version: " + version);
223
224        // Validate schema
225        if (schemaModel != null)
226            schemaModel.validate();
227
228        // Open KV transaction
229        final KVTransaction kvt = this.kvdb.createTransaction();
230        boolean success = false;
231        this.log.debug("creating transaction using "
232          + (version != 0 ? "schema version " + version : "highest recorded schema version"));
233        try {
234            Schema schema = null;
235            boolean firstAttempt = true;
236            while (true) {
237
238                // Get iterator over meta-data key/value pairs
239                final byte[] metaDataPrefix = new byte[] { METADATA_PREFIX };
240                final Iterator<KVPair> metaDataIterator = kvt.getRange(metaDataPrefix,
241                  ByteUtil.getKeyAfterPrefix(metaDataPrefix), false);
242
243                // Get format version; it should be first; if not found, database is uninitialized (and should be empty)
244                byte[] formatVersionBytes = null;
245                if (metaDataIterator.hasNext()) {
246                    final KVPair pair = metaDataIterator.next();
247                    if (!Arrays.equals(pair.getKey(), FORMAT_VERSION_KEY)) {
248
249                        // Check for old incompatible format
250                        if (kvt.get(ByteUtil.parse("008fb53fe9da1cc9")) != null)
251                            throw new InconsistentDatabaseException("old/incompatible database format; perform upgrade manually");
252
253                        // Unknown garbage
254                        throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage (key "
255                          + ByteUtil.toString(pair.getKey()) + ")");
256                    }
257                    formatVersionBytes = pair.getValue();
258                }
259
260                // Check for an uninitialized database
261                final boolean uninitialized = formatVersionBytes == null;
262                if (uninitialized) {
263
264                    // Sanity checks
265                    if (kvt.getAtLeast(new byte[0]) != null)
266                        throw new InconsistentDatabaseException("database is uninitialized but contains unrecognized garbage");
267                    if (kvt.getAtMost(new byte[] { (byte)0xff }) != null)
268                        throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getAtMost()");
269                    if (kvt.getRange(new byte[0], new byte[] { (byte)0xff }, false).hasNext())
270                        throw new InconsistentDatabaseException("inconsistent results from getAtLeast() and getRange()");
271
272                    // Initialize database
273                    this.log.info("detected an uninitialized database; initializing");
274                    final ByteWriter writer = new ByteWriter();
275                    UnsignedIntEncoder.write(writer, CURRENT_FORMAT_VERSION);
276                    kvt.put(FORMAT_VERSION_KEY.clone(), writer.getBytes());
277
278                    // Sanity check again
279                    formatVersionBytes = kvt.get(FORMAT_VERSION_KEY.clone());
280                    if (formatVersionBytes == null || ByteUtil.compare(formatVersionBytes, writer.getBytes()) != 0)
281                        throw new InconsistentDatabaseException("database failed basic read/write test");
282                    final KVPair lower = kvt.getAtLeast(new byte[0]);
283                    if (lower == null || !lower.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
284                        throw new InconsistentDatabaseException("database failed basic read/write test");
285                    final KVPair upper = kvt.getAtMost(new byte[] { (byte)0xff });
286                    if (upper == null || !upper.equals(new KVPair(FORMAT_VERSION_KEY, writer.getBytes())))
287                        throw new InconsistentDatabaseException("database failed basic read/write test");
288                }
289
290                // Verify format version
291                final int formatVersion;
292                try {
293                    formatVersion = UnsignedIntEncoder.read(new ByteReader(formatVersionBytes));
294                } catch (IllegalArgumentException e) {
295                    throw new InconsistentDatabaseException("database contains invalid encoded format version "
296                      + ByteUtil.toString(formatVersionBytes) + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
297                }
298                switch (formatVersion) {
299                case FORMAT_VERSION_1:
300                    break;
301                default:
302                    throw new InconsistentDatabaseException("database contains unrecognized format version "
303                      + formatVersion + " != " + CURRENT_FORMAT_VERSION + " under key " + ByteUtil.toString(FORMAT_VERSION_KEY));
304                }
305
306                // Read recorded database schema versions - should immediately follow FORMAT_VERSION_KEY
307                final TreeMap<Integer, byte[]> bytesMap = new TreeMap<>();
308                while (metaDataIterator.hasNext()) {
309                    final KVPair pair = metaDataIterator.next();
310
311                    // Sanity check
312                    if (ByteUtil.compare(pair.getKey(), SCHEMA_KEY_PREFIX) < 0) {
313                        throw new InconsistentDatabaseException("database contains unrecognized garbage key "
314                          + ByteUtil.toString(pair.getKey()));
315                    }
316
317                    // Stop at end of recorded schemas
318                    if (!ByteUtil.isPrefixOf(SCHEMA_KEY_PREFIX, pair.getKey()))
319                        break;
320
321                    // Decode schema version and get XML
322                    final int vers = UnsignedIntEncoder.read(new ByteReader(pair.getKey(), SCHEMA_KEY_PREFIX.length));
323                    bytesMap.put(vers, pair.getValue());
324                }
325
326                // Read and decode database schemas, avoiding rebuild if possible
327                schema = this.lastSchema;
328                if (schema != null && !schema.isSameVersions(bytesMap))
329                    schema = null;
330                if (schema == null) {
331                    try {
332                        schema = this.buildSchema(bytesMap);
333                    } catch (IllegalArgumentException e) {
334                        if (firstAttempt)
335                            throw new InconsistentDatabaseException("database contains invalid schema information", e);
336                        else
337                            throw new InvalidSchemaException("schema is not valid", e);
338                    }
339                }
340
341                // If no version specified, assume the highest recorded version
342                if (version == 0 && !bytesMap.isEmpty())
343                    version = bytesMap.lastKey();
344
345                // If transaction schema was not found in the database, add it and retry
346                if (!bytesMap.containsKey(version)) {
347
348                    // Log it
349                    if (bytesMap.isEmpty()) {
350                        if (!uninitialized)
351                            throw new InconsistentDatabaseException("database is initialized but contains zero schema versions");
352                    } else {
353                        this.log.info("schema version " + version
354                          + " not found in database; known versions are " + bytesMap.keySet());
355                    }
356
357                    // Check whether we can add a new schema version
358                    if (!allowNewSchema) {
359                        if (version == 0)
360                            throw new SchemaMismatchException("uninitialized database and no schema version was provided");
361                        else {
362                            throw new SchemaMismatchException("schema version " + version
363                              + " not found in database and allowNewSchema is false");
364                        }
365                    }
366                    if (schemaModel == null) {
367                        if (version == 0)
368                            throw new SchemaMismatchException("uninitialized database and no schema model was provided");
369                        else {
370                            throw new SchemaMismatchException("no schema model was provided but database does not contain"
371                              + " a recorded schema version " + version);
372                        }
373                    }
374
375                    // Record new schema in database
376                    this.log.info("recording new schema version " + version + " into database");
377                    this.writeSchema(kvt, version, schemaModel);
378
379                    // Try again
380                    schema = null;
381                    firstAttempt = false;
382                    continue;
383                }
384
385                // Compare transaction schema with the schema of the same version found in the database
386                this.log.debug("found schema version " + version + " in database; known versions are " + bytesMap.keySet());
387                final SchemaModel dbSchemaModel = schema.getVersion(version).getSchemaModel();
388                if (schemaModel != null && !schemaModel.equals(schema.getVersion(version).getSchemaModel())) {
389                    this.log.error("schema mismatch:\n  database schema:\n{}  provided schema:\n{}", dbSchemaModel, schemaModel);
390                    throw new IllegalArgumentException("the provided transaction schema does not match the schema with version "
391                      + version + " that is already recorded in the database");
392                }
393                break;
394            }
395
396            // Save schema for next time
397            this.lastSchema = schema;
398
399            // Create transaction
400            final Transaction tx = new Transaction(this, kvt, schema, version);
401            success = true;
402            return tx;
403        } finally {
404            if (!success) {
405                try {
406                    kvt.rollback();
407                } catch (KVTransactionException e) {
408                    // ignore
409                }
410            }
411        }
412    }

414    /**
415     * Validate a {@link SchemaModel}.
416     *
417     * <p>
418     * This method only performs "static" checks; it does not access the database and therefore
419     * does not validate the schema against existing schema versions previously recorded.
420     * It does however use the {@link FieldTypeRegistry} associated with this instance to look up field types.
421     * </p>
422     *
423     * <p>
424     * To validate a schema against the database contents as well, simply attempt to create a transaction
425     * via {@link #createTransaction createTransaction()}.
426     * </p>
427     *
428     * @param schemaModel schema to validate
429     * @throws InvalidSchemaException if {@code schemaModel} is invalid
430     * @throws IllegalArgumentException if {@code schemaModel} is null
431     */
432    public void validateSchema(SchemaModel schemaModel) {
433
434        // Sanity check
435        if (schemaModel == null)
436            throw new IllegalArgumentException("null schemaModel");
437
438        // Validate
439        schemaModel.validate();
440        try {
441            new SchemaVersion(1, new byte[0], schemaModel, this.fieldTypeRegistry);
442        } catch (IllegalArgumentException e) {
443            throw new InvalidSchemaException("invalid schema: " + e.getMessage(), e);
444        }
445    }

447    void copyMetaData(Transaction src, KVStore dst) {
448        for (Iterator<KVPair> i = src.kvt.getRange(new byte[] { METADATA_PREFIX }, VERSION_INDEX_PREFIX.clone(), false);
449          i.hasNext(); ) {
450            final KVPair pair = i.next();
451            dst.put(pair.getKey(), pair.getValue());
452        }
453    }
454
455    void reset(SnapshotTransaction tx) {
456        tx.kvt.removeRange(VERSION_INDEX_PREFIX.clone(), null);
457    }

459    static byte[] buildVersionIndexKey(ObjId id, int version) {
460        final ByteWriter writer = new ByteWriter(VERSION_INDEX_PREFIX.length + 1 + ObjId.NUM_BYTES);
461        writer.write(VERSION_INDEX_PREFIX);
462        UnsignedIntEncoder.write(writer, version);
463        id.writeTo(writer);
464        return writer.getBytes();
465    }
466
467    static NavigableMap<Integer, NavigableSet<ObjId>> getVersionIndex(Transaction tx) {
468        return new IndexMap<Integer, ObjId>(tx, VERSION_INDEX_PREFIX, new UnsignedIntType(), FieldType.OBJ_ID, false);
469    }

471    /**
472     * Build {@link Schema} object from a schema version XMLs.
473     *
474     * @throws InconsistentDatabaseException if any recorded schema version is invalid
475     */
476    private Schema buildSchema(SortedMap<Integer, byte[]> bytesMap) {
477        final TreeMap<Integer, SchemaVersion> versionMap = new TreeMap<>();
478        for (Map.Entry<Integer, byte[]> entry : bytesMap.entrySet()) {
479            final int version = entry.getKey();
480            final byte[] bytes = entry.getValue();
481            final SchemaModel schemaModel;
482            try {
483                schemaModel = this.decodeSchema(bytes);
484            } catch (InvalidSchemaException e) {
485                throw new InconsistentDatabaseException("found invalid schema version " + version + " recorded in database", e);
486            }
487            if (this.log.isTraceEnabled())
488                this.log.trace("read schema version {} from database:\n{}", version, schemaModel);
489            versionMap.put(version, new SchemaVersion(version, bytes, schemaModel, this.fieldTypeRegistry));
490        }
491        return new Schema(versionMap);
492    }

494    /**
495     * Decode and validate schema XML.
496     *
497     * @throws InvalidSchemaException if schema is invalid
498     */
499    private SchemaModel decodeSchema(byte[] value) {
500        try {
501            return SchemaModel.fromXML(new ByteArrayInputStream(value));
502        } catch (IOException e) {
503            throw new RuntimeException("unexpected exception", e);
504        }
505    }
506
507    /**
508     * Record the given schema into the database.
509     */
510    private void writeSchema(KVTransaction kvt, int version, SchemaModel schema) {
511
512        // Encode as XML
513        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
514        try {
515            schema.toXML(buf, false);
516        } catch (IOException e) {
517            throw new RuntimeException("unexpected exception", e);
518        }
519
520        // Write XML
521        final ByteWriter writer = new ByteWriter();
522        writer.write(SCHEMA_KEY_PREFIX);
523        UnsignedIntEncoder.write(writer, version);
524        kvt.put(writer.getBytes(), buf.toByteArray());
525    }
