032/**
033 * A JSimpleDB {@link Database} transaction.
034 *
035 * <p>
036 * Methods in this class can be divided into the following categories:
037 * </p>
038 *
039 * <p>
040 * <b>Transaction Meta-Data</b>
041 * <ul>
042 *  <li>{@link #getDatabase getDatabase()} - Get the associated {@link Database}
043 *  <li>{@link #getSchema getSchema()} - Get the database {@link Schema}, as seen by this transaction
044 *  <li>{@link #getSchemaVersion() getSchemaVersion()} - Get the {@link SchemaVersion} that will be used by this transaction
045 * </ul>
046 * </p>
047 *
048 * <p>
049 * <b>Transaction Lifecycle</b>
050 * <ul>
051 *  <li>{@link #commit commit()} - Commit transaction
052 *  <li>{@link #rollback rollback()} - Roll back transaction
053 *  <li>{@link #isValid isValid()} - Test transaction validity
054 *  <li>{@link #setTimeout setTimeout()} - Set transaction timeout
055 *  <li>{@link #setReadOnly setReadOnly()} - Set transaction to read-only
056 *  <li>{@link #setRollbackOnly setRollbackOnly()} - Set transaction for rollack only
057 *  <li>{@link #addCallback addCallback()} - Register a {@link Callback} on transaction completion
058 * </ul>
059 * </p>
060 *
061 * <p>
062 * <b>Object Lifecycle</b>
063 * <ul>
064 *  <li>{@link #create(int) create()} - Create a database object
065 *  <li>{@link #delete delete()} - Delete a database object
066 *  <li>{@link #exists exists()} - Test whether a database object exists
067 *  <li>{@link #addCreateListener addCreateListener()} - Register a {@link CreateListener} for notifications about new objects
068 *  <li>{@link #removeCreateListener removeCreateListener()} - Unregister a {@link CreateListener}
069 *  <li>{@link #addDeleteListener addDeleteListener()} - Register a {@link DeleteListener} for notifications about object deletions
070 *  <li>{@link #removeDeleteListener removeDeleteListener()} - Unregister a {@link DeleteListener}
071 * </ul>
072 * </p>
073 *
074 * <p>
075 * <b>Object Versioning</b>
076 * <ul>
077 *  <li>{@link #getSchemaVersion(ObjId) getSchemaVersion()} - Inspect an object's schema version
078 *  <li>{@link #updateSchemaVersion updateSchemaVersion()} - Update an object's schema version to match this transaction
079 *  <li>{@link #addVersionChangeListener addVersionChangeListener()} - Register a {@link VersionChangeListener} for notifications
080 *      about object version changes
081 *  <li>{@link #removeVersionChangeListener removeVersionChangeListener()} - Unregister a {@link VersionChangeListener}
082 * </ul>
083 *
084 * <p>
085 * <b>Object and Field Access</b>
086 * <ul>
087 *  <li>{@link #getAll getAll()} - Get all objects of a specific type
088 *  <li>{@link #readSimpleField readSimpleField()} - Read the value of a {@link SimpleField} in an object
089 *  <li>{@link #writeSimpleField writeSimpleField()} - Write the value of a {@link SimpleField} in an object
090 *  <li>{@link #readCounterField readCounterField()} - Read the value of a {@link CounterField} in an object
091 *  <li>{@link #writeCounterField writeCounterField()} - Write the value of a {@link CounterField} in an object
092 *  <li>{@link #adjustCounterField adjustCounterField()} - Adjust the value of a {@link CounterField} in an object
093 *  <li>{@link #readSetField readSetField()} - Access a {@link SetField} in an object as a {@link NavigableSet}
094 *  <li>{@link #readListField readListField()} - Access a {@link ListField} in an object as a {@link List}
095 *  <li>{@link #readMapField readMapField()} - Access a {@link MapField} in an object as a {@link NavigableMap}
096 *  <li>{@link #getKey getKey(ObjId)} - Get the {@link org.jsimpledb.kv.KVDatabase} key corresponding to an object
097 *  <li>{@link #getKey getKey(ObjId, int)} - Get the {@link org.jsimpledb.kv.KVDatabase}
098 *      key corresponding to a field in an object
099 * </ul>
100 * </p>
101 *
102 * <p>
103 * <b>Field Change Notifications</b>
104 * <ul>
105 *  <li>{@link #addSimpleFieldChangeListener addSimpleFieldChangeListener()} - Register a {@link SimpleFieldChangeListener} for
106 *      notifications of changes in a {@link SimpleField}, as seen through a path of object references
107 *  <li>{@link #addSetFieldChangeListener addSetFieldChangeListener()} - Register a {@link SetFieldChangeListener} for
108 *      notifications of changes in a {@link SetField}, as seen through a path of object references
109 *  <li>{@link #addListFieldChangeListener addListFieldChangeListener()} - Register a {@link ListFieldChangeListener} for
110 *      notifications of changes in a {@link ListField}, as seen through a path of object references
111 *  <li>{@link #addMapFieldChangeListener addMapFieldChangeListener()} - Register a {@link MapFieldChangeListener} for
112 *      notifications of changes in a {@link MapField}, as seen through a path of object references
113 *  <li>{@link #removeSimpleFieldChangeListener removeSimpleFieldChangeListener()} - Unregister a previously registered
114 *      {@link SimpleFieldChangeListener}
115 *  <li>{@link #removeSetFieldChangeListener removeSetFieldChangeListener()} - Unregister a previously registered
116 *      {@link SetFieldChangeListener}
117 *  <li>{@link #removeListFieldChangeListener removeListFieldChangeListener()} - Unregister a previously registered
118 *      {@link ListFieldChangeListener}
119 *  <li>{@link #removeMapFieldChangeListener removeMapFieldChangeListener()} - Unregister a previously registered
120 *      {@link MapFieldChangeListener}
121 * </ul>
122 * </p>
123 *
124 * <p>
125 * <b>Reference Path Queries</b>
126 * <ul>
127 *  <li>{@link #invertReferencePath invertReferencePath()} - Find all objects that refer to any element in a given set
128 *      of objects through a specified reference path
129 *  </li>
130 * </ul>
131 * </p>
132 *
133 * <p>
134 * <b>Index Queries</b>
135 * <ul>
136 *  <li>{@link #querySimpleField querySimpleField()} - Query the index associated with a {@link SimpleField}
137 *      to identify all values and all objects having those values
138 *  <li>{@link #querySetField querySetField()} - Query the index associated with a {@link SetField}
139 *      to identify all set elements and all objects having those elements in the set
140 *  <li>{@link #queryListField queryListField()} - Query the index associated with a {@link ListField}
141 *      to identify all list elements and all objects having those elements in the list
142 *  <li>{@link #queryListFieldEntries queryListFieldEntries()} - Query the index associated with a {@link ListField}
143 *      to identify all list elements and all objects with those elements in the list, as well as the associated list indicies
144 *  <li>{@link #queryMapFieldKey queryMapFieldKey()} - Query the index associated with a {@link MapField}
145 *      to identify all map keys and all objects having those keys in the map
146 *  <li>{@link #queryMapFieldKeyEntries queryMapFieldKeyEntries()} - Query the index associated with a {@link MapField}
147 *      to identify all map keys and all objects having those keys in the map, as well as the associated map values
148 *  <li>{@link #queryMapFieldValue queryMapFieldValue()} - Query the index associated with a {@link MapField}
149 *      to identify all map values and all objects having those values in the map
150 *  <li>{@link #queryMapFieldValueEntries queryMapFieldValueEntries()} - Query the index associated with a {@link MapField}
151 *      to identify all map values and all objects having those values in the map, as well as the associated map keys
152 * </ul>
153 * </p>
154 *
155 * <p>
156 * All methods returning a set of values return a {@link NavigableSet}.
157 * The {@link NavigableSets} utility class provides methods for the efficient {@link NavigableSets#intersection intersection},
158 * {@link NavigableSets#union union}, {@link NavigableSets#difference difference}, and
159 * {@link NavigableSets#symmetricDifference symmetric difference} of {@link NavigableSet}s containing the same element type,
160 * thereby providing the equivalent of traditional database joins.
161 * </p>
162 */

240    /**
241     * Commit this transaction.
242     *
243     * @throws StaleTransactionException if this transaction is no longer usable
244     * @throws RetryTransactionException from {@link KVTransaction#commit KVTransaction.commit()}
245     * @throws RollbackOnlyTransactionException if this instance has been {@linkplain #setRollbackOnly marked} rollback only;
246     *  this instance will be automatically rolled back
247     */
248    public synchronized void commit() {
249
250        // Sanity check
251        if (this.stale)
252            throw new StaleTransactionException(this);
253
254        // Rollback only?
255        if (this.rollbackOnly) {
256            this.log.info("commit() invoked on transaction " + this + " marked rollback-only, rolling back");
257            this.rollback();
258            throw new RollbackOnlyTransactionException(this);
259        }
260        this.stale = true;
261
262        // Do before completion callbacks
263        this.log.debug("commit() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
264        Callback failedCallback = null;
265        try {
266            for (Callback callback : this.callbacks) {
267                failedCallback = callback;
268                callback.beforeCommit(this.readOnly);
269            }
270            failedCallback = null;
271        } finally {
272            if (failedCallback != null) {
273                this.log.warn("error invoking beforeCommit() method on transaction "
274                  + this + " callback " + failedCallback + ", rolling back");
275            }
276            this.triggerBeforeCompletion();
277            if (failedCallback != null) {
278                try {
279                    this.kvt.rollback();
280                } finally {
281                    this.triggerAfterCompletion(false);
282                }
283            }
284        }
285
286        // Commit KVTransaction and trigger after completion callbacks
287        failedCallback = null;
288        try {
289            this.kvt.commit();
290            for (Callback callback : this.callbacks) {
291                failedCallback = callback;
292                callback.afterCommit();
293            }
294            failedCallback = null;
295        } finally {
296            if (failedCallback != null)
297                this.log.warn("error invoking afterCommit() method on transaction " + this + " callback " + failedCallback);
298            this.triggerAfterCompletion(true);
299        }
300    }
301
302    /**
303     * Roll back this transaction.
304     *
305     * @throws StaleTransactionException if this transaction is no longer usable
306     */
307    public synchronized void rollback() {
308
309        // Sanity check
310        if (this.stale)
311            throw new StaleTransactionException(this);
312        this.stale = true;
313        this.log.debug("rollback() invoked on" + (this.readOnly ? " read-only" : "") + " transaction " + this);
314
315        // Do before completion callbacks
316        this.triggerBeforeCompletion();
317
318        // Roll back KVTransaction and trigger after completion callbacks
319        try {
320            this.kvt.rollback();
321        } finally {
322            this.triggerAfterCompletion(false);
323        }
324    }
325
326    private /*synchronized*/ void triggerBeforeCompletion() {
327        for (Callback callback : this.callbacks) {
328            try {
329                callback.beforeCompletion();
330            } catch (Throwable t) {
331                this.log.error("error invoking beforeCompletion() method on transaction callback " + callback, t);
332            }
333        }
334    }
335
336    private /*synchronized*/ void triggerAfterCompletion(boolean committed) {
337        for (Callback callback : this.callbacks) {
338            try {
339                callback.afterCompletion(committed);
340            } catch (Throwable t) {
341                this.log.error("error invoking afterCompletion() method on transaction callback " + callback, t);
342            }
343        }
344    }

430    /**
431     * Create a new object with the given object ID, if it doesn't already exist. If it does exist, nothing happens.
432     *
433     * <p>
434     * If the object doesn't already exist, all fields are set to their default values and the object's
435     * schema version is set to {@linkplain #getSchemaVersion the version associated with this transaction}.
436     * </p>
437     *
438     * @param id object ID
439     * @return true if the object did not exist and was created, false if the object already existed
440     * @throws IllegalArgumentException if the storage ID associated with {@code id}
441     *  does not correspond to a known object type in this transaction's schema
442     * @throws IllegalArgumentException if {@code id} is null
443     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
444     * @throws StaleTransactionException if this transaction is no longer usable
445     */
446    public synchronized boolean create(ObjId id) {
447
448        // Sanity check
449        if (id == null)
450            throw new IllegalArgumentException("null id");
451        if (this.stale)
452            throw new StaleTransactionException(this);
453        if (this.readOnly)
454            throw new ReadOnlyTransactionException(this);
455
456        // Does object already exist?
457        if (this.exists(id))
458            return false;
459
460        // Initialize object
461        this.initialize(id, this.version.getSchemaItem(id.getStorageId(), ObjType.class));
462
463        // Done
464        return true;
465    }

467    /**
468     * Create a new object with a randomly assigned object ID and having the given type.
469     *
470     * <p>
471     * All fields will be set to their default values.
472     * The object's schema version will be set to {@linkplain #getSchemaVersion the version associated with this transaction}.
473     * </p>
474     *
475     * @param storageId object type storage ID
476     * @return object id of newly created object
477     * @throws IllegalArgumentException if {@code storageId} does not correspond to a known object type in this transaction's schema
478     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
479     * @throws StaleTransactionException if this transaction is no longer usable
480     */
481    public synchronized ObjId create(int storageId) {
482
483        // Sanity check
484        if (this.stale)
485            throw new StaleTransactionException(this);
486        if (this.readOnly)
487            throw new ReadOnlyTransactionException(this);
488        final ObjType objType = this.version.getSchemaItem(storageId, ObjType.class);
489
490        // Create a new, unique key
491        ObjId id;
492        final ByteWriter keyWriter = new ByteWriter();
493        int attempts = 0;
494        while (true) {
495            id = new ObjId(objType.storageId);
496            id.writeTo(keyWriter);
497            if (this.kvt.get(keyWriter.getBytes()) == null)
498                break;
499            if (++attempts == MAX_UNIQUE_KEY_ATTEMPTS) {
500                throw new DatabaseException("could not find a new, unused object ID after "
501                  + attempts + " attempts; is our source of randomness truly random?");
502            }
503            keyWriter.reset(0);
504        }
505
506        // Initialize object
507        this.initialize(id, objType);
508
509        // Done
510        return id;
511    }

513    private synchronized void initialize(ObjId id, ObjType objType) {
514
515        // Write object meta-data
516        ObjInfo.write(this, id, objType.version.versionNumber, false);
517
518        // Initialize counters to zero
519        if (!objType.counterFields.isEmpty()) {
520            final CountingKVStore ckv = this.getCountingKVTransaction();
521            for (CounterField field : objType.counterFields.values())
522                ckv.put(field.buildKey(id), ckv.encodeCounter(0));
523        }
524
525        // Write simple field index entries
526        for (SimpleField<?> field : objType.simpleFields.values()) {
527            if (field.indexed)
528                this.kvt.put(field.buildIndexKey(id, null), ByteUtil.EMPTY);
529        }
530
531        // Notify listeners
532        for (CreateListener listener : this.createListeners.toArray(new CreateListener[this.createListeners.size()]))
533            listener.onCreate(this, id);
534    }

536    /**
537     * Delete an object. Does nothing if object does not exist (e.g., has already been deleted).
538     *
539     * <p>
540     * This method does <i>not</i> change the object's schema version if it has a different version that this transaction.
541     * </p>
542     *
543     * @param id object ID of the object to delete
544     * @return true if object was found and deleted, false if object was not found
545     * @throws ReferencedObjectException if the object is referenced by some other object
546     *  through a reference field set to {@link DeleteAction#EXCEPTION}
547     * @throws IllegalArgumentException if {@code id} is null
548     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
549     * @throws StaleTransactionException if this transaction is no longer usable
550     */
551    public synchronized boolean delete(ObjId id) {
552
553        // Sanity check
554        if (id == null)
555            throw new IllegalArgumentException("null id");
556        if (this.stale)
557            throw new StaleTransactionException(this);
558        if (this.readOnly)
559            throw new ReadOnlyTransactionException(this);
560
561        // Handle recurive DeleteAction.DELETE without hogging Java stack
562        final ArrayList<ObjId> deletables = new ArrayList<>(1);
563        deletables.add(id);
564        boolean found = false;
565        while (true) {
566            final int size = deletables.size();
567            if (size == 0)
568                break;
569            id = deletables.remove(size - 1);
570            found |= this.doDelete(id, deletables);
571        }
572        return found;
573    }
574
575    private synchronized boolean doDelete(final ObjId id, List<ObjId> deletables) {
576
577        // Loop here to handle any mutations within delete notification listener callbacks
578        ObjInfo info;
579        while (true) {
580
581            // Get object info
582            try {
583                info = this.getObjectInfo(id, false);
584            } catch (DeletedObjectException e) {                    // possibly due to a cycle of DeleteAction.DELETE references
585                return false;
586            }
587
588            // Determine if any EXCEPTION reference fields refer to the object (from some other object); if so, throw exception
589            for (ReferenceFieldStorageInfo fieldInfo : this.schema.exceptionReferenceFieldStorageInfos) {
590                final NavigableSet<ObjId> referrers = this.findReferences(fieldInfo, id);
591                if (referrers == null)
592                    continue;
593                for (ObjId referrer : referrers) {
594                    if (!referrer.equals(id))
595                        throw new ReferencedObjectException(id, referrer, fieldInfo.storageId);
596                }
597            }
598
599            // Need to issue delete notifications?
600            if (info.isDeleteNotified() || this.deleteListeners.isEmpty())
601                break;
602
603            // Issue delete notifications and retry
604            ObjInfo.write(this, id, info.getVersionNumber(), true);
605            for (DeleteListener listener : this.deleteListeners.toArray(new DeleteListener[this.deleteListeners.size()]))
606                listener.onDelete(this, id);
607        }
608
609        // Delete object's simple field index entries
610        final ObjType type = info.getObjType();
611        for (SimpleField<?> field : type.simpleFields.values()) {
612            if (field.indexed)
613                this.kvt.remove(field.buildIndexKey(id, this.kvt.get(field.buildKey(id))));
614        }
615
616        // Delete object's complex field index entries
617        for (ComplexField<?> field : type.complexFields.values()) {
618            for (SimpleField<?> subField : field.getSubFields()) {
619                if (subField.indexed)
620                    field.removeIndexEntries(this, id, subField);
621            }
622        }
623
624        // Delete object version and all field content
625        final byte[] minKey = info.getId().getBytes();
626        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
627        this.kvt.removeRange(minKey, maxKey);
628
629        // Find all references to this object and deal with them
630        for (ReferenceFieldStorageInfo fieldInfo : this.schema.referenceFieldStorageInfos) {
631            if (fieldInfo.onDelete == DeleteAction.EXCEPTION)           // we already checked these
632                continue;
633            final NavigableSet<ObjId> referrers = this.findReferences(fieldInfo, id);
634            if (referrers == null)
635                continue;
636            switch (fieldInfo.onDelete) {
637            case NOTHING:
638                break;
639            case UNREFERENCE:
640            {
641                final int storageId = fieldInfo.storageId;
642                if (fieldInfo.isSubField()) {
643                    final ComplexFieldStorageInfo superFieldInfo = this.schema.verifyStorageInfo(
644                      fieldInfo.superFieldStorageId, ComplexFieldStorageInfo.class);
645                    superFieldInfo.unreferenceAll(this, storageId, id);
646                } else {
647                    for (ObjId referrer : referrers)
648                        this.writeSimpleField(referrer, storageId, null);
649                }
650                break;
651            }
652            case DELETE:
653                deletables.addAll(referrers);
654                break;
655            default:
656                throw new RuntimeException("unexpected case");
657            }
658        }
659
660        // Done
661        return true;
662    }

805    /**
806     * Update object to the current schema version.
807     *
808     * @param id object id
809     * @param info original object info
810     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
811     * @throws InvalidObjectVersionException if the schema version is invalid
812     * @throws IllegalArgumentException if {@code id} is null
813     */
814    private synchronized void updateVersion(ObjId id, ObjInfo info) {
815
816        // Sanity check
817        if (info.getVersionNumber() == this.version.versionNumber)
818            throw new IllegalArgumentException("object already at version " + this.version.versionNumber);
819        if (this.readOnly)
820            throw new ReadOnlyTransactionException(this);
821
822        // Get old and new types
823        final ObjType oldType = info.getObjType();
824        final ObjType newType;
825        try {
826            newType = this.version.getSchemaItem(id.getStorageId(), ObjType.class);
827        } catch (IllegalArgumentException e) {
828            throw new InconsistentDatabaseException("object " + id + " has an unrecognized storage ID", e);
829        }
830
831        // Gather removed fields' values here for user migration
832        final TreeMap<Integer, Object> oldFieldValues = new TreeMap<>();
833
834    //////// Update counter fields
835
836        // Get counter field storage IDs (old or new)
837        final TreeSet<Integer> counterFieldStorageIds = new TreeSet<>();
838        counterFieldStorageIds.addAll(oldType.counterFields.keySet());
839        counterFieldStorageIds.addAll(newType.counterFields.keySet());
840
841        // Update counter fields
842        for (int storageId : counterFieldStorageIds) {
843
844            // Get old and new counter fields having this storage ID
845            final CounterField oldField = oldType.counterFields.get(storageId);
846            final CounterField newField = newType.counterFields.get(storageId);
847
848            // Save old field values for version change notification
849            final byte[] key = Field.buildKey(id, storageId);
850            final byte[] oldValue = this.kvt.get(key);
851            if (oldField != null)
852                oldFieldValues.put(storageId, oldValue != null ? this.getCountingKVTransaction().decodeCounter(oldValue) : 0);
853
854            // Remove old value if field has disappeared, otherwise leave alone
855            if (oldValue != null && oldField != null && newField == null)
856                this.kvt.remove(key);
857        }
858
859    //////// Update simple fields and corresponding index entries
860
861        // Get simple field storage IDs (old or new)
862        final TreeSet<Integer> simpleFieldStorageIds = new TreeSet<>();
863        simpleFieldStorageIds.addAll(oldType.simpleFields.keySet());
864        simpleFieldStorageIds.addAll(newType.simpleFields.keySet());
865
866        // Update simple fields
867        for (int storageId : simpleFieldStorageIds) {
868
869            // Get old and new simple fields having this storage ID
870            final SimpleField<?> oldField = oldType.simpleFields.get(storageId);
871            final SimpleField<?> newField = newType.simpleFields.get(storageId);
872
873            // Save old field values for version change notification
874            final byte[] key = Field.buildKey(id, storageId);
875            final byte[] oldValue = this.kvt.get(key);
876            if (oldField != null) {
877                oldFieldValues.put(storageId,
878                  oldField.fieldType.read(new ByteReader(oldValue != null ? oldValue : oldField.fieldType.getDefaultValue())));
879            }
880
881            // Preserve the value if we can, otherwise, discard the old value (if any) leaving new value as new field default
882            boolean skipIndexUpdate = false;
883            if (oldField != null && newField != null
884              && newField.isSchemaChangeCompatible(oldField))                               // value is compatible so just leave it
885                skipIndexUpdate = newField.indexed == oldField.indexed;                     // index need not change either
886            else if (oldValue != null)
887                this.kvt.remove(key);                                                       // discard old value
888
889            // Remove old index entry
890            if (oldField != null && oldField.indexed && !skipIndexUpdate)
891                this.kvt.remove(oldField.buildIndexKey(id, oldValue));
892
893            // Add new index entry
894            if (newField != null && newField.indexed && !skipIndexUpdate)
895                this.kvt.put(newField.buildIndexKey(id, null), ByteUtil.EMPTY);
896        }
897
898    //////// Update complex fields and corresponding index entries
899
900        // Keep track of what to delete when we've done the migration notification
901        final ArrayList<ComplexField<?>> cleanupList = new ArrayList<>();
902
903        // Get complex field storage IDs (old or new)
904        final TreeSet<Integer> complexFieldStorageIds = new TreeSet<>();
905        complexFieldStorageIds.addAll(oldType.complexFields.keySet());
906        complexFieldStorageIds.addAll(newType.complexFields.keySet());
907
908        // Notes:
909        //
910        // - The only "migration" we support is adding/removing sub-field indexes; all other changes are equivalent to delete+add
911        // - New complex fields do not need to be explicitly initialized because their initial state is to have zero KV pairs
912        //
913        for (int storageId : complexFieldStorageIds) {
914
915            // Get old and new complex fields having this storage ID
916            final ComplexField<?> oldField = oldType.complexFields.get(storageId);
917            final ComplexField<?> newField = newType.complexFields.get(storageId);
918
919            // If there is no old field, new field is new and so it is already initialized (empty)
920            if (oldField == null)
921                continue;
922
923            // Save old field's value
924            oldFieldValues.put(storageId, oldField.getValue(this, id));
925
926            // Determine if the fields are compatible and field content can be preserved
927            final boolean compatible = newField != null && newField.isSchemaChangeCompatible(oldField);
928
929            // If fields are not compatible, delete old field when done, otherwise check if index(s) should be added/removed
930            if (!compatible)
931                cleanupList.add(oldField);
932            else
933                newField.updateSubFieldIndexes(this, oldField, id);
934        }
935
936        // Update object version
937        ObjInfo.write(this, id, this.version.versionNumber, info.isDeleteNotified());
938
939        // Notify about version update
940        try {
941            for (VersionChangeListener listener : this.versionChangeListeners)
942                listener.onVersionChange(this, id, info.getVersionNumber(), this.version.versionNumber, oldFieldValues);
943        } finally {
944            for (ComplexField<?> oldField : cleanupList)
945                oldField.deleteContent(this, id);
946        }
947    }

1520    /**
1521     * Monitor for changes within this transaction of the value of the given field, as seen through a path of references.
1522     *
1523     * <p>
1524     * When the specified field is changed, a listener notification will be delivered for each referring object that
1525     * refers to the object containing the changed field through the specified path of references. Notifications are
1526     * delivered at the end of the mutation operation just prior to returning to the caller. If the listener method
1527     * performs additional mutation(s) which are themselves being listened for, those notifications will also be delivered
1528     * prior to the returning to the original caller. Therefore, care must be taken to avoid change notification dependency
1529     * loops when listeners can themselves mutate fields, to avoid infinite loops.
1530     * </p>
1531     *
1532     * <p>
1533     * A referring object may refer to the changed object through more than one actual path of references matching {@code path};
1534     * if so, it will still appear only once in the {@link NavigableSet} provided to the listener (this is of course required
1535     * by set semantics).
1536     * </p>
1537     *
1538     * <p>
1539     * Although the reference back-tracking algorithm does consolidate multiple paths between the same two objects,
1540     * be careful to avoid an explosion of notifications when objects in the {@code path} have a high degree of fan-in.
1541     * </p>
1542     *
1543     * <p>
1544     * When a {@link ReferenceField} in {@code path} also happens to be the field being changed, then there is ambiguity
1545     * about how the set of referring objects is calculated: does it use the field's value before or after the change?
1546     * This API guarantees that the answer is "after the change"; however, if another listener on the same field is
1547     * invoked before {@code listener} and mutates any reference field(s) in {@code path}, then whether that additional
1548     * change is also be included in the calculation is undefined.
1549     * </p>
1550     *
1551     * <p>
1552     * Therefore, for consistency, avoid changing any {@link ReferenceField} from within a listener callback when that
1553     * field is also in some other listener's reference path, and both listeners are watching the same field.
1554     * </p>
1555     *
1556     * @param storageId storage ID of the field to monitor
1557     * @param path path of reference fields (represented by storage IDs) through which to monitor field
1558     * @param listener callback for notifications on changes in value
1559     * @throws UnknownFieldException if no {@link SimpleField} corresponding to {@code storageId} exists
1560     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
1561     * @throws IllegalArgumentException if {@code storageId} refers to a sub-field of a complex field
1562     * @throws IllegalArgumentException if {@code path} or {@code listener} is null
1563     * @throws StaleTransactionException if this transaction is no longer usable
1564     */
1565    public synchronized void addSimpleFieldChangeListener(int storageId, int[] path, SimpleFieldChangeListener listener) {
1566        this.validateChangeListener(SimpleField.class, storageId, path, listener);
1567        this.getMonitorsForField(storageId, true).add(new FieldMonitor(storageId, path, listener));
1568    }

1722        for (int pathStorageId : path)
1723            this.schema.verifyStorageInfo(pathStorageId, ReferenceFieldStorageInfo.class);
1724    }

1767    /**
1768     * Perform some action and, when entirely done (including re-entrant invocation), issue pending notifications to monitors.
1769     *
1770     * @param id object containing the mutated field; will be validated
1771     * @param mutation change to apply
1772     * @throws ReadOnlyTransactionException if this transaction has been {@linkplain #setReadOnly set read-only}
1773     * @throws StaleTransactionException if this transaction is no longer usable
1774     * @throws IllegalArgumentException if {@code id} is null
1775     */
1776    synchronized <V> V mutateAndNotify(ObjId id, Mutation<V> mutation) {
1777
1778        // Check that this transaction is still valid and the given object still exists.
1779        if (id == null)
1780            throw new IllegalArgumentException("null id");
1781        if (this.stale)
1782            throw new StaleTransactionException(this);
1783        if (this.readOnly)
1784            throw new ReadOnlyTransactionException(this);
1785        if (this.kvt.get(id.getBytes()) == null)
1786            throw new DeletedObjectException(id);
1787
1788        // If re-entrant invocation, we're already set up
1789        if (this.pendingNotifications.get() != null)
1790            return mutation.mutate();
1791
1792        // Set up pending report list, perform mutation, and then issue reports
1793        this.pendingNotifications.set(new TreeMap<Integer, ArrayList<FieldChangeNotifier>>());
1794        try {
1795            return mutation.mutate();
1796        } finally {
1797            try {
1798                final TreeMap<Integer, ArrayList<FieldChangeNotifier>> pendingNotificationMap = this.pendingNotifications.get();
1799                while (!pendingNotificationMap.isEmpty()) {
1800
1801                    // Get the next field with pending notifications
1802                    final Map.Entry<Integer, ArrayList<FieldChangeNotifier>> entry = pendingNotificationMap.pollFirstEntry();
1803                    final int storageId = entry.getKey();
1804
1805                    // For all pending notifications, back-track references and notify all field monitors for the field
1806                    for (FieldChangeNotifier notifier : entry.getValue()) {
1807                        assert notifier.getStorageId() == storageId;
1808                        final ArrayList<FieldMonitor> monitorList = new ArrayList<>(this.getMonitorsForField(storageId, false));
1809                        if (!monitorList.isEmpty())
1810                            this.notifyFieldMonitors(notifier, NavigableSets.singleton(notifier.getId()), monitorList, 0);
1811                    }
1812                }
1813            } finally {
1814                this.pendingNotifications.remove();
1815            }
1816        }
1817    }

1819    // Recursively back-track references along monitor paths and notify monitors when we reach the end (i.e., beginning)
1820    private void notifyFieldMonitors(FieldChangeNotifier notifier,
1821      NavigableSet<ObjId> objects, ArrayList<FieldMonitor> monitorList, int step) {
1822
1823        // Find the monitors for whom we have completed all the steps in their (inverse) path,
1824        // and group the remaining monitors by their next inverted reference path step.
1825        final HashMap<Integer, ArrayList<FieldMonitor>> remainingMonitorsMap = new HashMap<>();
1826        for (FieldMonitor monitor : monitorList) {
1827
1828            // Issue notification callback if we have back-tracked through the whole path
1829            if (monitor.path.length == step) {
1830                notifier.notify(this, monitor.listener, monitor.path, objects);
1831                continue;
1832            }
1833
1834            // Sort the unfinished monitors by their next (i.e., previous) reference field
1835            final int storageId = monitor.path[monitor.path.length - step - 1];
1836            ArrayList<FieldMonitor> remainingMonitors = remainingMonitorsMap.get(storageId);
1837            if (remainingMonitors == null) {
1838                remainingMonitors = new ArrayList<FieldMonitor>();
1839                remainingMonitorsMap.put(storageId, remainingMonitors);
1840            }
1841            remainingMonitors.add(monitor);
1842        }
1843
1844        // Invert references for each group of remaining monitors and recurse
1845        for (Map.Entry<Integer, ArrayList<FieldMonitor>> entry : remainingMonitorsMap.entrySet()) {
1846            final int storageId = entry.getKey();
1847
1848            // Gather all objects that refer to any object in our current "objects" set
1849            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
1850            for (ObjId object : objects) {
1851                final NavigableSet<ObjId> refs = this.queryIndex(storageId, FieldType.REFERENCE, FieldType.REFERENCE).get(object);
1852                if (refs != null)
1853                    refsList.add(refs);
1854            }
1855
1856            // Recurse on the union of those objects
1857            if (!refsList.isEmpty())
1858                this.notifyFieldMonitors(notifier, NavigableSets.union(refsList), entry.getValue(), step + 1);
1859        }
1860    }

1864    /**
1865     * Find all objects that refer to any object in the given target set through the specified path of references.
1866     *
1867     * @param path path of one or more reference fields (represented by storage IDs) through which to reach the target objects
1868     * @param targetObjects target objects
1869     * @return set of objects that refer to the {@code targetObjects} via {@code path}
1870     * @throws UnknownFieldException if {@code path} contains a storage ID that does not correspond to a {@link ReferenceField}
1871     * @throws IllegalArgumentException if {@code targetObjects} or {@code path} is null
1872     * @throws IllegalArgumentException if {@code path} is empty
1873     * @throws StaleTransactionException if this transaction is no longer usable
1874     */
1875    public NavigableSet<ObjId> invertReferencePath(int[] path, Iterable<ObjId> targetObjects) {
1876
1877        // Sanity check
1878        if (targetObjects == null)
1879            throw new IllegalArgumentException("null targetObjects");
1880        if (path == null)
1881            throw new IllegalArgumentException("null path");
1882        if (path.length == 0)
1883            throw new IllegalArgumentException("empty path");
1884
1885        // Verify all fields in the path are reference fields
1886        for (int pathStorageId : path)
1887            this.schema.verifyStorageInfo(pathStorageId, ReferenceFieldStorageInfo.class);
1888
1889        // Invert references
1890        NavigableSet<ObjId> result = null;
1891        for (int i = path.length - 1; i >= 0; i--) {
1892            final int storageId = path[i];
1893
1894            // Gather all objects that refer to any object in our current target objects set
1895            final ArrayList<NavigableSet<ObjId>> refsList = new ArrayList<>();
1896            for (ObjId id : targetObjects) {
1897                final NavigableSet<ObjId> refs = this.queryIndex(storageId, FieldType.REFERENCE, FieldType.REFERENCE).get(id);
1898                if (refs != null)
1899                    refsList.add(refs);
1900            }
1901            if (refsList.isEmpty())
1902                return NavigableSets.empty();
1903
1904            // Recurse on the union of those objects
1905            targetObjects = result = NavigableSets.union(refsList);
1906        }
1907
1908        // Done
1909        return result;
1910    }
