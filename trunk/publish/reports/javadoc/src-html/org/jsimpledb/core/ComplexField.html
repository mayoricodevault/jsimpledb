008package org.jsimpledb.core;
009
010import com.google.common.reflect.TypeToken;
011
012import java.util.Iterator;
013import java.util.List;
014
015import org.jsimpledb.kv.KVPair;
016import org.jsimpledb.kv.KVPairIterator;
017import org.jsimpledb.util.ByteReader;
018import org.jsimpledb.util.ByteUtil;
019import org.jsimpledb.util.ByteWriter;
020import org.jsimpledb.util.UnsignedIntEncoder;
021
022/**
023 * A complex {@link Field}, such as a collection or map field.
024 *
025 * @param <T> Java type for the field's values
026 */
027public abstract class ComplexField<T> extends Field<T> {
028
029    private final int storageIdLength;
030
031    /**
032     * Constructor.
033     *
034     * @param name the name of the field
035     * @param storageId field content storage ID
036     * @param version schema version
037     * @param typeToken Java type for the field's values
038     * @throws IllegalArgumentException if any parameter is null
039     * @throws IllegalArgumentException if {@code name} is invalid
040     * @throws IllegalArgumentException if {@code storageId} is non-positive
041     */
042    ComplexField(String name, int storageId, SchemaVersion version, TypeToken<T> typeToken) {
043        super(name, storageId, version, typeToken);
044        this.storageIdLength = UnsignedIntEncoder.encodeLength(storageId);
045    }
046
047    /**
048     * Get the sub-field(s) associated with this instance, ordered according to their meaning.
049     *
050     * @return unmodifiable list of simple fields
051     */
052    public abstract List<? extends SimpleField<?>> getSubFields();
053
054    /**
055     * Get the Java value of this field in the given object.
056     *
057     * @param tx transaction
058     * @param id object id
059     */
060    public abstract T getValue(Transaction tx, ObjId id);
061
062    abstract ComplexFieldStorageInfo toStorageInfo();
063
064    /**
065     * Check compatibility with another {@link ComplexField}.
066     * To be compatible, the two enties must be exactly the same in terms of data layouts and Java representations.
067     * This is in effect an {@code #equals equals()} test with respect to those aspects. Note that compatibililty
068     * does not necessarily imply the same sub-fields are indexed.
069     *
070     * <p>
071     * The implementation in {@link ComplexField} checks that {@code that} has the same Java type
072     * and as this instance and that all sub-fields have equivalent {@link FieldType}s.
073     * </p>
074     *
075     * @param that field to check for compatibility
076     * @throws NullPointerException if {@code that} is null
077     */
078    boolean isCompatible(ComplexField<?> that) {
079        if (that.getClass() != this.getClass() || that.storageId != this.storageId)
080            return false;
081        final List<? extends SimpleField<?>> thisSubFields = this.getSubFields();
082        final List<? extends SimpleField<?>> thatSubFields = that.getSubFields();
083        if (thisSubFields.size() != thatSubFields.size())
084            return false;
085        for (int i = 0; i < thisSubFields.size(); i++) {
086            if (!thisSubFields.get(i).fieldType.equals(thatSubFields.get(i).fieldType))
087                return false;
088        }
089        return true;
090    }
091
092    @Override
093    boolean isEquivalent(Field<?> field) {
094        if (field.getClass() != this.getClass())
095            return false;
096        final ComplexField<?> that = (ComplexField<?>)field;
097        final List<? extends SimpleField<?>> thisSubFields = this.getSubFields();
098        final List<? extends SimpleField<?>> thatSubFields = that.getSubFields();
099        if (thisSubFields.size() != thatSubFields.size())
100            return false;
101        for (int i = 0; i < thisSubFields.size(); i++) {
102            if (!thisSubFields.get(i).isEquivalent(thatSubFields.get(i)))
103                return false;
104        }
105        return true;
106    }
107
108    /**
109     * Delete all content (but not index entries) for the given object.
110     *
111     * @param tx transaction
112     * @param id object id
113     */
114    void deleteContent(Transaction tx, ObjId id) {
115        final byte[] minKey = this.buildContentPrefix(id);
116        final byte[] maxKey = ByteUtil.getKeyAfterPrefix(minKey);
117        this.deleteContent(tx, minKey, maxKey);
118    }
119
120    /**
121     * Delete all content (but not index entries) for the given object in the given key range
122     *
123     * @param tx transaction
124     * @param id object id
125     */
126    void deleteContent(Transaction tx, byte[] minKey, byte[] maxKey) {
127        tx.kvt.removeRange(minKey, maxKey);
128    }
129
130    /**
131     * Add an index entry corresponding to the given sub-field and content key/value pair.
132     *
133     * @param tx transaction
134     * @param id object id
135     * @param subField indexed sub-field
136     * @param contentKey the content key
137     * @param contentValue the value associated with the content key, or null if not needed
138     */
139    void addIndexEntry(Transaction tx, ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
140        tx.kvt.put(this.buildIndexEntry(id, subField, contentKey, contentValue), ByteUtil.EMPTY);
141    }
142
143    /**
144     * Remove an index entry corresponding to the given sub-field and content key/value pair.
145     *
146     * @param tx transaction
147     * @param id object id
148     * @param subField indexed sub-field
149     * @param contentKey the content key
150     * @param contentValue the value associated with the content key, or null if not needed
151     */
152    void removeIndexEntry(Transaction tx, ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
153        tx.kvt.remove(this.buildIndexEntry(id, subField, contentKey, contentValue));
154    }
155
156    private byte[] buildIndexEntry(ObjId id, SimpleField<?> subField, byte[] contentKey, byte[] contentValue) {
157        final ByteReader contentKeyReader = new ByteReader(contentKey);
158        contentKeyReader.skip(ObjId.NUM_BYTES + this.storageIdLength);                  // skip to content
159        final ByteWriter writer = new ByteWriter();
160        UnsignedIntEncoder.write(writer, subField.storageId);
161        this.buildIndexEntry(id, subField, contentKeyReader, contentValue, writer);
162        return writer.getBytes();
163    }
164
165    /**
166     * Build an index key for the given object, sub-field, and content key/value pair.
167     *
168     * @param id object id
169     * @param subField indexed sub-field
170     * @param reader reader of content key, positioned just after the object ID and the storage ID for this field
171     * @param value the value associated with the content key, or null if not needed
172     * @param writer writer for the index entry, with the sub-field's storage ID already written
173     */
174    abstract void buildIndexEntry(ObjId id, SimpleField<?> subField, ByteReader reader, byte[] value, ByteWriter writer);
175
176    /**
177     * Add or remove index entries for the given object as appropriate after a schema version change
178     * which changed only whether some or all sub-field(s) are indexed.
179     *
180     * @param kvt KV store
181     * @param oldField compatible field in older schema
182     * @param id object id
183     */
184    void updateSubFieldIndexes(Transaction tx, ComplexField<?> oldField, ObjId id) {
185        final Iterator<? extends SimpleField<?>> oldSubFields = oldField.getSubFields().iterator();
186        final Iterator<? extends SimpleField<?>> newSubFields = this.getSubFields().iterator();
187        while (oldSubFields.hasNext() || newSubFields.hasNext()) {
188            final SimpleField<?> oldSubField = oldSubFields.next();
189            final SimpleField<?> newSubField = newSubFields.next();
190            if (!oldSubField.indexed && newSubField.indexed)
191                this.addIndexEntries(tx, id, newSubField);
192            else if (oldSubField.indexed && !newSubField.indexed)
193                oldField.removeIndexEntries(tx, id, oldSubField);
194        }
195    }
196
197    /**
198     * Add all index entries for the given object and sub-field.
199     *
200     * @param tx transaction
201     * @param id object id
202     * @param subField sub-field of this field
203     */
204    void addIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField) {
205        for (KVPairIterator i = new KVPairIterator(tx.kvt, this.buildContentPrefix(id)); i.hasNext(); ) {
206            final KVPair pair = i.next();
207            this.addIndexEntry(tx, id, subField, pair.getKey(), pair.getValue());
208        }
209    }
210
211    /**
212     * Remove all index entries for the given object and sub-field.
213     *
214     * @param tx transaction
215     * @param id object id
216     * @param subField sub-field of this field
217     */
218    void removeIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField) {
219        final byte[] prefix = this.buildContentPrefix(id);
220        this.removeIndexEntries(tx, id, subField, prefix, ByteUtil.getKeyAfterPrefix(prefix));
221    }
222
223    /**
224     * Remove index entries for the given object and sub-field, restricted to the given key range.
225     *
226     * @param tx transaction
227     * @param id object id
228     * @param subField sub-field of this field
229     */
230    void removeIndexEntries(Transaction tx, ObjId id, SimpleField<?> subField, byte[] minKey, byte[] maxKey) {
231        for (KVPairIterator i = new KVPairIterator(tx.kvt, minKey, maxKey); i.hasNext(); ) {
232            final KVPair pair = i.next();
233            this.removeIndexEntry(tx, id, subField, pair.getKey(), pair.getValue());
234        }
235    }
236
237    /**
238     * Build the prefix of all content keys for this field in the given object.
239     */
240    byte[] buildContentPrefix(ObjId id) {
241        final ByteWriter writer = new ByteWriter();
242        id.writeTo(writer);
243        UnsignedIntEncoder.write(writer, this.storageId);
244        return writer.getBytes();
245    }
246}
