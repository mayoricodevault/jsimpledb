008package org.jsimpledb.core;
009
010import java.util.NavigableMap;
011import java.util.NavigableSet;
012
013import org.jsimpledb.index.Index;
014import org.jsimpledb.kv.KeyFilter;
015import org.jsimpledb.tuple.Tuple2;
016
017/**
018 * Core API {@link Index} implementation representing a index on a single field.
019 *
020 * <p>
021 * Instances are immutable.
022 * </p>
023 *
024 * @param <V> index value type
025 * @param <T> index target type
026 */
027public class CoreIndex<V, T> extends AbstractCoreIndex implements Index<V, T> {
028
029// Constructors
030
031    CoreIndex(Transaction tx, IndexView<V, T> indexView) {
032        super(tx, 2, indexView);
033    }
034
035// Methods
036
037    @Override
038    public CoreIndex<V, T> filter(int index, KeyFilter filter) {
039        return new CoreIndex<V, T>(this.tx, this.getIndexView().filter(index, filter));
040    }
041
042    @SuppressWarnings("unchecked")
043    IndexView<V, T> getIndexView() {
044        return (IndexView<V, T>)this.indexView;
045    }
046
047// Index
048
049    @Override
050    public NavigableSet<Tuple2<V, T>> asSet() {
051
052        // Get index view
053        final IndexView<V, T> iv = this.getIndexView();
054
055        // Create tuple field type
056        final Tuple2FieldType<V, T> tupleFieldType = new Tuple2FieldType<V, T>(iv.getValueType(), iv.getTargetType());
057
058        // Build set and apply filtering
059        IndexSet<Tuple2<V, T>> indexSet = new IndexSet<Tuple2<V, T>>(this.tx, tupleFieldType, iv.prefixMode, iv.prefix);
060        if (iv.hasFilters())
061            indexSet = indexSet.filterKeys(new IndexKeyFilter(this.tx, iv, 2));
062
063        // Done
064        return indexSet;
065    }
066
067    @Override
068    public NavigableMap<V, NavigableSet<T>> asMap() {
069
070        // Get index view
071        final IndexView<V, T> iv = this.getIndexView();
072
073        // Build map and apply filtering
074        IndexMap<V, NavigableSet<T>> indexMap = new IndexMap.OfValues<V, T>(this.tx, iv);
075        if (this.indexView.hasFilters())
076            indexMap = indexMap.filterKeys(new IndexKeyFilter(this.tx, iv, 1));
077
078        // Done
079        return indexMap;
080    }
081}
