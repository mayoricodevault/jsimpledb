008package org.jsimpledb.core;
009
010import com.google.common.base.Function;
011import com.google.common.collect.Lists;
012
013import java.util.Collection;
014import java.util.LinkedHashMap;
015import java.util.List;
016import java.util.Map;
017
018import org.dellroad.stuff.java.EnumUtil;
019
020/**
021 * The {@link FieldType} for {@link EnumField}s.
022 *
023 * <p>
024 * Each {@link EnumFieldType} instance has an ordered list of {@link String} identifiers; two {@link EnumFieldType}
025 * instances are not compatible unless they have identical identifier lists. The identifiers must be valid Java identifiers.
026 * </p>
027 *
028 * <p>
029 * Null values are supported by this class.
030 * </p>
031 */
032public class EnumFieldType extends NullSafeType<EnumValue> {
033
034    public static final String IDENT_PATTERN = "\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*";
035
036    /**
037     * Constructor to use when there is an associated {@link Enum} type;
038     * {@link #getEnumType} will return {@code enumType}.
039     *
040     * @param enumType Java {@link Enum} type from which to derive the ordered identifier list
041     * @throws NullPointerException if {@code enumType} is null
042     */
043    public <T extends Enum<T>> EnumFieldType(Class<T> enumType) {
044        super(new EnumType(enumType, enumType.getName(), EnumFieldType.getIdentifiers(enumType)));
045    }
046
047    /**
048     * Constructor to use when there is no corresponding {@link Enum} type;
049     * {@link #getEnumType} will return null.
050     *
051     * @param name name of this type
052     * @param idents ordered list of identifiers
053     * @throws IllegalArgumentException if {@code name} is null or invalid
054     * @throws IllegalArgumentException if {@code idents} is null or contains a duplicate or invalid identifier
055     */
056    public EnumFieldType(String name, List<String> idents) {
057        super(new EnumType(null, name, idents));
058    }
059
060    /**
061     * Get the identifiers associated with this instance in ordinal order.
062     *
063     * @return unmodifiable, ordinally ordered list of identifiers
064     */
065    public List<String> getIdentifiers() {
066        return ((EnumType)this.inner).getIdentifiers();
067    }
068
069    /**
070     * Get the {@link Enum} type associated with this instance, if known.
071     *
072     * @return associated {@link Enum} type, or null if actual {@link Enum} type is unknown
073     */
074    public Class<? extends Enum<?>> getEnumType() {
075        return ((EnumType)this.inner).getEnumType();
076    }
077
078    /**
079     * Create an {@link EnumFieldType} instance suitable for use with the given {@link Enum} type.
080     *
081     * @param enumType Java {@link Enum} type from which to derive the ordered identifier list
082     * @throws ClassCastException if {@code enumType} does not subclass {@link Enum}
083     * @throws NullPointerException if {@code enumType} is null
084     */
085    @SuppressWarnings("unchecked")
086    public static <T extends Enum<T>> EnumFieldType create(Class<?> enumType) {
087        return new EnumFieldType((Class<T>)enumType.asSubclass(Enum.class));
088    }
089
090    /**
091     * Create an {@link EnumFieldType} instance suitable for use with the given identifier list.
092     *
093     * <p>
094     * If {@code name} is the name of an {@link Enum} class with matching identifiers, then it will
095     * associated with the created instance and returned by its {@link #getEnumType} method.
096     * </p>
097     *
098     * @param name name of the type
099     * @param idents ordered list of identifiers
100     * @throws IllegalArgumentException if {@code name} is null or invalid
101     * @throws IllegalArgumentException if {@code idents} is null or contains a duplicate or invalid identifier
102     */
103    @SuppressWarnings("unchecked")
104    public static <T extends Enum<T>> EnumFieldType create(String name, List<String> idents) {
105        do {
106
107            // Search for Enum type
108            final Class<T> enumType;
109            try {
110                enumType = (Class<T>)Class.forName(name, false,
111                  Thread.currentThread().getContextClassLoader()).asSubclass(Enum.class);
112            } catch (Exception e) {
113                break;
114            }
115
116            // Validate it has the expected identifier list
117            if (!EnumFieldType.getIdentifiers(enumType).equals(idents))
118                break;
119
120            // We're good
121            return new EnumFieldType(enumType);
122        } while (false);
123
124        // Enum type not found or has incompatible identifier list
125        return new EnumFieldType(name, idents);
126    }
127
128    /**
129     * Validate a list of enum identifiers and build a mapping from identifier to corresponding {@link EnumValue}.
130     * The returned mapping will iterate the {@link EnumValue}s in ordinal order.
131     *
132     * @return ordinally ordered mapping from identifier to {@link EnumValue}
133     * @throws IllegalArgumentException if {@code idents} is null
134     * @throws IllegalArgumentException if any identifier in {@code idents} is null, duplicate, or not a valid Java identifier
135     */
136    public static Map<String, EnumValue> validateIdentifiers(List<String> idents) {
137        if (idents == null)
138            throw new IllegalArgumentException("null idents");
139        final LinkedHashMap<String, EnumValue> identifierMap = idents instanceof Collection ?
140          new LinkedHashMap<String, EnumValue>(((Collection<?>)idents).size()) : new LinkedHashMap<String, EnumValue>();
141        for (String ident : idents) {
142            final int index = identifierMap.size();
143            if (ident == null)
144                throw new IllegalArgumentException("invalid null enum identifier at index " + index);
145            if (ident.equals("null") || !ident.matches(IDENT_PATTERN))
146                throw new IllegalArgumentException("invalid enum identifier `" + ident + "' at index " + index);
147            final EnumValue otherValue = identifierMap.put(ident, new EnumValue(ident, index));
148            if (otherValue != null) {
149                throw new IllegalArgumentException("invalid duplicate enum identifier `" + ident
150                  + "' at indexes " + otherValue.getOrdinal() + " and " + index);
151            }
152        }
153        return identifierMap;
154    }
155
156    @SuppressWarnings("unchecked")
157    private static <T extends Enum<T>> List<String> getIdentifiers(Class<T> enumType) {
158        return Lists.transform(EnumUtil.getValues(enumType), new Function<T, String>() {
159            @Override
160            public String apply(T value) {
161                return value.name();
162            }
163        });
164    }
165}
