008package org.jsimpledb.core;
009
010import com.google.common.collect.Lists;
011import com.google.common.reflect.TypeToken;
012
013import java.util.ArrayList;
014import java.util.Collections;
015import java.util.HashMap;
016import java.util.Map;
017
018import org.dellroad.stuff.java.Primitive;
019import org.dellroad.stuff.java.PrimitiveSwitch;
020
021/**
022 * A registry of {@link FieldType}s.
023 */
024public class FieldTypeRegistry {
025
026    private final HashMap<String, FieldType<?>> typesByName = new HashMap<>();
027    private final HashMap<TypeToken<?>, ArrayList<FieldType<?>>> typesByType = new HashMap<>();
028
029    /**
030     * Constructor. Creates an instance with all of the pre-defined {@link FieldType}s (e.g., for
031     * types such as primitive types, {@link String}, etc.) already registered.
032     */
033    public FieldTypeRegistry() {
034        this.add(FieldType.VOID_WRAPPER);
035        this.add(FieldType.BOOLEAN);
036        this.add(FieldType.BOOLEAN_WRAPPER);
037        this.add(FieldType.BYTE);
038        this.add(FieldType.BYTE_WRAPPER);
039        this.add(FieldType.SHORT);
040        this.add(FieldType.SHORT_WRAPPER);
041        this.add(FieldType.CHARACTER);
042        this.add(FieldType.CHARACTER_WRAPPER);
043        this.add(FieldType.INTEGER);
044        this.add(FieldType.INTEGER_WRAPPER);
045        this.add(FieldType.FLOAT);
046        this.add(FieldType.FLOAT_WRAPPER);
047        this.add(FieldType.LONG);
048        this.add(FieldType.LONG_WRAPPER);
049        this.add(FieldType.DOUBLE);
050        this.add(FieldType.DOUBLE_WRAPPER);
051        this.add(FieldType.STRING);
052        this.add(FieldType.DATE);
053        this.add(FieldType.ENUM_VALUE);
054        this.add(FieldType.UUID);
055        this.add(FieldType.URI);
056        this.add(FieldType.FILE);
057        this.add(FieldType.PATTERN);
058    }
059
060    /**
061     * Add a user-defined {@link FieldType} to the registry.
062     *
063     * @param type type to add
064     * @throws IllegalArgumentException if {@code type} is null
065     * @throws IllegalArgumentException if {@code type} has a name that conflicts with an existing type
066     * @throws IllegalArgumentException if {@code type} has a name that ends with {@code []} (array name)
067     */
068    public synchronized void add(FieldType<?> type) {
069        if (type == null)
070            throw new IllegalArgumentException("null type");
071        if (type.name.endsWith(ArrayType.ARRAY_SUFFIX))
072            throw new IllegalArgumentException("illegal array type name `" + type.name + "'");
073        final FieldType<?> other = this.typesByName.get(type.name);
074        if (other != null)
075            throw new IllegalArgumentException("type name `" + type.name + "' conflicts with existing type " + other);
076        this.typesByName.put(type.name, type);
077        final TypeToken<?> typeToken = type.getTypeToken();
078        if (!this.typesByType.containsKey(typeToken))
079            this.typesByType.put(typeToken, Lists.newArrayList(Collections.<FieldType<?>>singleton(type)));
080        else
081            this.typesByType.get(typeToken).add(type);
082    }
083
084    /**
085     * Get the {@link FieldType} with the given name in this registry.
086     *
087     * @param name type name
088     * @return type with name {@code name}, or null if not found
089     * @throws IllegalArgumentException if {@code name} is null
090     */
091    public synchronized FieldType<?> getFieldType(final String name) {
092
093        // Sanity check
094        if (name == null)
095            throw new IllegalArgumentException("null name");
096
097        // Handle array types
098        if (name.endsWith(ArrayType.ARRAY_SUFFIX)) {
099            final FieldType<?> elementType = this.getFieldType(name.substring(0, name.length() - ArrayType.ARRAY_SUFFIX.length()));
100            return this.getArrayType(elementType);
101        }
102
103        // Handle non-array type
104        return this.typesByName.get(name);
105    }
106
107    /**
108     * Get the array {@link FieldType} with the given element type.
109     *
110     * @param elementType array element type
111     * @throws IllegalArgumentException if {@code elementType} is null
112     * @throws IllegalArgumentException if the resulting array type has too many dimensions
113     * @return array type with element type {@code elementType}
114     */
115    public FieldType<?> getArrayType(final FieldType<?> elementType) {
116
117        // Sanity check
118        if (elementType == null)
119            throw new IllegalArgumentException("null elementType");
120
121        // Create array type
122        final Primitive<?> primitive = Primitive.get(elementType.typeToken.getRawType());
123        final ArrayType<?, ?> notNullType = primitive == null ? this.createObjectArrayType(elementType) :
124          primitive.visit(new PrimitiveSwitch<ArrayType<?, ?>>() {
125            @Override
126            public ArrayType<?, ?> caseVoid() {
127                throw new IllegalArgumentException("cannot create array of type `" + elementType.name + "'");
128            }
129            @Override
130            public ArrayType<?, ?> caseBoolean() {
131                return new BooleanArrayType();
132            }
133            @Override
134            public ArrayType<?, ?> caseByte() {
135                return new ByteArrayType();
136            }
137            @Override
138            public ArrayType<?, ?> caseCharacter() {
139                return new CharacterArrayType();
140            }
141            @Override
142            public ArrayType<?, ?> caseShort() {
143                return new ShortArrayType();
144            }
145            @Override
146            public ArrayType<?, ?> caseInteger() {
147                return new IntegerArrayType();
148            }
149            @Override
150            public ArrayType<?, ?> caseFloat() {
151                return new FloatArrayType();
152            }
153            @Override
154            public ArrayType<?, ?> caseLong() {
155                return new LongArrayType();
156            }
157            @Override
158            public ArrayType<?, ?> caseDouble() {
159                return new DoubleArrayType();
160            }
161        });
162        return this.createNullSafeType(notNullType);
163    }
164
165    /**
166     * Get the (unique) {@link FieldType} in this registry that supports values of the given Java type, which must
167     * exactly match the {@link FieldType}'s {@linkplain FieldType#getTypeToken supported Java type}.
168     * If there is no such {@link FieldType}, null is returned; if there is more than one such {@link FieldType},
169     * an {@link IllegalArgumentException} is thrown.
170     *
171     * <b>Arrays</b>
172     *
173     * <p>
174     * Array types will be automatically created and registered on demand, assuming the base element
175     * type is already registered.
176     * </p>
177     *
178     * <p>
179     * Arrays are passed by value: i.e., the entire array is copied. Therefore, changes to array elements after
180     * getting or setting a field have no effect on the field's value.
181     * </p>
182     *
183     * <p>
184     * Arrays of references are not currently supported.
185     * </p>
186     *
187     * @param typeToken supported Java value type
188     * @return {@link FieldType} supporting Java values of type {@code typeToken}, or null if no such {@link FieldType} exists
189     * @throws IllegalArgumentException if {@code typeToken} is null
190     * @throws IllegalArgumentException if more than one {@link FieldType} supports {@code typeToken}
191     */
192    public synchronized FieldType<?> getFieldType(TypeToken<?> typeToken) {
193
194        // Sanity check
195        if (typeToken == null)
196            throw new IllegalArgumentException("null typeToken");
197
198        // Handle array types
199        final TypeToken<?> elementTypeToken = typeToken.getComponentType();
200        if (elementTypeToken != null)
201            return this.getArrayType(this.getFieldType(elementTypeToken));
202
203        // Handle non-array types
204        final ArrayList<FieldType<?>> types = this.typesByType.get(typeToken);
205        if (types != null) {
206            switch (types.size()) {
207            case 0:
208                break;
209            case 1:
210                return types.get(0);
211            default:
212                throw new IllegalArgumentException("multiple registered types support values of type " + typeToken + ": " + types);
213            }
214        }
215
216        // None found
217        return null;
218    }
219
220    /**
221     * Get all types registered with this instance.
222     *
223     * @return mapping from type name to type
224     */
225    public synchronized Map<String, FieldType<?>> getAll() {
226        return new HashMap<String, FieldType<?>>(this.typesByName);
227    }
228
229    // This method exists solely to bind the generic type parameters
230    private <E> ObjectArrayType<E> createObjectArrayType(FieldType<E> elementType) {
231        return new ObjectArrayType<E>(elementType);
232    }
233
234    // This method exists solely to bind the generic type parameters
235    private <T> NullSafeType<T> createNullSafeType(FieldType<T> notNullType) {
236        return new NullSafeType<T>(notNullType);
237    }
238}
