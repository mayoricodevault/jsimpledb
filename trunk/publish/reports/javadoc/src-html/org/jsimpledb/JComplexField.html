008package org.jsimpledb;
009
010import java.lang.reflect.Method;
011import java.util.List;
012
013import org.jsimpledb.schema.ComplexSchemaField;
014
015/**
016 * Represents a complex field in a {@link JClass}.
017 */
018public abstract class JComplexField extends JField {
019
020    JComplexField(JSimpleDB jdb, String name, int storageId, String description, Method getter) {
021        super(jdb, name, storageId, description, getter);
022        if (name == null)
023            throw new IllegalArgumentException("null name");
024    }
025
026    @Override
027    abstract ComplexSchemaField toSchemaItem(JSimpleDB jdb);
028
029    /**
030     * Get the sub-fields associated with this field.
031     *
032     * @return this instance's sub-fields (in order)
033     */
034    public abstract List<JSimpleField> getSubFields();
035
036    /**
037     * Get the sub-field with the given name.
038     *
039     * @param name sub-field name
040     * @return the sub-field with the specified name
041     * @throws IllegalArgumentException if {@code name} is invalid
042     */
043    public abstract JSimpleField getSubField(String name);
044
045    /**
046     * Get the sub-field with the given storage ID.
047     *
048     * @param storageId sub-field storage ID
049     * @throws IllegalArgumentException if not found
050     */
051    JSimpleField getSubField(int storageId) {
052        for (JSimpleField subField : this.getSubFields()) {
053            if (subField.storageId == storageId)
054                return subField;
055        }
056        throw new IllegalArgumentException("storage ID " + storageId + " not found");
057    }
058
059    /**
060     * Get the name of the given sub-field.
061     *
062     * @throws IllegalArgumentException if {@code subField} is not one of {@link #getSubFields}
063     */
064    abstract String getSubFieldName(JSimpleField subField);
065}
