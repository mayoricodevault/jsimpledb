008package org.jsimpledb;
009
010import com.google.common.reflect.TypeToken;
011
012import java.lang.reflect.Method;
013import java.util.Deque;
014import java.util.List;
015import java.util.Set;
016
017import org.jsimpledb.core.ObjId;
018import org.jsimpledb.schema.ComplexSchemaField;
019
020/**
021 * Represents a complex field in a {@link JClass}.
022 */
023public abstract class JComplexField extends JField {
024
025    JComplexField(String name, int storageId, String description, Method getter) {
026        super(name, storageId, description, getter);
027        if (name == null)
028            throw new IllegalArgumentException("null name");
029    }
030
031    @Override
032    abstract ComplexSchemaField toSchemaItem();
033
034    /**
035     * Get the sub-fields associated with this field.
036     */
037    public abstract List<JSimpleField> getSubFields();
038
039    /**
040     * Get the sub-field with the given name.
041     *
042     * @throws IllegalArgumentException if {@code name} is invalid
043     */
044    public abstract JSimpleField getSubField(String name);
045
046    /**
047     * Get the name of the given sub-field.
048     *
049     * @throws IllegalArgumentException if {@code subField} is not one of {@link #getSubFields}
050     */
051    abstract String getSubFieldName(JSimpleField subField);
052
053    /**
054     * Add any valid index entry return types for @IndexQuery-annotated methods that query the given indexed sub-field.
055     */
056    abstract <T> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, JSimpleField subField);
057
058    /**
059     * Determine the type of index query from the method return type, i.e., normal object query or some kind of index entry query.
060     *
061     * @return zero for a normal query or some field-specific value otherwise
062     */
063    abstract int getIndexEntryQueryType(TypeToken<?> queryObjectType);
064
065    /**
066     * Get the {@link JTransaction} method to invoke from generated classes for the given index entry query type.
067     */
068    abstract Method getIndexEntryQueryMethod(int queryType);
069
070    /**
071     * Recurse for copying between transactions.
072     */
073    abstract void copyRecurse(Set<ObjId> seen, JTransaction src, JTransaction dest,
074      ObjId id, JReferenceField subField, Deque<JReferenceField> nextFields);
075
076    void copyRecurse(Set<ObjId> seen, JTransaction src, JTransaction dest, Iterable<?> it, Deque<JReferenceField> nextFields) {
077        for (Object obj : it) {
078            if (obj != null)
079                src.copyTo(seen, dest, (ObjId)obj, nextFields);
080        }
081    }
082}
