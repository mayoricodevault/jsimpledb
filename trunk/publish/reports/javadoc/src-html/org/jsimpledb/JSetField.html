008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.List;
016
017import org.jsimpledb.change.FieldChange;
018import org.jsimpledb.change.ListFieldReplace;
019import org.jsimpledb.change.SetFieldAdd;
020import org.jsimpledb.change.SetFieldChange;
021import org.jsimpledb.change.SetFieldClear;
022import org.jsimpledb.change.SetFieldRemove;
023import org.jsimpledb.core.Transaction;
024import org.jsimpledb.schema.SetSchemaField;
025import org.objectweb.asm.ClassWriter;
026
027/**
028 * Represents a set field in a {@link JClass}.
029 */
030public class JSetField extends JCollectionField {
031
032    JSetField(String name, int storageId, JSimpleField elementField, String description, Method getter) {
033        super(name, storageId, elementField, description, getter);
034    }
035
036    @Override
037    SetSchemaField toSchemaItem() {
038        final SetSchemaField schemaField = new SetSchemaField();
039        super.initialize(schemaField);
040        return schemaField;
041    }
042
043    @Override
044    void outputMethods(ClassGenerator<?> generator, ClassWriter cw) {
045        this.outputReadMethod(generator, cw, ClassGenerator.READ_SET_FIELD_METHOD);
046    }
047
048    @Override
049    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
050        tx.addSetFieldChangeListener(this.storageId, path, listener);
051    }
052
053    @Override
054    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
055        this.addChangeParameterTypes(types, targetType, this.elementField.typeToken);
056    }
057
058    // This method exists solely to bind the generic type parameters
059    @SuppressWarnings("serial")
060    private <T, E> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<E> elementType) {
061        types.add(new TypeToken<FieldChange<T>>() { }
062          .where(new TypeParameter<T>() { }, targetType));
063        types.add(new TypeToken<SetFieldChange<T>>() { }
064          .where(new TypeParameter<T>() { }, targetType));
065        types.add(new TypeToken<SetFieldAdd<T, E>>() { }
066          .where(new TypeParameter<T>() { }, targetType)
067          .where(new TypeParameter<E>() { }, elementType.wrap()));
068        types.add(new TypeToken<SetFieldClear<T, E>>() { }
069          .where(new TypeParameter<T>() { }, targetType)
070          .where(new TypeParameter<E>() { }, elementType.wrap()));
071        types.add(new TypeToken<SetFieldRemove<T, E>>() { }
072          .where(new TypeParameter<T>() { }, targetType)
073          .where(new TypeParameter<E>() { }, elementType.wrap()));
074        types.add(new TypeToken<ListFieldReplace<T, E>>() { }
075          .where(new TypeParameter<T>() { }, targetType)
076          .where(new TypeParameter<E>() { }, elementType.wrap()));
077    }
078
079    @Override
080    <T> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, JSimpleField subField) {
081        // there are no index entry types for sets
082    }
083
084    @Override
085    int getIndexEntryQueryType(TypeToken<?> queryObjectType) {
086        return 0;
087    }
088
089    @Override
090    Method getIndexEntryQueryMethod(int queryType) {
091        throw new UnsupportedOperationException();
092    }
093
094    @Override
095    NavigableSetConverter<?, ?> getConverter(JSimpleDB jdb) {
096        final Converter<?, ?> elementConverter = this.elementField.getConverter(jdb);
097        if (elementConverter == null)
098            return null;
099        return this.createConverter(elementConverter);
100    }
101
102    // This method exists solely to bind the generic type parameters
103    private <X, Y> NavigableSetConverter<X, Y> createConverter(Converter<X, Y> elementConverter) {
104        return new NavigableSetConverter<X, Y>(elementConverter);
105    }
106}
