008package org.jsimpledb;

010/**
011 * Represents a {@link JObject} for which no Java model type is defined in the current schema.
012 *
013 * <p>
014 * Instances of this class are used to represent objects with a type that is defined in some older schema version
015 * but not in the current schema version. This situation can occur when a new schema is added that drops a previously
016 * defined Java model type, yet for which objects of that type still exist in the database. If encountered, these
017 * objects will be represented by instances of this class.
018 * </p>
019 *
020 * <p>
021 * All object fields are still accessible, but they must be accessed directly via the {@link JTransaction} field access methods
022 * with {@code upgradeVersion} set to false (otherwise, a {@link org.jsimpledb.core.TypeNotInSchemaVersionException} is thrown).
023 * </p>
024 */
025public abstract class UntypedJObject implements JObject {
026}
