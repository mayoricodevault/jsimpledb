008package org.jsimpledb.schema;
009
010import java.util.Map;
011
012import javax.xml.namespace.QName;
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.XMLStreamReader;
015import javax.xml.stream.XMLStreamWriter;
016
017import org.jsimpledb.core.InvalidSchemaException;
018
019/**
020 * A complex field in one version of a {@link SchemaObjectType}.
021 */
022public abstract class ComplexSchemaField extends SchemaField {
023
024    @Override
025    void validate() {
026        super.validate();
027        for (Map.Entry<String, SimpleSchemaField> entry : this.getSubFields().entrySet()) {
028            final String subFieldName = entry.getKey();
029            final SimpleSchemaField subField = entry.getValue();
030            subField.validate();
031            if (subField == null)
032                throw new InvalidSchemaException("invalid " + this + ": missing sub-field `" + subFieldName + "'");
033            if (!subFieldName.equals(subField.getName())) {
034                throw new InvalidSchemaException("sub-" + subField + " of " + this + " has the wrong name `"
035                  + subField.getName() + "' != `" + subFieldName + "'");
036            }
037        }
038    }
039
040    public abstract Map<String, SimpleSchemaField> getSubFields();
041
042    @Override
043    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
044        final ComplexSchemaField that = (ComplexSchemaField)that0;
045        if (!this.getSubFields().keySet().equals(that.getSubFields().keySet()))
046            return false;
047        for (String subFieldName : this.getSubFields().keySet()) {
048            final SimpleSchemaField thisSubField = this.getSubFields().get(subFieldName);
049            final SimpleSchemaField thatSubField = that.getSubFields().get(subFieldName);
050            if (!thisSubField.isCompatibleWith(thatSubField))
051                return false;
052        }
053        return true;
054    }
055
056    SimpleSchemaField readSubField(XMLStreamReader reader, int formatVersion, String name) throws XMLStreamException {
057        final SimpleSchemaField field = this.readMappedType(reader, false, SchemaModel.SIMPLE_FIELD_TAG_MAP);
058        field.readXML(reader, formatVersion);
059        if (field.getName() == null)
060            field.setName(name);
061        return field;
062    }
063
064    @Override
065    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
066        final QName tag = this.getXMLTag();
067        writer.writeStartElement(tag.getNamespaceURI(), tag.getLocalPart());
068        this.writeAttributes(writer);
069        for (SimpleSchemaField subField : this.getSubFields().values())
070            subField.writeXML(writer, false);                               // omit (redundant) names for sub-fields
071        writer.writeEndElement();
072    }
073
074    abstract QName getXMLTag();
075
076// Cloneable
077
078    @Override
079    public ComplexSchemaField clone() {
080        return (ComplexSchemaField)super.clone();
081    }
082}
