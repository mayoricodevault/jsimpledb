008package org.jsimpledb.schema;
009
010/**
011 * Adapter class for {@link SchemaFieldSwitch}.
012 *
013 * @param <R> switch method return type
014 */
015public class SchemaFieldSwitchAdapter<R> implements SchemaFieldSwitch<R> {
016
017    /**
018     * Handle a {@link SetSchemaField}.
019     *
020     * <p>
021     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
022     * {@link #caseCollectionSchemaField caseCollectionSchemaField()}.
023     * </p>
024     */
025    @Override
026    public R caseSetSchemaField(SetSchemaField field) {
027        return this.caseCollectionSchemaField(field);
028    }
029
030    /**
031     * Handle a {@link ListSchemaField}.
032     *
033     * <p>
034     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
035     * {@link #caseCollectionSchemaField caseCollectionSchemaField()}.
036     * </p>
037     */
038    @Override
039    public R caseListSchemaField(ListSchemaField field) {
040        return this.caseCollectionSchemaField(field);
041    }
042
043    /**
044     * Handle a {@link MapSchemaField}.
045     *
046     * <p>
047     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
048     * {@link #caseComplexSchemaField caseComplexSchemaField()}.
049     * </p>
050     */
051    @Override
052    public R caseMapSchemaField(MapSchemaField field) {
053        return this.caseComplexSchemaField(field);
054    }
055
056    /**
057     * Handle a {@link SimpleSchemaField}.
058     *
059     * <p>
060     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
061     * {@link #caseDefault caseDefault()}.
062     * </p>
063     */
064    @Override
065    public R caseSimpleSchemaField(SimpleSchemaField field) {
066        return this.caseDefault(field);
067    }
068
069    /**
070     * Handle a {@link ReferenceSchemaField}.
071     *
072     * <p>
073     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
074     * {@link #caseSimpleSchemaField caseSimpleSchemaField()}.
075     * </p>
076     */
077    @Override
078    public R caseReferenceSchemaField(ReferenceSchemaField field) {
079        return this.caseSimpleSchemaField(field);
080    }
081
082    /**
083     * Handle a {@link EnumSchemaField}.
084     *
085     * <p>
086     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to
087     * {@link #caseSimpleSchemaField caseSimpleSchemaField()}.
088     * </p>
089     */
090    @Override
091    public R caseEnumSchemaField(EnumSchemaField field) {
092        return this.caseSimpleSchemaField(field);
093    }
094
095    /**
096     * Handle a {@link CounterSchemaField}.
097     *
098     * <p>
099     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseDefault caseDefault()}.
100     * </p>
101     */
102    @Override
103    public R caseCounterSchemaField(CounterSchemaField field) {
104        return this.caseDefault(field);
105    }
106
107    /**
108     * Adapter class roll-up method.
109     *
110     * <p>
111     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseComplexSchemaField caseComplexSchemaField()}.
112     * </p>
113     *
114     * @param field visiting field
115     * @return visitor return value
116     */
117    protected R caseCollectionSchemaField(CollectionSchemaField field) {
118        return this.caseComplexSchemaField(field);
119    }
120
121    /**
122     * Adapter class roll-up method.
123     *
124     * <p>
125     * The implementation in {@link SchemaFieldSwitchAdapter} delegates to {@link #caseDefault caseDefault()}.
126     * </p>
127     *
128     * @param field visiting field
129     * @return visitor return value
130     */
131    protected R caseComplexSchemaField(ComplexSchemaField field) {
132        return this.caseDefault(field);
133    }
134
135    /**
136     * Adapter class roll-up method.
137     *
138     * <p>
139     * The implementation in {@link SchemaFieldSwitchAdapter} always throws {@link UnsupportedOperationException}.
140     * </p>
141     *
142     * @param field visiting field
143     * @return visitor return value
144     */
145    protected R caseDefault(SchemaField field) {
146        throw new UnsupportedOperationException("field type not handled: " + field);
147    }
148}
