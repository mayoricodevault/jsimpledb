008package org.jsimpledb.schema;
009
010import java.util.Iterator;
011import java.util.SortedMap;
012import java.util.TreeMap;
013
014import javax.validation.ConstraintValidatorContext;
015import javax.validation.Valid;
016import javax.validation.constraints.NotNull;
017
018import org.dellroad.stuff.validation.SelfValidates;
019import org.dellroad.stuff.validation.SelfValidating;
020import org.dellroad.stuff.validation.SelfValidationException;
021
022/**
023 * One object type in a {@link SchemaModel}.
024 */
025@SelfValidates
026public class SchemaObject extends AbstractSchemaItem implements SelfValidating {
027
028    private SortedMap<Integer, SchemaField> schemaFields = new TreeMap<>();
029
030    // Overridden for @NotNull annotation
031    @NotNull(message = "must have a name")
032    public String getName() {
033        return super.getName();
034    }
035
036    @NotNull
037    @Valid
038    public SortedMap<Integer, SchemaField> getSchemaFields() {
039        return this.schemaFields;
040    }
041    public void setSchemaFields(SortedMap<Integer, SchemaField> schemaFields) {
042        this.schemaFields = schemaFields;
043    }
044
045// Object
046
047    @Override
048    public String toString() {
049        return "object " + super.toString();
050    }
051
052    @Override
053    public boolean equals(Object obj) {
054        if (obj == this)
055            return true;
056        if (!super.equals(obj))
057            return false;
058        final SchemaObject that = (SchemaObject)obj;
059        return this.schemaFields.equals(that.schemaFields);
060    }
061
062    @Override
063    public int hashCode() {
064        return super.hashCode() ^ this.schemaFields.hashCode();
065    }
066
067// Cloneable
068
069    @Override
070    public SchemaObject clone() {
071        final SchemaObject clone = (SchemaObject)super.clone();
072        clone.schemaFields = new TreeMap<>();
073        for (SchemaField schemaField : this.schemaFields.values())
074            clone.addSchemaField(schemaField.clone());
075        return clone;
076    }
077
078// SelfValidating
079
080    @Override
081    public void checkValid(ConstraintValidatorContext context) throws SelfValidationException {
082        for (SchemaField field : this.getSchemaFields().values()) {
083            if (field.getName() == null)
084                throw new SelfValidationException(field + " must have a name");
085            if (field instanceof ComplexSchemaField) {
086                final ComplexSchemaField complexField = (ComplexSchemaField)field;
087                for (SimpleSchemaField subField : complexField.getSubFields().values()) {
088                    if (subField.getName() != null)
089                        throw new SelfValidationException(subField + " must not specify a name");
090                }
091            }
092        }
093    }
094
095// JiBX
096
097    public void addSchemaField(SchemaField field) {
098        this.schemaFields.put(field.getStorageId(), field);
099    }
100
101    public Iterator<SchemaField> iterateSchemaFields() {
102        return this.schemaFields.values().iterator();
103    }
104}
