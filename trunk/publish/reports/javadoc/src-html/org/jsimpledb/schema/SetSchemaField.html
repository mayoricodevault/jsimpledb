008package org.jsimpledb.schema;
009
010import javax.xml.namespace.QName;
011
012/**
013 * A set field in one version of a {@link SchemaObjectType}.
014 */
015public class SetSchemaField extends CollectionSchemaField {
016
017    @Override
018    public <R> R visit(SchemaFieldSwitch<R> target) {
019        return target.caseSetSchemaField(this);
020    }
021
022    @Override
023    QName getXMLTag() {
024        return SET_FIELD_TAG;
025    }
026
027// Object
028
029    @Override
030    public String toString() {
031        return "set " + super.toString();
032    }
033
034// Cloneable
035
036    @Override
037    public SetSchemaField clone() {
038        return (SetSchemaField)super.clone();
039    }
040}
