008package org.jsimpledb.schema;
009
010import javax.xml.stream.XMLStreamException;
011import javax.xml.stream.XMLStreamWriter;
012
013import org.jsimpledb.core.InvalidSchemaException;
014
015/**
016 * A counter field in a {@link SchemaObject}.
017 */
018public class CounterSchemaField extends SchemaField {
019
020    @Override
021    public void validate() {
022        super.validate();
023        if (this.getName() == null || this.getName().length() == 0)
024            throw new InvalidSchemaException(this + " must specify a name");
025    }
026
027    @Override
028    public <R> R visit(SchemaFieldSwitch<R> target) {
029        return target.caseCounterSchemaField(this);
030    }
031
032    @Override
033    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
034        writer.writeEmptyElement(COUNTER_FIELD_TAG.getNamespaceURI(), COUNTER_FIELD_TAG.getLocalPart());
035        this.writeAttributes(writer);
036    }
037
038// Object
039
040    @Override
041    public String toString() {
042        return "counter " + super.toString();
043    }
044
045// Cloneable
046
047    @Override
048    public CounterSchemaField clone() {
049        return (CounterSchemaField)super.clone();
050    }
051}
