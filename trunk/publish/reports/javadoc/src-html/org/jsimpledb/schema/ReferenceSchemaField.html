008package org.jsimpledb.schema;
009
010import javax.xml.stream.XMLStreamException;
011import javax.xml.stream.XMLStreamReader;
012import javax.xml.stream.XMLStreamWriter;
013
014import org.jsimpledb.core.DeleteAction;
015import org.jsimpledb.core.FieldType;
016import org.jsimpledb.core.InvalidSchemaException;
017
018/**
019 * A reference field in a {@link SchemaObject}.
020 */
021public class ReferenceSchemaField extends SimpleSchemaField {
022
023    private DeleteAction onDelete;
024
025    public ReferenceSchemaField() {
026        this.setType(FieldType.REFERENCE_TYPE_NAME);
027        this.setIndexed(true);
028    }
029
030    /**
031     * Get the desired behavior when an object referred to by this field is deleted.
032     */
033    public DeleteAction getOnDelete() {
034        return this.onDelete;
035    }
036    public void setOnDelete(DeleteAction onDelete) {
037        this.onDelete = onDelete;
038    }
039
040    @Override
041    public void validate() {
042        super.validate();
043        if (!FieldType.REFERENCE_TYPE_NAME.equals(this.getType())) {
044            throw new InvalidSchemaException("invalid " + this + ": reference fields must have type `"
045              + FieldType.REFERENCE_TYPE_NAME + "'");
046        }
047        if (!this.isIndexed())
048            throw new IllegalArgumentException("invalid " + this + ": reference fields must always be indexed");
049        if (this.onDelete == null)
050            throw new InvalidSchemaException("invalid " + this + ": no delete action specified");
051    }
052
053    @Override
054    public <R> R visit(SchemaFieldSwitch<R> target) {
055        return target.caseReferenceSchemaField(this);
056    }
057
058    @Override
059    void readAttributes(XMLStreamReader reader) throws XMLStreamException {
060        super.readAttributes(reader);
061        final String text = reader.getAttributeValue(ON_DELETE_ATTRIBUTE.getNamespaceURI(), ON_DELETE_ATTRIBUTE.getLocalPart());
062        DeleteAction action = DeleteAction.EXCEPTION;
063        if (text != null) {
064            if ((action = Enum.valueOf(DeleteAction.class, text)) == null) {
065                throw new XMLStreamException("invalid value `" + text
066                  + " for \"" + ON_DELETE_ATTRIBUTE.getLocalPart() + "\" attribute in " + this);
067            }
068        }
069        this.setOnDelete(action);
070    }
071
072    @Override
073    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
074        writer.writeEmptyElement(REFERENCE_FIELD_TAG.getNamespaceURI(), REFERENCE_FIELD_TAG.getLocalPart());
075        this.writeAttributes(writer);
076        if (this.onDelete != null)
077            writer.writeAttribute(ON_DELETE_ATTRIBUTE.getNamespaceURI(), ON_DELETE_ATTRIBUTE.getLocalPart(), this.onDelete.name());
078    }
079
080    @Override
081    void writeSimpleAttributes(XMLStreamWriter writer) throws XMLStreamException {
082        // don't need to write type or indexed
083    }
084
085// Object
086
087    @Override
088    public boolean equals(Object obj) {
089        if (obj == this)
090            return true;
091        if (!super.equals(obj))
092            return false;
093        final ReferenceSchemaField that = (ReferenceSchemaField)obj;
094        return this.onDelete == that.onDelete;
095    }
096
097    @Override
098    public int hashCode() {
099        return super.hashCode() ^ (this.onDelete != null ? this.onDelete.hashCode() : 0);
100    }
101
102// Cloneable
103
104    @Override
105    public ReferenceSchemaField clone() {
106        return (ReferenceSchemaField)super.clone();
107    }
108}
