008package org.jsimpledb.schema;
009
010import javax.validation.constraints.NotNull;
011
012import org.jsimpledb.core.DeleteAction;
013import org.jsimpledb.core.ReferenceType;
014
015/**
016 * A reference field in a {@link SchemaObject}.
017 */
018public class ReferenceSchemaField extends SimpleSchemaField {
019
020    private DeleteAction onDelete;
021
022    public ReferenceSchemaField() {
023        this.setType(ReferenceType.NAME);
024        this.setIndexed(true);
025    }
026
027    /**
028     * Get the desired behavior when an object referred to by this field is deleted.
029     */
030    @NotNull(message = "a non-null onDelete value must be specified")
031    public DeleteAction getOnDelete() {
032        return this.onDelete;
033    }
034    public void setOnDelete(DeleteAction onDelete) {
035        this.onDelete = onDelete;
036    }
037
038    @Override
039    public void setType(String type) {
040        if (!ReferenceType.NAME.equals(type))
041            throw new IllegalArgumentException("reference fields always have type `" + ReferenceType.NAME + "'");
042        super.setType(type);
043    }
044
045    @Override
046    public void setIndexed(boolean indexed) {
047        if (!indexed)
048            throw new IllegalArgumentException("reference fields are always indexed");
049        super.setIndexed(indexed);
050    }
051
052    @Override
053    public <R> R visit(SchemaFieldSwitch<R> target) {
054        return target.caseReferenceSchemaField(this);
055    }
056
057// Object
058
059    @Override
060    public boolean equals(Object obj) {
061        if (obj == this)
062            return true;
063        if (!super.equals(obj))
064            return false;
065        final ReferenceSchemaField that = (ReferenceSchemaField)obj;
066        return this.onDelete == that.onDelete;
067    }
068
069    @Override
070    public int hashCode() {
071        return super.hashCode() ^ (this.onDelete != null ? this.onDelete.hashCode() : 0);
072    }
073
074// Cloneable
075
076    @Override
077    public ReferenceSchemaField clone() {
078        return (ReferenceSchemaField)super.clone();
079    }
080}
