008package org.jsimpledb.schema;
009
010import java.util.HashMap;
011import java.util.List;
012import java.util.Map;
013import java.util.SortedMap;
014import java.util.TreeMap;
015
016import javax.xml.stream.XMLStreamException;
017import javax.xml.stream.XMLStreamReader;
018import javax.xml.stream.XMLStreamWriter;
019
020import org.jsimpledb.core.InvalidSchemaException;
021
022/**
023 * One object type in a {@link SchemaModel}.
024 */
025public class SchemaObjectType extends AbstractSchemaItem {
026
027    private /*final*/ TreeMap<Integer, SchemaField> schemaFields = new TreeMap<>();
028    private /*final*/ TreeMap<Integer, SchemaCompositeIndex> schemaCompositeIndexes = new TreeMap<>();
029
030    /**
031     * Get this object type's {@link SchemaField}s, indexed by storage ID.
032     */
033    public SortedMap<Integer, SchemaField> getSchemaFields() {
034        return this.schemaFields;
035    }
036
037    /**
038     * Get the {@link SchemaCompositeIndex}s defined on this object type's fields, indexed by storage ID.
039     */
040    public SortedMap<Integer, SchemaCompositeIndex> getSchemaCompositeIndexes() {
041        return this.schemaCompositeIndexes;
042    }
043
044    @Override
045    void validate() {
046        super.validate();
047
048        // Validate fields and verify field names are unique
049        final TreeMap<String, SchemaField> fieldsByName = new TreeMap<>();
050        for (SchemaField field : this.schemaFields.values()) {
051            field.validate();
052            final String fieldName = field.getName();
053            if (fieldsByName.put(fieldName, field) != null)
054                throw new InvalidSchemaException("duplicate field name `" + fieldName + "'");
055        }
056
057        // Validate composite indexes and verify index names are unique
058        final TreeMap<String, SchemaCompositeIndex> compositeIndexesByName = new TreeMap<>();
059        for (SchemaCompositeIndex index : this.schemaCompositeIndexes.values()) {
060            index.validate();
061            final String indexName = index.getName();
062            if (compositeIndexesByName.put(indexName, index) != null)
063                throw new InvalidSchemaException("duplicate composite index name `" + indexName + "'");
064        }
065
066        // Verify indexes index valid simple fields that are not sub-fields of complex fields
067        for (SchemaCompositeIndex index : this.schemaCompositeIndexes.values()) {
068            for (int storageId : index.getIndexedFields()) {
069                final SchemaField field = this.schemaFields.get(storageId);
070                if (!(field instanceof SimpleSchemaField))
071                    throw new InvalidSchemaException(index + " indexes unknown or invalid field with storage ID " + storageId);
072            }
073        }
074
075        // Verify there are no duplicate composite indexes
076        final HashMap<List<Integer>, SchemaCompositeIndex> compositeIndexFields = new HashMap<>();
077        for (SchemaCompositeIndex index : this.schemaCompositeIndexes.values()) {
078            final SchemaCompositeIndex previous = compositeIndexFields.put(index.getIndexedFields(), index);
079            if (previous != null)
080                throw new InvalidSchemaException("duplicate " + index + " (duplicates " + previous + ")");
081        }
082    }
083
084    @Override
085    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
086        this.schemaFields.clear();
087        this.schemaCompositeIndexes.clear();
088        boolean seenIndex = false;
089        for (AbstractSchemaItem item;
090          (item = this.readMappedType(reader, true, SchemaModel.FIELD_OR_COMPOSITE_INDEX_TAG_MAP)) != null; ) {
091            if (item instanceof SchemaField) {
092                if (seenIndex)
093                    throw new XMLStreamException("indexes must appear after fields", reader.getLocation());
094                final SchemaField field = (SchemaField)item;
095                field.readXML(reader, formatVersion);
096                final int storageId = field.getStorageId();
097                final SchemaField previous = this.schemaFields.put(storageId, field);
098                if (previous != null) {
099                    throw new XMLStreamException("duplicate use of storage ID " + storageId
100                      + " for both " + previous + " and " + field + " in " + this, reader.getLocation());
101                }
102            } else if (item instanceof SchemaCompositeIndex) {
103                final SchemaCompositeIndex index = (SchemaCompositeIndex)item;
104                index.readXML(reader, formatVersion);
105                final int storageId = index.getStorageId();
106                final SchemaCompositeIndex previous = this.schemaCompositeIndexes.put(storageId, index);
107                if (previous != null) {
108                    throw new XMLStreamException("duplicate use of storage ID " + storageId
109                      + " for both " + previous + " and " + index + " in " + this, reader.getLocation());
110                }
111                seenIndex = true;
112            } else
113                throw new RuntimeException("internal error");
114        }
115    }
116
117    @Override
118    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
119        if (this.schemaFields.isEmpty() && this.schemaCompositeIndexes.isEmpty()) {
120            writer.writeEmptyElement(OBJECT_TYPE_TAG.getNamespaceURI(), OBJECT_TYPE_TAG.getLocalPart());
121            this.writeAttributes(writer);
122            return;
123        }
124        writer.writeStartElement(OBJECT_TYPE_TAG.getNamespaceURI(), OBJECT_TYPE_TAG.getLocalPart());
125        this.writeAttributes(writer);
126        for (SchemaField schemaField : this.schemaFields.values())
127            schemaField.writeXML(writer);
128        for (SchemaCompositeIndex schemaCompositeIndex : this.schemaCompositeIndexes.values())
129            schemaCompositeIndex.writeXML(writer);
130        writer.writeEndElement();
131    }
132
133    @Override
134    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
135        final SchemaObjectType that = (SchemaObjectType)that0;
136        if (!this.schemaFields.keySet().equals(that.schemaFields.keySet()))
137            return false;
138        for (int storageId : this.schemaFields.keySet()) {
139            final SchemaField thisSchemaField = this.schemaFields.get(storageId);
140            final SchemaField thatSchemaField = that.schemaFields.get(storageId);
141            if (!thisSchemaField.isCompatibleWith(thatSchemaField))
142                return false;
143        }
144        if (!this.schemaCompositeIndexes.keySet().equals(that.schemaCompositeIndexes.keySet()))
145            return false;
146        for (int storageId : this.schemaCompositeIndexes.keySet()) {
147            final SchemaCompositeIndex thisSchemaIndex = this.schemaCompositeIndexes.get(storageId);
148            final SchemaCompositeIndex thatSchemaIndex = that.schemaCompositeIndexes.get(storageId);
149            if (!thisSchemaIndex.isCompatibleWith(thatSchemaIndex))
150                return false;
151        }
152        return true;
153    }
154
155// Object
156
157    @Override
158    public String toString() {
159        return "object " + super.toString();
160    }
161
162    @Override
163    public boolean equals(Object obj) {
164        if (obj == this)
165            return true;
166        if (!super.equals(obj))
167            return false;
168        final SchemaObjectType that = (SchemaObjectType)obj;
169        return this.schemaFields.equals(that.schemaFields) && this.schemaCompositeIndexes.equals(that.schemaCompositeIndexes);
170    }
171
172    @Override
173    public int hashCode() {
174        return super.hashCode() ^ this.schemaFields.hashCode() ^ this.schemaCompositeIndexes.hashCode();
175    }
176
177// Cloneable
178
179    @Override
180    @SuppressWarnings("unchecked")
181    public SchemaObjectType clone() {
182        final SchemaObjectType clone = (SchemaObjectType)super.clone();
183        clone.schemaFields = (TreeMap<Integer, SchemaField>)clone.schemaFields.clone();
184        for (Map.Entry<Integer, SchemaField> entry : clone.schemaFields.entrySet())
185            entry.setValue(entry.getValue().clone());
186        clone.schemaCompositeIndexes = (TreeMap<Integer, SchemaCompositeIndex>)clone.schemaCompositeIndexes.clone();
187        for (Map.Entry<Integer, SchemaCompositeIndex> entry : clone.schemaCompositeIndexes.entrySet())
188            entry.setValue(entry.getValue().clone());
189        return clone;
190    }
191}
