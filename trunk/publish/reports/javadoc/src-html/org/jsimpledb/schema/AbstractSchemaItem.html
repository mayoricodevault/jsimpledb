008package org.jsimpledb.schema;
009
010import javax.xml.stream.XMLStreamException;
011import javax.xml.stream.XMLStreamReader;
012import javax.xml.stream.XMLStreamWriter;
013
014import org.jsimpledb.core.InvalidSchemaException;
015import org.jsimpledb.util.AbstractXMLStreaming;
016
017/**
018 * Common superclass for {@link SchemaObject} and {@link SchemaField}.
019 */
020public abstract class AbstractSchemaItem extends AbstractXMLStreaming implements XMLConstants, Cloneable {
021
022    private String name;
023    private int storageId;
024
025    /**
026     * Get the name associated with this instance, if any.
027     */
028    public String getName() {
029        return this.name;
030    }
031    public void setName(String name) {
032        this.name = name;
033    }
034
035    /**
036     * Get the storage ID associated with this instance.
037     * Storage IDs must be positive values.
038     */
039    public int getStorageId() {
040        return this.storageId;
041    }
042    public void setStorageId(int storageId) {
043        this.storageId = storageId;
044    }
045
046    /**
047     * Validate this instance.
048     *
049     * @throws InvalidSchemaException if this instance in invalid
050     */
051    public void validate() {
052        if (this.storageId <= 0)
053            throw new InvalidSchemaException(this + " has an invalid storage ID; must be greater than zero");
054    }
055
056    void readXML(XMLStreamReader reader) throws XMLStreamException {
057        this.readAttributes(reader);
058        this.readSubElements(reader);
059    }
060
061    void readAttributes(XMLStreamReader reader) throws XMLStreamException {
062        final String text = reader.getAttributeValue(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart());
063        if (text != null) {
064            try {
065                this.setStorageId(Integer.valueOf(text));
066            } catch (NumberFormatException e) {
067                throw new XMLStreamException("invalid storage ID `" + text + "': must be a positive integer", reader.getLocation());
068            }
069        }
070        final String newName = reader.getAttributeValue(NAME_ATTRIBUTE.getNamespaceURI(), NAME_ATTRIBUTE.getLocalPart());
071        if (newName != null)
072            this.setName(newName);
073    }
074
075    void readSubElements(XMLStreamReader reader) throws XMLStreamException {
076        this.expect(reader, true);
077    }
078
079    abstract void writeXML(XMLStreamWriter writer) throws XMLStreamException;
080
081    void writeAttributes(XMLStreamWriter writer) throws XMLStreamException {
082        writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + this.storageId);
083        if (this.name != null)
084            writer.writeAttribute(NAME_ATTRIBUTE.getNamespaceURI(), NAME_ATTRIBUTE.getLocalPart(), this.name);
085    }
086
087// Object
088
089    @Override
090    public String toString() {
091        return "#" + this.storageId + (this.name != null ? " `" + this.name + "'" : "");
092    }
093
094    @Override
095    public boolean equals(Object obj) {
096        if (obj == this)
097            return true;
098        if (obj == null || obj.getClass() != this.getClass())
099            return false;
100        final AbstractSchemaItem that = (AbstractSchemaItem)obj;
101        return (this.name != null ? this.name.equals(that.name) : that.name == null) && this.storageId == that.storageId;
102    }
103
104    @Override
105    public int hashCode() {
106        return (this.name != null ? this.name.hashCode() : 0) ^ this.storageId;
107    }
108
109// Cloneable
110
111    /**
112     * Deep-clone this instance.
113     */
114    @Override
115    public AbstractSchemaItem clone() {
116        try {
117            return (AbstractSchemaItem)super.clone();
118        } catch (CloneNotSupportedException e) {
119            throw new RuntimeException(e);
120        }
121    }
122}
