008package org.jsimpledb.schema;
009
010import java.util.Comparator;
011import java.util.Map;
012
013import javax.xml.namespace.QName;
014import javax.xml.stream.XMLStreamException;
015import javax.xml.stream.XMLStreamReader;
016import javax.xml.stream.XMLStreamWriter;
017
018import org.jsimpledb.core.InvalidSchemaException;
019import org.jsimpledb.core.SchemaItem;
020import org.jsimpledb.util.AbstractXMLStreaming;
021import org.jsimpledb.util.Diffs;
022
023/**
024 * Common superclass for {@link SchemaObjectType} and {@link SchemaField}.
025 */
026public abstract class AbstractSchemaItem extends AbstractXMLStreaming implements XMLConstants, Cloneable {
027
028    private String name;
029    private int storageId;
030
031    /**
032     * Get the name associated with this instance, if any.
033     *
034     * @return the name of this instance, or null if it has none
035     */
036    public String getName() {
037        return this.name;
038    }
039    public void setName(String name) {
040        this.name = name;
041    }
042
043    /**
044     * Get the storage ID associated with this instance.
045     * Storage IDs must be positive values.
046     *
047     * @return the storage ID for this instance
048     */
049    public int getStorageId() {
050        return this.storageId;
051    }
052    public void setStorageId(int storageId) {
053        this.storageId = storageId;
054    }
055
056    void validate() {
057        if (name == null)
058            throw new InvalidSchemaException(this + " must specify a name");
059        if (!name.matches(SchemaItem.NAME_PATTERN))
060            throw new InvalidSchemaException(this + " has an invalid name `" + name + "'");
061        if (this.storageId <= 0)
062            throw new InvalidSchemaException(this + " has an invalid storage ID " + this.storageId + "; must be greater than zero");
063    }
064
065    /**
066     * Determine whether this instance is compatible with the given instance for use with the core API.
067     * Two instances are compatible if they are identical in all respects except for object and field names
068     * (to also include object and field names in the comparison, use {@link #equals equals()}).
069     * The core API uses storage IDs, not names, to identify objects and fields.
070     *
071     * @param that other schema object
072     * @return true if this and {@code that} are compatible
073     * @throws IllegalArgumentException if {@code that} is null
074     */
075    public final boolean isCompatibleWith(AbstractSchemaItem that) {
076        if (that == null)
077            throw new IllegalArgumentException("null that");
078        if (this.storageId != that.storageId)
079            return false;
080        if (this.getClass() != that.getClass())
081            return false;
082        return this.isCompatibleWithInternal(that);
083    }
084
085    abstract boolean isCompatibleWithInternal(AbstractSchemaItem that);
086
087// DiffGenerating
088
089    protected Diffs differencesFrom(AbstractSchemaItem that) {
090        if (that == null)
091            throw new IllegalArgumentException("null that");
092        final Diffs diffs = new Diffs();
093        if (!(this.name != null ? this.name.equals(that.name) : that.name == null)) {
094            diffs.add("changed name from " + (that.name != null ? "`" + that.name + "'" : null)
095              + " to " + (this.name != null ? "`" + this.name + "'" : null));
096        }
097        if (this.storageId != that.storageId)
098            diffs.add("changed storage ID from " + that.storageId + " to " + this.storageId);
099        return diffs;
100    }
101
102// XML Reading
103
104    /**
105     * Read in this item's XML.
106     *
107     * <p>
108     * The implementation in {@link AbstractSchemaItem} invokes {@link #readAttributes readAttributes()}
109     * followed by {@link #readSubElements readSubElements()}.
110     * </p>
111     *
112     * <p>
113     * Start state: positioned at opening XML tag.
114     * Return state: positioned at closing XML tag.
115     * </p>
116     */
117    void readXML(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
118        this.readAttributes(reader, formatVersion);
119        this.readSubElements(reader, formatVersion);
120    }
121
122    /**
123     * Read in this item's start tag attributes.
124     *
125     * <p>
126     * The implementation in {@link AbstractSchemaItem} reads in required storage ID and name attributes.
127     * </p>
128     *
129     * <p>
130     * Start state: positioned at opening XML tag.
131     * Return state: same.
132     * </p>
133     */
134    void readAttributes(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
135        final Integer storageIdAttr = this.getIntAttr(reader, STORAGE_ID_ATTRIBUTE, false);
136        if (storageIdAttr != null)
137            this.setStorageId(storageIdAttr);
138        final String nameAttr = this.getAttr(reader, NAME_ATTRIBUTE, false);
139        if (nameAttr != null)
140            this.setName(nameAttr);
141    }
142
143    /**
144     * Read in this item's sub-elements.
145     *
146     * <p>
147     * The implementation in {@link AbstractSchemaItem} expects no sub-elements.
148     * </p>
149     *
150     * <p>
151     * Start state: positioned at opening XML tag.
152     * Return state: positioned at closing XML tag.
153     * </p>
154     */
155    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
156        this.expectClose(reader);
157    }
158
159    /**
160     * Read an element found in the given map.
161     *
162     * @return element found, or null if closing XML tag encountered instead
163     */
164    <T> T readMappedType(XMLStreamReader reader, boolean closingOK, Map<QName, Class<? extends T>> tagMap)
165      throws XMLStreamException {
166
167        // Expect to see one of the map's XML tag keys
168        if (!this.expect(reader, closingOK, tagMap.keySet().toArray(new QName[tagMap.size()])))
169            return null;
170
171        // Instantiate the corresponding type
172        T obj = null;
173        for (Map.Entry<QName, Class<? extends T>> entry : tagMap.entrySet()) {
174            if (reader.getName().equals(entry.getKey())) {
175                try {
176                    return entry.getValue().newInstance();
177                } catch (InstantiationException | IllegalAccessException e) {
178                    throw new RuntimeException("unexpected exception", e);
179                }
180            }
181        }
182        throw new RuntimeException("internal error: didn't find " + reader.getName() + " in tagMap");
183    }
184
185// XML Writing
186
187    abstract void writeXML(XMLStreamWriter writer) throws XMLStreamException;
188
189    final void writeAttributes(XMLStreamWriter writer) throws XMLStreamException {
190        this.writeAttributes(writer, true);
191    }
192
193    void writeAttributes(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
194        writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + this.storageId);
195        if (includeName && this.name != null)
196            writer.writeAttribute(NAME_ATTRIBUTE.getNamespaceURI(), NAME_ATTRIBUTE.getLocalPart(), this.name);
197    }
198
199// Object
200
201    @Override
202    public String toString() {
203        return "#" + this.storageId + (this.name != null ? " `" + this.name + "'" : "");
204    }
205
206    @Override
207    public boolean equals(Object obj) {
208        if (obj == this)
209            return true;
210        if (obj == null || obj.getClass() != this.getClass())
211            return false;
212        final AbstractSchemaItem that = (AbstractSchemaItem)obj;
213        return (this.name != null ? this.name.equals(that.name) : that.name == null) && this.storageId == that.storageId;
214    }
215
216    @Override
217    public int hashCode() {
218        return (this.name != null ? this.name.hashCode() : 0) ^ this.storageId;
219    }
220
221// Cloneable
222
223    /**
224     * Deep-clone this instance.
225     */
226    @Override
227    public AbstractSchemaItem clone() {
228        try {
229            return (AbstractSchemaItem)super.clone();
230        } catch (CloneNotSupportedException e) {
231            throw new RuntimeException(e);
232        }
233    }
234
235// NameComparator
236
237    static class NameComparator implements Comparator<AbstractSchemaItem> {
238
239        @Override
240        public int compare(AbstractSchemaItem item1, AbstractSchemaItem item2) {
241            return item1.getName().compareTo(item2.getName());
242        }
243    }
244}
