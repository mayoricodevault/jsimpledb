008package org.jsimpledb.schema;
009
010import java.util.Map;
011
012import javax.xml.namespace.QName;
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.XMLStreamReader;
015import javax.xml.stream.XMLStreamWriter;
016
017import org.jsimpledb.core.InvalidSchemaException;
018import org.jsimpledb.core.SchemaItem;
019import org.jsimpledb.util.AbstractXMLStreaming;
020
021/**
022 * Common superclass for {@link SchemaObjectType} and {@link SchemaField}.
023 */
024public abstract class AbstractSchemaItem extends AbstractXMLStreaming implements XMLConstants, Cloneable {
025
026    private String name;
027    private int storageId;
028
029    /**
030     * Get the name associated with this instance, if any.
031     */
032    public String getName() {
033        return this.name;
034    }
035    public void setName(String name) {
036        this.name = name;
037    }
038
039    /**
040     * Get the storage ID associated with this instance.
041     * Storage IDs must be positive values.
042     */
043    public int getStorageId() {
044        return this.storageId;
045    }
046    public void setStorageId(int storageId) {
047        this.storageId = storageId;
048    }
049
050    void validate() {
051        if (name == null)
052            throw new InvalidSchemaException(this + " must specify a name");
053        if (!name.matches(SchemaItem.NAME_PATTERN))
054            throw new InvalidSchemaException(this + " has an invalid name `" + name + "'");
055        if (this.storageId <= 0)
056            throw new InvalidSchemaException(this + " has an invalid storage ID " + this.storageId + "; must be greater than zero");
057    }
058
059    /**
060     * Determine whether this instance is compatible with the given instance for use with the core API.
061     * Two instances are compatible if they are identical in all respects except for object and field names
062     * (to also include object and field names in the comparison, use {@link #equals equals()}).
063     * The core API uses storage IDs, not names, to identify objects and fields.
064     *
065     * @param that other schema object
066     * @throws IllegalArgumentException if {@code that} is null
067     */
068    public final boolean isCompatibleWith(AbstractSchemaItem that) {
069        if (that == null)
070            throw new IllegalArgumentException("null that");
071        if (this.storageId != that.storageId)
072            return false;
073        if (this.getClass() != that.getClass())
074            return false;
075        return this.isCompatibleWithInternal(that);
076    }
077
078    abstract boolean isCompatibleWithInternal(AbstractSchemaItem that);
079
080// XML Reading
081
082    /**
083     * Read in this item's XML.
084     *
085     * <p>
086     * The implementation in {@link AbstractSchemaItem} invokes {@link #readAttributes readAttributes()}
087     * followed by {@link #readSubElements readSubElements()}.
088     * </p>
089     *
090     * <p>
091     * Start state: positioned at opening XML tag.
092     * Return state: positioned at closing XML tag.
093     * </p>
094     */
095    void readXML(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
096        this.readAttributes(reader, formatVersion);
097        this.readSubElements(reader, formatVersion);
098    }
099
100    /**
101     * Read in this item's start tag attributes.
102     *
103     * <p>
104     * The implementation in {@link AbstractSchemaItem} reads in required storage ID and name attributes.
105     * </p>
106     *
107     * <p>
108     * Start state: positioned at opening XML tag.
109     * Return state: same.
110     * </p>
111     */
112    void readAttributes(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
113        final Integer storageIdAttr = this.getIntAttr(reader, STORAGE_ID_ATTRIBUTE, false);
114        if (storageIdAttr != null)
115            this.setStorageId(storageIdAttr);
116        final String nameAttr = this.getAttr(reader, NAME_ATTRIBUTE, false);
117        if (nameAttr != null)
118            this.setName(nameAttr);
119    }
120
121    /**
122     * Read in this item's sub-elements.
123     *
124     * <p>
125     * The implementation in {@link AbstractSchemaItem} expects no sub-elements.
126     * </p>
127     *
128     * <p>
129     * Start state: positioned at opening XML tag.
130     * Return state: positioned at closing XML tag.
131     * </p>
132     */
133    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
134        this.expectClose(reader);
135    }
136
137    /**
138     * Read an element found in the given map.
139     *
140     * @return element found, or null if closing XML tag encountered instead
141     */
142    <T> T readMappedType(XMLStreamReader reader, boolean closingOK, Map<QName, Class<? extends T>> tagMap)
143      throws XMLStreamException {
144
145        // Expect to see one of the map's XML tag keys
146        if (!this.expect(reader, closingOK, tagMap.keySet().toArray(new QName[tagMap.size()])))
147            return null;
148
149        // Instantiate the corresponding type
150        T obj = null;
151        for (Map.Entry<QName, Class<? extends T>> entry : tagMap.entrySet()) {
152            if (reader.getName().equals(entry.getKey())) {
153                try {
154                    return entry.getValue().newInstance();
155                } catch (InstantiationException | IllegalAccessException e) {
156                    throw new RuntimeException("unexpected exception", e);
157                }
158            }
159        }
160        throw new RuntimeException("internal error: didn't find " + reader.getName() + " in tagMap");
161    }
162
163// XML Writing
164
165    abstract void writeXML(XMLStreamWriter writer) throws XMLStreamException;
166
167    final void writeAttributes(XMLStreamWriter writer) throws XMLStreamException {
168        this.writeAttributes(writer, true);
169    }
170
171    void writeAttributes(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
172        writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + this.storageId);
173        if (includeName && this.name != null)
174            writer.writeAttribute(NAME_ATTRIBUTE.getNamespaceURI(), NAME_ATTRIBUTE.getLocalPart(), this.name);
175    }
176
177// Object
178
179    @Override
180    public String toString() {
181        return "#" + this.storageId + (this.name != null ? " `" + this.name + "'" : "");
182    }
183
184    @Override
185    public boolean equals(Object obj) {
186        if (obj == this)
187            return true;
188        if (obj == null || obj.getClass() != this.getClass())
189            return false;
190        final AbstractSchemaItem that = (AbstractSchemaItem)obj;
191        return (this.name != null ? this.name.equals(that.name) : that.name == null) && this.storageId == that.storageId;
192    }
193
194    @Override
195    public int hashCode() {
196        return (this.name != null ? this.name.hashCode() : 0) ^ this.storageId;
197    }
198
199// Cloneable
200
201    /**
202     * Deep-clone this instance.
203     */
204    @Override
205    public AbstractSchemaItem clone() {
206        try {
207            return (AbstractSchemaItem)super.clone();
208        } catch (CloneNotSupportedException e) {
209            throw new RuntimeException(e);
210        }
211    }
212}
