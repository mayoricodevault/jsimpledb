008package org.jsimpledb.schema;
009
010import java.util.ArrayList;
011import java.util.HashSet;
012import java.util.List;
013
014import javax.xml.stream.XMLStreamException;
015import javax.xml.stream.XMLStreamReader;
016import javax.xml.stream.XMLStreamWriter;
017
018import org.jsimpledb.core.Database;
019import org.jsimpledb.core.InvalidSchemaException;
020
021/**
022 * A composite index associated with a {@link SchemaObjectType}.
023 */
024public class SchemaCompositeIndex extends AbstractSchemaItem {
025
026    private /*final*/ ArrayList<Integer> indexedFields = new ArrayList<>();
027
028    /**
029     * Get the fields that comprise this index.
030     *
031     * @return storage IDs of indexed fields
032     */
033    public List<Integer> getIndexedFields() {
034        return this.indexedFields;
035    }
036
037    @Override
038    void validate() {
039        super.validate();
040        if (this.indexedFields.size() < 2 || this.indexedFields.size() > Database.MAX_INDEXED_FIELDS) {
041            throw new InvalidSchemaException("invalid " + this + ": between 2 and "
042              + Database.MAX_INDEXED_FIELDS + " fields must be specified");
043        }
044        final HashSet<Integer> idsSeen = new HashSet<Integer>();
045        for (int i = 0; i < this.indexedFields.size(); i++) {
046            final int storageId = this.indexedFields.get(i);
047            if (!idsSeen.add(storageId))
048                throw new InvalidSchemaException("invalid " + this + ": duplicate field in composite index: " + storageId);
049        }
050    }
051
052    @Override
053    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
054        final SchemaCompositeIndex that = (SchemaCompositeIndex)that0;
055        if (!this.indexedFields.equals(that.indexedFields))
056            return false;
057        return true;
058    }
059
060// XML Reading
061
062    @Override
063    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
064        this.indexedFields.clear();
065        while (this.expect(reader, true, INDEXED_FIELD_TAG)) {
066            this.indexedFields.add(this.getIntAttr(reader, STORAGE_ID_ATTRIBUTE));
067            this.expectClose(reader);   // </IndexedField>
068        }
069        this.indexedFields.trimToSize();
070    }
071
072// XML Writing
073
074    @Override
075    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
076        writer.writeStartElement(COMPOSITE_INDEX_TAG.getNamespaceURI(), COMPOSITE_INDEX_TAG.getLocalPart());
077        this.writeAttributes(writer, true);
078        for (int storageId : this.indexedFields) {
079            writer.writeEmptyElement(INDEXED_FIELD_TAG.getNamespaceURI(), INDEXED_FIELD_TAG.getLocalPart());
080            writer.writeAttribute(STORAGE_ID_ATTRIBUTE.getNamespaceURI(), STORAGE_ID_ATTRIBUTE.getLocalPart(), "" + storageId);
081        }
082        writer.writeEndElement();           // </CompositeIndex>
083    }
084
085// Object
086
087    @Override
088    public String toString() {
089        return "composite index " + super.toString();
090    }
091
092    @Override
093    public boolean equals(Object obj) {
094        if (obj == this)
095            return true;
096        if (!super.equals(obj))
097            return false;
098        final SchemaCompositeIndex that = (SchemaCompositeIndex)obj;
099        return this.indexedFields.equals(that.indexedFields);
100    }
101
102    @Override
103    public int hashCode() {
104        return super.hashCode() ^ this.indexedFields.hashCode();
105    }
106
107// Cloneable
108
109    @Override
110    @SuppressWarnings("unchecked")
111    public SchemaCompositeIndex clone() {
112        final SchemaCompositeIndex clone = (SchemaCompositeIndex)super.clone();
113        clone.indexedFields = (ArrayList<Integer>)clone.indexedFields.clone();
114        return clone;
115    }
116}
