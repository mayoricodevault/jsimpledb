008package org.jsimpledb.schema;
009
010import javax.xml.stream.XMLStreamException;
011import javax.xml.stream.XMLStreamReader;
012import javax.xml.stream.XMLStreamWriter;
013
014import org.jsimpledb.core.InvalidSchemaException;
015
016/**
017 * A simple field in a {@link SchemaObject}.
018 */
019public class SimpleSchemaField extends SchemaField {
020
021    private String type;
022    private boolean indexed;
023
024    /**
025     * Get the name of this field's type. For example {@code int} for primitive integer type.
026     */
027    public String getType() {
028        return this.type;
029    }
030    public void setType(String type) {
031        this.type = type;
032    }
033
034    /**
035     * Get whether this field is indexed or not.
036     */
037    public boolean isIndexed() {
038        return this.indexed;
039    }
040    public void setIndexed(boolean indexed) {
041        this.indexed = indexed;
042    }
043
044    @Override
045    public void validate() {
046        super.validate();
047        if (this.type == null)
048            throw new InvalidSchemaException("invalid " + this + ": no type specified");
049    }
050
051    @Override
052    public <R> R visit(SchemaFieldSwitch<R> target) {
053        return target.caseSimpleSchemaField(this);
054    }
055
056    @Override
057    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
058        final SimpleSchemaField that = (SimpleSchemaField)that0;
059        if (!this.type.equals(that.type))
060            return false;
061        if (this.indexed != that.indexed)
062            return false;
063        return true;
064    }
065
066    @Override
067    void readAttributes(XMLStreamReader reader) throws XMLStreamException {
068        super.readAttributes(reader);
069        final String text1 = reader.getAttributeValue(TYPE_ATTRIBUTE.getNamespaceURI(), TYPE_ATTRIBUTE.getLocalPart());
070        if (text1 != null)
071            this.setType(text1);
072        final String text2 = reader.getAttributeValue(INDEXED_ATTRIBUTE.getNamespaceURI(), INDEXED_ATTRIBUTE.getLocalPart());
073        if (text2 != null) {
074            switch (text2) {
075            case "true":
076            case "false":
077                this.setIndexed(Boolean.valueOf(text2));
078                break;
079            default:
080                throw new XMLStreamException("invalid boolean value `" + text2
081                  + " for \"" + INDEXED_ATTRIBUTE.getLocalPart() + "\" attribute in " + this);
082            }
083        }
084    }
085
086    @Override
087    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
088        writer.writeEmptyElement(SIMPLE_FIELD_TAG.getNamespaceURI(), SIMPLE_FIELD_TAG.getLocalPart());
089        this.writeAttributes(writer);
090    }
091
092    @Override
093    void writeAttributes(XMLStreamWriter writer) throws XMLStreamException {
094        super.writeAttributes(writer);
095        this.writeSimpleAttributes(writer);
096    }
097
098    void writeSimpleAttributes(XMLStreamWriter writer) throws XMLStreamException {
099        if (this.type != null)
100            writer.writeAttribute(TYPE_ATTRIBUTE.getNamespaceURI(), TYPE_ATTRIBUTE.getLocalPart(), this.type);
101        if (this.indexed)
102            writer.writeAttribute(INDEXED_ATTRIBUTE.getNamespaceURI(), INDEXED_ATTRIBUTE.getLocalPart(), "" + this.indexed);
103    }
104
105// Object
106
107    @Override
108    public String toString() {
109        return super.toString() + (this.type != null ? " of type " + this.type : "");
110    }
111
112    @Override
113    public boolean equals(Object obj) {
114        if (obj == this)
115            return true;
116        if (!super.equals(obj))
117            return false;
118        final SimpleSchemaField that = (SimpleSchemaField)obj;
119        return (this.type != null ? this.type.equals(that.type) : that.type == null) && this.indexed == that.indexed;
120    }
121
122    @Override
123    public int hashCode() {
124        return (this.type != null ? this.type.hashCode() : 0) ^ (this.indexed ? 1 : 0);
125    }
126
127// Cloneable
128
129    @Override
130    public SimpleSchemaField clone() {
131        return (SimpleSchemaField)super.clone();
132    }
133}
