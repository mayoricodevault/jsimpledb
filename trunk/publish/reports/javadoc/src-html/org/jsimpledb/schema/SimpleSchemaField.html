008package org.jsimpledb.schema;
009
010import java.util.regex.Pattern;
011
012import javax.xml.stream.XMLStreamException;
013import javax.xml.stream.XMLStreamReader;
014import javax.xml.stream.XMLStreamWriter;
015
016import org.jsimpledb.core.FieldType;
017import org.jsimpledb.core.InvalidSchemaException;
018import org.jsimpledb.util.DiffGenerating;
019import org.jsimpledb.util.Diffs;
020
021/**
022 * A simple field in a {@link SchemaObjectType}.
023 */
024public class SimpleSchemaField extends SchemaField implements DiffGenerating<SimpleSchemaField> {
025
026    private String type;
027    private long encodingSignature;
028    private boolean indexed;
029
030    /**
031     * Get the name of this field's type. For example {@code "int"} for primitive integer type,
032     * {@code "java.util.Date"} for the built-in {@link java.util.Date} type, any custom type name, etc.
033     *
034     * @return field type name
035     */
036    public String getType() {
037        return this.type;
038    }
039    public void setType(String type) {
040        this.type = type;
041    }
042
043    /**
044     * Get whether this field is indexed or not.
045     *
046     * @return true if this field is indexed
047     */
048    public boolean isIndexed() {
049        return this.indexed;
050    }
051    public void setIndexed(boolean indexed) {
052        this.indexed = indexed;
053    }
054
055    /**
056     * Get the encoding signature associated with this field's type.
057     *
058     * @return this field's encoding signature
059     * @see org.jsimpledb.core.FieldType
060     */
061    public long getEncodingSignature() {
062        return this.encodingSignature;
063    }
064    public void setEncodingSignature(long encodingSignature) {
065        this.encodingSignature = encodingSignature;
066    }
067
068    @Override
069    void validate() {
070        super.validate();
071        if (this.type == null)
072            throw new InvalidSchemaException("invalid " + this + ": no type specified");
073        if (!Pattern.compile(FieldType.NAME_PATTERN).matcher(this.type).matches()) {
074            throw new InvalidSchemaException("invalid " + super.toString() + " type `" + this.type
075              + "': does not match pattern \"" + FieldType.NAME_PATTERN + "\"");
076        }
077    }
078
079    @Override
080    public <R> R visit(SchemaFieldSwitch<R> target) {
081        return target.caseSimpleSchemaField(this);
082    }
083
084    @Override
085    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
086        final SimpleSchemaField that = (SimpleSchemaField)that0;
087        if (!this.type.equals(that.type))
088            return false;
089        if (this.encodingSignature != that.encodingSignature)
090            return false;
091        if (this.indexed != that.indexed)
092            return false;
093        return true;
094    }
095
096// DiffGenerating
097
098    @Override
099    public Diffs differencesFrom(SimpleSchemaField that) {
100        final Diffs diffs = new Diffs(super.differencesFrom(that));
101        if (!(this.type != null ? this.type.equals(that.type) : that.type == null))
102            diffs.add("changed field type from `" + that.type + "' to `" + this.type + "'");
103        if (this.encodingSignature != that.encodingSignature)
104            diffs.add("changed field type encoding signature from " + that.encodingSignature + " to " + this.encodingSignature);
105        if (this.indexed != that.indexed)
106            diffs.add((this.indexed ? "added" : "removed") + " index on field");
107        return diffs;
108    }
109
110// XML Reading
111
112    @Override
113    void readAttributes(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
114        super.readAttributes(reader, formatVersion);
115        final String typeAttr = this.getAttr(reader, TYPE_ATTRIBUTE, false);
116        if (typeAttr != null)
117            this.setType(typeAttr);
118        final Boolean indexedAttr = this.getBooleanAttr(reader, INDEXED_ATTRIBUTE, false);
119        if (indexedAttr != null)
120            this.setIndexed(indexedAttr);
121        final Long encodingSignatureAttr = this.getLongAttr(reader, ENCODING_SIGNATURE_ATTRIBUTE, false);
122        if (encodingSignatureAttr != null)
123            this.setEncodingSignature(encodingSignatureAttr);
124    }
125
126    @Override
127    final void writeXML(XMLStreamWriter writer) throws XMLStreamException {
128        this.writeXML(writer, true);
129    }
130
131// XML Writing
132
133    void writeXML(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
134        writer.writeEmptyElement(SIMPLE_FIELD_TAG.getNamespaceURI(), SIMPLE_FIELD_TAG.getLocalPart());
135        this.writeAttributes(writer, includeName);
136    }
137
138    @Override
139    final void writeAttributes(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
140        super.writeAttributes(writer, includeName);
141        this.writeSimpleAttributes(writer);
142        if (this.encodingSignature != 0) {
143            writer.writeAttribute(ENCODING_SIGNATURE_ATTRIBUTE.getNamespaceURI(), ENCODING_SIGNATURE_ATTRIBUTE.getLocalPart(),
144              "" + this.encodingSignature);
145        }
146    }
147
148    void writeSimpleAttributes(XMLStreamWriter writer) throws XMLStreamException {
149        if (this.type != null)
150            writer.writeAttribute(TYPE_ATTRIBUTE.getNamespaceURI(), TYPE_ATTRIBUTE.getLocalPart(), this.type);
151        if (this.indexed)
152            writer.writeAttribute(INDEXED_ATTRIBUTE.getNamespaceURI(), INDEXED_ATTRIBUTE.getLocalPart(), "" + this.indexed);
153    }
154
155// Object
156
157    @Override
158    public String toString() {
159        return super.toString()
160          + (this.type != null ? " of type " + this.type : "")
161          + (this.encodingSignature != 0 ? " (encoding " + this.encodingSignature + ")" : "");
162    }
163
164    @Override
165    public boolean equals(Object obj) {
166        if (obj == this)
167            return true;
168        if (!super.equals(obj))
169            return false;
170        final SimpleSchemaField that = (SimpleSchemaField)obj;
171        return (this.type != null ? this.type.equals(that.type) : that.type == null)
172          && this.encodingSignature == that.encodingSignature
173          && this.indexed == that.indexed;
174    }
175
176    @Override
177    public int hashCode() {
178        return super.hashCode()
179          ^ (this.type != null ? this.type.hashCode() : 0)
180          ^ ((Long)this.encodingSignature).hashCode()
181          ^ (this.indexed ? 1 : 0);
182    }
183
184// Cloneable
185
186    @Override
187    public SimpleSchemaField clone() {
188        return (SimpleSchemaField)super.clone();
189    }
190}
