008package org.jsimpledb.schema;
009
010import java.util.ArrayList;
011import java.util.List;
012
013import javax.xml.stream.XMLStreamException;
014import javax.xml.stream.XMLStreamReader;
015import javax.xml.stream.XMLStreamWriter;
016
017import org.dellroad.stuff.string.StringEncoder;
018import org.jsimpledb.core.EnumFieldType;
019import org.jsimpledb.core.InvalidSchemaException;
020
021/**
022 * An enum field in a {@link SchemaObjectType}.
023 */
024public class EnumSchemaField extends SimpleSchemaField {
025
026    private List<String> idents = new ArrayList<>();
027
028    /**
029     * Get the ordered list of identifiers.
030     */
031    public List<String> getIdentifiers() {
032        return this.idents;
033    }
034
035    @Override
036    void validate() {
037        super.validate();
038        try {
039            EnumFieldType.validateIdentifiers(this.idents);
040        } catch (IllegalArgumentException e) {
041            throw new InvalidSchemaException("invalid " + this + ": " + e.getMessage(), e);
042        }
043    }
044
045    @Override
046    public <R> R visit(SchemaFieldSwitch<R> target) {
047        return target.caseEnumSchemaField(this);
048    }
049
050    // For enum types, we don't care if the type names are different; we only care if the identifier sets are different.
051    // This allows enum types to change Java packages without creating an incompatible schema.
052    @Override
053    boolean isCompatibleWithInternal(AbstractSchemaItem that0) {
054        final EnumSchemaField that = (EnumSchemaField)that0;
055        if (this.isIndexed() != that.isIndexed())
056            return false;
057        if (!this.idents.equals(that.idents))
058            return false;
059        return true;
060    }
061
062    @Override
063    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
064        while (this.expect(reader, true, IDENTIFIER_TAG))
065            this.idents.add(reader.getElementText());
066    }
067
068    @Override
069    void writeXML(XMLStreamWriter writer, boolean includeName) throws XMLStreamException {
070        writer.writeStartElement(ENUM_FIELD_TAG.getNamespaceURI(), ENUM_FIELD_TAG.getLocalPart());
071        this.writeAttributes(writer, includeName);
072        for (String ident : this.idents) {
073            writer.writeStartElement(IDENTIFIER_TAG.getNamespaceURI(), IDENTIFIER_TAG.getLocalPart());
074            writer.writeCharacters(StringEncoder.encode(ident, false));
075            writer.writeEndElement();
076        }
077        writer.writeEndElement();
078    }
079
080// Object
081
082    @Override
083    public boolean equals(Object obj) {
084        if (obj == this)
085            return true;
086        if (!super.equals(obj))
087            return false;
088        final EnumSchemaField that = (EnumSchemaField)obj;
089        return this.idents.equals(that.idents);
090    }
091
092    @Override
093    public int hashCode() {
094        return super.hashCode() ^ this.idents.hashCode();
095    }
096
097// Cloneable
098
099    @Override
100    public EnumSchemaField clone() {
101        final EnumSchemaField clone = (EnumSchemaField)super.clone();
102        clone.idents = new ArrayList<>(this.idents);
103        return clone;
104    }
105}
