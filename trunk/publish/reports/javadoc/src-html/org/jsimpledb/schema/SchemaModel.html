008package org.jsimpledb.schema;
009
010import java.io.ByteArrayOutputStream;
011import java.io.IOException;
012import java.io.InputStream;
013import java.io.OutputStream;
014import java.nio.charset.Charset;
015import java.util.SortedMap;
016import java.util.TreeMap;
017
018import javax.xml.stream.XMLInputFactory;
019import javax.xml.stream.XMLOutputFactory;
020import javax.xml.stream.XMLStreamException;
021import javax.xml.stream.XMLStreamReader;
022import javax.xml.stream.XMLStreamWriter;
023
024import org.dellroad.stuff.xml.IndentXMLStreamWriter;
025import org.jsimpledb.core.InvalidSchemaException;
026import org.jsimpledb.util.AbstractXMLStreaming;
027
028/**
029 * Models one JSimpleDB {@link org.jsimpledb.core.Database} schema version.
030 */
031public class SchemaModel extends AbstractXMLStreaming implements XMLConstants, Cloneable {
032
033    private SortedMap<Integer, SchemaObject> schemaObjects = new TreeMap<>();
034
035    public SortedMap<Integer, SchemaObject> getSchemaObjects() {
036        return this.schemaObjects;
037    }
038    public void setSchemaObjects(SortedMap<Integer, SchemaObject> schemaObjects) {
039        this.schemaObjects = schemaObjects;
040    }
041
042    /**
043     * Serialize an instance to the given XML output.
044     *
045     * @param output XML output
046     * @param indent true to pretty print the XML
047     * @throws IOException if an I/O error occurs
048     */
049    public void toXML(OutputStream output, boolean indent) throws IOException {
050        try {
051            XMLStreamWriter writer = XMLOutputFactory.newInstance().createXMLStreamWriter(output, "UTF-8");
052            if (indent)
053                writer = new IndentXMLStreamWriter(writer);
054            writer.writeStartDocument("UTF-8", "1.0");
055            this.writeXML(writer);
056            writer.writeEndDocument();
057            writer.flush();
058        } catch (XMLStreamException e) {
059            if (e.getCause() instanceof IOException)
060                throw (IOException)e.getCause();
061            throw new RuntimeException("internal error", e);
062        }
063        output.flush();
064    }
065
066    /**
067     * Deserialize an instance from the given XML input and validate it.
068     *
069     * @param input XML input
070     * @throws IOException if an I/O error occurs
071     * @throws InvalidSchemaException if the XML input or decoded {@link SchemaModel} is invalid
072     */
073    public static SchemaModel fromXML(InputStream input) throws IOException {
074        final SchemaModel schemaModel = new SchemaModel();
075        try {
076            final XMLStreamReader reader = XMLInputFactory.newInstance().createXMLStreamReader(input);
077            schemaModel.readXML(reader);
078        } catch (XMLStreamException e) {
079            throw new InvalidSchemaException("error parsing schema model XML", e);
080        }
081        schemaModel.validate();
082        return schemaModel;
083    }
084
085    /**
086     * Validate this instance.
087     *
088     * @throws InvalidSchemaException if this instance is invalid
089     */
090    public void validate() {
091        for (SchemaObject schemaObject : this.schemaObjects.values())
092            schemaObject.validate();
093    }
094
095    /**
096     * Determine whether this schema is compatible with the given schema for use with the core API.
097     * Two instances are compatible if they are identical in all respects except for object and field names
098     * (to also include object and field names in the comparison, use {@link #equals equals()}).
099     * The core API uses storage IDs, not names, to identify objects and fields.
100     *
101     * @param that other schema
102     * @throws IllegalArgumentException if {@code that} is null
103     */
104    public boolean isCompatibleWith(SchemaModel that) {
105        if (that == null)
106            throw new IllegalArgumentException("null that");
107        if (!this.schemaObjects.keySet().equals(that.schemaObjects.keySet()))
108            return false;
109        for (int storageId : this.schemaObjects.keySet()) {
110            final SchemaObject thisSchemaObject = this.schemaObjects.get(storageId);
111            final SchemaObject thatSchemaObject = that.schemaObjects.get(storageId);
112            if (!thisSchemaObject.isCompatibleWith(thatSchemaObject))
113                return false;
114        }
115        return true;
116    }
117
118    void readXML(XMLStreamReader reader) throws XMLStreamException {
119        this.expect(reader, false, SCHEMA_MODEL_TAG);
120        while (this.expect(reader, true, OBJECT_TAG)) {
121            final SchemaObject schemaObject = new SchemaObject();
122            schemaObject.readXML(reader);
123            final int storageId = schemaObject.getStorageId();
124            final SchemaObject previous = this.schemaObjects.put(storageId, schemaObject);
125            if (previous != null) {
126                throw new InvalidSchemaException("duplicate use of storage ID " + storageId
127                  + " for both " + previous + " and " + schemaObject);
128            }
129        }
130    }
131
132    void writeXML(XMLStreamWriter writer) throws XMLStreamException {
133        writer.setDefaultNamespace(SCHEMA_MODEL_TAG.getNamespaceURI());
134        writer.writeStartElement(SCHEMA_MODEL_TAG.getNamespaceURI(), SCHEMA_MODEL_TAG.getLocalPart());
135        for (SchemaObject schemaObject : this.schemaObjects.values())
136            schemaObject.writeXML(writer);
137        writer.writeEndElement();
138    }
139
140// Object
141
142    @Override
143    public String toString() {
144        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
145        try {
146            this.toXML(buf, true);
147        } catch (IOException e) {
148            throw new RuntimeException(e);
149        }
150        return new String(buf.toByteArray(), Charset.forName("UTF-8"))
151          .replaceAll("(?s)<\\?xml version=\"1\\.0\" encoding=\"UTF-8\"\\?>\n", "");
152    }
153
154    @Override
155    public boolean equals(Object obj) {
156        if (obj == this)
157            return true;
158        if (obj == null || obj.getClass() != this.getClass())
159            return false;
160        final SchemaModel that = (SchemaModel)obj;
161        return this.schemaObjects.equals(that.schemaObjects);
162    }
163
164    @Override
165    public int hashCode() {
166        return this.schemaObjects.hashCode();
167    }
168
169// Cloneable
170
171    /**
172     * Deep-clone this instance.
173     */
174    @Override
175    public SchemaModel clone() {
176        SchemaModel clone;
177        try {
178            clone = (SchemaModel)super.clone();
179        } catch (CloneNotSupportedException e) {
180            throw new RuntimeException(e);
181        }
182        clone.schemaObjects = new TreeMap<>();
183        for (SchemaObject schemaObject : this.schemaObjects.values())
184            clone.schemaObjects.put(schemaObject.getStorageId(), schemaObject.clone());
185        return clone;
186    }
187}
