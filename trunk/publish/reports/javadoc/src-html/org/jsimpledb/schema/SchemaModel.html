008package org.jsimpledb.schema;
009
010import java.io.ByteArrayOutputStream;
011import java.io.IOException;
012import java.io.InputStream;
013import java.io.OutputStream;
014import java.nio.charset.Charset;
015import java.util.Iterator;
016import java.util.SortedMap;
017import java.util.TreeMap;
018
019import javax.validation.Valid;
020import javax.validation.constraints.NotNull;
021
022import org.dellroad.stuff.jibx.JiBXUtil;
023import org.jibx.runtime.JiBXException;
024
025/**
026 * Models one JSimpleDB {@link org.jsimpledb.core.Database} schema version.
027 */
028public class SchemaModel implements Cloneable {
029
030    private SortedMap<Integer, SchemaObject> schemaObjects = new TreeMap<>();
031
032    @NotNull
033    @Valid
034    public SortedMap<Integer, SchemaObject> getSchemaObjects() {
035        return this.schemaObjects;
036    }
037    public void setSchemaObjects(SortedMap<Integer, SchemaObject> schemaObjects) {
038        this.schemaObjects = schemaObjects;
039    }
040
041    /**
042     * Serialize an instance to the given XML output.
043     *
044     * @param output XML output
045     * @throws IOException if an I/O error occurs
046     */
047    public void toXML(OutputStream output) throws IOException {
048        try {
049            JiBXUtil.writeObject(this, output);
050        } catch (JiBXException e) {
051            throw new RuntimeException("internal error", e);
052        }
053        output.flush();
054    }
055
056    /**
057     * Deserialize an instance from the given XML input.
058     *
059     * @param input XML input
060     * @throws IOException if an I/O error occurs
061     * @throws IllegalArgumentException if the XML input is invalid
062     */
063    public static SchemaModel fromXML(InputStream input) throws IOException {
064        try {
065            return JiBXUtil.readObject(SchemaModel.class, input);
066        } catch (JiBXException e) {
067            throw new IllegalArgumentException("invalid XML", e);
068        }
069    }
070
071// Object
072
073    @Override
074    public String toString() {
075        final ByteArrayOutputStream buf = new ByteArrayOutputStream();
076        try {
077            this.toXML(buf);
078        } catch (IOException e) {
079            throw new RuntimeException(e);
080        }
081        return new String(buf.toByteArray(), Charset.forName("UTF-8"));
082    }
083
084    @Override
085    public boolean equals(Object obj) {
086        if (obj == this)
087            return true;
088        if (obj == null || obj.getClass() != this.getClass())
089            return false;
090        final SchemaModel that = (SchemaModel)obj;
091        return this.schemaObjects.equals(that.schemaObjects);
092    }
093
094    @Override
095    public int hashCode() {
096        return this.schemaObjects.hashCode();
097    }
098
099// Cloneable
100
101    /**
102     * Deep-clone this instance.
103     */
104    @Override
105    public SchemaModel clone() {
106        SchemaModel clone;
107        try {
108            clone = (SchemaModel)super.clone();
109        } catch (CloneNotSupportedException e) {
110            throw new RuntimeException(e);
111        }
112        clone.schemaObjects = new TreeMap<>();
113        for (SchemaObject schemaObject : this.schemaObjects.values())
114            clone.addSchemaObject(schemaObject.clone());
115        return clone;
116    }
117
118// JiBX
119
120    public void addSchemaObject(SchemaObject type) {
121        this.schemaObjects.put(type.getStorageId(), type);
122    }
123
124    public Iterator<SchemaObject> iterateSchemaObjects() {
125        return this.schemaObjects.values().iterator();
126    }
127}
