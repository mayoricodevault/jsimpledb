008package org.jsimpledb.schema;
009
010import java.util.LinkedHashMap;
011import java.util.Map;
012
013import javax.xml.namespace.QName;
014import javax.xml.stream.XMLStreamException;
015import javax.xml.stream.XMLStreamReader;
016
017import org.jsimpledb.core.MapField;
018import org.jsimpledb.util.DiffGenerating;
019import org.jsimpledb.util.Diffs;
020
021/**
022 * A map field in one version of a {@link SchemaObjectType}.
023 */
024public class MapSchemaField extends ComplexSchemaField implements DiffGenerating<MapSchemaField> {
025
026    private SimpleSchemaField keyField;
027    private SimpleSchemaField valueField;
028
029    public SimpleSchemaField getKeyField() {
030        return this.keyField;
031    }
032    public void setKeyField(SimpleSchemaField keyField) {
033        this.keyField = keyField;
034    }
035
036    public SimpleSchemaField getValueField() {
037        return this.valueField;
038    }
039    public void setValueField(SimpleSchemaField valueField) {
040        this.valueField = valueField;
041    }
042
043    @Override
044    public Map<String, SimpleSchemaField> getSubFields() {
045        final LinkedHashMap<String, SimpleSchemaField> map = new LinkedHashMap<String, SimpleSchemaField>(2);
046        map.put(MapField.KEY_FIELD_NAME, this.keyField);
047        map.put(MapField.VALUE_FIELD_NAME, this.valueField);
048        return map;
049    }
050
051    @Override
052    void readSubElements(XMLStreamReader reader, int formatVersion) throws XMLStreamException {
053        this.keyField = this.readSubField(reader, formatVersion, MapField.KEY_FIELD_NAME);
054        this.valueField = this.readSubField(reader, formatVersion, MapField.VALUE_FIELD_NAME);
055        this.expectClose(reader);
056    }
057
058    @Override
059    QName getXMLTag() {
060        return MAP_FIELD_TAG;
061    }
062
063    @Override
064    public <R> R visit(SchemaFieldSwitch<R> target) {
065        return target.caseMapSchemaField(this);
066    }
067
068// DiffGenerating
069
070    @Override
071    public Diffs differencesFrom(MapSchemaField that) {
072        final Diffs diffs = new Diffs(super.differencesFrom(that));
073        final Diffs keyDiffs = this.keyField.differencesFrom(that.keyField);
074        if (!keyDiffs.isEmpty())
075            diffs.add("changed key field", keyDiffs);
076        final Diffs valueDiffs = this.valueField.differencesFrom(that.valueField);
077        if (!valueDiffs.isEmpty())
078            diffs.add("changed value field", valueDiffs);
079        return diffs;
080    }
081
082// Object
083
084    @Override
085    public String toString() {
086        return "map " + super.toString() + " with key " + this.keyField + " and value " + this.valueField;
087    }
088
089    @Override
090    public boolean equals(Object obj) {
091        if (obj == this)
092            return true;
093        if (!super.equals(obj))
094            return false;
095        final MapSchemaField that = (MapSchemaField)obj;
096        return (this.keyField != null ? this.keyField.equals(that.keyField) : that.keyField == null)
097          && (this.valueField != null ? this.valueField.equals(that.valueField) : that.valueField == null);
098    }
099
100    @Override
101    public int hashCode() {
102        return super.hashCode()
103          ^ (this.keyField != null ? this.keyField.hashCode() : 0)
104          ^ (this.valueField != null ? this.valueField.hashCode() : 0);
105    }
106
107// Cloneable
108
109    @Override
110    public MapSchemaField clone() {
111        return (MapSchemaField)super.clone();
112    }
113}
