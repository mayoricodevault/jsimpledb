008package org.jsimpledb;
009
010import java.lang.reflect.Method;
011import java.nio.charset.Charset;
012import java.security.MessageDigest;
013import java.security.NoSuchAlgorithmException;
014
015import org.jsimpledb.core.ListField;
016import org.jsimpledb.core.MapField;
017import org.jsimpledb.core.SetField;
018
019/**
020 * Default {@link StorageIdGenerator} implementation.
021 *
022 * <p>
023 * This class hashes the names into storage ID's in the range {@value #MIN_STORAGE_ID} (inclusive) to
024 * {@value #MAX_STORAGE_ID} (exclusive); this corresponds to the range of values that are encoded in three bytes.
025 * This provides a target space of 65,280 possible storage IDs, so collisions should be extremely rare.
026 * </p>
027 */
028public class DefaultStorageIdGenerator implements StorageIdGenerator {
029
030    public static final int MIN_STORAGE_ID = 0x1fb;             // first value requiring three bytes
031    public static final int MAX_STORAGE_ID = 0x100fb;           // first value requiring four bytes
032
033    private final MessageDigest sha1;
034
035    public DefaultStorageIdGenerator() {
036        try {
037            this.sha1 = MessageDigest.getInstance("SHA-1");
038        } catch (NoSuchAlgorithmException e) {
039            throw new RuntimeException(e);
040        }
041    }
042
043    @Override
044    public int generateClassStorageId(Class<?> type, String name) {
045        return this.getStorageId("class:" + name);
046    }
047
048    @Override
049    public int generateCompositeIndexStorageId(Class<?> type, String name, int[] fields) {
050        return this.getStorageId("index:" + name);
051    }
052
053    @Override
054    public int generateFieldStorageId(Method getter, String name) {
055        return this.getStorageId("field:" + name);
056    }
057
058    @Override
059    public int generateSetElementStorageId(Method getter, String name) {
060        return this.getStorageId("field:" + name + "." + SetField.ELEMENT_FIELD_NAME);
061    }
062
063    @Override
064    public int generateListElementStorageId(Method getter, String name) {
065        return this.getStorageId("field:" + name + "." + ListField.ELEMENT_FIELD_NAME);
066    }
067
068    @Override
069    public int generateMapKeyStorageId(Method getter, String name) {
070        return this.getStorageId("field:" + name + "." + MapField.KEY_FIELD_NAME);
071    }
072
073    @Override
074    public int generateMapValueStorageId(Method getter, String name) {
075        return this.getStorageId("field:" + name + "." + MapField.VALUE_FIELD_NAME);
076    }
077
078    private int getStorageId(String string) {
079        this.sha1.reset();
080        final byte[] digest = this.sha1.digest(string.getBytes(Charset.forName("UTF-8")));
081        int value = 0;
082        for (int i = 0; i < 4; i++)
083            value = (value << 8) | (digest[i] & 0xff);
084        value &= 0x7fffffff;
085        return (value % (MAX_STORAGE_ID - MIN_STORAGE_ID)) + MIN_STORAGE_ID;
086    }
087}
