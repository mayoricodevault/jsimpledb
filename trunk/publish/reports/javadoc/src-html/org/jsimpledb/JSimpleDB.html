035/**
036 * Abstraction layer allowing access to a {@link Database} using normal Java objects.
037 *
038 * <p>
039 * This class provides a natural, Java-centric view on top of the core {@link Database} class. This is done via two main
040 * enhancements to the functionality provided by {@link Database}:
041 * <ul>
042 *  <li>{@linkplain org.jsimpledb.annotation Java annotations} are used to define all {@link Database} objects and fields,
043 *      to automatically register various listeners, to access indexed fields, etc.</li>
044 *  <li>{@link Database} objects and fields are represented using actual Java model objects, where the Java model classes
045 *      are automatically generated subclasses of the user-supplied Java bean model classes. That is, in the view that
046 *      a {@link JSimpleDB} provides, all {@link Database} object references (which are all {@link ObjId} objects)
047 *      are replaced by normal Java model objects of the appropriate type, and all {@link Database} fields are accessible
048 *      through the corresponding model objects' Java bean getter and setter methods, instead of directly through
049 *      a {@link Transaction} object. All Java model classes implement the {@link JObject} interface.</li>
050 * </ul>
051 * </p>
052 *
053 * <p>
054 * In addition, a {@link JSimpleDB} provides automatic incremental JSR 303 validation.
055 * </p>
056 *
057 * <p>
058 * Not counting "snapshot" objects (see below), this class guarantees that for each {@link ObjId}
059 * it will only create a single, globally unique, {@link JObject} Java model object.
060 * Therefore, the same Java model objects can be used in and out of any transaction, and can serve as
061 * unique database object identifiers (if you have not overridden {@link #equals equals()} in your model clases).
062 * Except for their associated {@link ObjId}s, the generated Java model objects are stateless; all database field state
063 * is contained within whichever transaction is {@linkplain JTransaction#getCurrent associated with the current thread}.
064 * </p>
065 *
066 * <p>
067 * "Snapshot" objects are in-memory copies of regular database objects that may be imported/exported to/from transactions.
068 * Snapshot {@link JObject}s are distinct from regular database {@link JObject}s; their state is contained in an in-memory
069 * {@link SnapshotJTransaction}. See {@link JObject#copyOut JObject.copyOut} and {@link JObject#copyIn JObject.copyIn}.
070 * </p>
071 *
072 * @see JObject
073 * @see JTransaction
074 * @see org.jsimpledb.annotation
075 */
076public class JSimpleDB {

099     * Create an instance using an initially empty, in-memory {@link SimpleKVDatabase}.
100     * Generates a database schema by introspecting the {@code classes}; schema version number {@code 1} is assumed.
101     *
102     * <p>
103     * This constructor can also be used just to validate the annotations on the given classes.
104     * </p>
105     *
106     * <p>
107     * All {@link JSimpleClass &#64;JSimpleClass}-annotated super-classes of any {@link JSimpleClass &#64;JSimpleClass}-annotated
108     * class in {@code classes} will be included even if not explicitly specified.
109     * </p>
110     *
111     * @param classes classes annotated with {@link JSimpleClass &#64;JSimpleClass}
112     *  and/or {@link JFieldType &#64;JFieldType} annotations
113     * @throws IllegalArgumentException if {@code classes} is null
114     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
115     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
116     */
117    public JSimpleDB(Iterable<Class<?>> classes) {
118        this(new Database(new SimpleKVDatabase()), 1, classes);

121    /**
122     * Primary constructor. Generates a database schema by introspecting the provided classes.
123     *
124     * <p>
125     * All {@link JSimpleClass &#64;JSimpleClass}-annotated super-classes of any {@link JSimpleClass &#64;JSimpleClass}-annotated
126     * class in {@code classes} will be included even if not explicitly specified.
127     * </p>
128     *
129     * @param database core database to use
130     * @param version schema version number of the schema derived from {@code classes},
131     *  or zero to use the highest version already recorded in the database
132     * @param classes classes annotated with {@link JSimpleClass &#64;JSimpleClass}
133     *  and/or {@link JFieldType &#64;JFieldType} annotations
134     * @throws IllegalArgumentException if any parameter is null
135     * @throws IllegalArgumentException if {@code version} is not greater than zero
136     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
137     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
138     */
139    public JSimpleDB(Database database, int version, Iterable<Class<?>> classes) {
140
141        // Initialize
142        if (database == null)
143            throw new IllegalArgumentException("null database");
144        if (version < 0)
145            throw new IllegalArgumentException("invalid schema version: " + version);
146        if (classes == null)
147            throw new IllegalArgumentException("null classes");
148        this.db = database;
149        this.version = version;
150
151        // Inventory classes; automatically add all @JSimpleClass-annotated superclasses of @JSimpleClass-annotated classes
152        final HashSet<Class<?>> jsimpleClasses = new HashSet<>();
153        final HashSet<Class<?>> jfieldTypeClasses = new HashSet<>();
154        for (Class<?> type : classes) {
155
156            // Sanity check
157            if (type == null)
158                throw new IllegalArgumentException("null class found in classes");
159
160            // Check for @JFieldType
161            if (type.isAnnotationPresent(JFieldType.class))
162                jfieldTypeClasses.add(type);
163
164            // Check for @JSimpleClass
165            if (type.isAnnotationPresent(JSimpleClass.class)) {
166
167                // Sanity check type
168                if (type.isPrimitive() || type.isInterface() || type.isArray()) {
169                    throw new IllegalArgumentException("illegal type " + type + " for @"
170                      + JSimpleClass.class.getSimpleName() + " annotation: not a normal class");
171                }
172
173                // Add type and all @JSimpleClass-annotated superclasses
174                jsimpleClasses.add(type);
175                for (Class<?> supertype = type.getSuperclass(); type != null; type = type.getSuperclass()) {
176                    if (supertype.getAnnotation(JSimpleClass.class) != null)
177                        jsimpleClasses.add(supertype);
178                }
179            }
180        }
181
182        // Register custom field types
183        for (Class<?> type : jfieldTypeClasses) {
184
185            // Get type name
186            final JFieldType fieldTypeAnnotation = type.getAnnotation(JFieldType.class);
187            final String name = fieldTypeAnnotation.name().length() != 0 ? fieldTypeAnnotation.name() : type.getName();
188            if (this.log.isTraceEnabled()) {
189                this.log.trace("found @" + JFieldType.class.getSimpleName()
190                  + " annotation on " + type + " defining field type `" + name + "'");
191            }
192
193            // Instantiate class
194            final Object obj;
195            try {
196                obj = type.newInstance();
197            } catch (Exception e) {
198                throw new DatabaseException("can't instantiate " + type, e);
199            }
200            if (!(obj instanceof FieldType)) {
201                throw new DatabaseException("invalid @" + JFieldType.class.getSimpleName()
202                  + " annotation on " + type + ": not a subclass of " + FieldType.class);
203            }
204
205            // Register field type
206            final FieldType<?> fieldType = (FieldType<?>)obj;
207            this.db.getFieldTypeRegistry().add(fieldType);
208            this.log.debug("registered new field type `" + fieldType.getName() + "' using " + type);
209        }
210
211        // Add Java model classes
212        for (Class<?> type : jsimpleClasses) {
213
214            // Create JClass
215            final JSimpleClass jclassAnnotation = type.getAnnotation(JSimpleClass.class);
216            final String name = jclassAnnotation.name().length() != 0 ? jclassAnnotation.name() : type.getSimpleName();
217            if (this.log.isTraceEnabled()) {
218                this.log.trace("found @" + JSimpleClass.class.getSimpleName() + " annotation on " + type
219                  + " defining object type `" + name + "'");
220            }
221            JClass<?> jclass;
222            try {
223                jclass = this.createJClass(name, jclassAnnotation.storageId(), Util.getWildcardedType(type));
224            } catch (IllegalArgumentException e) {
225                throw new IllegalArgumentException("invalid @" + JSimpleClass.class.getSimpleName()
226                  + " annotation on " + type + ": " + e, e);
227            }
228
229            // Add jclass
230            this.addJClass(jclass);
231            this.log.debug("added Java model class `" + jclass.name + "' with storage ID " + jclass.storageId);
232        }
233
234        // Create fields
235        for (JClass<?> jclass : this.jclasses.values())
236            jclass.createFields();
237
238        // Scan for other annotations
239        for (JClass<?> jclass : this.jclasses.values())
240            jclass.scanAnnotations();
241
242        // Index fields by storage ID
243        for (JClass<?> jclass : this.jclasses.values()) {
244            for (JField jfield : jclass.jfields.values()) {
245                this.jfields.put(jfield.storageId, jfield);
246                if (jfield instanceof JComplexField) {
247                    final JComplexField complexField = (JComplexField)jfield;
248                    for (JSimpleField subField : complexField.getSubFields())
249                        this.jfields.put(subField.storageId, subField);
250                }
251            }
252        }
253
254        // Validate schema
255        this.db.validateSchema(this.getSchemaModel());
256    }

298    /**
299     * Get the Java model class of the given {@link JObject}.
300     * If {@code jobj} is an instance of a JSimpleDB-generated subclass of a user-supplied Java model class,
301     * this returns the original Java model class. Otherwise, it returns {@code obj}'s type.
302     *
303     * @param jobj database instance
304     * @return lowest ancestor class of {@code jobj}'s class that is not a JSimpleDB-generated subclass
305     * @throws IllegalArgumentException if {@code jobj} is null
306     */
307    public static Class<?> getModelClass(JObject jobj) {
308        if (jobj == null)
309            throw new IllegalArgumentException("null jobj");
310        for (Class<?> type = jobj.getClass(); type != null; type = type.getSuperclass()) {
311            if (type.getName().indexOf(ClassGenerator.CLASSNAME_SUFFIX) == -1)
312                return type;
313        }
314        return null;
315    }

319    /**
320     * Create a new transaction. This does not invoke {@link JTransaction#setCurrent JTransaction.setCurrent()}: the
321     * caller is responsible for doing that if necessary.
322     *
323     * @param allowNewSchema whether creating a new schema version is allowed
324     * @param validationMode the {@link ValidationMode} to use for the new transaction
325     * @throws org.jsimpledb.InvalidSchemaException if {@code schemaModel} does not match what's recorded in the
326     *  database for the schema version provided to the constructor
327     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
328     *  is not recorded in the database and {@code allowNewSchema} is false
329     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
330     *  is not recorded in the database and {@code allowNewSchema} is true, but {@code schemaModel} is incompatible
331     *  with one or more previous schemas alread recorded in the database (i.e., the same storage ID is used
332     *  incompatibly between schema versions)
333     * @throws org.jsimpledb.InconsistentDatabaseException if inconsistent or invalid schema information is detected
334     *  in the database
335     * @throws IllegalArgumentException if {@code validationMode} is null
336     */
337    public JTransaction createTransaction(boolean allowNewSchema, ValidationMode validationMode) {
338        if (validationMode == null)
339            throw new IllegalArgumentException("null validationMode");
340        final Transaction tx = this.db.createTransaction(this.getSchemaModel(), this.version, allowNewSchema);
341        this.lastVersion = tx.getSchema().getSchemaVersions().lastKey();
342        return new JTransaction(this, tx, validationMode);
343    }

347    /**
348     * Get the {@link SchemaModel} associated with this instance, derived from the annotations on the scanned classes.
349     */
350    public SchemaModel getSchemaModel() {
351        if (this.schemaModel == null) {
352            final SchemaModel model = new SchemaModel();
353            for (JClass<?> jclass : this.jclasses.values()) {
354                final SchemaObject schemaObject = jclass.toSchemaItem();
355                model.getSchemaObjects().put(schemaObject.getStorageId(), schemaObject);
356            }
357            this.schemaModel = model;
358            this.log.debug("JSimpleDB schema generated from annotated classes:\n{}", this.schemaModel);
359        }
360        return this.schemaModel.clone();
361    }

432    /**
433     * Get all {@link JClass}es which sub-type the given type.
434     *
435     * @param typeToken type restriction, or null for no restrction
436     * @return list of {@link JClass}es whose type is {@code typeToken} or a sub-type (if not null), ordered by storage ID
437     */
438    @SuppressWarnings("unchecked")
439    public <T> List<JClass<? extends T>> getJClasses(TypeToken<T> typeToken) {
440        final ArrayList<JClass<? extends T>> list = new ArrayList<>();
441        for (JClass<?> jclass : this.jclasses.values()) {
442            if (typeToken == null || typeToken.isAssignableFrom(jclass.typeToken))
443                list.add((JClass<? extends T>)jclass);
444        }
445        return list;
446    }

508    /**
509     * Get the {@link JField} associated with the given storage ID.
510     *
511     * @param storageId field storage ID
512     * @param type required type, or null
513     * @return {@link JFiedl} instance
514     * @throws IllegalArgumentException if {@code storageId} does not represent a field
515     */
516    @SuppressWarnings("unchecked")
517    <T extends JField> T getJField(int storageId, Class<T> type) {
518        final JField jfield = this.jfields.get(storageId);
519        if (jfield == null)
520            throw new IllegalArgumentException("no JSimpleDB field associated with storage ID " + storageId);
521        return type != null ? type.cast(jfield) : (T)jfield;
522    }

524    // Add new JClass, checking for storage ID conflicts
525    void addJClass(JClass<?> jclass) {
526
527        // Check for storage ID conflict
528        final JClass<?> other = this.jclasses.get(jclass.storageId);
529        if (other != null) {
530            throw new IllegalArgumentException("illegal duplicate use of storage ID "
531              + jclass.storageId + " for both " + other + " and " + jclass);
532        }
533        this.jclasses.put(jclass.storageId, jclass);
534        this.jclassesByType.put(jclass.typeToken, jclass);      // this can never conflict, no need to check
535    }
