040/**
041 * A JSimpleDB database.
042 *
043 * <p>
044 * JSimpleDB consists of two main layers. The lower level "core" API provides a rigourous database abstraction on top of
045 * a key/value store. It supports simple fields of any atomic Java type, as well as list, set, and map complex fields,
046 * tightly controlled schema versioning, simple and composite indexes, and lifecycle and change notifications.
047 * The core API is provided via the {@link Database} class.
048 * </p>
049 *
050 * <p>
051 * This {@link JSimpleDB} class represents the upper layer. It sits on top of the core {@link Database} class and provides
052 * a fully type-safe Java view of a {@link Database}, where all access is through user-supplied Java model classes.
053 * In addition, a {@link JSimpleDB} provides automatic incremental JSR 303 validation. Database types and fields,
054 * as well as listener methods, are all declared using various {@linkplain org.jsimpledb.annotation Java annotations}.
055 * </p>
056 *
057 * <p>
058 * User-provided Java model classes are typically abstract and declare fields as abstract Java bean methods. In any case,
059 * {@link JSimpleDB} generates concrete subclasses at runtime; these runtime classes will implement the {@link JObject} interface.
060 * The corresponding Java model objects are stateless; all database field state is derived from whichever transaction is
061 * {@linkplain JTransaction#getCurrent associated with the current thread}.
062 * </p>
063 *
064 * <p>
065 * Not counting "snapshot" objects (see below), this class guarantees that for each {@link ObjId} there will only ever be
066 * a single, globally unique, {@link JObject} Java model object. Therefore, the same Java model objects can be used in and
067 * out of any transaction.
068 * </p>
069 *
070 * <p>
071 * "Snapshot" objects are in-memory copies of regular database objects that may be imported/exported to/from transactions.
072 * Snapshot {@link JObject}s are distinct from regular database {@link JObject}s; their state is contained in an in-memory
073 * {@link SnapshotJTransaction}. See {@link JObject#copyOut JObject.copyOut} and {@link JObject#copyIn JObject.copyIn}.
074 * </p>
075 *
076 * @see JObject
077 * @see JTransaction
078 * @see JSimpleDBFactory
079 * @see org.jsimpledb.annotation
080 */
081public class JSimpleDB {

142    /**
143     * Primary constructor.
144     *
145     * @param database core database to use
146     * @param version schema version number of the schema derived from {@code classes},
147     *  or zero to use the highest version already recorded in the database
148     * @param storageIdGenerator generator for auto-generated storage ID's, or null to disallow auto-generation of storage ID's
149     * @param classes classes annotated with {@link JSimpleClass &#64;JSimpleClass} annotations; non-annotated classes are ignored
150     * @throws IllegalArgumentException if {@code database} or {@code classes} is null
151     * @throws IllegalArgumentException if {@code version} is not greater than zero
152     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
153     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
154     */
155    JSimpleDB(Database database, int version, StorageIdGenerator storageIdGenerator, Iterable<? extends Class<?>> classes) {
156
157        // Initialize
158        if (database == null)
159            throw new IllegalArgumentException("null database");
160        if (version < 0)
161            throw new IllegalArgumentException("invalid schema version: " + version);
162        if (classes == null)
163            throw new IllegalArgumentException("null classes");
164        this.db = database;
165        this.configuredVersion = version;
166        this.storageIdGenerator = storageIdGenerator;
167
168        // Inventory classes; automatically add all @JSimpleClass-annotated superclasses of @JSimpleClass-annotated classes
169        final HashSet<Class<?>> jsimpleClasses = new HashSet<>();
170        for (Class<?> type : classes) {
171
172            // Sanity check
173            if (type == null)
174                throw new IllegalArgumentException("null class found in classes");
175
176            // Add type and all @JSimpleClass-annotated superclasses
177            do {
178
179                // Find annotation
180                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
181                if (annotation == null)
182                    continue;
183
184                // Sanity check type
185                if (type.isPrimitive() || type.isInterface() || type.isArray()) {
186                    throw new IllegalArgumentException("illegal type " + type + " for @"
187                      + JSimpleClass.class.getSimpleName() + " annotation: not a normal class");
188                }
189
190                // Add class
191                jsimpleClasses.add(type);
192            } while ((type = type.getSuperclass()) != null);
193        }
194
195        // Add Java model classes
196        for (Class<?> type : jsimpleClasses) {
197
198            // Create JClass
199            final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
200            final String name = annotation.name().length() != 0 ? annotation.name() : type.getSimpleName();
201            if (this.log.isTraceEnabled()) {
202                this.log.trace("found @" + JSimpleClass.class.getSimpleName() + " annotation on " + type
203                  + " defining object type `" + name + "'");
204            }
205
206            // Get storage ID
207            int storageId = annotation.storageId();
208            if (storageId == 0)
209                storageId = this.getStorageIdGenerator(annotation, type).generateClassStorageId(type, name);
210
211            // Create JClass
212            JClass<?> jclass;
213            try {
214                jclass = this.createJClass(name, storageId, type);
215            } catch (IllegalArgumentException e) {
216                throw new IllegalArgumentException("invalid @" + JSimpleClass.class.getSimpleName()
217                  + " annotation on " + type + ": " + e, e);
218            }
219
220            // Add jclass
221            this.addJClass(jclass);
222            this.log.debug("added Java model class `" + jclass.name + "' with storage ID " + jclass.storageId);
223        }
224
225        // Create fields
226        for (JClass<?> jclass : this.jclasses.values())
227            jclass.createFields(this);
228
229        // Create canonical field info structures
230        final HashMap<Integer, String> fieldDescriptionMap = new HashMap<>();
231        for (JClass<?> jclass : this.jclasses.values()) {
232            for (JField jfield : jclass.jfields.values()) {
233                if (jfield instanceof JComplexField) {
234                    final JComplexField complexField = (JComplexField)jfield;
235                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)jfield.toJFieldInfo();
236                    for (JSimpleField subField : complexField.getSubFields()) {
237                        this.addJFieldInfo(subField, subField.toJFieldInfo(jfield.storageId), fieldDescriptionMap);
238                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
239                        complexFieldInfo.getSubFieldInfos().add(subFieldInfo);
240                    }
241                    this.addJFieldInfo(complexField, complexFieldInfo, fieldDescriptionMap);
242                } else
243                    this.addJFieldInfo(jfield, jfield.toJFieldInfo(), fieldDescriptionMap);
244            }
245        }
246
247        // Witness all simple fields to corresponding simple field info's
248        for (JClass<?> jclass : this.jclasses.values()) {
249            for (JField jfield : jclass.jfields.values()) {
250                if (jfield instanceof JSimpleField) {
251                    final JSimpleField jsimpleField = (JSimpleField)jfield;
252                    final JSimpleFieldInfo jsimpleFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
253                    jsimpleFieldInfo.witness(jsimpleField);
254                }
255                if (jfield instanceof JComplexField) {
256                    final JComplexField complexField = (JComplexField)jfield;
257                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)this.jfieldInfos.get(jfield.storageId);
258                    for (int i = 0; i < complexField.getSubFields().size(); i++) {
259                        final JSimpleField subField = complexField.getSubFields().get(i);
260                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
261                        subFieldInfo.witness(subField);
262                    }
263                }
264            }
265        }
266
267        // Add composite indexes to class; like fields, indexes are inherited (duplicated) from superclasses
268        for (JClass<?> jclass : this.jclasses.values()) {
269            for (Class<?> type = jclass.type; type != null; type = type.getSuperclass()) {
270                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
271                if (annotation != null) {
272                    for (org.jsimpledb.annotation.JCompositeIndex indexAnnotation : annotation.compositeIndexes())
273                        jclass.addCompositeIndex(this, indexAnnotation);
274                }
275            }
276        }
277
278        // Create canonical info instances for indexes
279        final HashMap<Integer, String> indexDescriptionMap = new HashMap<>();
280        for (JClass<?> jclass : this.jclasses.values()) {
281            for (JCompositeIndex index : jclass.jcompositeIndexes.values()) {
282                final JCompositeIndexInfo indexInfo = index.toJCompositeIndexInfo();
283                for (JSimpleField jfield : index.jfields) {
284                    final JSimpleFieldInfo jfieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
285                    indexInfo.getJFieldInfos().add(jfieldInfo);
286                }
287                this.addJCompositeIndexInfo(index, indexInfo, indexDescriptionMap);
288            }
289        }
290
291        // Scan for other method-level annotations
292        for (JClass<?> jclass : this.jclasses.values())
293            jclass.scanAnnotations();
294
295        // Determine which JClass's have validation requirement(s) on creation
296        for (JClass<?> jclass : this.jclasses.values())
297            jclass.calculateInitialValidationRequirement();
298        boolean anyInitialValidation = false;
299        for (JClass<?> jclass : this.jclasses.values()) {
300            if ((anyInitialValidation |= jclass.requiresInitialValidation))
301                break;
302        }
303        this.anyJClassRequiresInitialValidation = anyInitialValidation;
304
305        // Detect whether we have any @OnCreate, @OnDelete, and/or @OnVersionChange methods
306        boolean anyOnCreateMethods = false;
307        boolean anyOnDeleteMethods = false;
308        boolean anyOnVersionChangeMethods = false;
309        for (JClass<?> jclass : this.jclasses.values()) {
310            anyOnCreateMethods |= !jclass.onCreateMethods.isEmpty();
311            anyOnDeleteMethods |= !jclass.onDeleteMethods.isEmpty();
312            anyOnVersionChangeMethods |= !jclass.onVersionChangeMethods.isEmpty();
313        }
314        this.hasOnCreateMethods = anyOnCreateMethods;
315        this.hasOnDeleteMethods = anyOnDeleteMethods;
316        this.hasOnVersionChangeMethods = anyOnVersionChangeMethods;
317
318        // Validate schema
319        this.db.validateSchema(this.getSchemaModel());
320    }

381    /**
382     * Get the Java model class of the given {@link JObject}.
383     * If {@code jobj} is an instance of a JSimpleDB-generated subclass of a user-supplied Java model class,
384     * this returns the original Java model class. Otherwise, it returns {@code obj}'s type.
385     *
386     * @param jobj database instance
387     * @return lowest ancestor class of {@code jobj}'s class that is not a JSimpleDB-generated subclass
388     * @throws IllegalArgumentException if {@code jobj} is null
389     */
390    public static Class<?> getModelClass(JObject jobj) {
391        if (jobj == null)
392            throw new IllegalArgumentException("null jobj");
393        for (Class<?> type = jobj.getClass(); type != null; type = type.getSuperclass()) {
394            if (type.getName().indexOf(ClassGenerator.CLASSNAME_SUFFIX) == -1)
395                return type;
396        }
397        return null;
398    }

402    /**
403     * Create a new transaction. This does not invoke {@link JTransaction#setCurrent JTransaction.setCurrent()}: the
404     * caller is responsible for doing that if necessary.
405     *
406     * @param allowNewSchema whether creating a new schema version is allowed
407     * @param validationMode the {@link ValidationMode} to use for the new transaction
408     * @throws org.jsimpledb.InvalidSchemaException if {@code schemaModel} does not match what's recorded in the
409     *  database for the schema version provided to the constructor
410     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
411     *  is not recorded in the database and {@code allowNewSchema} is false
412     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
413     *  is not recorded in the database and {@code allowNewSchema} is true, but {@code schemaModel} is incompatible
414     *  with one or more previous schemas alread recorded in the database (i.e., the same storage ID is used
415     *  incompatibly between schema versions)
416     * @throws org.jsimpledb.InconsistentDatabaseException if inconsistent or invalid schema information is detected
417     *  in the database
418     * @throws IllegalArgumentException if {@code validationMode} is null
419     */
420    public JTransaction createTransaction(boolean allowNewSchema, ValidationMode validationMode) {
421        if (validationMode == null)
422            throw new IllegalArgumentException("null validationMode");
423        final Transaction tx = this.db.createTransaction(this.getSchemaModel(), this.configuredVersion, allowNewSchema);
424        this.actualVersion = tx.getSchema().getSchemaVersions().lastKey();
425        return new JTransaction(this, tx, validationMode);
426    }

431     * Get the {@link SchemaModel} associated with this instance, derived from the annotations on the scanned classes.
432     */
433    public SchemaModel getSchemaModel() {
434        if (this.schemaModel == null) {
435            final SchemaModel model = new SchemaModel();
436            for (JClass<?> jclass : this.jclasses.values()) {
437                final SchemaObjectType schemaObjectType = jclass.toSchemaItem(this);
438                model.getSchemaObjectTypes().put(schemaObjectType.getStorageId(), schemaObjectType);
439            }
440            this.schemaModel = model;
441            this.log.debug("JSimpleDB schema generated from annotated classes:\n{}", this.schemaModel);
442        }
443        return this.schemaModel.clone();
444    }

522     * Get all {@link JClass}es which sub-type the given type.
523     *
524     * @param type type restriction, or null for no restrction
525     * @return list of {@link JClass}es whose type is {@code type} or a sub-type, ordered by storage ID
526     */
527    @SuppressWarnings("unchecked")
528    public <T> List<JClass<? extends T>> getJClasses(Class<T> type) {
529        final ArrayList<JClass<? extends T>> list = new ArrayList<>();
530        for (JClass<?> jclass : this.jclasses.values()) {
531            if (type == null || type.isAssignableFrom(jclass.type))
532                list.add((JClass<? extends T>)jclass);
533        }
534        return list;
535    }

538     * Generate the {@link KeyRanges} restricting objects to the specified type.
539     *
540     * @param type any Java type, or null for no restriction
541     * @return key restriction for {@code type}
542     */
543    KeyRanges keyRangesFor(Class<?> type) {
544        if (type == null)
545            return KeyRanges.FULL;
546        final ArrayList<KeyRange> list = new ArrayList<>(this.jclasses.size());
547        for (JClass<?> jclass : this.getJClasses(type))
548            list.add(ObjId.getKeyRange(jclass.storageId));
549        return new KeyRanges(list);

554    /**
555     * Get the Java model object that is has the given ID. This returns the object associated with normal
556     * (i.e., non-snapshot) {@link JTransaction}s. To get the object associated with a snapshot transaction,
557     * invoke {@link JTransaction#getJObject(ObjId) JTransaction.getJObject()} on the snapshot transaction.
558     *
559     * <p>
560     * This method guarantees that for any particular {@code id}, the same Java instance will always be returned.
561     * Note: while for any {@link ObjId} there is only one globally unique {@link JObject} per {@link JSimpleDB}
562     * shared by all {@link JTransaction}s (the object returned by this method), each {@link SnapshotJTransaction}
563     * maintains its own distinct pool of unique "snapshot" {@link JObject}s.
564     * </p>
565     *
566     * <p>
567     * A non-null object is always returned, but the corresponding object may not exist in a given transaction.
568     * In that case, attempts to access its fields will throw {@link org.jsimpledb.core.DeletedObjectException}.
569     * </p>
570     *
571     * @param id object ID
572     * @return Java model object
573     * @throws IllegalArgumentException if {@code id} is null
574     * @see JTransaction#getJObject JTransaction.getJObject()
575     */
576    public JObject getJObject(ObjId id) {
577        return this.jobjectCache.getJObject(id);
578    }

583     *
584     * @param id object ID
585     * @param type expected type

601     *
602     * @param startType starting Java type for the path
603     * @param path dot-separated path of zero or more reference fields, followed by a target field

618    /**
619     * Utility method to get all of the objects directly referenced by a given object via any field.
620     *
621     * <p>
622     * Note: the returned {@link Iterable} may contain duplicates; these can be eliminated using an
623     * {@link org.jsimpledb.core.ObjIdSet} if necessary.
624     * </p>
625     *
626     * @param jobj starting object
627     * @return all objects directly referenced by {@code jobj}
628     * @throws IllegalArgumentException if {@code jobj} is null
629     */
630    public Iterable<JObject> getReferencedObjects(final JObject jobj) {
631
632        // Sanity check
633        if (jobj == null)
634            throw new IllegalArgumentException("null jobj");
635        final ObjId id = jobj.getObjId();
636
637        // Visit fields
638        final ArrayList<Iterable<JObject>> iterables = new ArrayList<>();
639        for (JField jfield : this.getJClass(id).getJFieldsByStorageId().values()) {
640            jfield.visit(new JFieldSwitchAdapter<Void>() {
641
642                @Override
643                public Void caseJReferenceField(JReferenceField field) {
644                    final JObject ref = field.getValue(jobj);
645                    if (ref != null)
646                        iterables.add(Collections.singleton(ref));
647                    return null;
648                }
649
650                @Override
651                public Void caseJMapField(JMapField field) {
652                    if (field.getKeyField() instanceof JReferenceField)
653                        iterables.add(Iterables.filter(field.getValue(jobj).keySet(), JObject.class));
654                    if (field.getValueField() instanceof JReferenceField)
655                        iterables.add(Iterables.filter(field.getValue(jobj).values(), JObject.class));
656                    return null;
657                }
658
659                @Override
660                protected Void caseJCollectionField(JCollectionField field) {
661                    if (field.getElementField() instanceof JReferenceField)
662                        iterables.add(Iterables.filter(field.getValue(jobj), JObject.class));
663                    return null;
664                }
665
666                @Override
667                protected Void caseJField(JField field) {
668                    return null;
669                }
670            });
671        }
672
673        // Done
674        return Iterables.concat(iterables);
675    }

684    /**
685     * Get the {@link JFieldInfo} associated with the given storage ID.
686     *
687     * @param storageId field storage ID
688     * @param type required type
689     * @return {@link JField} instance
690     * @throws UnknownFieldException if {@code storageId} does not represent a field
691     */
692    <T extends JFieldInfo> T getJFieldInfo(int storageId, Class<T> type) {
693        if (type == null)
694            throw new IllegalArgumentException("null type");
695        final JFieldInfo jfieldInfo = this.jfieldInfos.get(storageId);
696        if (jfieldInfo == null) {
697            throw new UnknownFieldException(storageId, "no JSimpleDB field exists with storage ID "
698              + storageId + " in schema version " + this.actualVersion);
699        }
700        try {
701            return type.cast(jfieldInfo);
702        } catch (ClassCastException e) {
703            throw new UnknownFieldException(storageId, "no JSimpleDB fields exist with storage ID "
704              + storageId + " in schema version " + this.actualVersion + " (found field " + jfieldInfo + " instead)");
705        }
706    }

708    // Add new JClass, checking for storage ID conflicts
709    private void addJClass(JClass<?> jclass) {
710
711        // Check for storage ID conflict
712        final JClass<?> other = this.jclasses.get(jclass.storageId);
713        if (other != null) {
714            throw new IllegalArgumentException("illegal duplicate use of storage ID "
715              + jclass.storageId + " for both " + other + " and " + jclass);
716        }
717        this.jclasses.put(jclass.storageId, jclass);
718        assert !this.jclassesByType.containsKey(jclass.type);                   // this should never conflict, no need to check
719        this.jclassesByType.put(jclass.type, jclass);
720    }

722    // Add new JFieldInfo, checking for conflicts
723    private <T extends JFieldInfo> void addJFieldInfo(JField jfield, T fieldInfo, Map<Integer, String> descriptionMap) {
724        this.addInfo(jfield, fieldInfo, this.jfieldInfos, descriptionMap);
725    }

728    private void addJCompositeIndexInfo(JCompositeIndex index, JCompositeIndexInfo indexInfo, Map<Integer, String> descriptionMap) {
729        this.addInfo(index, indexInfo, this.jcompositeIndexInfos, descriptionMap);
730    }

732    // Add new info, checking for storage ID conflicts
733    private <T> void addInfo(JSchemaObject item, T info, Map<Integer, T> infoMap, Map<Integer, String> descriptionMap) {
734        final T existing = infoMap.get(item.storageId);
735        if (existing == null) {
736            infoMap.put(item.storageId, info);
737            descriptionMap.put(item.storageId, item.description);
738        } else if (!info.equals(existing)) {
739            throw new IllegalArgumentException("incompatible duplicate use of storage ID " + item.storageId
740              + " for " + descriptionMap.get(item.storageId) + " and " + item.description);
741        }
742    }
