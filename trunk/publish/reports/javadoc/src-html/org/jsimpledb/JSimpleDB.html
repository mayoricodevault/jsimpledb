038/**
039 * Abstraction layer allowing access to a {@link Database} using normal Java objects.
040 *
041 * <p>
042 * This class provides a natural, Java-centric view on top of the core {@link Database} class. This is done via two main
043 * enhancements to the functionality provided by {@link Database}:
044 * <ul>
045 *  <li>Java annotations are used to define all {@link Database} objects and fields, to automatically
046 *      register various listeners, and to define methods that access indexed fields.</li>
047 *  <li>{@link Database} objects and fields are represented using actual Java model objects, where the Java model classes
048 *      are automatically generated subclasses of the user-supplied Java bean model classes. That is, in the view that
049 *      a {@link JSimpleDB} provides, all {@link Database} object references (which are all {@link ObjId} objects)
050 *      are replaced by normal Java model objects of the appropriate type, and all {@link Database} fields are accessible
051 *      through the corresponding model objects' Java bean getter and setter methods, instead of directly through
052 *      a {@link Transaction} object.</li>
053 * </ul>
054 * </p>
055 *
056 * <p>
057 * In addition, a {@link JSimpleDB} provides automatic incremental JSR 303 validation.
058 * </p>
059 *
060 * <p>
061 * This class guarantees that for each {@link ObjId} it will only create a single, globally unique, Java model object.
062 * Therefore, the same Java model objects can be used in and out of any transaction, and can serve as
063 * unique database object identifiers (if you have not overridden {@link #equals equals()} in your model clases).
064 * Except for their associated {@link ObjId}s, the generated Java model objects are stateless; all database field state
065 * is contained within whichever transaction is {@linkplain JTransaction#getCurrent associated with the current thread}.
066 * All generated Java model classes will implement the {@link JObject} interface.
067 * </p>
068 */
069public class JSimpleDB {

091    /**
092     * Create an instance using an initially empty, in-memory {@link SimpleKVDatabase}.
093     * This constructor can also be used just to validate the given classes.
094     *
095     * @param classes classes annotated with {@link JSimpleClass} and/or {@link JFieldType} annotations
096     * @throws IllegalArgumentException if {@code classes} is null
097     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
098     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
099     */
100    public JSimpleDB(Iterable<Class<?>> classes) {
101        this(new Database(new SimpleKVDatabase()), 1, classes);
102    }

104    /**
105     * Primary constructor.
106     *
107     * @param database core database to use
108     * @param version schema version number of the schema derived from {@code classes}
109     * @param classes classes annotated with {@link JSimpleClass} and/or {@link JFieldType} annotations
110     * @throws IllegalArgumentException if any parameter is null
111     * @throws IllegalArgumentException if {@code version} is not greater than zero
112     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
113     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
114     */
115    public JSimpleDB(Database database, int version, Iterable<Class<?>> classes) {
116
117        // Initialize
118        if (database == null)
119            throw new IllegalArgumentException("null database");
120        if (version <= 0)
121            throw new IllegalArgumentException("invalid schema version: " + version);
122        if (classes == null)
123            throw new IllegalArgumentException("null classes");
124        this.db = database;
125        this.version = version;
126
127        // Scan for annotations
128        final HashSet<Class<?>> classesSeen = new HashSet<>();
129        for (Class<?> type : classes) {
130
131            // Sanity check
132            if (type == null)
133                throw new IllegalArgumentException("null class found in classes");
134
135            // Ignore duplicates
136            this.log.debug("checking " + type + " for JSimpleDB annotations");
137            if (!classesSeen.add(type)) {
138                this.log.debug("already seen " + type + ", ignoring");
139                continue;
140            }
141
142            // Look for custom field types
143            final JFieldType fieldTypeAnnotation = type.getAnnotation(JFieldType.class);
144            if (fieldTypeAnnotation != null) {
145
146                // Get type name
147                final String name = fieldTypeAnnotation.name().length() != 0 ? fieldTypeAnnotation.name() : type.getName();
148                this.log.debug("found @" + JFieldType.class.getSimpleName()
149                  + " annotation on " + type + " defining field type `" + name + "'");
150
151                // Instantiate class
152                final Object obj;
153                try {
154                    obj = type.newInstance();
155                } catch (Exception e) {
156                    throw new DatabaseException("can't instantiate " + type, e);
157                }
158                if (!(obj instanceof FieldType)) {
159                    throw new DatabaseException("invalid @" + JFieldType.class.getSimpleName()
160                      + " annotation on " + type + ": not a subclass of " + FieldType.class);
161                }
162
163                // Register field type
164                final FieldType<?> fieldType = (FieldType<?>)obj;
165                this.db.getFieldTypeRegistry().add(fieldType);
166                this.log.debug("registered new field type `" + fieldType.getName() + "' using " + type);
167            }
168
169            // Look for Java model classes
170            final JSimpleClass jclassAnnotation = type.getAnnotation(JSimpleClass.class);
171            if (jclassAnnotation != null) {
172
173                // Sanity check type
174                if (type == null || type.isPrimitive() || type.isInterface() || type.isArray()) {
175                    throw new IllegalArgumentException("illegal type " + type + " for @"
176                      + JSimpleClass.class.getSimpleName() + " annotation: not a normal class");
177                }
178            /* ???
179                for (TypeToken<?> superType : TypeToken.of(type).getTypes()) {
180                    if (superType.getRawType().getTypeParameters().length > 0) {
181                        throw new IllegalArgumentException("illegal class " + type.getName() + " for @"
182                          + JSimpleClass.class.getSimpleName() + " annotation: class has generic supertype " + superType);
183                    }
184                }
185            */
186
187                // Create JClass
188                final String name = jclassAnnotation.name().length() != 0 ? jclassAnnotation.name() : type.getSimpleName();
189                this.log.debug("found @" + JSimpleClass.class.getSimpleName() + " annotation on " + type
190                  + " defining object type `" + name + "'");
191                JClass<?> jclass;
192                try {
193                    jclass = this.createJClass(name, jclassAnnotation.storageId(), TypeToken.of(type));
194                } catch (IllegalArgumentException e) {
195                    throw new IllegalArgumentException("invalid @" + JSimpleClass.class.getSimpleName()
196                      + " annotation on " + type + ": " + e, e);
197                }
198
199                // Add jclass
200                this.addJClass(jclass);
201                this.log.debug("added model class `" + jclass.name + "' for object type " + jclass.storageId);
202            }
203        }
204
205        // Create fields
206        for (JClass<?> jclass : this.jclasses.values())
207            jclass.createFields();
208
209        // Scan for other annotations
210        for (JClass<?> jclass : this.jclasses.values())
211            jclass.scanAnnotations();
212
213        // Index fields by storage ID
214        for (JClass<?> jclass : this.jclasses.values()) {
215            for (JField jfield : jclass.jfields.values()) {
216                this.jfields.put(jfield.storageId, jfield);
217                if (jfield instanceof JComplexField) {
218                    final JComplexField complexField = (JComplexField)jfield;
219                    for (JSimpleField subField : complexField.getSubFields())
220                        this.jfields.put(subField.storageId, subField);
221                }
222            }
223        }
224
225        // Validate schema
226        this.db.validateSchema(this.getSchemaModel());
227    }

229    // This method exists solely to bind the generic type parameters
230    private <T> JClass<T> createJClass(String name, int storageId, TypeToken<T> typeToken) {
231        return new JClass<T>(this, name, storageId, typeToken);
232    }

236    /**
237     * Create a new transaction. This does not invoke {@link JTransaction#setCurrent JTransaction.setCurrent()}: the
238     * caller is responsible for doing that if necessary.
239     *
240     * @param allowNewSchema whether creating a new schema version is allowed
241     * @param validationMode the {@link ValidationMode} to use for the new transaction
242     * @throws org.jsimpledb.InvalidSchemaException if {@code schemaModel} does not match what's recorded in the
243     *  database for the schema version provided to the constructor
244     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
245     *  is not recorded in the database and {@code allowNewSchema} is false
246     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
247     *  is not recorded in the database and {@code allowNewSchema} is true, but {@code schemaModel} is incompatible
248     *  with one or more previous schemas alread recorded in the database (i.e., the same storage ID is used
249     *  incompatibly between schema versions)
250     * @throws org.jsimpledb.InconsistentDatabaseException if inconsistent or invalid schema information is detected
251     *  in the database
252     * @throws IllegalArgumentException if {@code validationMode} is null
253     */
254    public JTransaction createTransaction(boolean allowNewSchema, ValidationMode validationMode) {
255        if (validationMode == null)
256            throw new IllegalArgumentException("null validationMode");
257        final Transaction tx = this.db.createTransaction(this.getSchemaModel(), this.version, allowNewSchema);
258        return new JTransaction(this, tx, validationMode);
259    }

263    /**
264     * Get the {@link SchemaModel} associated with this instance, derived from the annotations on the scanned classes.
265     */
266    public SchemaModel getSchemaModel() {
267        if (this.schemaModel == null) {
268            final SchemaModel model = new SchemaModel();
269            for (JClass<?> jclass : this.jclasses.values()) {
270                SchemaObject schemaObject = jclass.toSchemaItem();
271                model.getSchemaObjects().put(schemaObject.getStorageId(), schemaObject);
272            }
273            this.schemaModel = model;
274            this.log.debug("generated schema:\n{}", this.schemaModel);
275        }
276        return this.schemaModel.clone();
277    }

279    /**
280     * Get a {@link NameIndex} based on {@linkplain #getSchemaModel this instance's schema model}.
281     */
282    public NameIndex getNameIndex() {
283        if (this.nameIndex == null)
284            this.nameIndex = new NameIndex(this.getSchemaModel());
285        return this.nameIndex;

308    /**
309     * Get the {@link JClass} modeled by the given type.
310     *
311     * @param typeToken an annotated Java object model type
312     * @return associated {@link JClass}
313     * @throws IllegalArgumentException if {@code type} is not a known Java object model type
314     */
315    @SuppressWarnings("unchecked")
316    public <T> JClass<T> getJClass(TypeToken<T> typeToken) {
317        final JClass<?> jclass = this.jclassesByType.get(typeToken);
318        if (jclass == null)
319            throw new IllegalArgumentException("java model type is not recognized: " + typeToken);
320        return (JClass<T>)jclass;
321    }

323    /**
324     * Get the {@link JClass} associated with the given storage ID.
325     *
326     * @param storageId object type storage ID
327     * @return {@link JClass} instance
328     * @throws IllegalArgumentException if {@code storageId} does not represent an object type
329     */
330    public JClass<?> getJClass(int storageId) {
331        final JClass<?> jclass = this.jclasses.get(storageId);
332        if (jclass == null)
333            throw new IllegalArgumentException("no JSimpleDB class associated with storage ID " + storageId);
334        return jclass;
335    }

339    /**
340     * Get the Java object used to represent the given object ID.
341     *
342     * <p>
343     * This method guarantees that for any particular {@code id}, the same Java instance will always be returned.
344     * </p>
345     *
346     * @param id object ID
347     * @return Java model object
348     */
349    public JObject getJObject(ObjId id) {
350        Throwable cause;
351        try {
352            return this.objectMap.get(id);
353        } catch (ExecutionException e) {
354            cause = e.getCause() != null ? e.getCause() : e;
355        } catch (UncheckedExecutionException e) {
356            cause = e.getCause() != null ? e.getCause() : e;
357        }
358        if (cause instanceof Error)
359            throw (Error)cause;
360        throw new DatabaseException("can't instantiate object for ID " + id, cause);
361    }
362
363    /**
364     * Get the Java object used to represent the given object ID, cast to the given type.
365     *
366     * @param id object ID
367     * @return Java model object
368     * @see #getJObject(ObjId)
369     * @throws IllegalArgumentException if {@code type} is null
370     */
371    public <T> T getJObject(ObjId id, Class<T> type) {
372        if (type == null)
373            throw new IllegalArgumentException("null type");
374        return type.cast(this.getJObject(id));
375    }

394    /**
395     * Get the {@link JField} associated with the given storage ID.
396     *
397     * @param storageId field storage ID
398     * @param type required type, or null
399     * @return {@link JFiedl} instance
400     * @throws IllegalArgumentException if {@code storageId} does not represent a field
401     */
402    @SuppressWarnings("unchecked")
403    <T extends JField> T getJField(int storageId, Class<T> type) {
404        final JField jfield = this.jfields.get(storageId);
405        if (jfield == null)
406            throw new IllegalArgumentException("no JSimpleDB field associated with storage ID " + storageId);
407        return type != null ? type.cast(jfield) : (T)jfield;
408    }

410    // Add new JClass, checking for storage ID conflicts
411    void addJClass(JClass<?> jclass) {
412
413        // Check for storage ID conflict
414        final JClass<?> other = this.jclasses.get(jclass.storageId);
415        if (other != null) {
416            throw new IllegalArgumentException("illegal duplicate use of storage ID "
417              + jclass.storageId + " for both " + other + " and " + jclass);
418        }
419        this.jclasses.put(jclass.storageId, jclass);
420        this.jclassesByType.put(jclass.typeToken, jclass);      // this can never conflict, no need to check
421    }
