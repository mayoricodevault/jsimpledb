040/**
041 * A JSimpleDB database.
042 *
043 * <p>
044 * JSimpleDB consists of two main layers. The lower level "core" API provides a rigourous database abstraction on top of
045 * a key/value store. It supports simple fields of any atomic Java type, as well as list, set, and map complex fields,
046 * tightly controlled schema versioning, simple and composite indexes, and lifecycle and change notifications.
047 * The core API is provided via the {@link Database} class.
048 * </p>
049 *
050 * <p>
051 * This {@link JSimpleDB} class represents the upper layer. It sits on top of the core {@link Database} class and provides
052 * a fully type-safe Java view of a {@link Database}, where all access is through user-supplied Java model classes.
053 * In addition, a {@link JSimpleDB} provides automatic incremental JSR 303 validation. Database types and fields,
054 * as well as listener methods, are all declared using various {@linkplain org.jsimpledb.annotation Java annotations}.
055 * </p>
056 *
057 * <p>
058 * User-provided Java model classes are typically abstract and declare fields as abstract Java bean methods. In any case,
059 * {@link JSimpleDB} generates concrete subclasses at runtime; these runtime classes will implement the {@link JObject} interface.
060 * The corresponding Java model objects are stateless; all database field state is derived from whichever transaction is
061 * {@linkplain JTransaction#getCurrent associated with the current thread}.
062 * </p>
063 *
064 * <p>
065 * Not counting "snapshot" objects (see below), this class guarantees that for each {@link ObjId} there will only ever be
066 * a single, globally unique, {@link JObject} Java model object. Therefore, the same Java model objects can be used in and
067 * out of any transaction.
068 * </p>
069 *
070 * <p>
071 * "Snapshot" objects are in-memory copies of regular database objects that may be imported/exported to/from transactions.
072 * Snapshot {@link JObject}s are distinct from regular database {@link JObject}s; their state is contained in an in-memory
073 * {@link SnapshotJTransaction}. See {@link JObject#copyOut JObject.copyOut} and {@link JObject#copyIn JObject.copyIn}.
074 * </p>
075 *
076 * @see JObject
077 * @see JTransaction
078 * @see JSimpleDBFactory
079 * @see org.jsimpledb.annotation
080 */
081public class JSimpleDB {

144    /**
145     * Primary constructor.
146     *
147     * @param database core database to use
148     * @param version schema version number of the schema derived from {@code classes},
149     *  or zero to use the highest version already recorded in the database
150     * @param storageIdGenerator generator for auto-generated storage ID's, or null to disallow auto-generation of storage ID's
151     * @param classes classes annotated with {@link JSimpleClass &#64;JSimpleClass} annotations; non-annotated classes are ignored
152     * @throws IllegalArgumentException if {@code database} or {@code classes} is null
153     * @throws IllegalArgumentException if {@code version} is not greater than zero
154     * @throws IllegalArgumentException if {@code classes} contains a null class or a class with invalid annotation(s)
155     * @throws InvalidSchemaException if the schema implied by {@code classes} is invalid
156     */
157    JSimpleDB(Database database, int version, StorageIdGenerator storageIdGenerator, Iterable<? extends Class<?>> classes) {
158
159        // Initialize
160        if (database == null)
161            throw new IllegalArgumentException("null database");
162        if (version < 0)
163            throw new IllegalArgumentException("invalid schema version: " + version);
164        if (classes == null)
165            throw new IllegalArgumentException("null classes");
166        this.db = database;
167        this.configuredVersion = version;
168        this.storageIdGenerator = storageIdGenerator;
169
170        // Inventory classes; automatically add all @JSimpleClass-annotated superclasses of @JSimpleClass-annotated classes
171        final HashSet<Class<?>> jsimpleClasses = new HashSet<>();
172        for (Class<?> type : classes) {
173
174            // Sanity check
175            if (type == null)
176                throw new IllegalArgumentException("null class found in classes");
177
178            // Add type and all @JSimpleClass-annotated superclasses
179            do {
180
181                // Find annotation
182                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
183                if (annotation == null)
184                    continue;
185
186                // Sanity check type
187                if (type.isPrimitive() || type.isInterface() || type.isArray()) {
188                    throw new IllegalArgumentException("illegal type " + type + " for @"
189                      + JSimpleClass.class.getSimpleName() + " annotation: not a normal class");
190                }
191
192                // Add class
193                jsimpleClasses.add(type);
194            } while ((type = type.getSuperclass()) != null);
195        }
196
197        // Add Java model classes
198        for (Class<?> type : jsimpleClasses) {
199
200            // Create JClass
201            final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
202            final String name = annotation.name().length() != 0 ? annotation.name() : type.getSimpleName();
203            if (this.log.isTraceEnabled()) {
204                this.log.trace("found @" + JSimpleClass.class.getSimpleName() + " annotation on " + type
205                  + " defining object type `" + name + "'");
206            }
207
208            // Get storage ID
209            int storageId = annotation.storageId();
210            if (storageId == 0)
211                storageId = this.getStorageIdGenerator(annotation, type).generateClassStorageId(type, name);
212
213            // Create JClass
214            JClass<?> jclass;
215            try {
216                jclass = this.createJClass(name, storageId, type);
217            } catch (IllegalArgumentException e) {
218                throw new IllegalArgumentException("invalid @" + JSimpleClass.class.getSimpleName()
219                  + " annotation on " + type + ": " + e, e);
220            }
221
222            // Add jclass
223            this.addJClass(jclass);
224            this.log.debug("added Java model class `" + jclass.name + "' with storage ID " + jclass.storageId);
225        }
226
227        // Inventory class generators
228        this.classGenerators = new ArrayList<>(this.jclasses.size() + 1);
229        for (JClass<?> jclass : this.jclasses.values())
230            this.classGenerators.add(jclass.classGenerator);
231        this.untypedClassGenerator = new ClassGenerator<UntypedJObject>(this, UntypedJObject.class);
232        this.classGenerators.add(this.untypedClassGenerator);
233
234        // Create fields
235        for (JClass<?> jclass : this.jclasses.values())
236            jclass.createFields(this);
237
238        // Create canonical field info structures
239        final HashMap<Integer, String> fieldDescriptionMap = new HashMap<>();
240        for (JClass<?> jclass : this.jclasses.values()) {
241            for (JField jfield : jclass.jfields.values()) {
242                if (jfield instanceof JComplexField) {
243                    final JComplexField complexField = (JComplexField)jfield;
244                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)jfield.toJFieldInfo();
245                    for (JSimpleField subField : complexField.getSubFields()) {
246                        this.addJFieldInfo(subField, subField.toJFieldInfo(jfield.storageId), fieldDescriptionMap);
247                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
248                        complexFieldInfo.getSubFieldInfos().add(subFieldInfo);
249                    }
250                    this.addJFieldInfo(complexField, complexFieldInfo, fieldDescriptionMap);
251                } else
252                    this.addJFieldInfo(jfield, jfield.toJFieldInfo(), fieldDescriptionMap);
253            }
254        }
255
256        // Witness all simple fields to corresponding simple field info's
257        for (JClass<?> jclass : this.jclasses.values()) {
258            for (JField jfield : jclass.jfields.values()) {
259                if (jfield instanceof JSimpleField) {
260                    final JSimpleField jsimpleField = (JSimpleField)jfield;
261                    final JSimpleFieldInfo jsimpleFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
262                    jsimpleFieldInfo.witness(jsimpleField);
263                }
264                if (jfield instanceof JComplexField) {
265                    final JComplexField complexField = (JComplexField)jfield;
266                    final JComplexFieldInfo complexFieldInfo = (JComplexFieldInfo)this.jfieldInfos.get(jfield.storageId);
267                    for (int i = 0; i < complexField.getSubFields().size(); i++) {
268                        final JSimpleField subField = complexField.getSubFields().get(i);
269                        final JSimpleFieldInfo subFieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(subField.storageId);
270                        subFieldInfo.witness(subField);
271                    }
272                }
273            }
274        }
275
276        // Add composite indexes to class; like fields, indexes are inherited (duplicated) from superclasses
277        for (JClass<?> jclass : this.jclasses.values()) {
278            for (Class<?> type = jclass.type; type != null; type = type.getSuperclass()) {
279                final JSimpleClass annotation = type.getAnnotation(JSimpleClass.class);
280                if (annotation != null) {
281                    for (org.jsimpledb.annotation.JCompositeIndex indexAnnotation : annotation.compositeIndexes())
282                        jclass.addCompositeIndex(this, indexAnnotation);
283                }
284            }
285        }
286
287        // Create canonical info instances for indexes
288        final HashMap<Integer, String> indexDescriptionMap = new HashMap<>();
289        for (JClass<?> jclass : this.jclasses.values()) {
290            for (JCompositeIndex index : jclass.jcompositeIndexes.values()) {
291                final JCompositeIndexInfo indexInfo = index.toJCompositeIndexInfo();
292                for (JSimpleField jfield : index.jfields) {
293                    final JSimpleFieldInfo jfieldInfo = (JSimpleFieldInfo)this.jfieldInfos.get(jfield.storageId);
294                    indexInfo.getJFieldInfos().add(jfieldInfo);
295                }
296                this.addJCompositeIndexInfo(index, indexInfo, indexDescriptionMap);
297            }
298        }
299
300        // Scan for other method-level annotations
301        for (JClass<?> jclass : this.jclasses.values())
302            jclass.scanAnnotations();
303
304        // Determine which JClass's have validation requirement(s) on creation
305        for (JClass<?> jclass : this.jclasses.values())
306            jclass.calculateValidationRequirement();
307        boolean anyValidation = false;
308        for (JClass<?> jclass : this.jclasses.values()) {
309            if ((anyValidation |= jclass.requiresValidation))
310                break;
311        }
312        this.anyJClassRequiresValidation = anyValidation;
313
314        // Detect whether we have any @OnCreate, @OnDelete, and/or @OnVersionChange methods
315        boolean anyOnCreateMethods = false;
316        boolean anyOnDeleteMethods = false;
317        boolean anyOnVersionChangeMethods = false;
318        for (JClass<?> jclass : this.jclasses.values()) {
319            anyOnCreateMethods |= !jclass.onCreateMethods.isEmpty();
320            anyOnDeleteMethods |= !jclass.onDeleteMethods.isEmpty();
321            anyOnVersionChangeMethods |= !jclass.onVersionChangeMethods.isEmpty();
322        }
323        this.hasOnCreateMethods = anyOnCreateMethods;
324        this.hasOnDeleteMethods = anyOnDeleteMethods;
325        this.hasOnVersionChangeMethods = anyOnVersionChangeMethods;
326
327        // Validate schema
328        this.db.validateSchema(this.getSchemaModel());
329    }

338            throw new IllegalArgumentException("invalid @" + annotation.annotationType().getSimpleName()

342        return this.storageIdGenerator;

390    /**
391     * Get the Java model class of the given {@link JObject}.
392     * If {@code jobj} is an instance of a JSimpleDB-generated subclass of a user-supplied Java model class,
393     * this returns the original Java model class. Otherwise, it returns {@code obj}'s type.
394     *
395     * @param jobj database instance
396     * @return lowest ancestor class of {@code jobj}'s class that is not a JSimpleDB-generated subclass
397     * @throws IllegalArgumentException if {@code jobj} is null
398     */
399    public static Class<?> getModelClass(JObject jobj) {
400        if (jobj == null)
401            throw new IllegalArgumentException("null jobj");
402        for (Class<?> type = jobj.getClass(); type != null; type = type.getSuperclass()) {
403            if (type.getName().indexOf(ClassGenerator.CLASSNAME_SUFFIX) == -1)
404                return type;
405        }
406        return null;
407    }

411    /**
412     * Create a new transaction. This does not invoke {@link JTransaction#setCurrent JTransaction.setCurrent()}: the
413     * caller is responsible for doing that if necessary.
414     *
415     * @param allowNewSchema whether creating a new schema version is allowed
416     * @param validationMode the {@link ValidationMode} to use for the new transaction
417     * @throws org.jsimpledb.InvalidSchemaException if {@code schemaModel} does not match what's recorded in the
418     *  database for the schema version provided to the constructor
419     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
420     *  is not recorded in the database and {@code allowNewSchema} is false
421     * @throws org.jsimpledb.InvalidSchemaException if the schema version provided to the constructor
422     *  is not recorded in the database and {@code allowNewSchema} is true, but {@code schemaModel} is incompatible
423     *  with one or more previous schemas alread recorded in the database (i.e., the same storage ID is used
424     *  incompatibly between schema versions)
425     * @throws org.jsimpledb.InconsistentDatabaseException if inconsistent or invalid schema information is detected
426     *  in the database
427     * @throws IllegalArgumentException if {@code validationMode} is null
428     */
429    public JTransaction createTransaction(boolean allowNewSchema, ValidationMode validationMode) {
430        if (validationMode == null)
431            throw new IllegalArgumentException("null validationMode");
432        final Transaction tx = this.db.createTransaction(this.getSchemaModel(), this.configuredVersion, allowNewSchema);
433        this.actualVersion = tx.getSchema().getSchemaVersions().lastKey();
434        return new JTransaction(this, tx, validationMode);
435    }

439    /**
440     * Get the {@link SchemaModel} associated with this instance, derived from the annotations on the scanned classes.
441     */
442    public SchemaModel getSchemaModel() {
443        if (this.schemaModel == null) {
444            final SchemaModel model = new SchemaModel();
445            for (JClass<?> jclass : this.jclasses.values()) {
446                final SchemaObjectType schemaObjectType = jclass.toSchemaItem(this);
447                model.getSchemaObjectTypes().put(schemaObjectType.getStorageId(), schemaObjectType);
448            }
449            this.schemaModel = model;
450            this.log.debug("JSimpleDB schema generated from annotated classes:\n{}", this.schemaModel);
451        }
452        return this.schemaModel.clone();
453    }

537    public <T> List<JClass<? extends T>> getJClasses(Class<T> type) {
538        final ArrayList<JClass<? extends T>> list = new ArrayList<>();
539        for (JClass<?> jclass : this.jclasses.values()) {
540            if (type == null || type.isAssignableFrom(jclass.type))
541                list.add((JClass<? extends T>)jclass);
542        }
543        return list;

548     *
549     * @param type any Java type, or null for no restriction
550     * @return key restriction for {@code type}
551     */
552    KeyRanges keyRangesFor(Class<?> type) {
553        if (type == null)
554            return KeyRanges.FULL;
555        final ArrayList<KeyRange> list = new ArrayList<>(this.jclasses.size());

563    /**
564     * Get the Java model object that is has the given ID. This returns the object associated with normal
565     * (i.e., non-snapshot) {@link JTransaction}s. To get the object associated with a snapshot transaction,
566     * invoke {@link JTransaction#getJObject(ObjId) JTransaction.getJObject()} on the snapshot transaction.
567     *
568     * <p>
569     * This method guarantees that for any particular {@code id}, the same Java instance will always be returned.
570     * Note: while for any {@link ObjId} there is only one globally unique {@link JObject} per {@link JSimpleDB}
571     * shared by all {@link JTransaction}s (the object returned by this method), each {@link SnapshotJTransaction}
572     * maintains its own distinct pool of unique "snapshot" {@link JObject}s.
573     * </p>
574     *
575     * <p>
576     * A non-null object is always returned, but the corresponding object may not exist in a given transaction.
577     * In that case, attempts to access its fields will throw {@link org.jsimpledb.core.DeletedObjectException}.
578     * Also, it's possible that {@code id} corresponds to an object type which no longer exists in the schema
579     * version associated with this instance. In that case, an {@link UntypedJObject} is returned.
580     * </p>
581     *
582     * @param id object ID
583     * @return Java model object
584     * @throws IllegalArgumentException if {@code id} is null
585     * @see JTransaction#getJObject JTransaction.getJObject()
586     */
587    public JObject getJObject(ObjId id) {
588        return this.jobjectCache.getJObject(id);
589    }

591    /**
592     * Get the Java object used to represent the given object ID, cast to the given type.
593     * This method just invoke {@link #getJObject(ObjId)} and then casts the result.
594     *
595     * @param id object ID
596     * @param type expected type
597     * @return Java model object
598     * @see #getJObject(ObjId)
599     * @throws ClassCastException if the Java model object does not have type {@code type}

629    /**
630     * Utility method to get all of the objects directly referenced by a given object via any field.
631     *
632     * <p>
633     * Note: the returned {@link Iterable} may contain duplicates; these can be eliminated using an
634     * {@link org.jsimpledb.core.ObjIdSet} if necessary.
635     * </p>
636     *
637     * @param jobj starting object
638     * @return all objects directly referenced by {@code jobj}
639     * @throws IllegalArgumentException if {@code jobj} is null
640     */
641    public Iterable<JObject> getReferencedObjects(final JObject jobj) {
642
643        // Sanity check
644        if (jobj == null)
645            throw new IllegalArgumentException("null jobj");
646        final ObjId id = jobj.getObjId();
647
648        // Visit fields
649        final ArrayList<Iterable<JObject>> iterables = new ArrayList<>();
650        for (JField jfield : this.getJClass(id).getJFieldsByStorageId().values()) {
651            jfield.visit(new JFieldSwitchAdapter<Void>() {
652
653                @Override
654                public Void caseJReferenceField(JReferenceField field) {
655                    final JObject ref = field.getValue(jobj);
656                    if (ref != null)
657                        iterables.add(Collections.singleton(ref));
658                    return null;
659                }
660
661                @Override
662                public Void caseJMapField(JMapField field) {
663                    if (field.getKeyField() instanceof JReferenceField)
664                        iterables.add(Iterables.filter(field.getValue(jobj).keySet(), JObject.class));
665                    if (field.getValueField() instanceof JReferenceField)
666                        iterables.add(Iterables.filter(field.getValue(jobj).values(), JObject.class));
667                    return null;
668                }
669
670                @Override
671                protected Void caseJCollectionField(JCollectionField field) {
672                    if (field.getElementField() instanceof JReferenceField)
673                        iterables.add(Iterables.filter(field.getValue(jobj), JObject.class));
674                    return null;
675                }
676
677                @Override
678                protected Void caseJField(JField field) {
679                    return null;
680                }
681            });
682        }
683
684        // Done
685        return Iterables.concat(iterables);
686    }

695    /**
696     * Get the {@link JFieldInfo} associated with the given storage ID.
697     *
698     * @param storageId field storage ID
699     * @param type required type
700     * @return {@link JField} instance
701     * @throws UnknownFieldException if {@code storageId} does not represent a field
702     */
703    <T extends JFieldInfo> T getJFieldInfo(int storageId, Class<T> type) {
704        if (type == null)
705            throw new IllegalArgumentException("null type");
706        final JFieldInfo jfieldInfo = this.jfieldInfos.get(storageId);
707        if (jfieldInfo == null) {
708            throw new UnknownFieldException(storageId, "no JSimpleDB field exists with storage ID "
709              + storageId + " in schema version " + this.actualVersion);
710        }
711        try {
712            return type.cast(jfieldInfo);
713        } catch (ClassCastException e) {
714            throw new UnknownFieldException(storageId, "no JSimpleDB fields exist with storage ID "
715              + storageId + " in schema version " + this.actualVersion + " (found field " + jfieldInfo + " instead)");
716        }
717    }

719    // Add new JClass, checking for storage ID conflicts
720    private void addJClass(JClass<?> jclass) {
721
722        // Check for storage ID conflict
723        final JClass<?> other = this.jclasses.get(jclass.storageId);
724        if (other != null) {
725            throw new IllegalArgumentException("illegal duplicate use of storage ID "
726              + jclass.storageId + " for both " + other + " and " + jclass);
727        }
728        this.jclasses.put(jclass.storageId, jclass);
729        assert !this.jclassesByType.containsKey(jclass.type);                   // this should never conflict, no need to check
730        this.jclassesByType.put(jclass.type, jclass);
731    }
732
733    // Add new JFieldInfo, checking for conflicts
734    private <T extends JFieldInfo> void addJFieldInfo(JField jfield, T fieldInfo, Map<Integer, String> descriptionMap) {
735        this.addInfo(jfield, fieldInfo, this.jfieldInfos, descriptionMap);
736    }
737
738    // Add new JCompositeIndexInfo, checking for conflicts
739    private void addJCompositeIndexInfo(JCompositeIndex index, JCompositeIndexInfo indexInfo, Map<Integer, String> descriptionMap) {
740        this.addInfo(index, indexInfo, this.jcompositeIndexInfos, descriptionMap);
741    }
742
743    // Add new info, checking for storage ID conflicts
744    private <T> void addInfo(JSchemaObject item, T info, Map<Integer, T> infoMap, Map<Integer, String> descriptionMap) {
745        final T existing = infoMap.get(item.storageId);
746        if (existing == null) {
747            infoMap.put(item.storageId, info);
748            descriptionMap.put(item.storageId, item.description);
749        } else if (!info.equals(existing)) {
750            throw new IllegalArgumentException("incompatible duplicate use of storage ID " + item.storageId
751              + " for " + descriptionMap.get(item.storageId) + " and " + item.description);
752        }
753    }

755// Loader
756
757    private class Loader extends ClassLoader {
758
759        // Set up class loader
760        public Loader() {
761            super(Thread.currentThread().getContextClassLoader());
762        }
763
764        // Find matching ClassGenerator, if any, otherwise defer to parent
765        @Override
766        protected Class<?> findClass(String name) throws ClassNotFoundException {
767            byte[] bytes = null;
768            for (ClassGenerator<?> generator : JSimpleDB.this.classGenerators) {
769                if (name.equals(generator.getClassName().replace('/', '.'))) {
770                    bytes = generator.generateBytecode();
771                    break;
772                }
773                if (name.equals(generator.getSnapshotClassName().replace('/', '.'))) {
774                    bytes = generator.generateSnapshotBytecode();
775                    break;
776                }
777            }
778            return bytes != null ? this.defineClass(name, bytes, 0, bytes.length) : super.findClass(name);
779        }
780    }
