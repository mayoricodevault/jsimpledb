008package org.jsimpledb;
009
010import org.jsimpledb.core.ObjId;
011
012/**
013 * Interface implemented by {@link JSimpleDB} Java model objects.
014 *
015 * <p>
016 * {@link JSimpleDB} automatically generates sub-classes of user-supplied Java model classes that implement this interface.
017 * There are two types of {@link JObject}s: regular instances that reflect the state of the {@link JTransaction}
018 * {@linkplain JTransaction#getCurrent associated} with the current thread, and "snapshot" instances that reflect
019 * the state of their associated {@link SnapshotJTransaction}.
020 * Therefore every {@link JObject} instance has an {@linkplain #getTransaction associated transaction}
021 * which, in the case of regular database objects only, may not exist.
022 * </p>
023 */
024public interface JObject {
025
026    /**
027     * Get this instance's unique JSimpleDB object identifier.
028     *
029     * <p>
030     * This method always succeeds.
031     * </p>
032     *
033     * @return unique database identifier for this instance
034     */
035    ObjId getObjId();
036
037    /**
038     * Get this instance's associated {@link JTransaction}.
039     *
040     * <p>
041     * If this is a regular database instance, this returns the {@link JTransaction}
042     * {@linkplain JTransaction#getCurrent associated} with the current thread. Otherwise, this instance
043     * is a snapshot instance and this method returns the associated {@link SnapshotJTransaction}.
044     * </p>
045     *
046     * @return the {@link JTransaction} that contains this instance's field state
047     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
048     *  associated with the current thread
049     */
050    JTransaction getTransaction();
051
052    /**
053     * Delete this instance, if it exists, in this instance's associated transaction.
054     *
055     * @return true if instance was deleted, false if it did not exist
056     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
057     *  associated with the current thread
058     * @throws org.jsimpledb.core.StaleTransactionException
059     *  if the transaction associated with the current thread is no longer usable
060     */
061    boolean delete();
062
063    /**
064     * Determine whether this instance still exists in its associated transaction.
065     *
066     * @return true if instance exists, otherwise false
067     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
068     *  associated with the current thread
069     * @throws org.jsimpledb.core.StaleTransactionException
070     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
071     */
072    boolean exists();
073
074    /**
075     * Recreate a deleted instance, if it does not exist, in its associated transaction.
076     * The fields of a recreated object are set to their initial values.
077     *
078     * @return true if instance was recreated, false if it already existed
079     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
080     *  associated with the current thread
081     * @throws org.jsimpledb.core.StaleTransactionException
082     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
083     */
084    boolean recreate();
085
086    /**
087     * Add this instance to the validation queue for validation in its associated transaction.
088     *
089     * <p>
090     * The actual validation will occur either during {@link JTransaction#commit}
091     * or at the next invocation of {@link JTransaction#validate}, whichever occurs first.
092     * </p>
093     *
094     * @throws org.jsimpledb.core.DeletedObjectException
095     *  if this object does not exist in the {@link JTransaction} associated with this instance
096     * @throws IllegalStateException if transaction commit is already in progress
097     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
098     *  associated with the current thread
099     * @throws org.jsimpledb.core.StaleTransactionException
100     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
101     */
102    void revalidate();
103
104    /**
105     * Update the schema version of this instance, if necessary, so that it matches the schema version
106     * of its associated transaction.
107     *
108     * <p>
109     * If a version change occurs, matching {@link org.jsimpledb.annotation.OnVersionChange &#64;OnVersionChange}
110     * methods will be invoked prior to this method returning.
111     * </p>
112     *
113     * @throws org.jsimpledb.core.DeletedObjectException
114     *  if this object does not exist in the {@link JTransaction} associated with this instance
115     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
116     *  associated with the current thread
117     * @throws org.jsimpledb.core.StaleTransactionException
118     *  if this is not a snapshot instance and the transaction associated with the current thread is no longer usable
119     */
120    void upgrade();
121
122    /**
123     * Copy this instance, and other instances it references, into another {@link JTransaction}.
124     *
125     * <p>
126     * This method will copy this object and all of its fields, along all other objects reachable through
127     * any of the specified {@linkplain ReferencePath reference paths}, into the specified {@link JTransaction}.
128     * If any object already exists in {@code dest}, it will be overwritten.
129     * </p>
130     *
131     * @param dest destination transaction for copies
132     * @param refPaths zero or more reference paths that refer to additional objects to be copied
133     * @return the copied version of this instance
134     * @throws org.jsimpledb.core.DeletedObjectException
135     *  if this object does not exist in the {@link JTransaction} associated with this instance
136     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
137     *  associated with the current thread
138     * @throws org.jsimpledb.core.SchemaMismatchException
139     *  if the schema corresponding to this object's version is not identical in both transactions
140     * @throws IllegalArgumentException if {@code dest} or {@code refPaths} is null
141     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
142     * @see #copyIn copyIn()
143     * @see #copyOut copyOut()
144     */
145    JObject copyTo(JTransaction dest, String... refPaths);
146
147    /**
148     * Snapshot this instance and other instances it references.
149     *
150     * <p>
151     * This method will copy this object and all of its fields, along all other objects reachable through
152     * any of the specified {@linkplain ReferencePath reference paths}, into the {@link SnapshotJTransaction}
153     * {@linkplain JTransaction#getSnapshotTransaction associated} with this instance's associated transaction.
154     * If any object already exists there, it will be overwritten.
155     * </p>
156     *
157     * <p>
158     * Normally this method would only be invoked on a regular database {@link JObject}.
159     * The returned object will always be a snapshot {@link JObject}.
160     * </p>
161     *
162     * <p>
163     * This is a convenience method, equivalent to:
164     *  <blockquote><code>
165     *  copyTo(JTransaction.getCurrent().getSnapshotTransaction(), refPaths);
166     *  </code></blockquote>
167     * </p>
168     *
169     * @param refPaths zero or more reference paths that refer to additional objects to be copied
170     * @return the snapshot {@link JObject} copy of this instance
171     * @throws org.jsimpledb.core.DeletedObjectException
172     *  if this object does not exist in the {@link JTransaction} associated with this instance
173     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
174     *  associated with the current thread
175     * @throws IllegalArgumentException if {@code refPaths} is null
176     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
177     * @see #copyIn copyIn()
178     */
179    JObject copyOut(String... refPaths);
180
181    /**
182     * Copy this instance and other instances it references into the transaction associated with the current thread.
183     *
184     * <p>
185     * This method will copy this object and all of its fields, along all other objects reachable through
186     * any of the specified {@linkplain ReferencePath reference paths}, into the {@link JTransaction}
187     * {@linkplain JTransaction#getCurrent associated} with the current thread.
188     * If any object already exists in the current thread's transaction, it will be overwritten.
189     * </p>
190     *
191     * <p>
192     * Normally this method would only be invoked on a snapshot {@link JObject}.
193     * The returned object will always be a regular database {@link JObject}.
194     * </p>
195     *
196     * <p>
197     * This is a convenience method, equivalent to {@code copyIn(JTransaction.getCurrent(), refPaths)}, but
198     * returning the narrower type {@link JObject}.
199     * </p>
200     *
201     * @param refPaths zero or more reference paths that refer to additional objects to be copied
202     * @return the regular database copy of this instance
203     * @throws org.jsimpledb.core.DeletedObjectException
204     *  if this object does not exist in the {@link JTransaction} associated with this instance
205     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
206     *  associated with the current thread
207     * @throws org.jsimpledb.core.SchemaMismatchException
208     *  if the schema corresponding to this object's version is not identical in both transactions
209     * @throws IllegalArgumentException if {@code refPaths} is null
210     * @throws IllegalArgumentException if any path in {@code refPaths} is invalid
211     * @see #copyOut copyOut()
212     */
213    JObject copyIn(String... refPaths);
214}
