008package org.jsimpledb;
009
010import org.jsimpledb.core.ObjId;
011
012/**
013 * Interface implemented by {@link JSimpleDB} Java model objects.
014 *
015 * <p>
016 * {@link JSimpleDB} automatically generates sub-classes of user-supplied Java model classes.
017 * These sub-classes know how to access database fields in the transaction associated with the current thread.
018 * In addition, these sub-classes implement this interface.
019 * </p>
020 */
021public interface JObject {
022
023    /**
024     * Get this instance's unique JSimpleDB object identifier.
025     *
026     * <p>
027     * This method will always succeed even if there is no transaction associated with the current thread.
028     * </p>
029     *
030     * @return unique database identifier for this instance
031     */
032    ObjId getObjId();
033
034    /**
035     * Delete this instance, if it exists, in
036     * {@linkplain JTransaction#getCurrent the transaction associated with the current thread}.
037     *
038     * @return true if instance was deleted, false if it did not exist
039     * @throws IllegalStateException if there is no {@link JTransaction} associated with the current thread
040     * @throws org.jsimpledb.StaleTransactionException
041     *  if the transaction associated with the current thread is no longer usable
042     */
043    boolean delete();
044
045    /**
046     * Determine whether this instance still exists in
047     * {@linkplain JTransaction#getCurrent the transaction associated with the current thread}.
048     *
049     * @return true if instance exists, otherwise false
050     * @throws IllegalStateException if there is no {@link JTransaction} associated with the current thread
051     * @throws org.jsimpledb.StaleTransactionException
052     *  if the transaction associated with the current thread is no longer usable
053     */
054    boolean exists();
055
056    /**
057     * Add this instance to the validation queue for validation in
058     * {@linkplain JTransaction#getCurrent the transaction associated with the current thread}.
059     *
060     * <p>
061     * The actual validation will occur either during {@link JTransaction#commit}
062     * or at the next invocation of {@link JTransaction#validate}, whichever occurs first.
063     * </p>
064     *
065     * @throws IllegalStateException if transaction commit is already in progress
066     * @throws IllegalStateException if there is no {@link JTransaction} associated with the current thread
067     * @throws org.jsimpledb.StaleTransactionException
068     *  if the transaction associated with the current thread is no longer usable
069     */
070    void revalidate();
071}
