008package org.jsimpledb;
009
010import org.jsimpledb.core.ObjId;
011import org.jsimpledb.core.Transaction;
012
013/**
014 * Represents a 64-bit counter value that can be adjusted concurrently by multiple transactions without locking.
015 * Supported by {@link org.jsimpledb.kv.KVDatabase}s whose {@linkplain org.jsimpledb.kv.KVDatabase#createTransaction transactions}
016 * implement the {@link org.jsimpledb.kv.CountingKVStore} interface.
017 *
018 * <p>
019 * To define a field of type {@link Counter}, annotate the field's getter method as a normal field using
020 * {@link org.jsimpledb.annotation.JField &#64;JField}. No setter method should be defined.
021 * </p>
022 *
023 * <p>
024 * If you access a {@link Counter} field while running on a non-supporting database,
025 * an {@link UnsupportedOperationException} will be thrown.
026 * </p>
027 *
028 * <p>
029 * During version change notification, counter fields appear as plain {@code long} values.
030 * </p>
031 */
032public class Counter {
033
034    private final Transaction tx;
035    private final ObjId id;
036    private final int storageId;
037
038    Counter(Transaction tx, ObjId id, int storageId) {
039        if (tx == null)
040            throw new IllegalArgumentException("null tx");
041        if (id == null)
042            throw new IllegalArgumentException("null id");
043        this.tx = tx;
044        this.id = id;
045        this.storageId = storageId;
046    }
047
048    /**
049     * Read the counter's current value. Invoking this method will typically disable the lock-free
050     * behavior of {@link #adjust adjust()} in the current transaction.
051     *
052     * @return current value of the counter
053     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
054     *  was read is no longer usable
055     * @throws DeletedObjectException if the object from which this instance was read no longer exists
056     */
057    public long get() {
058        return this.tx.readCounterField(this.id, this.storageId);
059    }
060
061    /**
062     * Set the counter's value. Invoking this method will typically disable the lock-free
063     * behavior of {@link #adjust adjust()} in the current transaction.
064     *
065     * @param value new value for the counter
066     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
067     *  was read is no longer usable
068     * @throws DeletedObjectException if the object from which this instance was read no longer exists
069     */
070    public void set(long value) {
071        this.tx.writeCounterField(this.id, this.storageId, value);
072    }
073
074    /**
075     * Adjust the counter's value by the specified amount.
076     *
077     * @param offset amount to add to counter
078     * @throws org.jsimpledb.kv.StaleTransactionException if the transaction from which this instance
079     *  was read is no longer usable
080     * @throws DeletedObjectException if the object from which this instance was read no longer exists
081     */
082    public void adjust(long offset) {
083        this.tx.adjustCounterField(this.id, this.storageId, offset);
084    }
085}
