008package org.jsimpledb.change;
009
010/**
011 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
012 * when an element is added to a set field.
013 *
014 * @param <T> the type of the object containing the changed field
015 * @param <E> the type of the changed set's elements
016 */
017public class SetFieldAdd<T, E> extends SetFieldChange<T> {
018
019    private final E element;
020
021    /**
022     * Constructor.
023     *
024     * @param jobj Java object containing the set field that changed
025     * @param element the element that was added
026     * @throws IllegalArgumentException if {@code jobj} is null
027     */
028    public SetFieldAdd(T jobj, E element) {
029        super(jobj);
030        this.element = element;
031    }
032
033    /**
034     * Get the element that was added to the set.
035     */
036    public E getElement() {
037        return this.element;
038    }
039
040    @Override
041    public boolean equals(Object obj) {
042        if (obj == this)
043            return true;
044        if (!super.equals(obj))
045            return false;
046        final SetFieldAdd<?, ?> that = (SetFieldAdd<?, ?>)obj;
047        return this.element != null ? this.element.equals(that.element) : that.element == null;
048    }
049
050    @Override
051    public int hashCode() {
052        return super.hashCode() ^ (this.element != null ? this.element.hashCode() : 0);
053    }
054
055    @Override
056    public String toString() {
057        return "SetFieldAdd[object=" + this.getObject() + ",element=" + this.element + "]";
058    }
059}
