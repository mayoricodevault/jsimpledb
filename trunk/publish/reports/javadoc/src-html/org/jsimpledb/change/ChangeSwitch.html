008package org.jsimpledb.change;
009
010/**
011 * Visitor pattern interface for the {@link Change} class hierarchy.
012 *
013 * @param <R> method return type
014 * @see Change#visit
015 */
016public interface ChangeSwitch<R> {
017
018    /**
019     * Handle an {@link ObjectCreate} event.
020     *
021     * @param change visiting change
022     * @param <T> changed object type
023     * @return visitor return value
024     */
025    <T> R caseObjectCreate(ObjectCreate<T> change);
026
027    /**
028     * Handle an {@link ObjectDelete} event.
029     *
030     * @param change visiting change
031     * @param <T> changed object type
032     * @return visitor return value
033     */
034    <T> R caseObjectDelete(ObjectDelete<T> change);
035
036    /**
037     * Handle a {@link ListFieldAdd} event.
038     *
039     * @param change visiting change
040     * @param <T> changed object type
041     * @param <E> changed list element type
042     * @return visitor return value
043     */
044    <T, E> R caseListFieldAdd(ListFieldAdd<T, E> change);
045
046    /**
047     * Handle a {@link ListFieldClear} event.
048     *
049     * @param change visiting change
050     * @param <T> changed object type
051     * @return visitor return value
052     */
053    <T> R caseListFieldClear(ListFieldClear<T> change);
054
055    /**
056     * Handle a {@link ListFieldRemove} event.
057     *
058     * @param change visiting change
059     * @param <T> changed object type
060     * @param <E> changed list element type
061     * @return visitor return value
062     */
063    <T, E> R caseListFieldRemove(ListFieldRemove<T, E> change);
064
065    /**
066     * Handle a {@link ListFieldReplace} event.
067     *
068     * @param change visiting change
069     * @param <T> changed object type
070     * @param <E> changed list element type
071     * @return visitor return value
072     */
073    <T, E> R caseListFieldReplace(ListFieldReplace<T, E> change);
074
075    /**
076     * Handle a {@link MapFieldAdd} event.
077     *
078     * @param change visiting change
079     * @param <T> changed object type
080     * @param <K> changed map key type
081     * @param <V> changed map value type
082     * @return visitor return value
083     */
084    <T, K, V> R caseMapFieldAdd(MapFieldAdd<T, K, V> change);
085
086    /**
087     * Handle a {@link MapFieldClear} event.
088     *
089     * @param change visiting change
090     * @param <T> changed object type
091     * @return visitor return value
092     */
093    <T> R caseMapFieldClear(MapFieldClear<T> change);
094
095    /**
096     * Handle a {@link MapFieldRemove} event.
097     *
098     * @param change visiting change
099     * @param <T> changed object type
100     * @param <K> changed map key type
101     * @param <V> changed map value type
102     * @return visitor return value
103     */
104    <T, K, V> R caseMapFieldRemove(MapFieldRemove<T, K, V> change);
105
106    /**
107     * Handle a {@link MapFieldReplace} event.
108     *
109     * @param change visiting change
110     * @param <T> changed object type
111     * @param <K> changed map key type
112     * @param <V> changed map value type
113     * @return visitor return value
114     */
115    <T, K, V> R caseMapFieldReplace(MapFieldReplace<T, K, V> change);
116
117    /**
118     * Handle a {@link SetFieldAdd} event.
119     *
120     * @param change visiting change
121     * @param <T> changed object type
122     * @param <E> changed set element type
123     * @return visitor return value
124     */
125    <T, E> R caseSetFieldAdd(SetFieldAdd<T, E> change);
126
127    /**
128     * Handle a {@link SetFieldClear} event.
129     *
130     * @param change visiting change
131     * @param <T> changed object type
132     * @return visitor return value
133     */
134    <T> R caseSetFieldClear(SetFieldClear<T> change);
135
136    /**
137     * Handle a {@link SetFieldRemove} event.
138     *
139     * @param change visiting change
140     * @param <T> changed object type
141     * @param <E> changed set element type
142     * @return visitor return value
143     */
144    <T, E> R caseSetFieldRemove(SetFieldRemove<T, E> change);
145
146    /**
147     * Handle a {@link SimpleFieldChange} event.
148     *
149     * @param change visiting change
150     * @param <T> changed object type
151     * @param <V> changed field type
152     * @return visitor return value
153     */
154    <T, V> R caseSimpleFieldChange(SimpleFieldChange<T, V> change);
155}
