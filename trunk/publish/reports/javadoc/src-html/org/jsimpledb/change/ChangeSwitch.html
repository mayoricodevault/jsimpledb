008package org.jsimpledb.change;
009
010/**
011 * Visitor pattern interface for the {@link Change} class hierarchy.
012 *
013 * @param <R> method return type
014 * @see Change#visit
015 */
016public interface ChangeSwitch<R> {
017
018    <T> R caseObjectCreate(ObjectCreate<T> change);
019    <T> R caseObjectDelete(ObjectDelete<T> change);
020    <T, E> R caseListFieldAdd(ListFieldAdd<T, E> change);
021    <T> R caseListFieldClear(ListFieldClear<T> change);
022    <T, E> R caseListFieldRemove(ListFieldRemove<T, E> change);
023    <T, E> R caseListFieldReplace(ListFieldReplace<T, E> change);
024    <T, K, V> R caseMapFieldAdd(MapFieldAdd<T, K, V> change);
025    <T> R caseMapFieldClear(MapFieldClear<T> change);
026    <T, K, V> R caseMapFieldRemove(MapFieldRemove<T, K, V> change);
027    <T, K, V> R caseMapFieldReplace(MapFieldReplace<T, K, V> change);
028    <T, E> R caseSetFieldAdd(SetFieldAdd<T, E> change);
029    <T> R caseSetFieldClear(SetFieldClear<T> change);
030    <T, E> R caseSetFieldRemove(SetFieldRemove<T, E> change);
031    <T, V> R caseSimpleFieldChange(SimpleFieldChange<T, V> change);
032}
