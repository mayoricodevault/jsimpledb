008package org.jsimpledb.change;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.JTransaction;
012import org.jsimpledb.ObjIdSet;
013
014/**
015 * Creates a new {@link Change} object based on an existing one where the {@link JObject}s referred to by the
016 * new {@link Change} are copies in a different transaction of the originals. This is useful to allow database
017 * change information to be accessed after the transaction in which the change occured has completed.
018 *
019 * <p>
020 * Each instance has an internal {@link ObjIdSet} used to avoid redundant copies, accessible via {@link #getObjIdSet}.
021 * </p>
022 */
023public class ChangeCopier implements ChangeSwitch<Change<?>> {
024
025    protected final JTransaction dest;
026    protected final ObjIdSet objIdSet = new ObjIdSet();
027
028    /**
029     * Primary constructor.
030     *
031     * @param dest destination transaction for copied {@link JObject}s
032     * @throws IllegalArgumentException if {@code dest} is null
033     */
034    public ChangeCopier(JTransaction dest) {
035        if (dest == null)
036            throw new IllegalArgumentException("null dest");
037        this.dest = dest;
038    }
039
040    /**
041     * "Snapshot" constructor for when the destination transaction is the "snapshot" transaction of the transaction
042     * associated with the current thread.
043     *
044     * <p>
045     * This is a convenience constructor, equivalent to:
046     *  <blockquote><code>
047     *  ChangeCopier(JTransaction.getCurrent().getSnapshotTransaction())
048     *  </code></blockquote>
049     * </p>
050     *
051     * @throws IllegalStateException if this is not a snapshot instance and there is no {@link JTransaction}
052     *  associated with the current thread
053     */
054    public ChangeCopier() {
055        this(JTransaction.getCurrent().getSnapshotTransaction());
056    }
057
058    /**
059     * Get the destination transaction configured in this instance.
060     */
061    public JTransaction getDestinationTransaction() {
062        return this.dest;
063    }
064
065    /**
066     * Get the {@link ObjIdSet} used by this instance.
067     */
068    public ObjIdSet getObjIdSet() {
069        return this.objIdSet;
070    }
071
072    @Override
073    public <T> ObjectCreate<T> caseObjectCreate(ObjectCreate<T> change) {
074        return new ObjectCreate<T>(this.copyIfReference(change.getObject()));
075    }
076
077    @Override
078    public <T> ObjectDelete<T> caseObjectDelete(ObjectDelete<T> change) {
079        return new ObjectDelete<T>(this.copyIfReference(change.getObject()));
080    }
081
082    @Override
083    public <T, E> ListFieldAdd<T, E> caseListFieldAdd(ListFieldAdd<T, E> change) {
084        return new ListFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
085          change.getIndex(), this.copyIfReference(change.getElement()));
086    }
087
088    @Override
089    public <T> ListFieldClear<T> caseListFieldClear(ListFieldClear<T> change) {
090        return new ListFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
091    }
092
093    @Override
094    public <T, E> ListFieldRemove<T, E> caseListFieldRemove(ListFieldRemove<T, E> change) {
095        return new ListFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
096          change.getIndex(), this.copyIfReference(change.getElement()));
097    }
098
099    @Override
100    public <T, E> ListFieldReplace<T, E> caseListFieldReplace(ListFieldReplace<T, E> change) {
101        return new ListFieldReplace<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
102          change.getIndex(), this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
103    }
104
105    @Override
106    public <T, K, V> MapFieldAdd<T, K, V> caseMapFieldAdd(MapFieldAdd<T, K, V> change) {
107        return new MapFieldAdd<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
108          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
109    }
110
111    @Override
112    public <T> MapFieldClear<T> caseMapFieldClear(MapFieldClear<T> change) {
113        return new MapFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
114    }
115
116    @Override
117    public <T, K, V> MapFieldRemove<T, K, V> caseMapFieldRemove(MapFieldRemove<T, K, V> change) {
118        return new MapFieldRemove<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
119          this.copyIfReference(change.getKey()), this.copyIfReference(change.getValue()));
120    }
121
122    @Override
123    public <T, K, V> MapFieldReplace<T, K, V> caseMapFieldReplace(MapFieldReplace<T, K, V> change) {
124        return new MapFieldReplace<T, K, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
125          this.copyIfReference(change.getKey()), this.copyIfReference(change.getOldValue()),
126          this.copyIfReference(change.getNewValue()));
127    }
128
129    @Override
130    public <T, E> SetFieldAdd<T, E> caseSetFieldAdd(SetFieldAdd<T, E> change) {
131        return new SetFieldAdd<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
132          this.copyIfReference(change.getElement()));
133    }
134
135    @Override
136    public <T> SetFieldClear<T> caseSetFieldClear(SetFieldClear<T> change) {
137        return new SetFieldClear<T>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName());
138    }
139
140    @Override
141    public <T, E> SetFieldRemove<T, E> caseSetFieldRemove(SetFieldRemove<T, E> change) {
142        return new SetFieldRemove<T, E>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
143          this.copyIfReference(change.getElement()));
144    }
145
146    @Override
147    public <T, V> SimpleFieldChange<T, V> caseSimpleFieldChange(SimpleFieldChange<T, V> change) {
148        return new SimpleFieldChange<T, V>(this.copyIfReference(change.getObject()), change.getStorageId(), change.getFieldName(),
149          this.copyIfReference(change.getOldValue()), this.copyIfReference(change.getNewValue()));
150    }
151
152    @SuppressWarnings("unchecked")
153    private <T> T copyIfReference(T obj) {
154        return obj instanceof JObject ? (T)this.copy((JObject)obj) : obj;
155    }
156
157    /**
158     * Copy the given {@link JObject} into the destination transaction.
159     *
160     * <p>
161     * The implementation in {@link ChangeCopier} invokes {@code jobj.copyTo(this.dest, null, this.getObjIdSet())} unless
162     * {@code jobj} does not exist, in which case it is not copied (but the
163     * {@linkplain JTransaction#getJObject(ObjId) corresponding} {@link JObject} is still returned).
164     * Subclasses may override to copy additional objects referenced by {@code jobj} as needed.
165     * </p>
166     *
167     * @param jobj original object
168     * @return copied object in {@link #dest}
169     * @throws IllegalArgumentException if {@code jobj} is null
170     */
171    @SuppressWarnings("unchecked")
172    protected JObject copy(JObject jobj) {
173        if (jobj == null)
174            throw new IllegalArgumentException("null jobj");
175        return !jobj.exists() ? this.dest.getJObject(jobj) : jobj.copyTo(this.dest, null, this.getObjIdSet());
176    }
177}
