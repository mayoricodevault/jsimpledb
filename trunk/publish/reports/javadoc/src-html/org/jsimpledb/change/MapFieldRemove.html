008package org.jsimpledb.change;
009
010import org.jsimpledb.JObject;
011import org.jsimpledb.JTransaction;
012
013/**
014 * Notification object that gets passed to {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated methods
015 * when a key/value pair is removed from a map field.
016 *
017 * @param <T> the type of the object containing the changed field
018 * @param <K> the type of the changed map's key
019 * @param <V> the type of the changed map's value
020 */
021public class MapFieldRemove<T, K, V> extends MapFieldChange<T> {
022
023    private final K key;
024    private final V value;
025
026    /**
027     * Constructor.
028     *
029     * @param jobj Java object containing the map field that changed
030     * @param storageId the storage ID of the affected field
031     * @param fieldName the name of the field that changed
032     * @param key the key of the removed key/value pair
033     * @param value the value of the removed key/value pair
034     * @throws IllegalArgumentException if {@code jobj} or {@code fieldName} is null
035     */
036    public MapFieldRemove(T jobj, int storageId, String fieldName, K key, V value) {
037        super(jobj, storageId, fieldName);
038        this.key = key;
039        this.value = value;
040    }
041
042    @Override
043    public <R> R visit(ChangeSwitch<R> target) {
044        return target.caseMapFieldRemove(this);
045    }
046
047    @Override
048    public void apply(JTransaction jtx, JObject jobj) {
049        jtx.readMapField(jobj, this.getStorageId(), false).remove(this.key);
050    }
051
052    /**
053     * Get the key of the key/value pair that was removed.
054     *
055     * @return the key of the removed key/value pair
056     */
057    public K getKey() {
058        return this.key;
059    }
060
061    /**
062     * Get the value of the key/value pair that was removed.
063     *
064     * @return the value of the removed key/value pair
065     */
066    public V getValue() {
067        return this.value;
068    }
069
070// Object
071
072    @Override
073    public boolean equals(Object obj) {
074        if (obj == this)
075            return true;
076        if (!super.equals(obj))
077            return false;
078        final MapFieldRemove<?, ?, ?> that = (MapFieldRemove<?, ?, ?>)obj;
079        return (this.key != null ? this.key.equals(that.key) : that.key == null)
080          && (this.value != null ? this.value.equals(that.value) : that.value == null);
081    }
082
083    @Override
084    public int hashCode() {
085        return super.hashCode() ^ (this.key != null ? this.key.hashCode() : 0) ^ (this.value != null ? this.value.hashCode() : 0);
086    }
087
088    @Override
089    public String toString() {
090        return "MapFieldRemove[object=" + this.getObject() + ",field=\"" + this.getFieldName() + "\",key="
091          + this.key + ",value=" + this.value + "]";
092    }
093}
