008package org.jsimpledb;
009
010/**
011 * Configures whether and how objects are enqueued for validation in a {@link JTransaction}.
012 *
013 * @see org.jsimpledb.JTransaction#validate
014 * @see org.jsimpledb.JObject#revalidate
015 * @see org.jsimpledb.annotation.Validate
016 */
017public enum ValidationMode {
018
019    /**
020     * Validation is disabled. No objects are enqueued for validation, even if {@link JObject#revalidate} is invoked.
021     */
022    DISABLED,
023
024    /**
025     * Objects are enqueued for validation only when {@link JObject#revalidate} is explicitly invoked.
026     */
027    MANUAL,
028
029    /**
030     * Objects are enqueued for validation automatically.
031     *
032     * <p>
033     * In this mode, objects are enqueued for validation whenever {@link JObject#revalidate} is invoked, or automatically when:
034     * <ul>
035     *  <li>An instance is {@linkplain org.jsimpledb.JTransaction#create created}, and the Java model type (or any super-type)
036     *      has a JSR 303 or {@link org.jsimpledb.annotation.Validate &#64;Validate} annotation on itself or a public method</li>
037     *  <li>An instance is {@linkplain org.jsimpledb.JObject#upgrade upgraded}, and the Java model type (or any super-type)
038     *      has a JSR 303 or {@link org.jsimpledb.annotation.Validate &#64;Validate} annotation on itself or a public method</li>
039     *  <li>A database field is modified, and the corresponding Java model `getter' method has any JSR 303 annotations</li>
040     * </ul>
041     *
042     * <p>
043     * Note that the presence of a {@link org.jsimpledb.annotation.Validate &#64;Validate} annotation on some method
044     * does <b>not</b> in itself result in automatic validation when any field changes. To achieve that, add an
045     * {@link org.jsimpledb.annotation.OnChange &#64;OnChange}-annotated method that invokes
046     * {@link org.jsimpledb.JObject#revalidate this.revalidate()}.
047     * </p>
048     */
049    AUTOMATIC;
050}
