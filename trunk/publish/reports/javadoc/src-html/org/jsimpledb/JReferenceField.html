008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeToken;
012
013import java.lang.reflect.Method;
014
015import org.jsimpledb.core.DeleteAction;
016import org.jsimpledb.core.FieldType;
017import org.jsimpledb.schema.ReferenceSchemaField;
018
019/**
020 * Represents a reference field in a {@link JClass} or a reference sub-field of a complex field in a {@link JClass}.
021 */
022public class JReferenceField extends JSimpleField {
023
024    final DeleteAction onDelete;
025
026    JReferenceField(String name, int storageId, String description,
027      TypeToken<?> typeToken, DeleteAction onDelete, Method getter, Method setter) {
028        super(name, storageId, typeToken, FieldType.REFERENCE_TYPE_NAME, true, description, getter, setter);
029        this.onDelete = onDelete;
030    }
031
032    /**
033     * Get the {@link DeleteAction} configured for this field.
034     */
035    public DeleteAction getOnDelete() {
036        return this.onDelete;
037    }
038
039    @Override
040    ReferenceSchemaField toSchemaItem() {
041        final ReferenceSchemaField schemaField = new ReferenceSchemaField();
042        super.initialize(schemaField);
043        schemaField.setOnDelete(this.onDelete);
044        return schemaField;
045    }
046
047    @Override
048    Converter<?, ?> getConverter(JTransaction jtx) {
049        return jtx.referenceConverter.reverse();
050    }
051}
