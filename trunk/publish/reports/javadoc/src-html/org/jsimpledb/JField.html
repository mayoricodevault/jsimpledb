008package org.jsimpledb;
009
010import com.google.common.reflect.TypeToken;
011
012import java.lang.reflect.Method;
013
014import org.dellroad.stuff.java.Primitive;
015import org.jsimpledb.schema.SchemaField;
016import org.objectweb.asm.ClassWriter;
017import org.objectweb.asm.MethodVisitor;
018import org.objectweb.asm.Opcodes;
019import org.objectweb.asm.Type;
020
021/**
022 * Represents a field in a Java model object.
023 */
024public abstract class JField extends JSchemaObject {
025
026    final Method getter;
027    final boolean requiresValidation;
028
029    JSchemaObject parent;
030
031    JField(String name, int storageId, String description, Method getter) {
032        super(name, storageId, description);
033        this.getter = getter;
034
035        // Check for validation
036        this.requiresValidation = getter != null && Util.requiresValidation(getter);
037    }
038
039    @Override
040    abstract SchemaField toSchemaItem(JSimpleDB jdb);
041
042    /**
043     * Get the getter method associated with this field.
044     *
045     * @return field property getter method, or null if this field is a sub-field of a complex field
046     */
047    public Method getGetter() {
048        return this.getter;
049    }
050
051    /**
052     * Get the Java value of this field in the given Java model object.
053     * Does not alter the schema version of the object.
054     *
055     * @param jtx transaction
056     * @param jobj object containing this field
057     * @throws DeletedObjectException if {@code jobj} does not exist in {@code jtx}
058     * @throws StaleTransactionException if {@code jtx} is no longer usable
059     * @throws IllegalArgumentException if either parameter is null
060     */
061    public abstract Object getValue(JTransaction jtx, JObject jobj);
062
063    /**
064     * Apply visitor pattern.
065     *
066     * @param target target to invoke
067     * @return value from the method of {@code target} corresponding to this instance's type
068     * @throws NullPointerException if {@code target} is null
069     */
070    public abstract <R> R visit(JFieldSwitch<R> target);
071
072    abstract void outputMethods(ClassGenerator<?> generator, ClassWriter cw);
073
074    /**
075     * Generate bytecode for getter method override.
076     */
077    void outputReadMethod(final ClassGenerator<?> generator, ClassWriter cw, final Method readMethod) {
078
079        // Get property type
080        final TypeToken<?> propertyType = TypeToken.of(this.getter.getReturnType());
081
082        // Generate method override
083        generator.overrideBeanMethod(cw, this.getter, this.storageId, new ClassGenerator.CodeEmitter() {
084            @Override
085            public void emit(MethodVisitor mv) {
086
087                // Push "true"
088                mv.visitInsn(Opcodes.ICONST_1);
089
090                // Invoke JTransaction.readXXX()
091                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
092                  readMethod.getName(), Type.getMethodDescriptor(readMethod));
093
094                // Cast result value
095                mv.visitTypeInsn(Opcodes.CHECKCAST, Type.getInternalName(propertyType.wrap().getRawType()));
096
097                // Unwrap result if necessary
098                if (propertyType.isPrimitive())
099                    generator.unwrap(mv, Primitive.get(propertyType.getRawType()));
100            }
101        });
102    }
103
104    /**
105     * Create a {@link JFieldInfo} instance that corresponds to this instance.
106     */
107    abstract JFieldInfo toJFieldInfo();
108}
