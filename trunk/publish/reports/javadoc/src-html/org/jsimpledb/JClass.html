060    /**
061     * Constructor.
062     *
063     * @param jdb the associated {@link JSimpleDB}
064     * @param name the name of the object type
065     * @param storageId object type storage ID
066     * @param type object type Java model class
067     * @throws IllegalArgumentException if any parameter is null
068     * @throws IllegalArgumentException if {@code storageId} is non-positive
069     */
070    JClass(JSimpleDB jdb, String name, int storageId, TypeToken<T> typeToken) {
071        super(name, storageId, "object type `" + name + "' (" + typeToken + ")");
072        if (jdb == null)
073            throw new IllegalArgumentException("null jdb");
074        if (name == null)
075            throw new IllegalArgumentException("null name");
076        this.jdb = jdb;
077        this.typeToken = typeToken;
078    }

080    // Get generated subclass' constructor
081    Constructor<? extends T> getConstructor() {
082        if (this.constructor == null) {
083            try {
084                this.constructor = this.getSubclass().getConstructor(ObjId.class);
085            } catch (NoSuchMethodException e) {
086                throw new RuntimeException("internal error", e);
087            }
088            this.constructor.setAccessible(true);
089        }
090        return this.constructor;
091    }
092
093    // Get generated snapshot subclass' constructor
094    Constructor<? extends T> getSnapshotConstructor() {
095        if (this.snapshotConstructor == null) {
096            try {
097                this.snapshotConstructor = this.getSnapshotSubclass().getConstructor(ObjId.class, SnapshotJTransaction.class);
098            } catch (NoSuchMethodException e) {
099                throw new RuntimeException("internal error", e);
100            }
101            this.snapshotConstructor.setAccessible(true);
102        }
103        return this.snapshotConstructor;
104    }
105
106    // Get generated subclass
107    Class<? extends T> getSubclass() {
108        if (this.subclass == null)
109            this.subclass = new ClassGenerator<T>(this).generateClass();
110        return this.subclass;
111    }
112
113    // Get generated snapshot subclass
114    Class<? extends T> getSnapshotSubclass() {
115        if (this.snapshotSubclass == null)
116            this.snapshotSubclass = new ClassGenerator<T>(this).generateSnapshotClass();
117        return this.snapshotSubclass;
118    }

156    void createFields() {
157
158        // Scan for Simple fields
159        final JFieldScanner<T> simpleFieldScanner = new JFieldScanner<T>(this);
160        for (JFieldScanner<T>.MethodInfo info : simpleFieldScanner.findAnnotatedMethods()) {
161
162            // Get info
163            final org.jsimpledb.annotation.JField annotation = info.getAnnotation();
164            final Method getter = info.getMethod();
165            final String description = simpleFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
166            final String fieldName = this.getFieldName(annotation.name(), info, description);
167            final TypeToken<?> fieldTypeToken = TypeToken.of(getter.getGenericReturnType());
168            if (this.log.isTraceEnabled())
169                this.log.trace("found " + description);
170
171            // Handle Counter fields
172            if (fieldTypeToken.equals(TypeToken.of(Counter.class))) {
173
174                // Sanity check annotation
175                if (annotation.type().length() != 0)
176                    throw new IllegalArgumentException("invalid " + description + ": counter fields must not specify a type");
177                if (annotation.indexed())
178                    throw new IllegalArgumentException("invalid " + description + ": counter fields cannot be indexed");
179
180                // Create counter field
181                final JCounterField jfield = new JCounterField(fieldName, annotation.storageId(),
182                  "counter field `" + fieldName + "' of object type `" + this.name + "'", getter);
183                jfield.parent = this;
184
185                // Add field
186                this.addField(jfield);
187                if (this.log.isTraceEnabled())
188                    this.log.trace("added counter field `" + fieldName + "' to object type `" + this.name + "'");
189                continue;
190            }
191
192            // Find corresponding setter method
193            final Matcher matcher = Pattern.compile("(is|get)(.+)").matcher(getter.getName());
194            if (!matcher.matches()) {
195                throw new IllegalArgumentException("invalid " + description
196                  + ": can't infer setter method name because getter method name does not follow Java bean naming conventions");
197            }
198            final String setterName = "set" + matcher.group(2);
199            Method setter = null;
200            for (TypeToken<?> superType : this.typeToken.getTypes()) {
201                try {
202                    setter = superType.getRawType().getMethod(setterName, fieldTypeToken.getRawType());
203                } catch (NoSuchMethodException e) {
204                    continue;
205                }
206                break;
207            }
208            if (setter == null) {
209                throw new IllegalArgumentException("invalid " + description
210                  + ": can't find any corresponding setter method " + setterName + "() taking " + fieldTypeToken);
211            }
212
213            // Create simple field
214            final JSimpleField jfield = this.createSimpleField(description, fieldTypeToken,
215              fieldName, annotation.type(), annotation.storageId(), annotation.indexed(), annotation.onDelete(),
216              getter, setter, "field `" + fieldName + "' of object type `" + this.name + "'");
217            jfield.parent = this;
218
219            // Add field
220            this.addField(jfield);
221            if (this.log.isTraceEnabled())
222                this.log.trace("added simple field `" + fieldName + "' to object type `" + this.name + "'");
223        }
224
225        // Scan for Set fields
226        final JSetFieldScanner<T> setFieldScanner = new JSetFieldScanner<T>(this);
227        for (JSetFieldScanner<T>.MethodInfo info : setFieldScanner.findAnnotatedMethods()) {
228
229            // Get info
230            final org.jsimpledb.annotation.JSetField annotation = info.getAnnotation();
231            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
232            final Method getter = info.getMethod();
233            final String description = setFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
234            final String fieldName = this.getFieldName(annotation.name(), info, description);
235            if (this.log.isTraceEnabled())
236                this.log.trace("found " + description);
237
238            // Get element type (the raw return type has already been validated by the annotation scanner)
239            final TypeToken<?> elementType = this.getParameterType(description, getter, 0);
240
241            // Create element sub-field
242            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
243              elementType, null, elementAnnotation.type(), elementAnnotation.storageId(), elementAnnotation.indexed(),
244              elementAnnotation.onDelete(), null, null,
245              "element field of set field `" + fieldName + "' in object type `" + this.name + "'");
246
247            // Create set field
248            final JSetField jfield = new JSetField(fieldName, annotation.storageId(),
249              elementField, "set field `" + fieldName + "' in object type `" + this.name + "'", getter);
250            elementField.parent = jfield;
251
252            // Add field
253            this.addField(jfield);
254            if (this.log.isTraceEnabled())
255                this.log.trace("added set field `" + fieldName + "' to object type `" + this.name + "'");
256        }
257
258        // Scan for List fields
259        final JListFieldScanner<T> listFieldScanner = new JListFieldScanner<T>(this);
260        for (JListFieldScanner<T>.MethodInfo info : listFieldScanner.findAnnotatedMethods()) {
261
262            // Get info
263            final org.jsimpledb.annotation.JListField annotation = info.getAnnotation();
264            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
265            final Method getter = info.getMethod();
266            final String description = listFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
267            final String fieldName = this.getFieldName(annotation.name(), info, description);
268            if (this.log.isTraceEnabled())
269                this.log.trace("found " + description);
270
271            // Get element type (the raw return type has already been validated by the annotation scanner)
272            final TypeToken<?> elementType = this.getParameterType(description, getter, 0);
273
274            // Create element sub-field
275            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
276              elementType, null, elementAnnotation.type(), elementAnnotation.storageId(), elementAnnotation.indexed(),
277              elementAnnotation.onDelete(), null, null,
278              "element field of list field `" + fieldName + "' in object type `" + this.name + "'");
279
280            // Create list field
281            final JListField jfield = new JListField(fieldName, annotation.storageId(),
282              elementField, "list field `" + fieldName + "' in object type `" + this.name + "'", getter);
283            elementField.parent = jfield;
284
285            // Add field
286            this.addField(jfield);
287            if (this.log.isTraceEnabled())
288                this.log.trace("added list field `" + fieldName + "' to object type `" + this.name + "'");
289        }
290
291        // Scan for Map fields
292        final JMapFieldScanner<T> mapFieldScanner = new JMapFieldScanner<T>(this);
293        for (JMapFieldScanner<T>.MethodInfo info : mapFieldScanner.findAnnotatedMethods()) {
294
295            // Get info
296            final org.jsimpledb.annotation.JMapField annotation = info.getAnnotation();
297            final org.jsimpledb.annotation.JField keyAnnotation = annotation.key();
298            final org.jsimpledb.annotation.JField valueAnnotation = annotation.value();
299            final Method getter = info.getMethod();
300            final String description = mapFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
301            final String fieldName = this.getFieldName(annotation.name(), info, description);
302            if (this.log.isTraceEnabled())
303                this.log.trace("found " + description);
304
305            // Get key and value types (the raw return type has already been validated by the annotation scanner)
306            final TypeToken<?> keyType = this.getParameterType(description, getter, 0);
307            final TypeToken<?> valueType = this.getParameterType(description, getter, 1);
308
309            // Create key and value sub-fields
310            final JSimpleField keyField = this.createSimpleField("key() property of " + description,
311              keyType, null, keyAnnotation.type(), keyAnnotation.storageId(), keyAnnotation.indexed(),
312              keyAnnotation.onDelete(), null, null,
313              "key field of map field `" + fieldName + "' in object type `" + this.name + "'");
314            final JSimpleField valueField = this.createSimpleField("value() property of " + description,
315              valueType, null, valueAnnotation.type(), valueAnnotation.storageId(), valueAnnotation.indexed(),
316              valueAnnotation.onDelete(), null, null,
317              "value field of map field `" + fieldName + "' in object type `" + this.name + "'");
318
319            // Create map field
320            final JMapField jfield = new JMapField(fieldName, annotation.storageId(),
321              keyField, valueField, "map field `" + fieldName + "' in object type `" + this.name + "'", getter);
322            keyField.parent = jfield;
323            valueField.parent = jfield;
324
325            // Add field
326            this.addField(jfield);
327            if (this.log.isTraceEnabled())
328                this.log.trace("added map field `" + fieldName + "' to object type `" + this.name + "'");
329        }
330    }

332    void scanAnnotations() {
333        this.onCreateMethods = new OnCreateScanner<T>(this).findAnnotatedMethods();
334        this.onDeleteMethods = new OnDeleteScanner<T>(this).findAnnotatedMethods();
335        this.onChangeMethods = new OnChangeScanner<T>(this).findAnnotatedMethods();
336        this.validateMethods = new ValidateScanner<T>(this).findAnnotatedMethods();
337        final OnVersionChangeScanner<T> onVersionChangeScanner = new OnVersionChangeScanner<T>(this);
338        this.onVersionChangeMethods = new ArrayList<>(onVersionChangeScanner.findAnnotatedMethods());
339        Collections.sort(this.onVersionChangeMethods, onVersionChangeScanner);
340        this.indexQueryMethods = new IndexQueryScanner<T>(this).findAnnotatedMethods();
341    }

343    @Override
344    SchemaObject toSchemaItem() {
345        final SchemaObject schemaObject = new SchemaObject();
346        super.initialize(schemaObject);
347        for (JField field : this.jfields.values()) {
348            SchemaField schemaField = field.toSchemaItem();
349            schemaObject.getSchemaFields().put(schemaField.getStorageId(), schemaField);
350        }
351        return schemaObject;
352    }

354    // Add new JField (and sub-fields, if any), checking for name and storage ID conflicts
355    private void addField(JField jfield) {
356
357        // Check for storage ID conflict
358        JField other = this.jfields.get(jfield.storageId);
359        if (other != null) {
360            throw new IllegalArgumentException("illegal duplicate use of storage ID "
361              + jfield.storageId + " for both " + other + " and " + jfield);
362        }
363        this.jfields.put(jfield.storageId, jfield);
364
365        // Check for name conflict
366        if ((other = this.jfieldsByName.get(jfield.name)) != null)
367            throw new IllegalArgumentException("illegal duplicate use of field name `" + jfield.name + "' in " + this);
368        this.jfieldsByName.put(jfield.name, jfield);
369    }

371    // Get field name, deriving it from the getter property name if necessary
372    private String getFieldName(String fieldName, AnnotationScanner<T, ?>.MethodInfo info, String description) {
373        if (fieldName.length() > 0)
374            return fieldName;
375        try {
376            return info.getMethodPropertyName();
377        } catch (IllegalArgumentException e) {
378            throw new IllegalArgumentException("invalid " + description + ": can't infer field name: " + e, e);
379        }
380    }

382    // Get the n'th generic type parameter
383    private TypeToken<?> getParameterType(String description, Method method, int index) {
384        try {
385            return Util.getTypeParameter(TypeToken.of(method.getGenericReturnType()), index);
386        } catch (IllegalArgumentException e) {
387            throw new IllegalArgumentException("invalid " + description + ": invalid method return type: " + e.getMessage(), e);
388        }
389    }

391    // Create a simple field, either regular object field or sub-field of complex field
392    private JSimpleField createSimpleField(String description, TypeToken<?> fieldTypeToken, String fieldName,
393      String typeName, int storageId, boolean indexed, DeleteAction onDelete, Method getter, Method setter,
394      String fieldDescription) {
395
396        // Complex sub-field?
397        final boolean isSubField = getter == null;
398
399        // Empty type name same as null
400        if (typeName.length() == 0)
401            typeName = null;
402
403        // See if field type encompasses one or more JClass types and is therefore a reference type
404        boolean isReferenceType = false;
405        for (JClass<?> jclass : this.jdb.jclasses.values()) {
406            if (fieldTypeToken.isAssignableFrom(jclass.typeToken)) {
407                isReferenceType = true;
408                break;
409            }
410        }
411
412        // See if field type is a simple type, known either by explicitly-given name or type
413        FieldType<?> nonReferenceType = null;
414        if (typeName != null) {
415
416            // Field type is explicitly specified by name
417            if ((nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(typeName)) == null)
418                throw new IllegalArgumentException("invalid " + description + ": unknown simple field type `" + typeName + "'");
419
420            // Verify field type matches what we expect
421            final TypeToken<?> expectedType = isSubField ? nonReferenceType.getTypeToken().wrap() : nonReferenceType.getTypeToken();
422            if (!expectedType.equals(fieldTypeToken)) {
423                throw new IllegalArgumentException("invalid " + description + ": field type `" + typeName
424                  + "' supports values of type " + nonReferenceType.getTypeToken() + " but " + fieldTypeToken
425                  + " is required (according to the getter method's return type)");
426            }
427        } else {
428
429            // Try to find a field type supporting getter method return type
430            final List<? extends FieldType<?>> fieldTypes = this.jdb.db.getFieldTypeRegistry().getFieldTypes(fieldTypeToken);
431            switch (fieldTypes.size()) {
432            case 0:
433                nonReferenceType = null;
434                break;
435            case 1:
436                nonReferenceType = fieldTypes.get(0);
437                break;
438            default:
439                if (!isReferenceType) {
440                    throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
441                      + " because type " + fieldTypeToken + " is supported by more than one registered simple field type");
442                }
443                break;
444            }
445
446            // Check for enum types
447            if (nonReferenceType == null && Enum.class.isAssignableFrom(fieldTypeToken.getRawType()))
448                nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(TypeToken.of(EnumValue.class));
449        }
450
451        // If field type neither refers to a JClass type, or is a registered field type, fail
452        if (!isReferenceType && nonReferenceType == null) {
453            throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
454              + " because no known type supports values of type " + fieldTypeToken);
455        }
456
457        // Handle ambiguity between reference vs. non-reference
458        if (isReferenceType && nonReferenceType != null) {
459
460            // If an explicit type name was provided, assume they want the specified non-reference type
461            if (typeName != null)
462                isReferenceType = false;
463            else {
464                throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
465                  + " because type " + fieldTypeToken + " is ambiguous, being both a @" + JSimpleClass.class.getSimpleName()
466                  + " reference type and a simple Java type supported by type `" + nonReferenceType + "'");
467            }
468        }
469
470        // Create simple or reference field
471        try {
472            return isReferenceType ?
473              new JReferenceField(fieldName, storageId, fieldDescription, fieldTypeToken, onDelete, getter, setter) :
474              new JSimpleField(fieldName, storageId, fieldTypeToken,
475                nonReferenceType.getName(), indexed, fieldDescription, getter, setter);
476        } catch (IllegalArgumentException e) {
477            throw new IllegalArgumentException("invalid " + description + ": " + e, e);
478        }
479    }
