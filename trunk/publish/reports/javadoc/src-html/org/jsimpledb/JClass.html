029/**
030 * Information about a Java class that is used to represent a specific JSimpleDB object type.
031 *
032 * @param <T> the Java class
033 */
034public class JClass<T> extends JSchemaObject {

036    final Logger log = LoggerFactory.getLogger(this.getClass());

038    final JSimpleDB jdb;
039    final TypeToken<T> typeToken;

053    /**
054     * Constructor.
055     *
056     * @param jdb the associated {@link JSimpleDB}
057     * @param name the name of the object type
058     * @param storageId object type storage ID
059     * @param type object type Java model class
060     * @throws IllegalArgumentException if any parameter is null
061     * @throws IllegalArgumentException if {@code storageId} is non-positive
062     */
063    JClass(JSimpleDB jdb, String name, int storageId, TypeToken<T> typeToken) {
064        super(name, storageId, "object type `" + name + "' (" + typeToken + ")");
065        if (jdb == null)
066            throw new IllegalArgumentException("null jdb");
067        if (name == null)
068            throw new IllegalArgumentException("null name");
069        this.jdb = jdb;
070        this.typeToken = typeToken;
071    }
072
073    // Get generated subclass' constructor
074    Constructor<? extends T> getConstructor() {
075        if (this.constructor == null) {
076            try {
077                this.constructor = this.getSubclass().getConstructor(ObjId.class);
078            } catch (NoSuchMethodException e) {
079                throw new RuntimeException("internal error", e);
080            }
081            this.constructor.setAccessible(true);
082        }
083        return this.constructor;
084    }
085
086    // Get generated subclass
087    Class<? extends T> getSubclass() {
088        if (this.subclass == null)
089            this.subclass = new ClassGenerator<T>(this).generateClass();
090        return this.subclass;
091    }

129    void createFields() {
130
131        // Scan for Simple fields
132        final JFieldScanner<T> simpleFieldScanner = new JFieldScanner<T>(this);
133        for (JFieldScanner<T>.MethodInfo info : simpleFieldScanner.findAnnotatedMethods()) {
134
135            // Get info
136            final org.jsimpledb.annotation.JField annotation = info.getAnnotation();
137            final Method getter = info.getMethod();
138            final String description = simpleFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
139            final String fieldName = this.getFieldName(annotation.name(), info, description);
140            final TypeToken<?> fieldTypeToken = TypeToken.of(getter.getGenericReturnType());
141            this.log.debug("found " + description);
142
143            // Find corresponding setter method
144            final Matcher matcher = Pattern.compile("(is|get)(.+)").matcher(getter.getName());
145            if (!matcher.matches()) {
146                throw new IllegalArgumentException("invalid " + description
147                  + ": can't infer setter method name because getter method name does not follow Java bean naming conventions");
148            }
149            final String setterName = "set" + matcher.group(2);
150            Method setter = null;
151            for (TypeToken<?> superType : this.typeToken.getTypes()) {
152                try {
153                    setter = superType.getRawType().getMethod(setterName, fieldTypeToken.getRawType());
154                } catch (NoSuchMethodException e) {
155                    continue;
156                }
157                break;
158            }
159            if (setter == null) {
160                throw new IllegalArgumentException("invalid " + description
161                  + ": can't find any corresponding setter method " + setterName + "() taking " + fieldTypeToken);
162            }
163
164            // Create simple field
165            final JSimpleField jfield = this.createSimpleField(description, fieldTypeToken,
166              fieldName, annotation.type(), annotation.storageId(), annotation.indexed(), annotation.onDelete(),
167              getter, setter, "field `" + fieldName + "' of object type `" + this.name + "'");
168            jfield.parent = this;
169
170            // Add field
171            this.addField(jfield);
172            this.log.debug("added simple field `" + fieldName + "' to object type `" + this.name + "'");
173        }
174
175        // Scan for Set fields
176        final JSetFieldScanner<T> setFieldScanner = new JSetFieldScanner<T>(this);
177        for (JSetFieldScanner<T>.MethodInfo info : setFieldScanner.findAnnotatedMethods()) {
178
179            // Get info
180            final org.jsimpledb.annotation.JSetField annotation = info.getAnnotation();
181            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
182            final Method getter = info.getMethod();
183            final String description = setFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
184            final String fieldName = this.getFieldName(annotation.name(), info, description);
185            this.log.debug("found " + description);
186
187            // Get element type (the raw return type has already been validated by the annotation scanner)
188            final TypeToken<?> elementType = this.getParameterType(description, getter, 0);
189
190            // Create element sub-field
191            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
192              elementType, null, elementAnnotation.type(), elementAnnotation.storageId(), elementAnnotation.indexed(),
193              elementAnnotation.onDelete(), null, null,
194              "element field of set field `" + fieldName + "' in object type `" + this.name + "'");
195
196            // Create set field
197            final JSetField jfield = new JSetField(fieldName, annotation.storageId(),
198              elementField, "set field `" + fieldName + "' in object type `" + this.name + "'", getter);
199            elementField.parent = jfield;
200
201            // Add field
202            this.addField(jfield);
203            this.log.debug("added set field `" + fieldName + "' to object type `" + this.name + "'");
204        }
205
206        // Scan for List fields
207        final JListFieldScanner<T> listFieldScanner = new JListFieldScanner<T>(this);
208        for (JListFieldScanner<T>.MethodInfo info : listFieldScanner.findAnnotatedMethods()) {
209
210            // Get info
211            final org.jsimpledb.annotation.JListField annotation = info.getAnnotation();
212            final org.jsimpledb.annotation.JField elementAnnotation = annotation.element();
213            final Method getter = info.getMethod();
214            final String description = listFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
215            final String fieldName = this.getFieldName(annotation.name(), info, description);
216            this.log.debug("found " + description);
217
218            // Get element type (the raw return type has already been validated by the annotation scanner)
219            final TypeToken<?> elementType = this.getParameterType(description, getter, 0);
220
221            // Create element sub-field
222            final JSimpleField elementField = this.createSimpleField("element() property of " + description,
223              elementType, null, elementAnnotation.type(), elementAnnotation.storageId(), elementAnnotation.indexed(),
224              elementAnnotation.onDelete(), null, null,
225              "element field of list field `" + fieldName + "' in object type `" + this.name + "'");
226
227            // Create list field
228            final JListField jfield = new JListField(fieldName, annotation.storageId(),
229              elementField, "list field `" + fieldName + "' in object type `" + this.name + "'", getter);
230            elementField.parent = jfield;
231
232            // Add field
233            this.addField(jfield);
234            this.log.debug("added list field `" + fieldName + "' to object type `" + this.name + "'");
235        }
236
237        // Scan for Map fields
238        final JMapFieldScanner<T> mapFieldScanner = new JMapFieldScanner<T>(this);
239        for (JMapFieldScanner<T>.MethodInfo info : mapFieldScanner.findAnnotatedMethods()) {
240
241            // Get info
242            final org.jsimpledb.annotation.JMapField annotation = info.getAnnotation();
243            final org.jsimpledb.annotation.JField keyAnnotation = annotation.key();
244            final org.jsimpledb.annotation.JField valueAnnotation = annotation.value();
245            final Method getter = info.getMethod();
246            final String description = mapFieldScanner.getAnnotationDescription() + " annotation on method " + getter;
247            final String fieldName = this.getFieldName(annotation.name(), info, description);
248            this.log.debug("found " + description);
249
250            // Get key and value types (the raw return type has already been validated by the annotation scanner)
251            final TypeToken<?> keyType = this.getParameterType(description, getter, 0);
252            final TypeToken<?> valueType = this.getParameterType(description, getter, 1);
253
254            // Create key and value sub-fields
255            final JSimpleField keyField = this.createSimpleField("key() property of " + description,
256              keyType, null, keyAnnotation.type(), keyAnnotation.storageId(), keyAnnotation.indexed(),
257              keyAnnotation.onDelete(), null, null,
258              "key field of map field `" + fieldName + "' in object type `" + this.name + "'");
259            final JSimpleField valueField = this.createSimpleField("value() property of " + description,
260              valueType, null, valueAnnotation.type(), valueAnnotation.storageId(), valueAnnotation.indexed(),
261              valueAnnotation.onDelete(), null, null,
262              "value field of map field `" + fieldName + "' in object type `" + this.name + "'");
263
264            // Create map field
265            final JMapField jfield = new JMapField(fieldName, annotation.storageId(),
266              keyField, valueField, "map field `" + fieldName + "' in object type `" + this.name + "'", getter);
267            keyField.parent = jfield;
268            valueField.parent = jfield;
269
270            // Add field
271            this.addField(jfield);
272            this.log.debug("added map field `" + fieldName + "' to object type `" + this.name + "'");
273        }
274    }

277        this.onCreateMethods = new OnCreateScanner<T>(this).findAnnotatedMethods();
278        this.onDeleteMethods = new OnDeleteScanner<T>(this).findAnnotatedMethods();
279        this.onChangeMethods = new OnChangeScanner<T>(this).findAnnotatedMethods();
280        this.onVersionChangeMethods = new OnVersionChangeScanner<T>(this).findAnnotatedMethods();
281        this.indexQueryMethods = new IndexQueryScanner<T>(this).findAnnotatedMethods();
282    }

284    @Override
285    SchemaObject toSchemaItem() {
286        final SchemaObject schemaObject = new SchemaObject();
287        super.initialize(schemaObject);
288        for (JField field : this.jfields.values())
289            schemaObject.addSchemaField(field.toSchemaItem());
290        return schemaObject;

293    // Add new JField (and sub-fields, if any), checking for name and storage ID conflicts
294    private void addField(JField jfield) {
295
296        // Check for storage ID conflict
297        JField other = this.jfields.get(jfield.storageId);
298        if (other != null) {
299            throw new IllegalArgumentException("illegal duplicate use of storage ID "
300              + jfield.storageId + " for both " + other + " and " + jfield);
301        }
302        this.jfields.put(jfield.storageId, jfield);
303
304        // Check for name conflict
305        if ((other = this.jfieldsByName.get(jfield.name)) != null)
306            throw new IllegalArgumentException("illegal duplicate use of field name `" + jfield.name + "' in " + this);
307        this.jfieldsByName.put(jfield.name, jfield);
308    }

310    // Get field name, deriving it from the getter property name if necessary
311    private String getFieldName(String fieldName, AnnotationScanner<T, ?>.MethodInfo info, String description) {
312        if (fieldName.length() > 0)
313            return fieldName;
314        try {
315            return info.getMethodPropertyName();
316        } catch (IllegalArgumentException e) {
317            throw new IllegalArgumentException("invalid " + description + ": can't infer field name: " + e, e);
318        }
319    }

321    // Get the n'th generic type parameter
322    private TypeToken<?> getParameterType(String description, Method method, int index) {
323        try {
324            return Util.getTypeParameter(TypeToken.of(method.getGenericReturnType()), index);
325        } catch (IllegalArgumentException e) {
326            throw new IllegalArgumentException("invalid " + description + ": invalid method return type: " + e.getMessage(), e);
327        }
328    }

330    // Create a simple field, either regular object field or sub-field of complex field
331    private JSimpleField createSimpleField(String description, TypeToken<?> fieldTypeToken, String fieldName,
332      String typeName, int storageId, boolean indexed, DeleteAction onDelete, Method getter, Method setter,
333      String fieldDescription) {
334
335        // Complex sub-field?
336        final boolean isSubField = getter == null;
337
338        // Empty type name same as null
339        if (typeName.length() == 0)
340            typeName = null;
341
342        // See if field type encompasses one or more JClass types and is therefore a reference type
343        boolean isReferenceType = false;
344        for (JClass<?> jclass : this.jdb.jclasses.values()) {
345            if (fieldTypeToken.isAssignableFrom(jclass.typeToken)) {
346                isReferenceType = true;
347                break;
348            }
349        }
350
351        // See if field type is a simple type, known either by explicitly-given name or type
352        FieldType<?> nonReferenceType = null;
353        if (typeName != null) {
354
355            // Field type is explicitly specified by name
356            if ((nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(typeName)) == null)
357                throw new IllegalArgumentException("invalid " + description + ": unknown simple field type `" + typeName + "'");
358
359            // Verify field type matches what we expect
360            final TypeToken<?> expectedType = isSubField ? nonReferenceType.getTypeToken().wrap() : nonReferenceType.getTypeToken();
361            if (!expectedType.equals(fieldTypeToken)) {
362                throw new IllegalArgumentException("invalid " + description + ": field type `" + typeName
363                  + "' supports values of type " + nonReferenceType.getTypeToken() + " but " + fieldTypeToken
364                  + " is required (according to the getter method's return type)");
365            }
366        } else {
367
368            // Try to find a field type supporting getter method return type
369            try {
370                nonReferenceType = this.jdb.db.getFieldTypeRegistry().getFieldType(fieldTypeToken);
371            } catch (IllegalArgumentException e) {
372                if (!isReferenceType) {
373                    throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
374                      + " because type " + fieldTypeToken + " is supported by more than one registered simple field type", e);
375                }
376            }
377        }
378
379        // If field type neither refers to a JClass type, or is a registered field type, fail
380        if (!isReferenceType && nonReferenceType == null) {
381            throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
382              + " because no known type supports values of type " + fieldTypeToken);
383        }
384
385        // Handle ambiguity between reference vs. non-reference
386        if (isReferenceType && nonReferenceType != null) {
387
388            // If an explicit type name was provided, assume they want the specified non-reference type
389            if (typeName != null)
390                isReferenceType = false;
391            else {
392                throw new IllegalArgumentException("invalid " + description + ": an explicit type() must be specified"
393                  + " because type " + fieldTypeToken + " is ambiguous, being both a @" + JSimpleClass.class.getSimpleName()
394                  + " reference type and a simple Java type supported by type `" + nonReferenceType + "'");
395            }
396        }
397
398        // Create simple or reference field
399        try {
400            return isReferenceType ?
401              new JReferenceField(fieldName, storageId, fieldDescription, fieldTypeToken, onDelete, getter, setter) :
402              new JSimpleField(fieldName, storageId, fieldTypeToken,
403                nonReferenceType.getName(), indexed, fieldDescription, getter, setter);
404        } catch (IllegalArgumentException e) {
405            throw new IllegalArgumentException("invalid " + description + ": " + e, e);
406        }
407    }
