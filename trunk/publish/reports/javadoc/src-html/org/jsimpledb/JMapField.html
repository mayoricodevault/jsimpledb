008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.Arrays;
016import java.util.Deque;
017import java.util.List;
018import java.util.NavigableMap;
019import java.util.NavigableSet;
020import java.util.Set;
021
022import org.jsimpledb.change.MapFieldAdd;
023import org.jsimpledb.change.MapFieldClear;
024import org.jsimpledb.change.MapFieldRemove;
025import org.jsimpledb.change.MapFieldReplace;
026import org.jsimpledb.core.MapField;
027import org.jsimpledb.core.ObjId;
028import org.jsimpledb.core.Transaction;
029import org.jsimpledb.schema.MapSchemaField;
030import org.objectweb.asm.ClassWriter;
031
032/**
033 * Represents a map field in a {@link JClass}.
034 */
035public class JMapField extends JComplexField {
036
037    private static final int KEY_INDEX_ENTRY_QUERY = 1;
038    private static final int VALUE_INDEX_ENTRY_QUERY = 2;
039
040    final JSimpleField keyField;
041    final JSimpleField valueField;
042
043    JMapField(String name, int storageId, JSimpleField keyField, JSimpleField valueField, String description, Method getter) {
044        super(name, storageId, description, getter);
045        if (keyField == null)
046            throw new IllegalArgumentException("null keyField");
047        if (valueField == null)
048            throw new IllegalArgumentException("null valueField");
049        this.keyField = keyField;
050        this.valueField = valueField;
051    }
052
053    /**
054     * Get the key sub-field.
055     */
056    public JSimpleField getKeyField() {
057        return this.keyField;
058    }
059
060    /**
061     * Get the value sub-field.
062     */
063    public JSimpleField getValueField() {
064        return this.valueField;
065    }
066
067    @Override
068    public NavigableMap<?, ?> getValue(JTransaction jtx, JObject jobj) {
069        if (jtx == null)
070            throw new IllegalArgumentException("null jtx");
071        if (jobj == null)
072            throw new IllegalArgumentException("null jobj");
073        return jtx.readMapField(jobj, this.storageId, false);
074    }
075
076    @Override
077    public List<JSimpleField> getSubFields() {
078        return Arrays.asList(this.keyField, this.valueField);
079    }
080
081    @Override
082    public JSimpleField getSubField(String name) {
083        if (MapField.KEY_FIELD_NAME.equals(name))
084            return this.keyField;
085        if (MapField.VALUE_FIELD_NAME.equals(name))
086            return this.valueField;
087        throw new IllegalArgumentException("unknown sub-field `" + name
088          + "' (did you mean `" + MapField.KEY_FIELD_NAME + "' or `" + MapField.VALUE_FIELD_NAME + "' instead?)");
089    }
090
091    @Override
092    String getSubFieldName(JSimpleField subField) {
093        if (subField == this.keyField)
094            return MapField.KEY_FIELD_NAME;
095        if (subField == this.valueField)
096            return MapField.VALUE_FIELD_NAME;
097        throw new IllegalArgumentException("unknown sub-field");
098    }
099
100    @Override
101    MapSchemaField toSchemaItem() {
102        final MapSchemaField schemaField = new MapSchemaField();
103        super.initialize(schemaField);
104        schemaField.setKeyField(this.keyField.toSchemaItem());
105        schemaField.setValueField(this.valueField.toSchemaItem());
106        return schemaField;
107    }
108
109    @Override
110    void outputMethods(ClassGenerator<?> generator, ClassWriter cw) {
111        this.outputReadMethod(generator, cw, ClassGenerator.READ_MAP_FIELD_METHOD);
112    }
113
114    @Override
115    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
116        tx.addMapFieldChangeListener(this.storageId, path, listener);
117    }
118
119    @Override
120    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
121        this.addChangeParameterTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
122    }
123
124    // This method exists solely to bind the generic type parameters
125    @SuppressWarnings("serial")
126    private <T, K, V> void addChangeParameterTypes(List<TypeToken<?>> types,
127      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
128        types.add(new TypeToken<MapFieldAdd<T, K, V>>() { }
129          .where(new TypeParameter<T>() { }, targetType)
130          .where(new TypeParameter<K>() { }, keyType.wrap())
131          .where(new TypeParameter<V>() { }, valueType.wrap()));
132        types.add(new TypeToken<MapFieldClear<T>>() { }
133          .where(new TypeParameter<T>() { }, targetType));
134        types.add(new TypeToken<MapFieldRemove<T, K, V>>() { }
135          .where(new TypeParameter<T>() { }, targetType)
136          .where(new TypeParameter<K>() { }, keyType.wrap())
137          .where(new TypeParameter<V>() { }, valueType.wrap()));
138        types.add(new TypeToken<MapFieldReplace<T, K, V>>() { }
139          .where(new TypeParameter<T>() { }, targetType)
140          .where(new TypeParameter<K>() { }, keyType.wrap())
141          .where(new TypeParameter<V>() { }, valueType.wrap()));
142    }
143
144    @Override
145    <T> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, JSimpleField subField) {
146        if (subField == this.keyField)
147            this.addKeyIndexEntryReturnTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
148        else if (subField == this.valueField)
149            this.addValueIndexEntryReturnTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
150        else
151            throw new RuntimeException();
152    }
153
154    // This method exists solely to bind the generic type parameters
155    @SuppressWarnings("serial")
156    private <T, K, V> void addKeyIndexEntryReturnTypes(List<TypeToken<?>> types,
157      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
158        types.add(new TypeToken<NavigableMap<K, NavigableSet<MapKeyIndexEntry<T, V>>>>() { }
159          .where(new TypeParameter<T>() { }, targetType)
160          .where(new TypeParameter<K>() { }, keyType.wrap())
161          .where(new TypeParameter<V>() { }, valueType.wrap()));
162    }
163
164    // This method exists solely to bind the generic type parameters
165    @SuppressWarnings("serial")
166    private <T, K, V> void addValueIndexEntryReturnTypes(List<TypeToken<?>> types,
167      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
168        types.add(new TypeToken<NavigableMap<V, NavigableSet<MapValueIndexEntry<T, K>>>>() { }
169          .where(new TypeParameter<T>() { }, targetType)
170          .where(new TypeParameter<K>() { }, keyType.wrap())
171          .where(new TypeParameter<V>() { }, valueType.wrap()));
172    }
173
174    @Override
175    int getIndexEntryQueryType(TypeToken<?> queryObjectType) {
176        return queryObjectType.getRawType().equals(MapKeyIndexEntry.class) ? KEY_INDEX_ENTRY_QUERY :
177          queryObjectType.getRawType().equals(MapValueIndexEntry.class) ? VALUE_INDEX_ENTRY_QUERY : 0;
178    }
179
180    @Override
181    Method getIndexEntryQueryMethod(int queryType) {
182        switch (queryType) {
183        case KEY_INDEX_ENTRY_QUERY:
184            return ClassGenerator.QUERY_MAP_FIELD_KEY_ENTRIES_METHOD;
185        case VALUE_INDEX_ENTRY_QUERY:
186            return ClassGenerator.QUERY_MAP_FIELD_VALUE_ENTRIES_METHOD;
187        default:
188            throw new UnsupportedOperationException();
189        }
190    }
191
192    @Override
193    NavigableMapConverter<?, ?, ?, ?> getConverter(JTransaction jtx) {
194        final Converter<?, ?> keyConverter = this.keyField.getConverter(jtx);
195        final Converter<?, ?> valueConverter = this.valueField.getConverter(jtx);
196        if (keyConverter == null && valueConverter == null)
197            return null;
198        return this.createConverter(keyConverter != null ? keyConverter : Converter.identity(),
199          valueConverter != null ? valueConverter : Converter.identity());
200    }
201
202    // This method exists solely to bind the generic type parameters
203    private <K, V, WK, WV> NavigableMapConverter<K, V, WK, WV> createConverter(
204      Converter<K, WK> keyConverter, Converter<V, WV> valueConverter) {
205        return new NavigableMapConverter<K, V, WK, WV>(keyConverter, valueConverter);
206    }
207
208    @Override
209    void copyRecurse(Set<ObjId> seen, JTransaction srcTx, JTransaction dstTx,
210      ObjId id, JReferenceField subField, Deque<JReferenceField> nextFields) {
211        final NavigableMap<?, ?> map = srcTx.tx.readMapField(id, this.storageId, false);
212        if (subField == this.keyField)
213            this.copyRecurse(seen, srcTx, dstTx, map.keySet(), nextFields);
214        else if (subField == this.valueField)
215            this.copyRecurse(seen, srcTx, dstTx, map.values(), nextFields);
216        else
217            throw new RuntimeException();
218    }
219}
