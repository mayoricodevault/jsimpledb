008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.Arrays;
016import java.util.List;
017import java.util.NavigableMap;
018import java.util.NavigableSet;
019
020import org.jsimpledb.change.FieldChange;
021import org.jsimpledb.change.MapFieldAdd;
022import org.jsimpledb.change.MapFieldChange;
023import org.jsimpledb.change.MapFieldClear;
024import org.jsimpledb.change.MapFieldRemove;
025import org.jsimpledb.change.MapFieldReplace;
026import org.jsimpledb.core.MapField;
027import org.jsimpledb.core.Transaction;
028import org.jsimpledb.schema.MapSchemaField;
029import org.objectweb.asm.ClassWriter;
030
031/**
032 * Represents a map field in a {@link JClass}.
033 */
034public class JMapField extends JComplexField {
035
036    private static final int KEY_INDEX_ENTRY_QUERY = 1;
037    private static final int VALUE_INDEX_ENTRY_QUERY = 2;
038
039    final JSimpleField keyField;
040    final JSimpleField valueField;
041
042    JMapField(String name, int storageId, JSimpleField keyField, JSimpleField valueField, String description, Method getter) {
043        super(name, storageId, description, getter);
044        if (keyField == null)
045            throw new IllegalArgumentException("null keyField");
046        if (valueField == null)
047            throw new IllegalArgumentException("null valueField");
048        this.keyField = keyField;
049        this.valueField = valueField;
050    }
051
052    @Override
053    public List<JSimpleField> getSubFields() {
054        return Arrays.asList(this.keyField, this.valueField);
055    }
056
057    @Override
058    public JSimpleField getSubField(String name) {
059        if (MapField.KEY_FIELD_NAME.equals(name))
060            return this.keyField;
061        if (MapField.VALUE_FIELD_NAME.equals(name))
062            return this.valueField;
063        throw new IllegalArgumentException("unknown sub-field `" + name
064          + "' (did you mean `" + MapField.KEY_FIELD_NAME + "' or `" + MapField.VALUE_FIELD_NAME + "' instead?)");
065    }
066
067    @Override
068    String getSubFieldName(JSimpleField subField) {
069        if (subField == this.keyField)
070            return MapField.KEY_FIELD_NAME;
071        if (subField == this.valueField)
072            return MapField.VALUE_FIELD_NAME;
073        throw new IllegalArgumentException("unknown sub-field");
074    }
075
076    @Override
077    MapSchemaField toSchemaItem() {
078        final MapSchemaField schemaField = new MapSchemaField();
079        super.initialize(schemaField);
080        schemaField.setKeyField(this.keyField.toSchemaItem());
081        schemaField.setValueField(this.valueField.toSchemaItem());
082        return schemaField;
083    }
084
085    @Override
086    void outputMethods(ClassGenerator<?> generator, ClassWriter cw) {
087        this.outputReadMethod(generator, cw, ClassGenerator.READ_MAP_FIELD_METHOD);
088    }
089
090    @Override
091    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
092        tx.addMapFieldChangeListener(this.storageId, path, listener);
093    }
094
095    @Override
096    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
097        this.addChangeParameterTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
098    }
099
100    // This method exists solely to bind the generic type parameters
101    @SuppressWarnings("serial")
102    private <T, K, V> void addChangeParameterTypes(List<TypeToken<?>> types,
103      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
104        types.add(new TypeToken<FieldChange<T>>() { }
105          .where(new TypeParameter<T>() { }, targetType));
106        types.add(new TypeToken<MapFieldChange<T>>() { }
107          .where(new TypeParameter<T>() { }, targetType));
108        types.add(new TypeToken<MapFieldAdd<T, K, V>>() { }
109          .where(new TypeParameter<T>() { }, targetType)
110          .where(new TypeParameter<K>() { }, keyType.wrap())
111          .where(new TypeParameter<V>() { }, valueType.wrap()));
112        types.add(new TypeToken<MapFieldClear<T, K, V>>() { }
113          .where(new TypeParameter<T>() { }, targetType)
114          .where(new TypeParameter<K>() { }, keyType.wrap())
115          .where(new TypeParameter<V>() { }, valueType.wrap()));
116        types.add(new TypeToken<MapFieldRemove<T, K, V>>() { }
117          .where(new TypeParameter<T>() { }, targetType)
118          .where(new TypeParameter<K>() { }, keyType.wrap())
119          .where(new TypeParameter<V>() { }, valueType.wrap()));
120        types.add(new TypeToken<MapFieldReplace<T, K, V>>() { }
121          .where(new TypeParameter<T>() { }, targetType)
122          .where(new TypeParameter<K>() { }, keyType.wrap())
123          .where(new TypeParameter<V>() { }, valueType.wrap()));
124    }
125
126    @Override
127    <T> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, JSimpleField subField) {
128        if (subField == this.keyField)
129            this.addKeyIndexEntryReturnTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
130        else if (subField == this.valueField)
131            this.addValueIndexEntryReturnTypes(types, targetType, this.keyField.typeToken, this.valueField.typeToken);
132        else
133            throw new RuntimeException();
134    }
135
136    // This method exists solely to bind the generic type parameters
137    @SuppressWarnings("serial")
138    private <T, K, V> void addKeyIndexEntryReturnTypes(List<TypeToken<?>> types,
139      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
140        types.add(new TypeToken<NavigableMap<K, NavigableSet<MapKeyIndexEntry<T, V>>>>() { }
141          .where(new TypeParameter<T>() { }, targetType)
142          .where(new TypeParameter<K>() { }, keyType.wrap())
143          .where(new TypeParameter<V>() { }, valueType.wrap()));
144    }
145
146    // This method exists solely to bind the generic type parameters
147    @SuppressWarnings("serial")
148    private <T, K, V> void addValueIndexEntryReturnTypes(List<TypeToken<?>> types,
149      TypeToken<T> targetType, TypeToken<K> keyType, TypeToken<V> valueType) {
150        types.add(new TypeToken<NavigableMap<V, NavigableSet<MapValueIndexEntry<T, K>>>>() { }
151          .where(new TypeParameter<T>() { }, targetType)
152          .where(new TypeParameter<K>() { }, keyType.wrap())
153          .where(new TypeParameter<V>() { }, valueType.wrap()));
154    }
155
156    @Override
157    int getIndexEntryQueryType(TypeToken<?> queryObjectType) {
158        return queryObjectType.getRawType().equals(MapKeyIndexEntry.class) ? KEY_INDEX_ENTRY_QUERY :
159          queryObjectType.getRawType().equals(MapValueIndexEntry.class) ? VALUE_INDEX_ENTRY_QUERY : 0;
160    }
161
162    @Override
163    Method getIndexEntryQueryMethod(int queryType) {
164        switch (queryType) {
165        case KEY_INDEX_ENTRY_QUERY:
166            return ClassGenerator.QUERY_MAP_FIELD_KEY_ENTRIES_METHOD;
167        case VALUE_INDEX_ENTRY_QUERY:
168            return ClassGenerator.QUERY_MAP_FIELD_VALUE_ENTRIES_METHOD;
169        default:
170            throw new UnsupportedOperationException();
171        }
172    }
173
174    @Override
175    NavigableMapConverter<?, ?, ?, ?> getConverter(JSimpleDB jdb) {
176        final Converter<?, ?> keyConverter = this.keyField.getConverter(jdb);
177        final Converter<?, ?> valueConverter = this.valueField.getConverter(jdb);
178        if (keyConverter == null && valueConverter == null)
179            return null;
180        return this.createConverter(keyConverter != null ? keyConverter : Converter.identity(),
181          valueConverter != null ? valueConverter : Converter.identity());
182    }
183
184    // This method exists solely to bind the generic type parameters
185    private <K, V, WK, WV> NavigableMapConverter<K, V, WK, WV> createConverter(
186      Converter<K, WK> keyConverter, Converter<V, WV> valueConverter) {
187        return new NavigableMapConverter<K, V, WK, WV>(keyConverter, valueConverter);
188    }
189}
