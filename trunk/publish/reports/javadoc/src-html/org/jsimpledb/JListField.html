008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.Deque;
016import java.util.List;
017import java.util.NavigableMap;
018import java.util.NavigableSet;
019import java.util.Set;
020
021import org.jsimpledb.change.ListFieldAdd;
022import org.jsimpledb.change.ListFieldClear;
023import org.jsimpledb.change.ListFieldRemove;
024import org.jsimpledb.change.ListFieldReplace;
025import org.jsimpledb.core.ObjId;
026import org.jsimpledb.core.Transaction;
027import org.jsimpledb.schema.ListSchemaField;
028import org.objectweb.asm.ClassWriter;
029
030/**
031 * Represents a list field in a {@link JClass}.
032 */
033public class JListField extends JCollectionField {
034
035    private static final int LIST_INDEX_ENTRY_QUERY = 1;
036
037    JListField(String name, int storageId, JSimpleField elementField, String description, Method getter) {
038        super(name, storageId, elementField, description, getter);
039    }
040
041    @Override
042    public List<?> getValue(JTransaction jtx, JObject jobj) {
043        if (jtx == null)
044            throw new IllegalArgumentException("null jtx");
045        if (jobj == null)
046            throw new IllegalArgumentException("null jobj");
047        return jtx.readListField(jobj, this.storageId, false);
048    }
049
050    @Override
051    ListSchemaField toSchemaItem() {
052        final ListSchemaField schemaField = new ListSchemaField();
053        super.initialize(schemaField);
054        return schemaField;
055    }
056
057    @Override
058    void outputMethods(ClassGenerator<?> generator, ClassWriter cw) {
059        this.outputReadMethod(generator, cw, ClassGenerator.READ_LIST_FIELD_METHOD);
060    }
061
062    @Override
063    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
064        tx.addListFieldChangeListener(this.storageId, path, listener);
065    }
066
067    @Override
068    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
069        this.addChangeParameterTypes(types, targetType, this.elementField.typeToken);
070    }
071
072    // This method exists solely to bind the generic type parameters
073    @SuppressWarnings("serial")
074    private <T, E> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<E> elementType) {
075        types.add(new TypeToken<ListFieldAdd<T, E>>() { }
076          .where(new TypeParameter<T>() { }, targetType)
077          .where(new TypeParameter<E>() { }, elementType.wrap()));
078        types.add(new TypeToken<ListFieldClear<T>>() { }
079          .where(new TypeParameter<T>() { }, targetType));
080        types.add(new TypeToken<ListFieldRemove<T, E>>() { }
081          .where(new TypeParameter<T>() { }, targetType)
082          .where(new TypeParameter<E>() { }, elementType.wrap()));
083        types.add(new TypeToken<ListFieldReplace<T, E>>() { }
084          .where(new TypeParameter<T>() { }, targetType)
085          .where(new TypeParameter<E>() { }, elementType.wrap()));
086    }
087
088    @Override
089    <T> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, JSimpleField subField) {
090        assert subField == this.elementField;
091        this.addIndexEntryReturnTypes(types, targetType, this.elementField.typeToken);
092    }
093
094    // This method exists solely to bind the generic type parameters
095    @SuppressWarnings("serial")
096    private <T, E> void addIndexEntryReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<E> elementType) {
097        types.add(new TypeToken<NavigableMap<E, NavigableSet<ListIndexEntry<T>>>>() { }
098          .where(new TypeParameter<T>() { }, targetType)
099          .where(new TypeParameter<E>() { }, elementType.wrap()));
100    }
101
102    @Override
103    int getIndexEntryQueryType(TypeToken<?> queryObjectType) {
104        return queryObjectType.getRawType().equals(ListIndexEntry.class) ? LIST_INDEX_ENTRY_QUERY : 0;
105    }
106
107    @Override
108    Method getIndexEntryQueryMethod(int queryType) {
109        switch (queryType) {
110        case LIST_INDEX_ENTRY_QUERY:
111            return ClassGenerator.QUERY_LIST_FIELD_ENTRIES_METHOD;
112        default:
113            throw new UnsupportedOperationException();
114        }
115    }
116
117    @Override
118    ListConverter<?, ?> getConverter(JTransaction jtx) {
119        final Converter<?, ?> elementConverter = this.elementField.getConverter(jtx);
120        if (elementConverter == null)
121            return null;
122        return this.createConverter(elementConverter);
123    }
124
125    // This method exists solely to bind the generic type parameters
126    private <X, Y> ListConverter<X, Y> createConverter(Converter<X, Y> elementConverter) {
127        return new ListConverter<X, Y>(elementConverter);
128    }
129
130    @Override
131    void copyRecurse(Set<ObjId> seen, JTransaction srcTx, JTransaction dstTx,
132      ObjId id, JReferenceField subField, Deque<JReferenceField> nextFields) {
133        assert subField == this.elementField;
134        this.copyRecurse(seen, srcTx, dstTx, srcTx.tx.readListField(id, this.storageId, false), nextFields);
135    }
136}
