008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.List;
016import java.util.NavigableMap;
017import java.util.NavigableSet;
018
019import org.dellroad.stuff.java.Primitive;
020import org.jsimpledb.change.SimpleFieldChange;
021import org.jsimpledb.core.Transaction;
022import org.jsimpledb.schema.SimpleSchemaField;
023import org.objectweb.asm.ClassWriter;
024import org.objectweb.asm.MethodVisitor;
025import org.objectweb.asm.Opcodes;
026import org.objectweb.asm.Type;
027
028/**
029 * Represents a simple field in a {@link JClass} or a simple sub-field of a complex field in a {@link JClass}.
030 */
031public class JSimpleField extends JField {
032
033    final TypeToken<?> typeToken;
034    final String typeName;
035    final boolean indexed;
036    final Method setter;
037
038    JSimpleField(String name, int storageId, TypeToken<?> typeToken,
039      String typeName, boolean indexed, String description, Method getter, Method setter) {
040        super(name, storageId, description, getter);
041        if (typeName == null)
042            throw new IllegalArgumentException("null typeName");
043        if (typeToken == null)
044            throw new IllegalArgumentException("null typeToken");
045        this.typeToken = typeToken;
046        this.typeName = typeName;
047        this.indexed = indexed;
048        this.setter = setter;
049    }
050
051    /**
052     * Get name of this field's {@link org.jsimpledb.core.FieldType}.
053     */
054    public String getTypeName() {
055        return this.typeName;
056    }
057
058    /**
059     * Get whether this field is indexed.
060     */
061    public boolean isIndexed() {
062        return this.indexed;
063    }
064
065    /**
066     * Get the setter method associated with this field.
067     *
068     * @return field property setter method, or null if this field is a sub-field of a complex field
069     */
070    public Method getSetter() {
071        return this.setter;
072    }
073
074    @Override
075    SimpleSchemaField toSchemaItem() {
076        final SimpleSchemaField schemaField = new SimpleSchemaField();
077        this.initialize(schemaField);
078        return schemaField;
079    }
080
081    void initialize(SimpleSchemaField schemaField) {
082        super.initialize(schemaField);
083        schemaField.setType(this.typeName);
084        schemaField.setIndexed(this.indexed);
085    }
086
087    @Override
088    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
089
090        // Getter
091        this.outputReadMethod(generator, cw, ClassGenerator.READ_SIMPLE_FIELD_METHOD);
092
093        // Setter
094        final Method writeMethod = ClassGenerator.WRITE_SIMPLE_FIELD_METHOD;
095        generator.overrideBeanMethod(cw, this.setter, this.storageId, new ClassGenerator.CodeEmitter() {
096            @Override
097            public void emit(MethodVisitor mv) {
098
099                // Push field value
100                mv.visitVarInsn(Type.getType(JSimpleField.this.typeToken.getRawType()).getOpcode(Opcodes.ILOAD), 1);
101
102                // Wrap result if needed
103                if (JSimpleField.this.typeToken.isPrimitive())
104                    generator.wrap(mv, Primitive.get(JSimpleField.this.typeToken.getRawType()));
105
106                // Invoke Transaction.writeSimpleField()
107                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
108                  writeMethod.getName(), Type.getMethodDescriptor(writeMethod));
109            }
110        });
111    }
112
113    @Override
114    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
115        tx.addSimpleFieldChangeListener(this.storageId, path, listener);
116    }
117
118    @Override
119    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
120        this.addChangeParameterTypes(types, targetType, this.typeToken);
121    }
122
123    // This method exists solely to bind the generic type parameters
124    @SuppressWarnings("serial")
125    private <T, V> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<V> fieldType) {
126        types.add(new TypeToken<SimpleFieldChange<T, V>>() { }
127          .where(new TypeParameter<T>() { }, targetType)
128          .where(new TypeParameter<V>() { }, fieldType.wrap()));
129    }
130
131    /**
132     * Add valid return types for @IndexQuery-annotated methods that query this indexed field.
133     */
134    <T> void addIndexReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
135        this.addIndexReturnTypes(types, targetType, this.typeToken);
136    }
137
138    // This method exists solely to bind the generic type parameters
139    @SuppressWarnings("serial")
140    private <T, V> void addIndexReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<V> fieldType) {
141        types.add(new TypeToken<NavigableMap<V, NavigableSet<T>>>() { }
142          .where(new TypeParameter<V>() { }, fieldType.wrap())
143          .where(new TypeParameter<T>() { }, targetType));
144    }
145
146    @Override
147    @SuppressWarnings({ "unchecked", "rawtypes" })
148    Converter<?, ?> getConverter(JTransaction jtx) {
149        if (Enum.class.isAssignableFrom(this.typeToken.getRawType()))
150            return this.createEnumConverter((TypeToken<Enum>)this.typeToken);
151        return null;
152    }
153
154    // This method exists solely to bind the generic type parameters
155    private <T extends Enum<T>> EnumConverter<T> createEnumConverter(TypeToken<T> etype) {
156        return new EnumConverter<T>(etype);
157    }
158}
