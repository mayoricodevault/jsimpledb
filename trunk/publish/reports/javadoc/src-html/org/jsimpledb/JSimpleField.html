008package org.jsimpledb;
009
010import com.google.common.reflect.TypeToken;
011
012import java.lang.reflect.Method;
013
014import org.dellroad.stuff.java.Primitive;
015import org.jsimpledb.schema.SimpleSchemaField;
016import org.objectweb.asm.ClassWriter;
017import org.objectweb.asm.MethodVisitor;
018import org.objectweb.asm.Opcodes;
019import org.objectweb.asm.Type;
020
021/**
022 * Represents a simple field in a {@link JClass} or a simple sub-field of a complex field in a {@link JClass}.
023 */
024public class JSimpleField extends JField {
025
026    final TypeToken<?> typeToken;
027    final String typeName;
028    final boolean indexed;
029    final Method setter;
030
031    JSimpleField(String name, int storageId, TypeToken<?> typeToken,
032      String typeName, boolean indexed, String description, Method getter, Method setter) {
033        super(name, storageId, description, getter);
034        if (typeName == null)
035            throw new IllegalArgumentException("null typeName");
036        if (typeToken == null)
037            throw new IllegalArgumentException("null typeToken");
038        this.typeToken = typeToken;
039        this.typeName = typeName;
040        this.indexed = indexed;
041        this.setter = setter;
042    }
043
044    /**
045     * Get the {@link JComplexField} of which this instance is a sub-field, if any.
046     *
047     * @return parent {@link JComplexField}, or null if this instance is not a sub-field
048     */
049    public JComplexField getParentField() {
050        return this.parent instanceof JComplexField ? (JComplexField)this.parent : null;
051    }
052
053    /**
054     * Get the type of this field.
055     */
056    public TypeToken<?> getType() {
057        return this.typeToken;
058    }
059
060    /**
061     * Get name of this field's {@link org.jsimpledb.core.FieldType}.
062     */
063    public String getTypeName() {
064        return this.typeName;
065    }
066
067    /**
068     * Get whether this field is indexed.
069     */
070    public boolean isIndexed() {
071        return this.indexed;
072    }
073
074    /**
075     * Get the setter method associated with this field.
076     *
077     * @return field property setter method, or null if this field is a sub-field of a complex field
078     */
079    public Method getSetter() {
080        return this.setter;
081    }
082
083    @Override
084    public Object getValue(JTransaction jtx, JObject jobj) {
085        if (jtx == null)
086            throw new IllegalArgumentException("null jtx");
087        if (jobj == null)
088            throw new IllegalArgumentException("null jobj");
089        return jtx.readSimpleField(jobj, this.storageId, false);
090    }
091
092    @Override
093    public <R> R visit(JFieldSwitch<R> target) {
094        return target.caseJSimpleField(this);
095    }
096
097    /**
098     * Set the Java value of this field in the given object.
099     * Does not alter the schema version of the object.
100     *
101     * @param jtx transaction
102     * @param jobj object containing this field
103     * @param value new value
104     * @throws DeletedObjectException if {@code jobj} does not exist in {@code jtx}
105     * @throws StaleTransactionException if {@code jtx} is no longer usable
106     * @throws IllegalArgumentException if {@code value} is not an appropriate value for this field
107     * @throws IllegalArgumentException if {@code jtx} or {@code jobj} is null
108     */
109    public void setValue(JTransaction jtx, JObject jobj, Object value) {
110        if (jtx == null)
111            throw new IllegalArgumentException("null jtx");
112        if (jobj == null)
113            throw new IllegalArgumentException("null jobj");
114        jtx.writeSimpleField(jobj, this.storageId, value, false);
115    }
116
117    @Override
118    SimpleSchemaField toSchemaItem(JSimpleDB jdb) {
119        final SimpleSchemaField schemaField = new SimpleSchemaField();
120        this.initialize(jdb, schemaField);
121        return schemaField;
122    }
123
124    void initialize(JSimpleDB jdb, SimpleSchemaField schemaField) {
125        super.initialize(jdb, schemaField);
126        schemaField.setType(this.typeName);
127        schemaField.setIndexed(this.indexed);
128    }
129
130    @Override
131    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
132
133        // Getter
134        this.outputReadMethod(generator, cw, ClassGenerator.READ_SIMPLE_FIELD_METHOD);
135
136        // Setter
137        final Method writeMethod = ClassGenerator.WRITE_SIMPLE_FIELD_METHOD;
138        generator.overrideBeanMethod(cw, this.setter, this.storageId, new ClassGenerator.CodeEmitter() {
139            @Override
140            public void emit(MethodVisitor mv) {
141
142                // Push field value
143                mv.visitVarInsn(Type.getType(JSimpleField.this.typeToken.getRawType()).getOpcode(Opcodes.ILOAD), 1);
144
145                // Wrap result if needed
146                if (JSimpleField.this.typeToken.isPrimitive())
147                    generator.wrap(mv, Primitive.get(JSimpleField.this.typeToken.getRawType()));
148
149                // Push "true"
150                mv.visitInsn(Opcodes.ICONST_1);
151
152                // Invoke Transaction.writeSimpleField()
153                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
154                  writeMethod.getName(), Type.getMethodDescriptor(writeMethod));
155            }
156        });
157    }
158
159    @Override
160    final JSimpleFieldInfo toJFieldInfo() {
161        return this.toJFieldInfo(0);
162    }
163
164    JSimpleFieldInfo toJFieldInfo(int parentStorageId) {
165        return new JSimpleFieldInfo(this, parentStorageId);
166    }
167}
