008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeParameter;
012import com.google.common.reflect.TypeToken;
013
014import java.lang.reflect.Method;
015import java.util.List;
016import java.util.NavigableMap;
017import java.util.NavigableSet;
018
019import org.dellroad.stuff.java.Primitive;
020import org.jsimpledb.change.SimpleFieldChange;
021import org.jsimpledb.core.Transaction;
022import org.jsimpledb.schema.SimpleSchemaField;
023import org.objectweb.asm.ClassWriter;
024import org.objectweb.asm.MethodVisitor;
025import org.objectweb.asm.Opcodes;
026import org.objectweb.asm.Type;
027
028/**
029 * Represents a simple field in a {@link JClass} or a simple sub-field of a complex field in a {@link JClass}.
030 */
031public class JSimpleField extends JField {
032
033    final TypeToken<?> typeToken;
034    final String typeName;
035    final boolean indexed;
036    final Method setter;
037
038    JSimpleField(String name, int storageId, TypeToken<?> typeToken,
039      String typeName, boolean indexed, String description, Method getter, Method setter) {
040        super(name, storageId, description, getter);
041        if (typeName == null)
042            throw new IllegalArgumentException("null typeName");
043        if (typeToken == null)
044            throw new IllegalArgumentException("null typeToken");
045        this.typeToken = typeToken;
046        this.typeName = typeName;
047        this.indexed = indexed;
048        this.setter = setter;
049    }
050
051    /**
052     * Get the {@link JComplexField} of which this instance is a sub-field, if any.
053     *
054     * @return parent {@link JComplexField}, or null if this instance is not a sub-field
055     */
056    public JComplexField getParentField() {
057        return this.parent instanceof JComplexField ? (JComplexField)this.parent : null;
058    }
059
060    /**
061     * Get name of this field's {@link org.jsimpledb.core.FieldType}.
062     */
063    public String getTypeName() {
064        return this.typeName;
065    }
066
067    /**
068     * Get whether this field is indexed.
069     */
070    public boolean isIndexed() {
071        return this.indexed;
072    }
073
074    /**
075     * Get the setter method associated with this field.
076     *
077     * @return field property setter method, or null if this field is a sub-field of a complex field
078     */
079    public Method getSetter() {
080        return this.setter;
081    }
082
083    @Override
084    public Object getValue(JTransaction jtx, JObject jobj) {
085        if (jtx == null)
086            throw new IllegalArgumentException("null jtx");
087        if (jobj == null)
088            throw new IllegalArgumentException("null jobj");
089        return jtx.readSimpleField(jobj, this.storageId, false);
090    }
091
092    /**
093     * Set the Java value of this field in the given object.
094     * Does not alter the schema version of the object.
095     *
096     * @param jtx transaction
097     * @param jobj object containing this field
098     * @param value new value
099     * @throws DeletedObjectException if {@code jobj} does not exist in {@code jtx}
100     * @throws StaleTransactionException if {@code jtx} is no longer usable
101     * @throws IllegalArgumentException if {@code value} is not an appropriate value for this field
102     * @throws IllegalArgumentException if {@code jtx} or {@code jobj} is null
103     */
104    public void setValue(JTransaction jtx, JObject jobj, Object value) {
105        if (jtx == null)
106            throw new IllegalArgumentException("null jtx");
107        if (jobj == null)
108            throw new IllegalArgumentException("null jobj");
109        jtx.writeSimpleField(jobj, this.storageId, value, false);
110    }
111
112    /**
113     * Get a {@link Converter} that converts from core {@link org.jsimpledb.core.Database} values
114     * to this field's Java model values.
115     *
116     * <p>
117     * Currently, these are the only field types that require conversion:
118     * <ul>
119     *  <li>Reference fields require conversion from {@link org.jsimpledb.core.ObjId} to Java model object
120     *  (i.e., {@link JObject})</li>
121     *  <li>{@link Enum} types require conversion from {@link org.jsimpledb.core.EnumValue} to Java {@link Enum} values</li>
122     * </ul>
123     * </p>
124     *
125     * @return value {@link Converter} for this field, or null if no conversion is required
126     */
127    @Override
128    @SuppressWarnings({ "unchecked", "rawtypes" })
129    public Converter<?, ?> getConverter(JTransaction jtx) {
130        if (Enum.class.isAssignableFrom(this.typeToken.getRawType()))
131            return new EnumConverter(this.typeToken.getRawType());
132        return null;
133    }
134
135    @Override
136    SimpleSchemaField toSchemaItem() {
137        final SimpleSchemaField schemaField = new SimpleSchemaField();
138        this.initialize(schemaField);
139        return schemaField;
140    }
141
142    void initialize(SimpleSchemaField schemaField) {
143        super.initialize(schemaField);
144        if (this.parent instanceof JComplexField)
145            schemaField.setName(null);
146        schemaField.setType(this.typeName);
147        schemaField.setIndexed(this.indexed);
148    }
149
150    @Override
151    void outputMethods(final ClassGenerator<?> generator, ClassWriter cw) {
152
153        // Getter
154        this.outputReadMethod(generator, cw, ClassGenerator.READ_SIMPLE_FIELD_METHOD);
155
156        // Setter
157        final Method writeMethod = ClassGenerator.WRITE_SIMPLE_FIELD_METHOD;
158        generator.overrideBeanMethod(cw, this.setter, this.storageId, new ClassGenerator.CodeEmitter() {
159            @Override
160            public void emit(MethodVisitor mv) {
161
162                // Push field value
163                mv.visitVarInsn(Type.getType(JSimpleField.this.typeToken.getRawType()).getOpcode(Opcodes.ILOAD), 1);
164
165                // Wrap result if needed
166                if (JSimpleField.this.typeToken.isPrimitive())
167                    generator.wrap(mv, Primitive.get(JSimpleField.this.typeToken.getRawType()));
168
169                // Push "true"
170                mv.visitInsn(Opcodes.ICONST_1);
171
172                // Invoke Transaction.writeSimpleField()
173                mv.visitMethodInsn(Opcodes.INVOKEVIRTUAL, Type.getInternalName(JTransaction.class),
174                  writeMethod.getName(), Type.getMethodDescriptor(writeMethod));
175            }
176        });
177    }
178
179    @Override
180    void registerChangeListener(Transaction tx, int[] path, AllChangesListener listener) {
181        tx.addSimpleFieldChangeListener(this.storageId, path, listener);
182    }
183
184    @Override
185    <T> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
186        this.addChangeParameterTypes(types, targetType, this.typeToken);
187    }
188
189    // This method exists solely to bind the generic type parameters
190    @SuppressWarnings("serial")
191    private <T, V> void addChangeParameterTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<V> fieldType) {
192        types.add(new TypeToken<SimpleFieldChange<T, V>>() { }
193          .where(new TypeParameter<T>() { }, targetType)
194          .where(new TypeParameter<V>() { }, fieldType.wrap()));
195    }
196
197    /**
198     * Add valid return types for @IndexQuery-annotated methods that query this indexed field.
199     */
200    <T> void addIndexReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType) {
201        this.addIndexReturnTypes(types, targetType, this.typeToken);
202    }
203
204    // This method exists solely to bind the generic type parameters
205    @SuppressWarnings("serial")
206    private <T, V> void addIndexReturnTypes(List<TypeToken<?>> types, TypeToken<T> targetType, TypeToken<V> fieldType) {
207        types.add(new TypeToken<NavigableMap<V, NavigableSet<T>>>() { }
208          .where(new TypeParameter<V>() { }, fieldType.wrap())
209          .where(new TypeParameter<T>() { }, targetType));
210    }
211}
