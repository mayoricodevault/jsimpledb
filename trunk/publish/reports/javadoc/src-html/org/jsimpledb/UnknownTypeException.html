008package org.jsimpledb;
009
010import org.jsimpledb.core.ObjId;
011
012/**
013 * Thrown when attempting to get the {@link JObject} associated with an {@link ObjId} that
014 * has an unrecognized type (i.e., storage ID).
015 *
016 * <p>
017 * This exception should not occur during normal usage. It may occur when using both the higher-level {@link JSimpleDB} Java model
018 * object API and the lower level core database API at the same time in an incompatible way.
019 * </p>
020 *
021 * @see JSimpleDB#getJObject
022 */
023@SuppressWarnings("serial")
024public class UnknownTypeException extends JSimpleDBException {
025
026    private final ObjId id;
027    private final int version;
028
029    /**
030     * Constructor.
031     */
032    public UnknownTypeException(ObjId id, int version) {
033        this(id, version, "no model class with storage ID " + id.getStorageId() + " exists in database schema version " + version);
034    }
035
036    /**
037     * Constructor.
038     */
039    public UnknownTypeException(ObjId id, int version, String message) {
040        super(message);
041        this.id = id;
042        this.version = version;
043    }
044
045    /**
046     * Get the {@link ObjId} that has the unrecognized type.
047     */
048    public ObjId getObjId() {
049        return this.id;
050    }
051
052    /**
053     * Get the {@link JSimpleDB}'s schema version, which did not contain the unrecognized type.
054     */
055    public int getSchemaVersion() {
056        return this.version;
057    }
058}
