010import java.io.BufferedInputStream;
011import java.io.BufferedOutputStream;
012import java.io.File;
013import java.io.FileInputStream;
014import java.io.FileOutputStream;

016import java.util.HashSet;
017
018import org.apache.tools.ant.AntClassLoader;
019import org.apache.tools.ant.BuildException;
020import org.apache.tools.ant.Task;
021import org.apache.tools.ant.types.Path;
022import org.apache.tools.ant.types.Reference;
023import org.jsimpledb.DefaultStorageIdGenerator;
024import org.jsimpledb.JSimpleDBFactory;
025import org.jsimpledb.StorageIdGenerator;
026import org.jsimpledb.annotation.JFieldType;

034/**
035 * Ant task for schema XML generation and/or verification.
036 *
037 * <p>
038 * This task scans the configured classpath for classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
039 * and {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations and either writes the generated schema
040 * to an XML file, or verifies the schema from an existing XML file.
041 * </p>
042 *
043 * <p>
044 * The following tasks are supported:
045 * </p>
046 *
047 * <div style="margin-left: 20px;">
048 * <table border="1" cellpadding="3" cellspacing="0" summary="Supported Tasks">
049 * <tr style="bgcolor:#ccffcc">
050 *  <th align="left">Attribute</th>
051 *  <th align="left">Required?</th>
052 *  <th align="left">Description</th>
053 * </tr>
054 * <tr>
055 *  <td>{@code mode}</td>
056 *  <td>No</td>
057 *  <td>
058 *      <p>
059 *      Set to {@code generate} to generate a new XML file, or {@code verify} to verify an existing XML file.
060 *      </p>
061 *
062 *      <p>
063 *      Default is {@code verify}.
064 *      </p>
065 * </td>
066 * </tr>
067 * <tr>
068 *  <td>{@code file}</td>
069 *  <td>Yes</td>
070 *  <td>
071 *      <p>
072 *      The XML file to generate or verify.
073 *      </p>
074 * </td>
075 * </tr>
076 * <tr>
077 *  <td>{@code matchNames}</td>
078 *  <td>No</td>
079 *  <td>
080 *      <p>
081 *      Whether to verify not only schema compatibility but also that the two schemas are identical, i.e.,
082 *      the same names are used for object types, fields, and composite indexes.
083 *      </p>
084 *
085 *      <p>
086 *      Two schemas that are equivalent except for names are considered compatible, because the core API uses
087 *      storage ID's, not names. However, if names change then some JSimpleDB layer operations, such as index
088 *      queries and reference path inversion, may need to be updated.
089 *      </p>
090 *
091 *      <p>
092 *      Default is {@code true}. Ignored unless {@code mode} is {@code verify}.
093 *      </p>
094 * </td>
095 * </tr>
096 * <tr>
097 *  <td>{@code failOnError}</td>
098 *  <td>No</td>
099 *  <td>
100 *      <p>
101 *      Whether to fail if verification fails.
102 *      </p>
103 *
104 *      <p>
105 *      Default is {@code true}. Ignored unless {@code mode} is {@code verify}.
106 *      </p>
107 * </td>
108 * </tr>
109 * <tr>
110 *  <td>{@code verifiedProperty}</td>
111 *  <td>No</td>
112 *  <td>
113 *      <p>
114 *      The name of an ant property to set to {@code true} or {@code false} depending on whether
115 *      verification succeeded or failed. Useful when {@code failOnError} is set to {@code false}
116 *      and you want to handle the failure elsewhere in the build file.
117 *      </p>
118 *
119 *      <p>
120 *      Default is to not set any property.
121 *      </p>
122 * </td>
123 * </tr>
124 * <tr>
125 *  <td>{@code classpath} or {@code classpathref}</td>
126 *  <td>Yes</td>
127 *  <td>
128 *      <p>
129 *      Specifies the search path for classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
130 *      annotations.
131 *      </p>
132 * </td>
133 * </tr>
134 * <tr>
135 *  <td>{@code packages}</td>
136 *  <td>Yes</td>
137 *  <td>
138 *      <p>
139 *      Specifies one or more Java package names (separated by commas and/or whitespace) under which to look
140 *      for classes with {@link org.jsimpledb.annotation.JSimpleClass &#64;JSimpleClass}
141 *      and {@link org.jsimpledb.annotation.JFieldType &#64;JFieldType} annotations.
142 *      </p>
143 * </td>
144 * </tr>
145 * <tr>
146 *  <td>{@code storageIdGeneratorClass}</td>
147 *  <td>No</td>
148 *  <td>
149 *      <p>
150 *      Specifies the name of an optional custom {@link StorageIdGenerator} class.
151 *      </p>
152 *
153 *      <p>
154 *      By default, a {@link DefaultStorageIdGenerator} is used.
155 *      </p>
156 * </td>
157 * </tr>
158 * </table>
159 * </div>
160 *
161 * <p>
162 * Example:
163 * <pre>
164 *  &lt;project xmlns:jsimpledb="urn:org.dellroad.jsimpledb" ... &gt;
165 *      ...
166 *      &lt;taskdef uri="urn:org.dellroad.jsimpledb" name="schema"
167 *        classname="org.jsimpledb.ant.SchemaGeneratorTask" classpathref="jsimpledb.classpath"/&gt;
168 *      &lt;jsimpledb:schema mode="verify" classpathref="myclasses.classpath"
169 *        file="expected-schema.xml" packages="com.example.model"/&gt;
170 * </pre>
171 *
172 * @see org.jsimpledb.JSimpleDB
173 * @see org.jsimpledb.schema.SchemaModel
174 */
175public class SchemaGeneratorTask extends Task {
176
177    public static final String MODE_VERIFY = "verify";
178    public static final String MODE_GENERATE = "generate";
179
180    private String[] packages;
181    private String mode = MODE_VERIFY;
182    private boolean matchNames = true;
183    private boolean failOnError = true;
184    private String verifiedProperty;
185    private File file;
186    private Path classPath;
187    private String storageIdGeneratorClassName = DefaultStorageIdGenerator.class.getName();
188
189    public void setPackages(String packages) {
190        this.packages = packages.split("[\\s,]");
191    }
192
193    public void setMode(String mode) {
194        this.mode = mode;
195    }
196
197    public void setMatchNames(boolean matchNames) {
198        this.matchNames = matchNames;
199    }
200
201    public void setFailOnError(boolean failOnError) {
202        this.failOnError = failOnError;
203    }
204
205    public void setVerifiedProperty(String verifiedProperty) {
206        this.verifiedProperty = verifiedProperty;
207    }
208
209    public void setFile(File file) {
210        this.file = file;
211    }
212
213    public Path createClasspath() {
214        this.classPath = new Path(this.getProject());
215        return this.classPath;
216    }
217
218    public void setClasspath(Path classPath) {
219        this.classPath = classPath;
220    }
221
222    public void setClasspathRef(Reference ref) {
223        this.classPath = (Path)ref.getReferencedObject(this.getProject());
224    }
225
226    public void setStorageIdGeneratorClass(String storageIdGeneratorClassName) {
227        this.storageIdGeneratorClassName = storageIdGeneratorClassName;
228    }
229
230    /**
231     * @throws BuildException if operation fails
232     */
233    @Override
234    public void execute() {
235
236        // Sanity check
237        if (this.file == null)
238            throw new BuildException("`file' attribute is required specifying output/verify file");
239        final boolean generate;
240        switch (this.mode) {
241        case MODE_VERIFY:
242            generate = false;
243            break;
244        case MODE_GENERATE:
245            generate = true;
246            break;
247        default:
248            throw new BuildException("`mode' attribute must be one of `" + MODE_VERIFY + "' or `" + MODE_GENERATE + "'");
249        }
250        if (this.packages == null)
251            throw new BuildException("`packages' attribute is required specifying packages to scan for Java model classes");
252        if (this.classPath == null)
253            throw new BuildException("`classpath' attribute is required specifying search path for scanned classes");
254
255        // Create directory containing file
256        if (generate && this.file.getParent() != null && !this.file.getParentFile().exists() && !this.file.getParentFile().mkdirs())
257            throw new BuildException("error creating directory `" + this.file.getParentFile() + "'");
258
259        // Set up mysterious classloader stuff
260        final AntClassLoader loader = this.getProject().createClassLoader(this.classPath);
261        final ClassLoader currentLoader = this.getClass().getClassLoader();
262        if (currentLoader != null)
263            loader.setParent(currentLoader);
264        loader.setThreadContextLoader();
265        try {
266
267            // Scan for @JSimpleClass classes
268            final HashSet<Class<?>> modelClasses = new HashSet<>();
269            for (String className : new JSimpleDBClassScanner().scanForClasses(this.packages)) {
270                this.log("adding JSimpleDB schema class " + className);
271                try {
272                    modelClasses.add(Class.forName(className, false, Thread.currentThread().getContextClassLoader()));
273                } catch (ClassNotFoundException e) {
274                    throw new BuildException("failed to load class `" + className + "'", e);
275                }
276            }
277
278            // Scan for @JFieldType classes
279            final HashSet<FieldType<?>> fieldTypes = new HashSet<>();
280            for (String className : new JSimpleDBFieldTypeScanner().scanForClasses(this.packages)) {
281                this.log("adding JSimpleDB field type class " + className);
282                try {
283                    fieldTypes.add(this.asFieldTypeClass(
284                      Class.forName(className, false, Thread.currentThread().getContextClassLoader())).newInstance());
285                } catch (Exception e) {
286                    throw new BuildException("failed to instantiate class `" + className + "'", e);
287                }
288            }
289
290            // Instantiate StorageIdGenerator
291            final StorageIdGenerator storageIdGenerator;
292            try {
293                storageIdGenerator = Class.forName(this.storageIdGeneratorClassName,
294                  false, Thread.currentThread().getContextClassLoader()).asSubclass(StorageIdGenerator.class).newInstance();
295            } catch (Exception e) {
296                throw new BuildException("failed to instantiate class `" + storageIdGeneratorClassName + "'", e);
297            }
298
299            // Set up database and configure field type classes
300            final Database db = new Database(new SimpleKVDatabase());
301            for (FieldType<?> fieldType : fieldTypes) {
302                try {
303                    db.getFieldTypeRegistry().add(fieldType);
304                } catch (Exception e) {
305                    throw new BuildException("failed to register custom field type " + fieldType.getClass().getName(), e);
306                }
307            }
308
309            // Set up factory
310            final JSimpleDBFactory factory = new JSimpleDBFactory();
311            factory.setDatabase(db);
312            factory.setSchemaVersion(1);
313            factory.setStorageIdGenerator(storageIdGenerator);
314            factory.setModelClasses(modelClasses);
315
316            // Build schema model
317            this.log("generating JSimpleDB schema from schema classes");
318            final SchemaModel schemaModel;
319            try {
320                schemaModel = factory.newJSimpleDB().getSchemaModel();
321            } catch (Exception e) {
322                throw new BuildException("schema generation failed: " + e, e);
323            }
324
325            // Verify or generate
326            if (generate) {
327
328                // Write schema model to file
329                this.log("writing JSimpleDB schema to `" + this.file + "'");
330                try (BufferedOutputStream output = new BufferedOutputStream(new FileOutputStream(this.file))) {
331                    schemaModel.toXML(output, true);
332                } catch (IOException e) {
333                    throw new BuildException("error writing schema to `" + this.file + "': " + e, e);
334                }
335            } else {
336
337                // Read file
338                this.log("verifying JSimpleDB schema matches `" + this.file + "'");
339                final SchemaModel verifyModel;
340                try (BufferedInputStream input = new BufferedInputStream(new FileInputStream(this.file))) {
341                    verifyModel = SchemaModel.fromXML(input);
342                } catch (IOException e) {
343                    throw new BuildException("error reading schema from `" + this.file + "': " + e, e);
344                }
345
346                // Compare
347                final boolean verified = matchNames ? schemaModel.equals(verifyModel) : schemaModel.isCompatibleWith(verifyModel);
348                if (this.verifiedProperty != null)
349                    this.getProject().setProperty(this.verifiedProperty, "" + verified);
350                this.log("schema verification " + (verified ? "succeeded" : "failed"));
351                if (!verified) {
352                    this.log(schemaModel.differencesFrom(verifyModel).toString());
353                    if (this.failOnError)
354                        throw new BuildException("schema verification failed");
355                }
356            }
357        } finally {
358            loader.resetThreadContextLoader();
359            loader.cleanup();
360        }
361    }
362
363    @SuppressWarnings("unchecked")
364    private Class<? extends FieldType<?>> asFieldTypeClass(Class<?> klass) {
365        try {
366            return (Class<? extends FieldType<?>>)klass.asSubclass(FieldType.class);
367        } catch (ClassCastException e) {
368            throw new BuildException("invalid @" + JFieldType.class.getSimpleName() + " annotation on "
369              + klass + ": type is not a subclass of " + FieldType.class);
370        }
371    }
372}
