008package org.jsimpledb.parse.expr;
009
010import org.jsimpledb.parse.ParseSession;
011
012/**
013 * Parses logical OR expressions of the form {@code x || y}.
014 */
015public class LogicalOrParser extends BinaryExprParser {
016
017    public static final LogicalOrParser INSTANCE = new LogicalOrParser();
018
019    public LogicalOrParser() {
020        super(LogicalAndParser.INSTANCE, Op.LOGICAL_OR);
021    }
022
023    // Overridden to provide short-circuit logic
024    @Override
025    protected Node createNode(final Op op, final Node lhNode, final Node rhNode) {
026        return new Node() {
027            @Override
028            public Value evaluate(ParseSession session) {
029                for (Node node : new Node[] { lhNode, rhNode }) {
030                    if (node.evaluate(session).checkBoolean(session, "logical `or'"))
031                        return new ConstValue(true);
032                }
033                return new ConstValue(false);
034            }
035        };
036    }
037}
