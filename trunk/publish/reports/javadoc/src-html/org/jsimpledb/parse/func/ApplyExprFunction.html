008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseException;
012import org.jsimpledb.parse.ParseSession;
013import org.jsimpledb.parse.expr.AtomParser;
014import org.jsimpledb.parse.expr.ExprParser;
015import org.jsimpledb.parse.expr.Node;
016import org.jsimpledb.parse.expr.Value;
017import org.jsimpledb.parse.expr.VarNode;
018
019public abstract class ApplyExprFunction extends AbstractFunction {
020
021    protected ApplyExprFunction(String name) {
022        super(name);
023    }
024
025    @Override
026    public ParamInfo parseParams(ParseSession session, ParseContext ctx, boolean complete) {
027
028        // Get items
029        final int mark = ctx.getIndex();
030        if (ctx.tryLiteral(")"))
031            throw new ParseException(ctx, "three parameters required for " + this.name + "()");
032        final Node items = new ExprParser().parse(session, ctx, complete);
033
034        // Get variable
035        ctx.skipWhitespace();
036        if (!ctx.tryLiteral(","))
037            throw new ParseException(ctx, "expected `,'").addCompletion(", ");
038        this.spaceParser.parse(ctx, complete);
039        final Node param1 = new AtomParser().parse(session, ctx, complete);
040        if (!(param1 instanceof VarNode)) {
041            ctx.setIndex(mark);
042            throw new ParseException(ctx, "expected variable as second parameter to " + this.name + "()");
043        }
044        final String variable = ((VarNode)param1).getName();
045
046        // Get expression
047        ctx.skipWhitespace();
048        if (!ctx.tryLiteral(","))
049            throw new ParseException(ctx, "expected `,'").addCompletion(", ");
050        this.spaceParser.parse(ctx, complete);
051        final Node expr = new ExprParser().parse(session, ctx, complete);
052
053        // Finish parse
054        ctx.skipWhitespace();
055        if (!ctx.tryLiteral(")"))
056            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
057
058        // Done
059        return new ParamInfo(variable, items, expr);
060    }
061
062    @Override
063    public final Value apply(ParseSession session, Object params) {
064        return this.apply(session, (ParamInfo)params);
065    }
066
067    /**
068     * Apply this function.
069     *
070     * @param session parse session
071     * @param params function parameters
072     * @return the value of this function
073     */
074    protected abstract Value apply(ParseSession session, ParamInfo params);
075
076    /**
077     * Set the variable to the given value and then evaluate the expression.
078     * Any previous value of the variable is saved prior and restored after.
079     *
080     * @param session parse session
081     * @param variable the name of the variable to use
082     * @param value the value to set to the variable
083     * @param expr the expression to evaluate
084     * @return the value of the expression
085     */
086    protected Value evaluate(ParseSession session, String variable, Value value, Node expr) {
087
088        // Save previous variable value, if any
089        final boolean hasPreviousValue = session.getVars().containsKey(variable);
090        final Value previousValue = hasPreviousValue ? session.getVars().get(variable) : null;
091
092        // Assign variable
093        session.getVars().put(variable, value);
094        final Value result;
095        try {
096
097            // Evaluate expression
098            result = expr.evaluate(session);
099
100        } finally {
101
102            // Restore previous variable value
103            if (hasPreviousValue)
104                session.getVars().put(variable, previousValue);
105        }
106
107        // Done
108        return result;
109    }
110
111// ParamInfo
112
113    static class ParamInfo {
114
115        private final String variable;
116        private final Node items;
117        private final Node expr;
118
119        ParamInfo(String variable, Node items, Node expr) {
120            this.variable = variable;
121            this.items = items;
122            this.expr = expr;
123        }
124
125        public String getVariable() {
126            return this.variable;
127        }
128
129        public Node getItems() {
130            return this.items;
131        }
132
133        public Node getExpr() {
134            return this.expr;
135        }
136    }
137}
