008package org.jsimpledb.parse.expr;
009
010import com.google.common.base.Function;
011import com.google.common.collect.Lists;
012
013import java.lang.reflect.InvocationTargetException;
014import java.lang.reflect.Method;
015import java.util.LinkedHashSet;
016import java.util.List;
017import java.util.Set;
018
019import org.jsimpledb.parse.ParseSession;
020
021/**
022 * {@link Node} that invokes a Java method when evaluated.
023 */
024public class MethodInvokeNode implements Node {
025
026    private final Class<?> klass;
027    private final Node node;
028    private final String name;
029    private final List<Node> paramNodes;
030
031    /**
032     * Constructor for static method invocation.
033     *
034     * @param klass class containing static method
035     * @param name static method name
036     * @param paramNodes method parameters
037     */
038    public MethodInvokeNode(Class<?> klass, String name, List<Node> paramNodes) {
039        this(name, paramNodes, klass, null);
040        if (klass == null)
041            throw new IllegalArgumentException("null klass");
042    }
043
044    /**
045     * Constructor for instance method invocation.
046     *
047     * @param node node evaluating to target object
048     * @param name instance method name
049     * @param paramNodes method parameters
050     */
051    public MethodInvokeNode(Node node, String name, List<Node> paramNodes) {
052        this(name, paramNodes, null, node);
053        if (node == null)
054            throw new IllegalArgumentException("null node");
055    }
056
057    private MethodInvokeNode(String name, List<Node> paramNodes, Class<?> klass, Node node) {
058        if (name == null)
059            throw new IllegalArgumentException("null target");
060        if (paramNodes == null)
061            throw new IllegalArgumentException("null paramNodes");
062        for (Node paramNode : paramNodes) {
063            if (paramNode == null)
064                throw new IllegalArgumentException("null paramNode in list");
065        }
066        this.klass = klass;
067        this.node = node;
068        this.name = name;
069        this.paramNodes = paramNodes;
070    }
071
072    @Override
073    public Value evaluate(final ParseSession session) {
074
075        // Evaluate params
076        final Object[] params = Lists.transform(paramNodes, new Function<Node, Object>() {
077            @Override
078            public Object apply(Node param) {
079                return param.evaluate(session).get(session);
080            }
081        }).toArray();
082
083        // Handle static method
084        if (this.klass != null)
085            return this.invokeMethod(this.klass, null, name, params);
086
087        // Handle instance method
088        final Object obj = this.node.evaluate(session).checkNotNull(session, "method " + name + "() invocation");
089        return this.invokeMethod(obj.getClass(), obj, name, params);
090    }
091
092    private Value invokeMethod(Class<?> cl, Object obj, String name, Object[] params) {
093
094        // Try interface methods
095        for (Class<?> iface : this.addInterfaces(cl, new LinkedHashSet<Class<?>>())) {
096            for (Method method : iface.getMethods()) {
097                final Value value = this.tryMethod(method, obj, name, params);
098                if (value != null)
099                    return value;
100            }
101        }
102
103        // Try class methods
104        for (Method method : cl.getMethods()) {
105            final Value value = this.tryMethod(method, obj, name, params);
106            if (value != null)
107                return value;
108        }
109
110        // Not found
111        throw new EvalException("no compatible method `" + name + "()' found in " + cl);
112    }
113
114    private Set<Class<?>> addInterfaces(Class<?> cl, Set<Class<?>> interfaces) {
115        for (Class<?> iface : cl.getInterfaces()) {
116            interfaces.add(iface);
117            this.addInterfaces(iface, interfaces);
118        }
119        if (cl.getSuperclass() != null)
120            this.addInterfaces(cl.getSuperclass(), interfaces);
121        return interfaces;
122    }
123
124    private Value tryMethod(Method method, Object obj, String name, Object[] params) {
125        if (!method.getName().equals(name))
126            return null;
127        final Class<?>[] ptypes = method.getParameterTypes();
128        if (method.isVarArgs()) {
129            if (params.length < ptypes.length - 1)
130                return null;
131            Object[] newParams = new Object[ptypes.length];
132            System.arraycopy(params, 0, newParams, 0, ptypes.length - 1);
133            Object[] varargs = new Object[params.length - (ptypes.length - 1)];
134            System.arraycopy(params, ptypes.length - 1, varargs, 0, varargs.length);
135            newParams[ptypes.length - 1] = varargs;
136            params = newParams;
137        } else if (params.length != ptypes.length)
138            return null;
139        final Object result;
140        try {
141            result = method.invoke(obj, params);
142        } catch (IllegalArgumentException e) {
143            return null;                            // a parameter type didn't match -> wrong method
144        } catch (Exception e) {
145            final Throwable t = e instanceof InvocationTargetException ?
146              ((InvocationTargetException)e).getTargetException() : e;
147            throw new EvalException("error invoking method `" + name + "()' on "
148              + (obj != null ? "object of type " + obj.getClass().getName() : method.getDeclaringClass()) + ": " + t, t);
149        }
150        return result != null || method.getReturnType() != Void.TYPE ? new ConstValue(result) : Value.NO_VALUE;
151    }
152}
