008package org.jsimpledb.parse.expr;
009
010import org.jsimpledb.parse.ParseContext;
011import org.jsimpledb.parse.ParseException;
012import org.jsimpledb.parse.ParseSession;
013import org.jsimpledb.parse.Parser;
014import org.jsimpledb.parse.SpaceParser;
015
016public class ConditionalParser implements Parser<Node> {
017
018    public static final ConditionalParser INSTANCE = new ConditionalParser();
019
020    private final SpaceParser spaceParser = new SpaceParser();
021
022    @Override
023    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
024
025        // Parse test
026        final Node test = LogicalOrParser.INSTANCE.parse(session, ctx, complete);
027
028        // Parse '?'
029        this.spaceParser.parse(ctx, complete);
030        if (!ctx.tryLiteral("?"))
031            return test;
032
033        // Parse true result
034        this.spaceParser.parse(ctx, complete);
035        final Node ifTrue = LogicalOrParser.INSTANCE.parse(session, ctx, complete);
036
037        // Parse ':'
038        this.spaceParser.parse(ctx, complete);
039        if (!ctx.tryLiteral(":"))
040            throw new ParseException(ctx).addCompletion(": ");
041
042        // Parse false result
043        this.spaceParser.parse(ctx, complete);
044        final Node ifFalse = LogicalOrParser.INSTANCE.parse(session, ctx, complete);
045
046        // Return node
047        return new Node() {
048            @Override
049            public Value evaluate(ParseSession session) {
050                return test.evaluate(session).checkBoolean(session, "conditional") ?
051                  ifTrue.evaluate(session) : ifFalse.evaluate(session);
052            }
053        };
054    }
055}
