008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.core.ObjId;
012import org.jsimpledb.core.Transaction;
013import org.jsimpledb.parse.ObjTypeParser;
014import org.jsimpledb.parse.ParseContext;
015import org.jsimpledb.parse.ParseException;
016import org.jsimpledb.parse.ParseSession;
017import org.jsimpledb.parse.SpaceParser;
018import org.jsimpledb.parse.expr.AssignmentExprParser;
019import org.jsimpledb.parse.expr.Node;
020import org.jsimpledb.parse.expr.Value;
021
022@Function
023public class CreateFunction extends AbstractFunction {
024
025    private final SpaceParser spaceParser = new SpaceParser();
026
027    public CreateFunction() {
028        super("create");
029    }
030
031    @Override
032    public String getHelpSummary() {
033        return "create a new object instance";
034    }
035
036    @Override
037    public String getUsage() {
038        return "create(type [, version ])";
039    }
040
041    @Override
042    public String getHelpDetail() {
043        return "Creates and returns a new instance of the specified type. The optional `version' parameter forces"
044          + " a specific schema version.";
045    }
046
047    @Override
048    public ParamInfo parseParams(ParseSession session, ParseContext ctx, boolean complete) {
049
050        // Get object type
051        if (ctx.tryLiteral(")"))
052            throw new ParseException(ctx, "type parameter required");
053        final int storageId = new ObjTypeParser().parse(session, ctx, complete).getStorageId();
054
055        // Check for optional version
056        this.spaceParser.parse(ctx, complete);
057        if (ctx.tryLiteral(")"))
058            return new ParamInfo(storageId);
059        if (!ctx.tryLiteral(","))
060            throw new ParseException(ctx, "expected `,' between function parameters").addCompletion(", ");
061
062        // Get version
063        this.spaceParser.parse(ctx, complete);
064        final Node version = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
065        this.spaceParser.parse(ctx, complete);
066
067        // Finish parse
068        ctx.skipWhitespace();
069        if (!ctx.tryLiteral(")"))
070            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
071
072        // Done
073        return new ParamInfo(storageId, version);
074    }
075
076    @Override
077    public Value apply(ParseSession session, Object params) {
078        final ParamInfo info = (ParamInfo)params;
079        final int storageId = info.getStorageId();
080        final Node version = info.getVersion();
081
082        // Create object
083        final Transaction tx = session.getTransaction();
084        final ObjId id = version != null ?
085          tx.create(storageId, version.evaluate(session).checkIntegral(session, "create()")) : tx.create(storageId);
086        return new Value(session.hasJSimpleDB() ? JTransaction.getCurrent().getJObject(id) : id);
087    }
088
089// ParamInfo
090
091    private static class ParamInfo {
092
093        private final int storageId;
094        private final Node version;
095
096        ParamInfo(int storageId) {
097            this(storageId, null);
098        }
099
100        ParamInfo(int storageId, Node version) {
101            this.storageId = storageId;
102            this.version = version;
103        }
104
105        public int getStorageId() {
106            return this.storageId;
107        }
108
109        public Node getVersion() {
110            return this.version;
111        }
112    }
113}
