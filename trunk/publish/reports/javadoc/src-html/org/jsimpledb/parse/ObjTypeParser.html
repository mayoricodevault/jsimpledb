008package org.jsimpledb.parse;
009
010import com.google.common.reflect.TypeToken;
011
012import java.util.regex.Matcher;
013
014import org.jsimpledb.core.Database;
015import org.jsimpledb.core.ObjType;
016import org.jsimpledb.core.SchemaVersion;
017import org.jsimpledb.core.Transaction;
018import org.jsimpledb.core.UnknownTypeException;
019import org.jsimpledb.schema.NameIndex;
020import org.jsimpledb.schema.SchemaObjectType;
021
022/**
023 * Parses an object type name.
024 *
025 * <p>
026 * Syntax examples:
027 * <ul>
028 *  <li><code>100</code> - object type with schema ID 100</li>
029 *  <li><code>Person</code> - `Person' object type defined in the current schema version</li>
030 *  <li><code>Person#12</code> - `Person' object type defined in schema version 12</li>
031 * </p>
032 */
033public class ObjTypeParser implements Parser<ObjType> {
034
035    @Override
036    public ObjType parse(ParseSession session, ParseContext ctx, boolean complete) {
037
038        // Try to parse as an integer
039        final Transaction tx = session.getTransaction();
040        final Database db = session.getDatabase();
041        final int startIndex = ctx.getIndex();
042        try {
043            final int storageId = db.getFieldTypeRegistry().getFieldType(TypeToken.of(Integer.TYPE)).fromParseableString(ctx);
044            return tx.getSchemaVersion().getObjType(storageId);
045        } catch (IllegalArgumentException | UnknownTypeException e) {
046            // ignore
047        }
048        ctx.setIndex(startIndex);
049
050        // Try to parse as an object type name with optional #version suffix
051        final Matcher matcher;
052        try {
053            matcher = ctx.matchPrefix("(\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*)(#([0-9]+))?");
054        } catch (IllegalArgumentException e) {
055            throw new ParseException(ctx, "invalid object type").addCompletions(session.getNameIndex().getSchemaObjectTypeNames());
056        }
057        final String typeName = matcher.group(1);
058        final String versionString = matcher.group(3);
059
060        // Get specified schema version and corresponding name index
061        final SchemaVersion version;
062        final NameIndex nameIndex;
063        if (versionString != null) {
064            try {
065                version = tx.getSchema().getVersion(Integer.parseInt(versionString));
066            } catch (IllegalArgumentException e) {
067                ctx.setIndex(startIndex);
068                throw new ParseException(ctx, "invalid object type schema version `" + versionString + "'");
069            }
070            nameIndex = new NameIndex(version.getSchemaModel());
071        } else {
072            version = tx.getSchemaVersion();
073            nameIndex = session.getNameIndex();
074        }
075
076        // Find type by name
077        final SchemaObjectType schemaObjectType = nameIndex.getSchemaObjectType(typeName);
078        if (schemaObjectType == null) {
079            throw new ParseException(ctx, "unknown object type `" + typeName + "'")
080               .addCompletions(ParseUtil.complete(nameIndex.getSchemaObjectTypeNames(), typeName));
081        }
082        return version.getObjType(schemaObjectType.getStorageId());
083    }
084}
