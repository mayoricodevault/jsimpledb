008package org.jsimpledb.parse;
009
010import org.jsimpledb.core.DeletedObjectException;
011import org.jsimpledb.core.ObjId;
012import org.jsimpledb.core.ObjType;
013import org.jsimpledb.core.Schema;
014import org.jsimpledb.core.Transaction;
015
016/**
017 * Utility class holding meta-data about a database object.
018 */
019public class ObjInfo {
020
021    private final ObjId id;
022    private final Schema schema;
023    private final ObjType type;
024
025    /**
026     * Constructor.
027     *
028     * @param tx database transaction
029     * @param id the ID of the object to query
030     * @throws DeletedObjectException if object does not exist
031     */
032    public ObjInfo(Transaction tx, ObjId id) {
033        this.id = id;
034        this.schema = tx.getSchemas().getVersion(tx.getSchemaVersion(id));
035        this.type = this.schema.getObjType(id.getStorageId());
036    }
037
038    public ObjId getObjId() {
039        return this.id;
040    }
041
042    public Schema getSchema() {
043        return this.schema;
044    }
045
046    public ObjType getObjType() {
047        return this.type;
048    }
049
050    @Override
051    public String toString() {
052        return this.id + " type " + this.type.getName() + "#" + this.type.getStorageId()
053          + " version " + this.schema.getVersionNumber();
054    }
055
056    /**
057     * Get object meta-data.
058     *
059     * @param session parse session
060     * @param id the ID of the object to query
061     * @return object info, or null if object doesn't exist
062     */
063    public static ObjInfo getObjInfo(ParseSession session, ObjId id) {
064        try {
065            return new ObjInfo(session.getTransaction(), id);
066        } catch (DeletedObjectException e) {
067            return null;
068        }
069    }
070}
