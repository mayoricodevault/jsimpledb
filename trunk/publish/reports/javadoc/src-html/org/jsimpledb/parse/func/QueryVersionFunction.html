008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.core.CoreIndex;
012import org.jsimpledb.core.ObjId;
013import org.jsimpledb.core.ObjType;
014import org.jsimpledb.kv.KeyRanges;
015import org.jsimpledb.parse.ObjTypeParser;
016import org.jsimpledb.parse.ParseContext;
017import org.jsimpledb.parse.ParseException;
018import org.jsimpledb.parse.ParseSession;
019import org.jsimpledb.parse.expr.AbstractValue;
020import org.jsimpledb.parse.expr.Node;
021import org.jsimpledb.parse.expr.Value;
022
023@Function
024public class QueryVersionFunction extends AbstractFunction {
025
026    public QueryVersionFunction() {
027        super("queryVersion");
028    }
029
030    @Override
031    public String getHelpSummary() {
032        return "queries the object version index";
033    }
034
035    @Override
036    public String getUsage() {
037        return "queryVersion([object-type])";
038    }
039
040    @Override
041    public String getHelpDetail() {
042        return "Queries the index of object versions, returning a map from object version to the set of objects with that version."
043          + " An optional object type restricts returned objects.";
044    }
045
046    // Returns either null, ObjType, or Node
047    @Override
048    public Object parseParams(final ParseSession session, final ParseContext ctx, final boolean complete) {
049
050        // Check existence of parameter
051        if (ctx.tryLiteral(")"))
052            return null;
053
054        // Attempt to parse either type name or Java expression (hopefully evaluating to Class<?>)
055        Object result;
056        final int typeStart = ctx.getIndex();
057        try {
058            result = new ObjTypeParser().parse(session, ctx, complete);
059            ctx.skipWhitespace();
060            if (!ctx.tryLiteral(")"))
061                throw new ParseException(ctx);
062        } catch (ParseException e) {
063            ctx.setIndex(typeStart);
064            result = this.parseExpressionParams(session, ctx, complete, 0, 1, 1)[0];
065        }
066
067        // Done
068        return result;
069    }
070
071    @Override
072    public Value apply(ParseSession session, final Object param) {
073        return new AbstractValue() {
074            @Override
075            public Object get(ParseSession session) {
076                if (session.hasJSimpleDB()) {
077                    final Class<?> type =
078                      param instanceof ObjType ?
079                        session.getJSimpleDB().getJClass(((ObjType)param).getStorageId()).getType() :
080                      param instanceof Node ?
081                        ((Node)param).evaluate(session).checkType(session, QueryVersionFunction.this.getName(), Class.class) :
082                        Object.class;
083                    return JTransaction.getCurrent().queryVersion(type);
084                } else {
085                    CoreIndex<Integer, ObjId> index = session.getTransaction().queryVersion();
086                    final int storageId =
087                      param instanceof Node ?
088                        ((Node)param).evaluate(session).checkType(session, QueryVersionFunction.this.getName(), Integer.class) :
089                      param instanceof ObjType ?
090                        ((ObjType)param).getStorageId() :
091                        -1;
092                    if (storageId != -1)
093                        index = index.filter(1, new KeyRanges(ObjId.getKeyRange(storageId)));
094                    return index.asMap();
095                }
096            }
097        };
098    }
099}
