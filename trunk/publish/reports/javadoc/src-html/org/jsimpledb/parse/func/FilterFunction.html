008package org.jsimpledb.parse.func;
009
010import com.google.common.base.Predicate;
011import com.google.common.collect.Iterables;
012
013import java.util.Map;
014
015import org.jsimpledb.parse.ParseSession;
016import org.jsimpledb.parse.expr.EvalException;
017import org.jsimpledb.parse.expr.Value;
018
019@Function
020public class FilterFunction extends ApplyExprFunction {
021
022    public FilterFunction() {
023        super("filter");
024    }
025
026    @Override
027    public String getHelpSummary() {
028        return "filters a collection";
029    }
030
031    @Override
032    public String getUsage() {
033        return "filters(items, variable, expression)";
034    }
035
036    @Override
037    public String getHelpDetail() {
038        return "Creates a filtered view of an Iterable, where items are included only if when the item is assigned to the"
039          + " specified variable the specified expression evaluates to true. Maps are also supported, in which case the map's"
040          + " entrySet() is filtered.";
041    }
042
043    @Override
044    protected Value apply(ParseSession session, final ParamInfo params) {
045        return new Value(null) {
046            @Override
047            public Object get(final ParseSession session) {
048
049                // Evaluate items
050                Object items = params.getItems().evaluate(session).checkNotNull(session, "filter()");
051                if (items instanceof Map)
052                    items = ((Map<?, ?>)items).entrySet();
053                if (!(items instanceof Iterable)) {
054                    throw new EvalException("invalid filter() operation on non-Iterable object of type "
055                      + items.getClass().getName());
056                }
057
058                // Return filtered view
059                return Iterables.filter((Iterable<?>)items, new Predicate<Object>() {
060                    @Override
061                    public boolean apply(Object item) {
062                        return FilterFunction.this.evaluate(session, params.getVariable(),
063                          new Value(item), params.getExpr()).checkBoolean(session, "filter()");
064                    }
065                });
066            }
067        };
068    }
069}
