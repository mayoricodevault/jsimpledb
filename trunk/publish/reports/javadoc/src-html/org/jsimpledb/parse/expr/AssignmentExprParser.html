008package org.jsimpledb.parse.expr;
009
010import org.jsimpledb.parse.ParseSession;
011
012public class AssignmentExprParser extends BinaryExprParser {
013
014    public static final AssignmentExprParser INSTANCE = new AssignmentExprParser();
015
016    public AssignmentExprParser() {
017        super(ConditionalParser.INSTANCE, false, Op.EQUALS, Op.PLUS_EQUALS, Op.MINUS_EQUALS, Op.MULTIPLY_EQUALS,
018          Op.DIVIDE_EQUALS, Op.MODULO_EQUALS, Op.AND_EQUALS, Op.XOR_EQUALS, Op.OR_EQUALS, Op.LSHIFT_EQUALS,
019          Op.RSHIFT_EQUALS, Op.URSHIFT_EQUALS);
020    }
021
022    @Override
023    protected Node createNode(final Op op, final Node lhs, final Node rhs) {
024        return new Node() {
025            @Override
026            public Value evaluate(ParseSession session) {
027
028                // Get left-hand value
029                final Value oldValue = lhs.evaluate(session);
030                final Setter setter = oldValue.getSetter();
031                if (setter == null)
032                    throw new EvalException("invalid assignment to non-assignable value");
033
034                // Calculate new value
035                Value newValue = rhs.evaluate(session);
036                if (op != Op.EQUALS)
037                    newValue = Op.forSymbol(op.getSymbol().replaceAll("=", "")).apply(session, oldValue, newValue);
038
039                // Assign new value
040                setter.set(session, newValue);
041
042                // Done
043                return newValue;
044            }
045        };
046    }
047}
