008package org.jsimpledb.parse.expr;
009
010import java.util.TreeSet;
011import java.util.regex.Matcher;
012import java.util.regex.Pattern;
013
014import org.dellroad.stuff.java.Primitive;
015import org.dellroad.stuff.string.StringEncoder;
016import org.jsimpledb.JTransaction;
017import org.jsimpledb.core.FieldType;
018import org.jsimpledb.core.ObjId;
019import org.jsimpledb.parse.ObjIdParser;
020import org.jsimpledb.parse.ParseContext;
021import org.jsimpledb.parse.ParseException;
022import org.jsimpledb.parse.ParseSession;
023import org.jsimpledb.parse.ParseUtil;
024import org.jsimpledb.parse.Parser;
025import org.jsimpledb.parse.SpaceParser;
026
027/**
028 * The lowest parse level.
029 */
030public class AtomParser implements Parser<Node> {
031
032    public static final AtomParser INSTANCE = new AtomParser();
033
034    private final SpaceParser spaceParser = new SpaceParser();
035    private final TreeSet<String> identifierCompletions;
036
037    public AtomParser() {
038        this(null);
039    }
040
041    /**
042     * Constructor.
043     *
044     * @param identifierCompletions set of valid identifiers tab completions, or null to allow any identifiers
045     */
046    public AtomParser(Iterable<String> identifierCompletions) {
047        if (identifierCompletions != null) {
048            this.identifierCompletions = new TreeSet<String>();
049            for (String identifierCompletion : identifierCompletions)
050                this.identifierCompletions.add(identifierCompletion);
051        } else
052            this.identifierCompletions = null;
053    }
054
055    @Override
056    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
057        final int mark = ctx.getIndex();
058
059        // Check for parenthesized expression
060        if (ctx.tryLiteral("(")) {
061            this.spaceParser.parse(ctx, complete);
062            final Node node = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
063            this.spaceParser.parse(ctx, complete);
064            if (!ctx.tryLiteral(")"))
065                throw new ParseException(ctx).addCompletion(") ");
066            return node;
067        }
068
069        // Try to match null
070        if (this.tryWord(ctx, "null") != null)
071            return new LiteralNode(null);
072
073        // Try to match boolean
074        final String booleanMatch = this.tryWord(ctx, "false|true");
075        if (booleanMatch != null)
076            return new LiteralNode(Boolean.valueOf(booleanMatch));
077
078        // Try to match double, float, int, or long literal
079        final int start = ctx.getIndex();
080        final String floatMatch = this.tryWord(ctx, Primitive.FLOAT.getParsePattern() + "[fF]");
081        if (floatMatch != null) {
082            try {
083                return new LiteralNode(Primitive.FLOAT.parseValue(floatMatch.substring(0, floatMatch.length() - 1)));
084            } catch (IllegalArgumentException e) {
085                ctx.setIndex(start);
086            }
087        }
088        final String longMatch = this.tryWord(ctx, Primitive.LONG.getParsePattern() + "[Ll]");
089        if (longMatch != null) {
090            try {
091                return new LiteralNode(Primitive.LONG.parseValue(longMatch.substring(0, longMatch.length() - 1)));
092            } catch (IllegalArgumentException e) {
093                ctx.setIndex(start);
094            }
095        }
096        final String intMatch = this.tryFollow(ctx, Primitive.INTEGER.getParsePattern(), "[^.\\p{javaJavaIdentifierPart}]");
097        if (intMatch != null) {
098            try {
099                return new LiteralNode(Primitive.INTEGER.parseValue(intMatch));
100            } catch (IllegalArgumentException e) {
101                ctx.setIndex(start);
102            }
103        }
104        final String doubleMatch = this.tryFollow(ctx, Primitive.DOUBLE.getParsePattern(), "[^.\\p{javaJavaIdentifierPart}]");
105        if (doubleMatch != null) {
106            try {
107                return new LiteralNode(Primitive.DOUBLE.parseValue(doubleMatch));
108            } catch (IllegalArgumentException e) {
109                ctx.setIndex(start);
110            }
111        }
112
113        // Try to match a char literal
114        String match = this.tryWord(ctx,
115          StringEncoder.ENQUOTE_PATTERN.toString().replaceAll("\"", "'").replaceAll("\\*", ""));    // kludge
116        if (match != null) {
117            match = match.substring(1, match.length() - 1);
118            if (match.length() > 0 && match.charAt(0) == '\'') {
119                ctx.setIndex(start);
120                throw new ParseException(ctx, "invalid character: contains unescaped single quote");
121            }
122            match = StringEncoder.decode(match.replaceAll(Pattern.quote("\\'"), Matcher.quoteReplacement("'")));
123            if (match.length() != 1) {
124                ctx.setIndex(start);
125                throw new ParseException(ctx, "invalid character: quotes must contain exactly one character");
126            }
127            return new LiteralNode(match.charAt(0));
128        }
129
130        // Try to match a string literal
131        final Matcher stringMatch = ctx.tryPattern(StringEncoder.ENQUOTE_PATTERN);
132        if (stringMatch != null)
133            return new LiteralNode(new String(StringEncoder.dequote(stringMatch.group())));
134
135        // Try to type from type registry within curly braces
136        int fieldTypeStart = ctx.getIndex();
137        final Matcher braceMatch = ctx.tryPattern("\\{(" + FieldType.NAME_PATTERN + ")\\}");
138        if (braceMatch != null) {
139            final String fieldTypeName = braceMatch.group(1);
140            final FieldType<?> fieldType = session.getDatabase().getFieldTypeRegistry().getFieldType(fieldTypeName);
141            if (fieldType == null) {
142                ctx.setIndex(fieldTypeStart);
143                throw new ParseException(ctx, "unknown simple field type `" + fieldTypeName + "'");
144            }
145            fieldTypeStart = ctx.getIndex();
146            try {
147                return new LiteralNode(fieldType.fromParseableString(ctx));
148            } catch (IllegalArgumentException e) {
149                ctx.setIndex(fieldTypeStart);
150                throw new ParseException(ctx, "invalid value for type `" + fieldTypeName + "'");
151            }
152        }
153
154        // Try to match variable
155        if (ctx.tryLiteral("$")) {
156            final Matcher varMatcher = ctx.tryPattern(AbstractNamed.NAME_PATTERN);
157            if (varMatcher == null)
158                throw new ParseException(ctx).addCompletions(session.getVars().keySet());
159            final String name = varMatcher.group();
160            if (ctx.isEOF() && complete)
161                throw new ParseException(ctx).addCompletions(ParseUtil.complete(session.getVars().keySet(), name));
162            return new VarNode(name);
163        }
164
165        // Try to match object literal
166        if (ctx.tryLiteral("@")) {
167            final ObjId id = new ObjIdParser().parse(session, ctx, complete);
168            return !session.hasJSimpleDB() ? new LiteralNode(id) : new Node() {
169                @Override
170                public Value evaluate(ParseSession session) {
171                    return new ConstValue(JTransaction.getCurrent().getJObject(id));
172                }
173            };
174        }
175
176        // Try to match identifier; support tab-completion of configured identifiers, if any
177        final Matcher identMatcher = ctx.tryPattern(IdentNode.NAME_PATTERN);
178        if (identMatcher != null) {
179            final String name = identMatcher.group();
180            if (complete && ctx.isEOF() && this.identifierCompletions != null && !name.equals("new"))
181                throw new ParseException(ctx).addCompletions(ParseUtil.complete(this.identifierCompletions, name));
182            return new IdentNode(name);
183        }
184        if (complete && ctx.isEOF() && this.identifierCompletions != null)
185            throw new ParseException(ctx).addCompletions(this.identifierCompletions);
186
187        // No match
188        throw new ParseException(ctx);
189    }
190
191// Match a pattern not followed by a identifier letter
192
193    private String tryWord(ParseContext ctx, Pattern pattern) {
194        return this.tryWord(ctx, pattern.toString());
195    }
196
197    private String tryWord(ParseContext ctx, String pattern) {
198        return this.tryFollow(ctx, pattern, "[^\\p{javaJavaIdentifierPart}]");
199    }
200
201    private String tryFollow(ParseContext ctx, Pattern pattern, String follow) {
202        return this.tryFollow(ctx, pattern.toString(), follow);
203    }
204
205    private String tryFollow(ParseContext ctx, String pattern, String follow) {
206        final Matcher matcher = Pattern.compile("(" + pattern + ")(" + follow + "(?s:.*))?").matcher(ctx.getInput());
207        if (!matcher.matches())
208            return null;
209        final String match = matcher.group(1);
210        ctx.setIndex(ctx.getIndex() + match.length());
211        return match;
212    }
213}
