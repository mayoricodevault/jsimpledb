010import java.lang.reflect.InvocationTargetException;
011import java.util.ArrayList;
012import java.util.LinkedHashSet;
013import java.util.Set;
014import java.util.SortedMap;
015import java.util.TreeMap;
016
017import org.jsimpledb.JSimpleDB;
018import org.jsimpledb.Session;
019import org.jsimpledb.core.Database;

021import org.jsimpledb.parse.expr.Value;

023import org.jsimpledb.parse.func.Function;

026/**
027 * A {@link Session} with support for parsing Java expressions.
028 */
029public class ParseSession extends Session {
030
031    private final LinkedHashSet<String> imports = new LinkedHashSet<>();
032    private final TreeMap<String, AbstractFunction> functions = new TreeMap<>();
033    private final TreeMap<String, Value> variables = new TreeMap<>();
034
035// Constructors
036
037    /**
038     * Constructor for core API level access.
039     *
040     * @param db core database
041     * @throws IllegalArgumentException if {@code db} is null
042     */
043    public ParseSession(Database db) {
044        super(db);
045        this.imports.add("java.lang.*");
046    }
047
048    /**
049     * Constructor for {@link JSimpleDB} level access.
050     *
051     * @param jdb database
052     * @throws IllegalArgumentException if {@code jdb} is null
053     */
054    public ParseSession(JSimpleDB jdb) {
055        super(jdb);
056        this.imports.add("java.lang.*");
057    }
058
059// Accessors
060
061    /**
062     * Get currently configured Java imports.
063     *
064     * <p>
065     * Each entry should of the form {@code foo.bar.Name} or {@code foo.bar.*}.
066     * </p>
067     */
068    public Set<String> getImports() {
069        return this.imports;
070    }
071
072    /**
073     * Get the {@link AbstractFunction}s registered with this instance.
074     */
075    public SortedMap<String, AbstractFunction> getFunctions() {
076        return this.functions;
077    }
078
079    /**
080     * Get all variables set on this instance.
081     */
082    public SortedMap<String, Value> getVars() {
083        return this.variables;
084    }
085
086// Function registration
087
088    /**
089     * Register the standard built-in functions such as {@code all()}, {@code foreach()}, etc.
090     */
091    public void registerStandardFunctions() {
092        for (String className : new AnnotatedClassScanner(Function.class).scanForClasses(Function.class.getPackage().getName())) {
093            final Class<?> cl;
094            try {
095                cl = Class.forName(className, false, Thread.currentThread().getContextClassLoader());
096            } catch (ClassNotFoundException e) {
097                throw new RuntimeException("failed to load class `" + className + "'", e);
098            }
099            this.registerFunction(cl);
100        }
101    }
102
103    /**
104     * Create an instance of the specified class and register it as an {@link AbstractFunction}.
105     * as appropriate. The class must have a public constructor taking either a single {@link ParseSession} parameter
106     * or no parameters; they will be tried in that order.
107     *
108     * @param cl function class
109     * @throws IllegalArgumentException if {@code cl} has no suitable constructor
110     * @throws IllegalArgumentException if {@code cl} instantiation fails
111     * @throws IllegalArgumentException if {@code cl} does not subclass {@link AbstractFunction}
112     */
113    public void registerFunction(Class<?> cl) {
114        if (!AbstractFunction.class.isAssignableFrom(cl))
115            throw new IllegalArgumentException(cl + " does not subclass " + AbstractFunction.class.getName());
116        final AbstractFunction function = this.instantiate(cl.asSubclass(AbstractFunction.class));
117        this.functions.put(function.getName(), function);
118    }
119
120    /**
121     * Instantiate an instance of the given class.
122     * The class must have a public constructor taking either a single {@link ParseSession} parameter
123     * or no parameters; they will be tried in that order.
124     */
125    private <T> T instantiate(Class<T> cl) {
126        Throwable failure;
127        try {
128            return cl.getConstructor(ParseSession.class).newInstance(this);
129        } catch (NoSuchMethodException e) {
130            try {
131                return cl.getConstructor().newInstance();
132            } catch (NoSuchMethodException e2) {
133                throw new IllegalArgumentException("no suitable constructor found in class " + cl.getName());
134            } catch (Exception e2) {
135                failure = e2;
136            }
137        } catch (Exception e) {
138            failure = e;
139        }
140        if (failure instanceof InvocationTargetException)
141            failure = failure.getCause();
142        throw new IllegalArgumentException("unable to instantiate class " + cl.getName() + ": " + failure, failure);
143    }
144
145// Class name resolution
146
147    /**
148     * Resolve a class name against this instance's currently configured class imports.
149     *
150     * @return resolved class, or null if not found
151     */
152    public Class<?> resolveClass(final String name) {
153        final int firstDot = name.indexOf('.');
154        final String firstPart = firstDot != -1 ? name.substring(0, firstDot - 1) : name;
155        final ArrayList<String> packages = new ArrayList<>(this.imports.size() + 1);
156        packages.add(null);
157        packages.addAll(this.imports);
158        for (String pkg : packages) {
159
160            // Get absolute class name
161            String className;
162            if (pkg == null)
163                className = name;
164            else if (pkg.endsWith(".*"))
165                className = pkg.substring(0, pkg.length() - 1) + name;
166            else {
167                if (!firstPart.equals(pkg.substring(pkg.lastIndexOf('.') + 1, pkg.length() - 2)))
168                    continue;
169                className = pkg.substring(0, pkg.length() - 2 - firstPart.length()) + name;
170            }
171
172            // Try package vs. nested classes
173            while (true) {
174                try {
175                    return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
176                } catch (ClassNotFoundException e) {
177                    // not found
178                }
179                final int lastDot = className.lastIndexOf('.');
180                if (lastDot == -1)
181                    break;
182                className = className.substring(0, lastDot) + "$" + className.substring(lastDot + 1);
183            }
184        }
185        return null;
186    }
187
188    /**
189     * Relativize the given class's name, so that it is as short as possible given the configured imports.
190     * For example, for class {@link String} this will return {@code String}, but for class {@link ArrayList}
191     * this will return {@code java.util.ArrayList} unless {@code java.util.*} has been imported.
192     *
193     * @param klass class whose name to relativize
194     * @throws IllegalArgumentException if {@code klass} is null
195     */
196    public String relativizeClassName(Class<?> klass) {
197        if (klass == null)
198            throw new IllegalArgumentException("null klass");
199        final String name = klass.getName();
200        for (int pos = name.lastIndexOf('.'); pos > 0; pos = name.lastIndexOf('.', pos - 1)) {
201            final String shortName = name.substring(pos + 1);
202            if (this.resolveClass(shortName) == klass)
203                return shortName;
204        }
205        return klass.getName();
206    }
207
208// Action
209
210    /**
211     * Perform the given action. This is a convenience method, equivalent to: {@code perform(null, action)}
212     *
213     * @param action action to perform
214     * @throws IllegalArgumentException if {@code action} is null
215     */
216    public boolean perform(Action action) {
217        return this.perform(null, action);
218    }
219
220    /**
221     * Perform the given action within the given existing transaction, if any, otherwise within a new transaction.
222     * If {@code tx} is not null, it will used and left open when this method returns. Otherwise,
223     * if there is already an open transaction associated with this instance, it will be used;
224     * otherwise, a new transaction is created for the duration of {@code action} and then committed.
225     *
226     * <p>
227     * If {@code tx} is not null and there is already an open transaction associated with this instance and they
228     * are not the same transaction, an {@link IllegalStateException} is thrown.
229     * </p>
230     *
231     * @param tx transaction in which to perform the action, or null to create a new one (if necessary)
232     * @param action action to perform
233     * @throws IllegalStateException if {@code tx} conflict with the already an open transaction associated with this instance
234     * @throws IllegalArgumentException if {@code action} is null
235     */
236    public boolean perform(Transaction tx, final Action action) {
237        return this.perform(tx, new Session.Action() {
238            @Override
239            public void run(Session session) throws Exception {
240                action.run((ParseSession)session);
241            }
242        });
243    }
244
245    /**
246     * Callback interface used by {@link ParseSession#perform ParseSession.perform()}.
247     */
248    public interface Action {
249
250        /**
251         * Perform some action using the given {@link ParseSession} while a transaction is open.
252         */
253        void run(ParseSession session) throws Exception;
254    }
255}
