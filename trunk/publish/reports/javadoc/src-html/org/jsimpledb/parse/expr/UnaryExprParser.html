008package org.jsimpledb.parse.expr;
009
010import java.util.regex.Matcher;
011
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseSession;
014import org.jsimpledb.parse.Parser;
015import org.jsimpledb.parse.SpaceParser;
016
017/**
018 * Parses unary expressions using one of {@code !}, {@code ~}, {@code +}, {@code -}, {@code ++}, {@code --}.
019 */
020public class UnaryExprParser implements Parser<Node> {
021
022    public static final UnaryExprParser INSTANCE = new UnaryExprParser();
023
024    private final SpaceParser spaceParser = new SpaceParser();
025
026    @Override
027    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
028
029        // Parse operator, if any
030        final Matcher matcher = ctx.tryPattern("\\+{1,2}|-{1,2}|!|~");
031        if (matcher == null)
032            return BaseExprParser.INSTANCE.parse(session, ctx, complete);
033        final String opsym = matcher.group();
034        this.spaceParser.parse(ctx, complete);
035
036        // Parse argument (this recursion will give the correct right-to-left association)
037        final Node arg = this.parse(session, ctx, complete);
038
039        // Proceed with parse
040        switch (opsym) {
041        case "!":
042            return this.createUnaryOpNode(Op.LOGICAL_NOT, arg);
043        case "~":
044            return this.createUnaryOpNode(Op.INVERT, arg);
045        case "+":
046            return this.createUnaryOpNode(Op.UNARY_PLUS, arg);
047        case "-":
048            return this.createUnaryOpNode(Op.UNARY_MINUS, arg);
049        case "++":
050            return this.createPrecrementNode("increment", arg, true);
051        case "--":
052            return this.createPrecrementNode("decrement", arg, false);
053        default:
054            throw new RuntimeException("internal error: " + opsym);
055        }
056    }
057
058    private Node createUnaryOpNode(final Op op, final Node arg) {
059        return new Node() {
060            @Override
061            public Value evaluate(ParseSession session) {
062                return op.apply(session, arg.evaluate(session));
063            }
064        };
065    }
066
067    private Node createPrecrementNode(final String operation, final Node node, final boolean increment) {
068        return new Node() {
069            @Override
070            public Value evaluate(ParseSession session) {
071                return node.evaluate(session).xxcrement(session, "pre-" + operation, increment);
072            }
073        };
074    }
075}
