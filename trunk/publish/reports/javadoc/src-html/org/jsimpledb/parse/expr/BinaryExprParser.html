008package org.jsimpledb.parse.expr;
009
010import java.util.ArrayList;
011import java.util.Arrays;
012import java.util.List;
013
014import org.jsimpledb.parse.ParseContext;
015import org.jsimpledb.parse.ParseException;
016import org.jsimpledb.parse.ParseSession;
017import org.jsimpledb.parse.Parser;
018import org.jsimpledb.parse.SpaceParser;
019
020/**
021 * Support superclass for tail-recursive binary expression parsers for expressions of the form {@code ARG1 OP ARG2}.
022 *
023 * @see <a href="http://en.wikipedia.org/wiki/Tail_recursive_parser">Tail Recursive Parser</a>
024 */
025public abstract class BinaryExprParser implements Parser<Node> {
026
027    private final SpaceParser spaceParser = new SpaceParser();
028    private final Parser<? extends Node> lowerLevel;
029    private final boolean leftAssociative;
030    private final List<Op> ops;
031
032    /**
033     * Primary constructor.
034     *
035     * @param lowerLevel next lower parse level
036     * @param leftAssociative true for left-to-right association, false for right-to-left
037     * @param ops operations, with operations that have other operations as prefixes listed first
038     */
039    protected BinaryExprParser(Parser<? extends Node> lowerLevel, boolean leftAssociative, Op... ops) {
040        if (lowerLevel == null)
041            throw new IllegalArgumentException("null lowerLevel");
042        if (ops == null || ops.length == 0)
043            throw new IllegalArgumentException("null/empty ops");
044        this.lowerLevel = lowerLevel;
045        this.leftAssociative = leftAssociative;
046        this.ops = Arrays.asList(ops);
047    }
048
049    /**
050     * Convenience constructor for left-associative operators.
051     *
052     * @param lowerLevel next lower parse level
053     * @param ops operations, with operations that have other operations as prefixes listed first
054     */
055    protected BinaryExprParser(Parser<? extends Node> lowerLevel, Op... ops) {
056        this(lowerLevel, true, ops);
057    }
058
059    @Override
060    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
061
062        // Gather sub-nodes and intervening operators
063        final ArrayList<Node> nodeList = new ArrayList<>(2);
064        final ArrayList<Op> opList = new ArrayList<>(1);
065        nodeList.add(this.lowerLevel.parse(session, ctx, complete));
066        while (true) {
067            this.spaceParser.parse(ctx, complete);
068            final int mark = ctx.getIndex();
069            Op op = null;
070        candidateLoop:
071            for (Op candidate : this.ops) {
072                if (ctx.tryLiteral(candidate.getSymbol())) {
073                    op = candidate;
074                    break;
075                }
076            }
077            if (op == null)
078                break;
079            this.spaceParser.parse(ctx, complete);
080            final Node rhs;
081            try {
082                rhs = this.lowerLevel.parse(session, ctx, complete);
083            } catch (ParseException e) {
084                if (complete && !e.getCompletions().isEmpty())
085                    throw e;
086                ctx.setIndex(mark);                             // backtrack
087                break;
088            }
089            nodeList.add(rhs);
090            opList.add(op);
091        }
092        if (nodeList.size() == 1)
093            return nodeList.get(0);
094
095        // Build AST
096        Node node;
097        if (this.leftAssociative) {
098            node = this.createNode(opList.get(0), nodeList.get(0), nodeList.get(1));
099            for (int i = 1; i < nodeList.size() - 1; i++)
100                node = this.createNode(opList.get(i), node, nodeList.get(i + 1));
101        } else {
102            final int max = nodeList.size();
103            node = this.createNode(opList.get(max - 2), nodeList.get(max - 2), nodeList.get(max - 1));
104            for (int i = max - 3; i >= 0; i--)
105                node = this.createNode(opList.get(i), nodeList.get(i + 1), node);
106        }
107
108        // Done
109        return node;
110    }
111
112    protected Node createNode(final Op op, final Node lhs, final Node rhs) {
113        return new Node() {
114            @Override
115            public Value evaluate(ParseSession session) {
116                return op.apply(session, lhs.evaluate(session), rhs.evaluate(session));
117            }
118        };
119    }
120}
