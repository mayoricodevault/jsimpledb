008package org.jsimpledb.parse.expr;
009
010import java.util.regex.Matcher;
011
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseSession;
014import org.jsimpledb.parse.Parser;
015
016/**
017 * Parses {@code instanceof} expressions.
018 */
019public class InstanceofParser implements Parser<Node> {
020
021    public static final InstanceofParser INSTANCE = new InstanceofParser();
022
023    @Override
024    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
025
026        // Parse left-hand side
027        final Node node = ShiftExprParser.INSTANCE.parse(session, ctx, complete);
028
029        // Try to parse 'instanceof some.class.name'
030        final Matcher matcher = ctx.tryPattern("\\s*instanceof\\s+"
031          + "(" + IdentNode.NAME_PATTERN + "(\\s*\\.\\s*" + IdentNode.NAME_PATTERN + ")*)\\s*");
032        if (matcher == null)
033            return node;
034        final String className = matcher.group(1);
035
036        // Resolve class name
037        final Class<?> cl = session.resolveClass(className);
038        if (cl == null)
039            throw new EvalException("unknown class `" + className + "'");     // TODO: tab-completions
040
041        // Return node that compares value's type to class
042        return new Node() {
043            @Override
044            public Value evaluate(final ParseSession session) {
045                return new ConstValue(cl.isInstance(node.evaluate(session).get(session)));
046            }
047        };
048    }
049}
