045/**
046 * Parses basic Java expressions such as {@link Class} literals, {@code new} expressions, auto-increment expressions,
047 * array access, field access, method invocation, etc.
048 *
049 * <p>
050 * Includes these extensions:
051 * <ul>
052 *  <li>Access to Java bean methods via "property syntax", e.g., {@code foo.name = "fred"} means {@code foo.setName("fred")}</li>
053 *  <li>Access database object fields via "property syntax" given an object ID, e.g., <code>@fc21bf6d8930a215.name</code></li>
054 *  <li>Array syntax for {@link Map} and {@link List} value access, e.g., {@code mymap[key] = value} means
055 *      {@code mymap.put(key, value)}, and {@code mylist[12] = "abc"} means {@code mylist.set(12, "abc")}</li>
056 * </ul>
057 */
058public class BaseExprParser implements Parser<Node> {
059
060    public static final BaseExprParser INSTANCE = new BaseExprParser();
061
062    private final SpaceParser spaceParser = new SpaceParser();
063
064    @Override
065    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
066
067        // Parse initial atom
068        Node node = new AtomParser(
069          Iterables.transform(session.getFunctions().keySet(), new AddSuffixFunction("("))).parse(session, ctx, complete);
070
071        // Handle new
072        if (node instanceof IdentNode && ((IdentNode)node).getName().equals("new")) {
073
074            // Get class name (or array type base), which must be a sequence of identifiers connected via dots
075            new SpaceParser(true).parse(ctx, complete);
076            final Matcher firstMatcher = ctx.tryPattern(IdentNode.NAME_PATTERN);
077            if (firstMatcher == null)
078                throw new ParseException(ctx, "expected class name");
079            String className = firstMatcher.group();
080            while (true) {
081                final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
082                if (matcher == null)
083                    break;
084                className += "." + matcher.group(1);
085            }
086
087            // Get array dimensions, if any
088            final ArrayList<Node> dims = new ArrayList<>();
089            boolean sawNull = false;
090            while (true) {
091                ctx.skipWhitespace();
092                if (!ctx.tryLiteral("["))
093                    break;
094                ctx.skipWhitespace();
095                if (ctx.tryLiteral("]")) {
096                    dims.add(null);
097                    sawNull = true;
098                    continue;
099                }
100                if (!sawNull) {
101                    dims.add(ExprParser.INSTANCE.parse(session, ctx, complete));
102                    ctx.skipWhitespace();
103                }
104                if (!ctx.tryLiteral("]"))
105                    throw new ParseException(ctx, "expected `]'").addCompletion("]");
106            }
107
108            // Resolve (base) class name
109            final Class<?> baseClass;
110            final Primitive<?> primitive;
111            if (!dims.isEmpty() && (primitive = Primitive.forName(className)) != null) {
112                if (primitive == Primitive.VOID)
113                    throw new ParseException(ctx, "illegal instantiation of void array");
114                baseClass = primitive.getType();
115            } else {
116                baseClass = session.resolveClass(className);
117                if (baseClass == null)
118                    throw new ParseException(ctx, "unknown class `" + className + "'");     // TODO: tab-completions
119            }
120
121            // Handle non-array
122            if (dims.isEmpty()) {
123
124                // Parse parameters
125                if (!ctx.tryLiteral("("))
126                    throw new ParseException(ctx, "expected `('").addCompletion("(");
127                ctx.skipWhitespace();
128                final List<Node> paramNodes = BaseExprParser.parseParams(session, ctx, complete);
129
130                // Return constructor invocation node
131                node = new Node() {
132                    @Override
133                    public Value evaluate(final ParseSession session) {
134                        if (baseClass.isInterface())
135                            throw new EvalException("invalid instantiation of " + baseClass);
136                        final Object[] params = Lists.transform(paramNodes, new com.google.common.base.Function<Node, Object>() {
137                            @Override
138                            public Object apply(Node param) {
139                                return param.evaluate(session).get(session);
140                            }
141                        }).toArray();
142                        for (Constructor<?> constructor : baseClass.getConstructors()) {
143                            final Class<?>[] ptypes = constructor.getParameterTypes();
144                            if (ptypes.length != params.length)
145                                continue;
146                            try {
147                                return new ConstValue(constructor.newInstance(params));
148                            } catch (IllegalArgumentException e) {
149                                continue;                               // wrong method, a parameter type didn't match
150                            } catch (Exception e) {
151                                final Throwable t = e instanceof InvocationTargetException ?
152                                  ((InvocationTargetException)e).getTargetException() : e;
153                                throw new EvalException("error invoking constructor `" + baseClass.getName() + "()': " + t, t);
154                            }
155                        }
156                        throw new EvalException("no compatible constructor found in " + baseClass);
157                    }
158                };
159            } else {                                        // handle array
160
161                // Check number of dimensions
162                if (dims.size() > 255)
163                    throw new ParseException(ctx, "too many array dimensions (" + dims.size() + " > 255)");
164
165                // Array literal must be present if and only if no dimensions are given
166                final List<?> literal = dims.get(0) == null ?
167                  this.parseArrayLiteral(session, ctx, complete, this.getArrayClass(baseClass, dims.size() - 1)) : null;
168
169                // Return array instantiation invocation node
170                node = new Node() {
171                    @Override
172                    public Value evaluate(final ParseSession session) {
173                        final Class<?> elemType = BaseExprParser.this.getArrayClass(baseClass, dims.size() - 1);
174                        return new ConstValue(literal != null ?
175                          this.createLiteral(session, elemType, literal) : this.createEmpty(session, elemType, dims));
176                    }
177
178                    private Object createEmpty(ParseSession session, Class<?> elemType, List<Node> dims) {
179                        final int length = dims.get(0).evaluate(session).checkIntegral(session, "array creation");
180                        final Object array = Array.newInstance(elemType, length);
181                        final List<Node> remainingDims = dims.subList(1, dims.size());
182                        if (!remainingDims.isEmpty() && remainingDims.get(0) != null) {
183                            for (int i = 0; i < length; i++)
184                                Array.set(array, i, this.createEmpty(session, elemType.getComponentType(), remainingDims));
185                        }
186                        return array;
187                    }
188
189                    private Object createLiteral(ParseSession session, Class<?> elemType, List<?> values) {
190                        final int length = values.size();
191                        final Object array = Array.newInstance(elemType, length);
192                        for (int i = 0; i < length; i++) {
193                            if (elemType.isArray()) {
194                                Array.set(array, i, this.createLiteral(session,
195                                  elemType.getComponentType(), (List<?>)values.get(i)));
196                            } else {
197                                try {
198                                    Array.set(array, i, ((Node)values.get(i)).evaluate(session).get(session));
199                                } catch (Exception e) {
200                                    throw new EvalException("error setting array value: " + e, e);
201                                }
202                            }
203                        }
204                        return array;
205                    }
206                };
207            }
208        }
209
210        // Repeatedly parse operators (this gives left-to-right association)
211        while (true) {
212
213            // Parse operator, if any (one of `[', `.', `(', `++', or `--')
214            final Matcher opMatcher = ctx.tryPattern("\\s*(\\[|\\.|\\(|\\+{2}|-{2})");
215            if (opMatcher == null)
216                return node;
217            final String opsym = opMatcher.group(1);
218            final int mark = ctx.getIndex();
219
220            // Handle operators
221            switch (opsym) {
222            case "(":
223            {
224                // Atom must be an identifier, the name of a global function being invoked
225                if (!(node instanceof IdentNode))
226                    throw new ParseException(ctx);
227                final String functionName = ((IdentNode)node).getName();
228                final AbstractFunction function = session.getFunctions().get(functionName);
229                if (function == null) {
230                    throw new ParseException(ctx, "unknown function `" + functionName + "()'")
231                      .addCompletions(ParseUtil.complete(session.getFunctions().keySet(), functionName));
232                }
233
234                // Parse function parameters
235                ctx.skipWhitespace();
236                final Object params = function.parseParams(session, ctx, complete);
237
238                // Return node that applies the function to the parameters
239                node = new Node() {
240                    @Override
241                    public Value evaluate(ParseSession session) {
242                        return function.apply(session, params);
243                    }
244                };
245                break;
246            }
247            case ".":
248            {
249                // Parse next atom - it must be an identifier, either a method or property name
250                ctx.skipWhitespace();
251                final Node memberNode = AtomParser.INSTANCE.parse(session, ctx, complete);
252                if (!(memberNode instanceof IdentNode)) {
253                    ctx.setIndex(mark);
254                    throw new ParseException(ctx);
255                }
256                String member = ((IdentNode)memberNode).getName();
257
258                // If first atom was an identifier, this must be a class name followed by a field or method name
259                Class<?> cl = null;
260                if (node instanceof IdentNode) {
261
262                    // Keep parsing identifiers as long as we can; after each identifier, try to resolve a class name
263                    final TreeMap<Integer, Class<?>> classes = new TreeMap<>();
264                    String idents = ((IdentNode)node).getName();
265                    while (true) {
266                        classes.put(ctx.getIndex(), session.resolveClass(idents));
267                        final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
268                        if (matcher == null)
269                            break;
270                        idents += "." + member;
271                        member = matcher.group(1);
272                    }
273
274                    // Use the longest class name resolved, if any
275                    for (Map.Entry<Integer, Class<?>> entry : classes.descendingMap().entrySet()) {
276                        if (entry.getValue() != null) {
277                            cl = entry.getValue();
278                            ctx.setIndex(entry.getKey());
279                            break;
280                        }
281                    }
282                    if (cl == null)
283                        throw new ParseException(ctx, "unknown class `" + idents + "'");        // TODO: tab-completions
284                }
285
286                // Handle property access
287                if (ctx.tryPattern("\\s*\\(") == null) {                                        // not a method call
288                    final String propertyName = member;
289                    final Node target = node;
290
291                    // Handle "class" when preceded by a class name
292                    if (cl != null && propertyName.equals("class")) {
293                        node = new LiteralNode(cl);
294                        break;
295                    }
296
297                    // Handle static fields
298                    if (cl != null) {
299
300                        // Get static field
301                        final Field field;
302                        try {
303                            field = cl.getField(propertyName);
304                        } catch (NoSuchFieldException e) {
305                            throw new ParseException(ctx, "class `" + cl.getName() + "' has no field named `"
306                              + propertyName + "'", e);
307                        }
308                        if ((field.getModifiers() & Modifier.STATIC) == 0) {
309                            throw new ParseException(ctx, "field `" + propertyName + "' in class `" + cl.getName()
310                              + "' is not a static field");
311                        }
312
313                        // Return node accessing it
314                        node = new ConstNode(new StaticFieldValue(field));
315                        break;
316                    }
317
318                    // Must be object property access
319                    node = new Node() {
320                        @Override
321                        public Value evaluate(ParseSession session) {
322                            return BaseExprParser.this.evaluateProperty(session, target.evaluate(session), propertyName);
323                        }
324                    };
325                    break;
326                }
327
328                // Handle method call
329                node = cl != null ?
330                  new MethodInvokeNode(cl, member, BaseExprParser.parseParams(session, ctx, complete)) :
331                  new MethodInvokeNode(node, member, BaseExprParser.parseParams(session, ctx, complete));
332                break;
333            }
334            case "[":
335            {
336                this.spaceParser.parse(ctx, complete);
337                final Node index = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
338                this.spaceParser.parse(ctx, complete);
339                if (!ctx.tryLiteral("]"))
340                    throw new ParseException(ctx).addCompletion("] ");
341                final Node target = node;
342                node = new Node() {
343                    @Override
344                    public Value evaluate(ParseSession session) {
345                        return Op.ARRAY_ACCESS.apply(session, target.evaluate(session), index.evaluate(session));
346                    }
347                };
348                break;
349            }
350            case "++":
351                node = this.createPostcrementNode("increment", node, true);
352                break;
353            case "--":
354                node = this.createPostcrementNode("decrement", node, false);
355                break;
356            default:
357                throw new RuntimeException("internal error: " + opsym);
358            }
359        }
360    }

362    // Parse array literal
363    private List<?> parseArrayLiteral(ParseSession session, ParseContext ctx, boolean complete, Class<?> elemType) {
364        final ArrayList<Object> list = new ArrayList<>();
365        this.spaceParser.parse(ctx, complete);
366        if (!ctx.tryLiteral("{"))
367            throw new ParseException(ctx).addCompletion("{ ");
368        this.spaceParser.parse(ctx, complete);
369        if (ctx.tryLiteral("}"))
370            return list;
371        while (true) {
372            list.add(elemType.isArray() ?
373              this.parseArrayLiteral(session, ctx, complete, elemType.getComponentType()) :
374              ExprParser.INSTANCE.parse(session, ctx, complete));
375            ctx.skipWhitespace();
376            if (ctx.tryLiteral("}"))
377                break;
378            if (!ctx.tryLiteral(","))
379                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
380            this.spaceParser.parse(ctx, complete);
381        }
382        return list;
383    }

385    // Get the array class with the given base type and dimensions
386    private Class<?> getArrayClass(Class<?> base, int dimensions) {
387        if (dimensions == 0)
388            return base;
389        final String suffix = base.isArray() ? base.getName() :
390          base.isPrimitive() ? "" + Primitive.get(base).getLetter() : "L" + base.getName() + ";";
391        final StringBuilder buf = new StringBuilder(dimensions + suffix.length());
392        while (dimensions-- > 0)
393            buf.append('[');
394        buf.append(suffix);
395        final String className = buf.toString();
396        try {
397            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
398        } catch (Exception e) {
399            throw new EvalException("can't load array class `" + className + "'");
400        }
401    }

403    // Parse method parameters; we assume opening `(' has just been parsed
404    static List<Node> parseParams(ParseSession session, ParseContext ctx, boolean complete) {
405        final ArrayList<Node> params = new ArrayList<Node>();
406        final SpaceParser spaceParser = new SpaceParser();
407        spaceParser.parse(ctx, complete);
408        if (ctx.tryLiteral(")"))
409            return params;
410        while (true) {
411            params.add(ExprParser.INSTANCE.parse(session, ctx, complete));
412            spaceParser.parse(ctx, complete);
413            if (ctx.tryLiteral(")"))
414                break;
415            if (!ctx.tryLiteral(","))
416                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
417            spaceParser.parse(ctx, complete);
418        }
419        return params;
420    }

422    private Value evaluateProperty(ParseSession session, Value value, final String name) {
423
424        // Evaluate target
425        final Object target = value.checkNotNull(session, "property `" + name + "' access");
426        final Class<?> cl = target.getClass();
427
428        // Handle properties of database objects (i.e., database fields)
429        if (session.hasJSimpleDB() && target instanceof JObject) {
430
431            // Get object and ID
432            final JObject jobj = (JObject)target;
433            final ObjId id = jobj.getObjId();
434
435            // Resolve JField
436            JField jfield0;
437            try {
438                jfield0 = ParseUtil.resolveJField(session, id, name);
439            } catch (IllegalArgumentException e) {
440                jfield0 = null;
441            }
442            final JField jfield = jfield0;
443
444            // Return value reflecting the field if the field was found
445            if (jfield instanceof JSimpleField)
446                return new JSimpleFieldValue(jobj, (JSimpleField)jfield);
447            else if (jfield != null)
448                return new JFieldValue(jobj, jfield);
449        } else if (!session.hasJSimpleDB() && target instanceof ObjId) {
450            final ObjId id = (ObjId)target;
451
452            // Resolve field
453            org.jsimpledb.core.Field<?> field0;
454            try {
455                field0 = ParseUtil.resolveField(session, id, name);
456            } catch (IllegalArgumentException e) {
457                field0 = null;
458            }
459            final org.jsimpledb.core.Field<?> field = field0;
460
461            // Return value reflecting the field if the field was found
462            if (field instanceof SimpleField)
463                return new SimpleFieldValue(id, (SimpleField<?>)field);
464            else if (field != null)
465                return new FieldValue(id, field);
466        }
467
468        // Try bean property accessed via bean methods
469        final BeanInfo beanInfo;
470        try {
471            beanInfo = Introspector.getBeanInfo(cl);
472        } catch (IntrospectionException e) {
473            throw new EvalException("error introspecting class `" + cl.getName() + "': " + e, e);
474        }
475        for (PropertyDescriptor propertyDescriptor : beanInfo.getPropertyDescriptors()) {
476            if (propertyDescriptor instanceof IndexedPropertyDescriptor)
477                continue;
478            if (!propertyDescriptor.getName().equals(name))
479                continue;
480            final Method getter = this.makeAccessible(propertyDescriptor.getReadMethod());
481            final Method setter = this.makeAccessible(propertyDescriptor.getWriteMethod());
482            if (getter != null && setter != null)
483                return new MutableBeanPropertyValue(target, propertyDescriptor.getName(), getter, setter);
484            else if (getter != null)
485                return new BeanPropertyValue(target, propertyDescriptor.getName(), getter);
486        }
487
488        // Try instance field
489        /*final*/ Field javaField;
490        try {
491            javaField = cl.getField(name);
492        } catch (NoSuchFieldException e) {
493            javaField = null;
494        }
495        if (javaField != null)
496            return new ObjectFieldValue(target, javaField);
497
498        // Try array.length
499        if (target.getClass().isArray() && name.equals("length"))
500            return new ConstValue(Array.getLength(target));
501
502        // Not found
503        throw new EvalException("property `" + name + "' not found in " + cl);
504    }

506    // Workaround the problem where non-public class C implements public method M of interface I.
507    // In that case, invoking method C.M results in IllegalAccessException; instead, you have to invoke I.M.
508    private Method makeAccessible(Method method) {
509        if (method == null)
510            return null;
511        Class<?> cl = method.getDeclaringClass();
512        if ((cl.getModifiers() & Modifier.PUBLIC) != 0)
513            return method;
514        do {
515            for (Class<?> iface : cl.getInterfaces()) {
516                try {
517                    return iface.getMethod(method.getName(), method.getParameterTypes());
518                } catch (NoSuchMethodException e) {
519                    continue;
520                }
521            }
522        } while ((cl = cl.getSuperclass()) != null);
523        return null;
524    }

526    private Node createPostcrementNode(final String operation, final Node node, final boolean increment) {
527        return new Node() {
528            @Override
529            public Value evaluate(ParseSession session) {
530                return node.evaluate(session).xxcrement(session, "post-" + operation, increment);
531            }
532        };
533    }
