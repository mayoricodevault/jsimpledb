053    @Override
054    public Node parse(ParseSession session, ParseContext ctx, boolean complete) {
055
056        // Parse initial atom
057        Node node = new AtomParser(
058          Iterables.transform(session.getFunctions().keySet(), new AddSuffixFunction("("))).parse(session, ctx, complete);
059
060        // Handle new
061        if (node instanceof IdentNode && ((IdentNode)node).getName().equals("new")) {
062
063            // Get class name (or array type base), which must be a sequence of identifiers connected via dots
064            new SpaceParser(true).parse(ctx, complete);
065            final Matcher firstMatcher = ctx.tryPattern(IdentNode.NAME_PATTERN);
066            if (firstMatcher == null)
067                throw new ParseException(ctx, "expected class name");
068            String className = firstMatcher.group();
069            while (true) {
070                final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
071                if (matcher == null)
072                    break;
073                className += "." + matcher.group(1);
074            }
075
076            // Get array dimensions, if any
077            final ArrayList<Node> dims = new ArrayList<>();
078            boolean sawNull = false;
079            while (true) {
080                ctx.skipWhitespace();
081                if (!ctx.tryLiteral("["))
082                    break;
083                ctx.skipWhitespace();
084                if (ctx.tryLiteral("]")) {
085                    dims.add(null);
086                    sawNull = true;
087                    continue;
088                }
089                if (!sawNull) {
090                    dims.add(ExprParser.INSTANCE.parse(session, ctx, complete));
091                    ctx.skipWhitespace();
092                }
093                if (!ctx.tryLiteral("]"))
094                    throw new ParseException(ctx, "expected `]'").addCompletion("]");
095            }
096
097            // Resolve (base) class name
098            final Class<?> baseClass;
099            final Primitive<?> primitive;
100            if (!dims.isEmpty() && (primitive = Primitive.forName(className)) != null) {
101                if (primitive == Primitive.VOID)
102                    throw new ParseException(ctx, "illegal instantiation of void array");
103                baseClass = primitive.getType();
104            } else {
105                baseClass = session.resolveClass(className);
106                if (baseClass == null)
107                    throw new ParseException(ctx, "unknown class `" + className + "'");     // TODO: tab-completions
108            }
109
110            // Handle non-array
111            if (dims.isEmpty()) {
112
113                // Parse parameters
114                if (!ctx.tryLiteral("("))
115                    throw new ParseException(ctx, "expected `('").addCompletion("(");
116                ctx.skipWhitespace();
117                final List<Node> paramNodes = BaseExprParser.parseParams(session, ctx, complete);
118
119                // Return constructor invocation node
120                node = new Node() {
121                    @Override
122                    public Value evaluate(final ParseSession session) {
123                        if (baseClass.isInterface())
124                            throw new EvalException("invalid instantiation of " + baseClass);
125                        final Object[] params = Lists.transform(paramNodes, new com.google.common.base.Function<Node, Object>() {
126                            @Override
127                            public Object apply(Node param) {
128                                return param.evaluate(session).get(session);
129                            }
130                        }).toArray();
131                        for (Constructor<?> constructor : baseClass.getConstructors()) {
132                            final Class<?>[] ptypes = constructor.getParameterTypes();
133                            if (ptypes.length != params.length)
134                                continue;
135                            try {
136                                return new Value(constructor.newInstance(params));
137                            } catch (IllegalArgumentException e) {
138                                continue;                               // wrong method, a parameter type didn't match
139                            } catch (Exception e) {
140                                final Throwable t = e instanceof InvocationTargetException ?
141                                  ((InvocationTargetException)e).getTargetException() : e;
142                                throw new EvalException("error invoking constructor `" + baseClass.getName() + "()': " + t, t);
143                            }
144                        }
145                        throw new EvalException("no compatible constructor found in " + baseClass);
146                    }
147                };
148            } else {                                        // handle array
149
150                // Check number of dimensions
151                if (dims.size() > 255)
152                    throw new ParseException(ctx, "too many array dimensions (" + dims.size() + " > 255)");
153
154                // Array literal must be present if and only if no dimensions are given
155                final List<?> literal = dims.get(0) == null ?
156                  this.parseArrayLiteral(session, ctx, complete, this.getArrayClass(baseClass, dims.size() - 1)) : null;
157
158                // Return array instantiation invocation node
159                node = new Node() {
160                    @Override
161                    public Value evaluate(final ParseSession session) {
162                        final Class<?> elemType = BaseExprParser.this.getArrayClass(baseClass, dims.size() - 1);
163                        return new Value(literal != null ?
164                          this.createLiteral(session, elemType, literal) : this.createEmpty(session, elemType, dims));
165                    }
166
167                    private Object createEmpty(ParseSession session, Class<?> elemType, List<Node> dims) {
168                        final int length = dims.get(0).evaluate(session).checkIntegral(session, "array creation");
169                        final Object array = Array.newInstance(elemType, length);
170                        final List<Node> remainingDims = dims.subList(1, dims.size());
171                        if (!remainingDims.isEmpty() && remainingDims.get(0) != null) {
172                            for (int i = 0; i < length; i++)
173                                Array.set(array, i, this.createEmpty(session, elemType.getComponentType(), remainingDims));
174                        }
175                        return array;
176                    }
177
178                    private Object createLiteral(ParseSession session, Class<?> elemType, List<?> values) {
179                        final int length = values.size();
180                        final Object array = Array.newInstance(elemType, length);
181                        for (int i = 0; i < length; i++) {
182                            if (elemType.isArray()) {
183                                Array.set(array, i, this.createLiteral(session,
184                                  elemType.getComponentType(), (List<?>)values.get(i)));
185                            } else {
186                                try {
187                                    Array.set(array, i, ((Node)values.get(i)).evaluate(session).get(session));
188                                } catch (Exception e) {
189                                    throw new EvalException("error setting array value: " + e, e);
190                                }
191                            }
192                        }
193                        return array;
194                    }
195                };
196            }
197        }
198
199        // Repeatedly parse operators (this gives left-to-right association)
200        while (true) {
201
202            // Parse operator, if any (one of `[', `.', `(', `++', or `--')
203            final Matcher opMatcher = ctx.tryPattern("\\s*(\\[|\\.|\\(|\\+{2}|-{2})");
204            if (opMatcher == null)
205                return node;
206            final String opsym = opMatcher.group(1);
207            final int mark = ctx.getIndex();
208
209            // Handle operators
210            switch (opsym) {
211            case "(":
212            {
213                // Atom must be an identifier, the name of a global function being invoked
214                if (!(node instanceof IdentNode))
215                    throw new ParseException(ctx);
216                final String functionName = ((IdentNode)node).getName();
217                final AbstractFunction function = session.getFunctions().get(functionName);
218                if (function == null) {
219                    throw new ParseException(ctx, "unknown function `" + functionName + "()'")
220                      .addCompletions(ParseUtil.complete(session.getFunctions().keySet(), functionName));
221                }
222
223                // Parse function parameters
224                ctx.skipWhitespace();
225                final Object params = function.parseParams(session, ctx, complete);
226
227                // Return node that applies the function to the parameters
228                node = new Node() {
229                    @Override
230                    public Value evaluate(ParseSession session) {
231                        return function.apply(session, params);
232                    }
233                };
234                break;
235            }
236            case ".":
237            {
238                // Parse next atom - it must be an identifier, either a method or property name
239                ctx.skipWhitespace();
240                final Node memberNode = AtomParser.INSTANCE.parse(session, ctx, complete);
241                if (!(memberNode instanceof IdentNode)) {
242                    ctx.setIndex(mark);
243                    throw new ParseException(ctx);
244                }
245                String member = ((IdentNode)memberNode).getName();
246
247                // If first atom was an identifier, this must be a class name followed by a field or method name
248                Class<?> cl = null;
249                if (node instanceof IdentNode) {
250
251                    // Keep parsing identifiers as long as we can; after each identifier, try to resolve a class name
252                    final TreeMap<Integer, Class<?>> classes = new TreeMap<>();
253                    String idents = ((IdentNode)node).getName();
254                    while (true) {
255                        classes.put(ctx.getIndex(), session.resolveClass(idents));
256                        final Matcher matcher = ctx.tryPattern("\\s*\\.\\s*(" + IdentNode.NAME_PATTERN + ")");
257                        if (matcher == null)
258                            break;
259                        idents += "." + member;
260                        member = matcher.group(1);
261                    }
262
263                    // Use the longest class name resolved, if any
264                    for (Map.Entry<Integer, Class<?>> entry : classes.descendingMap().entrySet()) {
265                        if (entry.getValue() != null) {
266                            cl = entry.getValue();
267                            ctx.setIndex(entry.getKey());
268                            break;
269                        }
270                    }
271                    if (cl == null)
272                        throw new ParseException(ctx, "unknown class `" + idents + "'");        // TODO: tab-completions
273                }
274
275                // Handle property access
276                if (ctx.tryPattern("\\s*\\(") == null) {
277                    final String propertyName = member;
278                    final Node target = node;
279                    final Class<?> cl2 = cl;
280                    node = new Node() {
281                        @Override
282                        public Value evaluate(ParseSession session) {
283                            if (cl2 != null) {
284                                return new Value(propertyName.equals("class") ?
285                                  cl2 : BaseExprParser.this.readStaticField(cl2, propertyName));
286                            }
287                            return BaseExprParser.this.evaluateProperty(session, target.evaluate(session), propertyName);
288                        }
289                    };
290                    break;
291                }
292
293                // Handle method call
294                node = this.createMethodInvokeNode(cl != null ? cl : node,
295                  member, BaseExprParser.parseParams(session, ctx, complete));
296                break;
297            }
298            case "[":
299            {
300                this.spaceParser.parse(ctx, complete);
301                final Node index = AssignmentExprParser.INSTANCE.parse(session, ctx, complete);
302                this.spaceParser.parse(ctx, complete);
303                if (!ctx.tryLiteral("]"))
304                    throw new ParseException(ctx).addCompletion("] ");
305                final Node target = node;
306                node = new Node() {
307                    @Override
308                    public Value evaluate(ParseSession session) {
309                        return Op.ARRAY_ACCESS.apply(session, target.evaluate(session), index.evaluate(session));
310                    }
311                };
312                break;
313            }
314            case "++":
315                node = this.createPostcrementNode("increment", node, true);
316                break;
317            case "--":
318                node = this.createPostcrementNode("decrement", node, false);
319                break;
320            default:
321                throw new RuntimeException("internal error: " + opsym);
322            }
323        }
324    }

326    // Parse array literal
327    private List<?> parseArrayLiteral(ParseSession session, ParseContext ctx, boolean complete, Class<?> elemType) {
328        final ArrayList<Object> list = new ArrayList<>();
329        this.spaceParser.parse(ctx, complete);
330        if (!ctx.tryLiteral("{"))
331            throw new ParseException(ctx).addCompletion("{ ");
332        this.spaceParser.parse(ctx, complete);
333        if (ctx.tryLiteral("}"))
334            return list;
335        while (true) {
336            list.add(elemType.isArray() ?
337              this.parseArrayLiteral(session, ctx, complete, elemType.getComponentType()) :
338              ExprParser.INSTANCE.parse(session, ctx, complete));
339            ctx.skipWhitespace();
340            if (ctx.tryLiteral("}"))
341                break;
342            if (!ctx.tryLiteral(","))
343                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
344            this.spaceParser.parse(ctx, complete);
345        }
346        return list;
347    }

349    // Get the array class with the given base type and dimensions
350    private Class<?> getArrayClass(Class<?> base, int dimensions) {
351        if (dimensions == 0)
352            return base;
353        final String suffix = base.isArray() ? base.getName() :
354          base.isPrimitive() ? "" + Primitive.get(base).getLetter() : "L" + base.getName() + ";";
355        final StringBuilder buf = new StringBuilder(dimensions + suffix.length());
356        while (dimensions-- > 0)
357            buf.append('[');
358        buf.append(suffix);
359        final String className = buf.toString();
360        try {
361            return Class.forName(className, false, Thread.currentThread().getContextClassLoader());
362        } catch (Exception e) {
363            throw new EvalException("can't load array class `" + className + "'");
364        }
365    }

367    // Parse method parameters; we assume opening `(' has just been parsed
368    static List<Node> parseParams(ParseSession session, ParseContext ctx, boolean complete) {
369        final ArrayList<Node> params = new ArrayList<Node>();
370        final SpaceParser spaceParser = new SpaceParser();
371        spaceParser.parse(ctx, complete);
372        if (ctx.tryLiteral(")"))
373            return params;
374        while (true) {
375            params.add(ExprParser.INSTANCE.parse(session, ctx, complete));
376            spaceParser.parse(ctx, complete);
377            if (ctx.tryLiteral(")"))
378                break;
379            if (!ctx.tryLiteral(","))
380                throw new ParseException(ctx, "expected `,'").addCompletion(", ");
381            spaceParser.parse(ctx, complete);
382        }
383        return params;
384    }

386    private Object readStaticField(Class<?> cl, String name) {
387        try {
388            return cl.getField(name).get(null);
389        } catch (NoSuchFieldException e) {
390            throw new EvalException("no field `" + name + "' found in class `" + cl.getName() + "'", e);
391        } catch (NullPointerException e) {
392            throw new EvalException("field `" + name + "' in class `" + cl.getName() + "' is not a static field");
393        } catch (Exception e) {
394            throw new EvalException("error reading field `" + name + "' in class `" + cl.getName() + "': " + e, e);
395        }
396    }

398    private Value evaluateProperty(ParseSession session, Value value, final String name) {
399
400        // Evaluate target
401        final Object target = value.checkNotNull(session, "property `" + name + "' access");
402        final Class<?> cl = target.getClass();
403
404        // Handle properties of database objects (i.e., database fields)
405        if (session.hasJSimpleDB() && target instanceof JObject) {
406
407            // Get object and ID
408            final JObject jobj = (JObject)target;
409            final ObjId id = jobj.getObjId();
410
411            // Resolve JField
412            JField jfield0;
413            try {
414                jfield0 = ParseUtil.resolveJField(session, id, name);
415            } catch (IllegalArgumentException e) {
416                jfield0 = null;
417            }
418            final JField jfield = jfield0;
419
420            // Return value if found
421            if (jfield != null) {
422                return new Value(null, jfield instanceof JSimpleField ? new Setter() {
423                    @Override
424                    public void set(ParseSession session, Value value) {
425                        final Object obj = value.get(session);
426                        try {
427                            ((JSimpleField)jfield).setValue(JTransaction.getCurrent(), jobj, obj);
428                        } catch (IllegalArgumentException e) {
429                            throw new EvalException("invalid value of type " + (obj != null ? obj.getClass().getName() : "null")
430                              + " for field `" + jfield.getName() + "'", e);
431                        }
432                    }
433                } : null) {
434                    @Override
435                    public Object get(ParseSession session) {
436                        try {
437                            return jfield.getValue(JTransaction.getCurrent(), jobj);
438                        } catch (Exception e) {
439                            throw new EvalException("error reading field `" + name + "' from object " + id + ": "
440                              + (e.getMessage() != null ? e.getMessage() : e));
441                        }
442                    }
443                };
444            }
445        } else if (!session.hasJSimpleDB() && target instanceof ObjId) {
446            final ObjId id = (ObjId)target;
447
448            // Resolve field
449            org.jsimpledb.core.Field<?> field0;
450            try {
451                field0 = ParseUtil.resolveField(session, id, name);
452            } catch (IllegalArgumentException e) {
453                field0 = null;
454            }
455            final org.jsimpledb.core.Field<?> field = field0;
456
457            // Return value if found
458            if (field != null) {
459                return new Value(field instanceof SimpleField ? new Setter() {
460                    @Override
461                    public void set(ParseSession session, Value value) {
462                        final Object obj = value.get(session);
463                        try {
464                            session.getTransaction().writeSimpleField(id, field.getStorageId(), obj, false);
465                        } catch (IllegalArgumentException e) {
466                            throw new EvalException("invalid value of type " + (obj != null ? obj.getClass().getName() : "null")
467                              + " for " + field, e);
468                        }
469                    }
470                } : null) {
471                    @Override
472                    public Object get(ParseSession session) {
473                        try {
474                            return field.getValue(session.getTransaction(), id);
475                        } catch (Exception e) {
476                            throw new EvalException("error reading field `" + name + "' from object " + id + ": "
477                              + (e.getMessage() != null ? e.getMessage() : e));
478                        }
479                    }
480                };
481            }
482        }
483
484        // Try bean property accessed via bean methods
485        final BeanInfo beanInfo;
486        try {
487            beanInfo = Introspector.getBeanInfo(cl);
488        } catch (IntrospectionException e) {
489            throw new EvalException("error introspecting class `" + cl.getName() + "': " + e, e);
490        }
491        for (PropertyDescriptor pd : beanInfo.getPropertyDescriptors()) {
492            if (pd instanceof IndexedPropertyDescriptor)
493                continue;
494            if (!pd.getName().equals(name))
495                continue;
496            final Method readMethod = pd.getReadMethod();
497            final Method writeMethod = pd.getWriteMethod();
498            return new Value(null, writeMethod != null ? new Setter() {
499                @Override
500                public void set(ParseSession session, Value value) {
501                    final Object obj = value.get(session);
502                    try {
503                        writeMethod.invoke(target, obj);
504                    } catch (Exception e) {
505                        final Throwable t = e instanceof InvocationTargetException ?
506                          ((InvocationTargetException)e).getTargetException() : e;
507                        throw new EvalException("error writing property `" + name + "' to object of type "
508                          + cl.getName() + ": " + t, t);
509                    }
510                }
511            } : null) {
512                @Override
513                public Object get(ParseSession session) {
514                    try {
515                        return readMethod.invoke(target);
516                    } catch (Exception e) {
517                        final Throwable t = e instanceof InvocationTargetException ?
518                          ((InvocationTargetException)e).getTargetException() : e;
519                        throw new EvalException("error reading property `" + name + "' from object of type "
520                          + cl.getName() + ": " + t, t);
521                    }
522                }
523            };
524        }
525
526        // Try instance field
527        /*final*/ Field javaField;
528        try {
529            javaField = cl.getField(name);
530        } catch (NoSuchFieldException e) {
531            javaField = null;
532        }
533        if (javaField != null) {
534            final Field javaField2 = javaField;
535            return new DynamicValue() {
536                @Override
537                public void set(ParseSession session, Value value) {
538                    final Object obj = value.get(session);
539                    try {
540                        javaField2.set(target, obj);
541                    } catch (Exception e) {
542                        throw new EvalException("error setting field `" + name + "' in object of type "
543                          + cl.getName() + ": " + e, e);
544                    }
545                }
546                @Override
547                public Object get(ParseSession session) {
548                    try {
549                        return javaField2.get(target);
550                    } catch (Exception e) {
551                        throw new EvalException("error reading field `" + name + "' in object of type "
552                          + cl.getName() + ": " + e, e);
553                    }
554                }
555            };
556        }
557
558        // Handle array.length
559        if (target.getClass().isArray() && name.equals("length"))
560            return new Value(Array.getLength(target));
561
562        // Not found
563        throw new EvalException("property `" + name + "' not found in " + cl);
564    }

566    private Node createMethodInvokeNode(final Object target, final String name, final List<Node> paramNodes) {
567        return new Node() {
568            @Override
569            public Value evaluate(final ParseSession session) {
570
571                // Evaluate params
572                final Object[] params = Lists.transform(paramNodes, new com.google.common.base.Function<Node, Object>() {
573                    @Override
574                    public Object apply(Node param) {
575                        return param.evaluate(session).get(session);
576                    }
577                }).toArray();
578
579                // Handle static method
580                if (!(target instanceof Node))
581                    return this.invokeMethod((Class<?>)target, null, name, params);
582
583                // Handle instance method
584                final Object obj = ((Node)target).evaluate(session).checkNotNull(session, "method " + name + "() invocation");
585                return this.invokeMethod(obj.getClass(), obj, name, params);
586            }
587
588            private Value invokeMethod(Class<?> cl, Object obj, String name, Object[] params) {
589
590                // Try interface methods
591                for (Class<?> iface : this.addInterfaces(cl, new LinkedHashSet<Class<?>>())) {
592                    for (Method method : iface.getMethods()) {
593                        final Value value = this.tryMethod(method, obj, name, params);
594                        if (value != null)
595                            return value;
596                    }
597                }
598
599                // Try class methods
600                for (Method method : cl.getMethods()) {
601                    final Value value = this.tryMethod(method, obj, name, params);
602                    if (value != null)
603                        return value;
604                }
605
606                // Not found
607                throw new EvalException("no compatible method `" + name + "()' found in " + cl);
608            }
609
610            private Set<Class<?>> addInterfaces(Class<?> cl, Set<Class<?>> interfaces) {
611                for (Class<?> iface : cl.getInterfaces()) {
612                    interfaces.add(iface);
613                    this.addInterfaces(iface, interfaces);
614                }
615                if (cl.getSuperclass() != null)
616                    this.addInterfaces(cl.getSuperclass(), interfaces);
617                return interfaces;
618            }
619
620            private Value tryMethod(Method method, Object obj, String name, Object[] params) {
621                if (!method.getName().equals(name))
622                    return null;
623                final Class<?>[] ptypes = method.getParameterTypes();
624                if (method.isVarArgs()) {
625                    if (params.length < ptypes.length - 1)
626                        return null;
627                    Object[] newParams = new Object[ptypes.length];
628                    System.arraycopy(params, 0, newParams, 0, ptypes.length - 1);
629                    Object[] varargs = new Object[params.length - (ptypes.length - 1)];
630                    System.arraycopy(params, ptypes.length - 1, varargs, 0, varargs.length);
631                    newParams[ptypes.length - 1] = varargs;
632                    params = newParams;
633                } else if (params.length != ptypes.length)
634                    return null;
635                final Object result;
636                try {
637                    result = method.invoke(obj, params);
638                } catch (IllegalArgumentException e) {
639                    return null;                            // a parameter type didn't match -> wrong method
640                } catch (Exception e) {
641                    final Throwable t = e instanceof InvocationTargetException ?
642                      ((InvocationTargetException)e).getTargetException() : e;
643                    throw new EvalException("error invoking method `" + name + "()' on "
644                      + (obj != null ? "object of type " + obj.getClass().getName() : method.getDeclaringClass()) + ": " + t, t);
645                }
646                return result != null || method.getReturnType() != Void.TYPE ? new Value(result) : Value.NO_VALUE;
647            }
648        };
649    }

651    private Node createPostcrementNode(final String operation, final Node node, final boolean increment) {
652        return new Node() {
653            @Override
654            public Value evaluate(ParseSession session) {
655                final Value oldValue = node.evaluate(session);
656                oldValue.xxcrement(session, "post-" + operation, increment);
657                return oldValue;
658            }
659        };
