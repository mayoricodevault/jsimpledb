008package org.jsimpledb.parse;
009
010import com.google.common.base.Predicate;
011import com.google.common.collect.Iterables;
012
013import java.util.NoSuchElementException;
014import java.util.regex.Matcher;
015
016import org.jsimpledb.core.ComplexField;
017import org.jsimpledb.core.Field;
018import org.jsimpledb.core.FieldSwitchAdapter;
019import org.jsimpledb.core.ObjType;
020import org.jsimpledb.core.SimpleField;
021
022/**
023 * Parses an indexed field.
024 *
025 * <p>
026 * Syntax examples:
027 * <ul>
028 *  <li><code>Person.lastName</code></li>
029 *  <li><code>Person.friends.element</code></li>
030 *  <li><code>Person.grades.key</code></li>
031 * </p>
032 */
033public class IndexedFieldParser implements Parser<IndexedFieldParser.Result> {
034
035    private final SpaceParser spaceParser = new SpaceParser();
036
037    @Override
038    public Result parse(final ParseSession session, final ParseContext ctx, final boolean complete) {
039
040        // Get object type
041        final int typeStart = ctx.getIndex();
042        final ObjType objType = new ObjTypeParser().parse(session, ctx, complete);
043        final String typeName = ctx.getOriginalInput().substring(typeStart, ctx.getIndex()).trim();
044
045        // Get indexed field
046        ctx.skipWhitespace();
047        if (!ctx.tryLiteral("."))
048            throw new ParseException(ctx, "expected field name").addCompletion(".");
049        ctx.skipWhitespace();
050        final Matcher fieldMatcher = ctx.tryPattern("\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*");
051        if (fieldMatcher == null) {
052            throw new ParseException(ctx, "expected field name").addCompletions(Iterables.transform(Iterables.filter(
053               objType.getFields().values(), new HasIndexedPredicate()), new FieldNameFunction()));
054        }
055        final String fieldName = fieldMatcher.group();
056        final Field<?> field;
057        try {
058            field = ParseUtil.resolveField(session, objType, fieldName);
059        } catch (IllegalArgumentException e) {
060            throw new ParseException(ctx, e.getMessage()).addCompletions(ParseUtil.complete(Iterables.transform(Iterables.filter(
061              objType.getFields().values(), new HasIndexedPredicate()), new FieldNameFunction()), fieldName));
062        }
063
064        // Get sub-field if field is a complex field
065        return field.visit(new FieldSwitchAdapter<Result>() {
066
067            @Override
068            protected <T> Result caseComplexField(ComplexField<T> field) {
069                ctx.skipWhitespace();
070                if (!ctx.tryLiteral(".")) {
071                    throw new ParseException(ctx, "expected sub-field name").addCompletion(".");
072                }
073                final Matcher subfieldMatcher = ctx.tryPattern("\\p{javaJavaIdentifierStart}\\p{javaJavaIdentifierPart}*");
074                if (subfieldMatcher == null) {
075                    throw new ParseException(ctx, "expected sub-field name").addCompletions(Iterables.transform(Iterables.filter(
076                       field.getSubFields(), new IsIndexedPredicate()), new FieldNameFunction()));
077                }
078                final String subName = subfieldMatcher.group();
079                final SimpleField<?> subField;
080                try {
081                    subField = Iterables.find(field.getSubFields(), new HasNamePredicate(subName));
082                } catch (NoSuchElementException e) {
083                    throw new ParseException(ctx, "unknown sub-field `" + subName + "' of complex field `" + fieldName + "'")
084                      .addCompletions(ParseUtil.complete(Iterables.transform(Iterables.filter(
085                        field.getSubFields(), new IsIndexedPredicate()), new FieldNameFunction()), fieldName));
086                }
087                return new Result(fieldName + "." + subName, this.verifyIndexedSimple(subField), field);
088            }
089
090            @Override
091            protected <T> Result caseField(Field<T> field) {
092                return new Result(fieldName, this.verifyIndexedSimple(field));
093            }
094
095            private SimpleField<?> verifyIndexedSimple(Field<?> field) {
096                if (!(field instanceof SimpleField) || !((SimpleField<?>)field).isIndexed())
097                    throw new ParseException(ctx, "expected indexed field");
098                return (SimpleField<?>)field;
099            }
100        });
101    }
102
103// Return type
104
105    /**
106     * Result returned from an {@link IndexedFieldParser} parse.
107     */
108    public static class Result {
109
110        private final String fieldName;
111        private final SimpleField<?> field;
112        private final ComplexField<?> parentField;
113
114        Result(String fieldName, SimpleField<?> field) {
115            this(fieldName, field, null);
116        }
117
118        Result(String fieldName, SimpleField<?> field, ComplexField<?> parentField) {
119            this.fieldName = fieldName;
120            this.field = field;
121            this.parentField = parentField;
122        }
123
124        /**
125         * Get the field's name.
126         */
127        public String getFieldName() {
128            return this.fieldName;
129        }
130
131        /**
132         * Get the indexed field.
133         */
134        public SimpleField<?> getField() {
135            return this.field;
136        }
137
138        /**
139         * Get the indexed field's parent field, if any.
140         *
141         * @return indexed field's parent field, or null if the indexed field is not a sub-field
142         */
143        public ComplexField<?> getParentField() {
144            return this.parentField;
145        }
146    }
147
148// Functions and Predicates
149
150    private static class FieldNameFunction implements com.google.common.base.Function<Field<?>, String> {
151        @Override
152        public String apply(Field<?> field) {
153            return field.getName();
154        }
155    }
156
157    private static class HasNamePredicate implements Predicate<Field<?>> {
158
159        private final String name;
160
161        HasNamePredicate(String name) {
162            this.name = name;
163        }
164
165        @Override
166        public boolean apply(Field<?> field) {
167            return field.getName().equals(this.name);
168        }
169    }
170
171    private static class IsIndexedPredicate implements Predicate<SimpleField<?>> {
172        @Override
173        public boolean apply(SimpleField<?> field) {
174            return field.isIndexed();
175        }
176    }
177
178    private static class HasIndexedPredicate implements Predicate<Field<?>> {
179        @Override
180        public boolean apply(Field<?> field) {
181            return field instanceof SimpleField ? ((SimpleField<?>)field).isIndexed() :
182              field instanceof ComplexField ? Iterables.any(((ComplexField<?>)field).getSubFields(), new IsIndexedPredicate()) :
183              false;
184        }
185    }
186}
