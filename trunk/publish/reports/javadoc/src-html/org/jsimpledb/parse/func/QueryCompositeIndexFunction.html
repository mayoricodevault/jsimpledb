008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.parse.CompositeIndexParser;
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseSession;
014import org.jsimpledb.parse.SpaceParser;
015import org.jsimpledb.parse.expr.AbstractValue;
016import org.jsimpledb.parse.expr.Value;
017
018@Function
019public class QueryCompositeIndexFunction extends AbstractQueryFunction {
020
021    private final SpaceParser spaceParser = new SpaceParser();
022
023    public QueryCompositeIndexFunction() {
024        super("queryCompositeIndex", 1, 1);
025    }
026
027    @Override
028    public String getHelpSummary() {
029        return "queries a composite index";
030    }
031
032    @Override
033    public String getUsage() {
034        return this.getName() + "(object-type, index-name, value-type, ...) (JSimpleDB mode only)\n"
035          + "       " + this.getName() + "(type-name.index-name)"
036          + "       " + this.getName() + "(storage-id)";
037    }
038
039    @Override
040    public String getHelpDetail() {
041        return "Queries a composite index. The object-type is the type of object to be queried, i.e.,"
042          + " the object type on which the composite index is defined, or any super-type or sub-type; a strict"
043          + " sub-type will cause the returned index to be restricted to that sub-type. The index-name"
044          + " is the name of the compositeindex. The value-type(s) are the indexed fields' value type(s);"
045          + " in the case of reference fields, a super-type or more restrictive sub-type may also be specified,"
046          + " otherwise the field type must exactly match the field."
047          + "\n\nThe first form is only valid in JSimpleDB mode; the second and third forms may be used in either JSimpleDB"
048          + " mode or Core API mode.";
049    }
050
051    @Override
052    protected int parseName(ParseSession session, ParseContext ctx, boolean complete) {
053        return new CompositeIndexParser().parse(session, ctx, complete).getStorageId();
054    }
055
056    @Override
057    protected Value apply(ParseSession session, final int storageId) {
058        return new AbstractValue() {
059            @Override
060            public Object get(ParseSession session) {
061                return session.getTransaction().queryIndex(storageId);
062            }
063        };
064    }
065
066    @Override
067    protected Value apply(ParseSession session, final Class<?> objectType, final String indexName, final Class<?>[] valueTypes) {
068        return new AbstractValue() {
069            @Override
070            public Object get(ParseSession session) {
071                switch (valueTypes.length) {
072                case 2:
073                    return JTransaction.getCurrent().queryCompositeIndex(objectType, indexName, valueTypes[0], valueTypes[1]);
074                case 3:
075                    return JTransaction.getCurrent().queryCompositeIndex(objectType,
076                      indexName, valueTypes[0], valueTypes[1], valueTypes[2]);
077                // COMPOSITE-INDEX
078                default:
079                    throw new IllegalArgumentException("wrong number of value types (" + valueTypes.length
080                      + ") provided for composite index `" + indexName + "'");
081                }
082            }
083        };
084    }
085}
