008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.parse.IndexedFieldParser;
012import org.jsimpledb.parse.ParseContext;
013import org.jsimpledb.parse.ParseException;
014import org.jsimpledb.parse.ParseSession;
015import org.jsimpledb.parse.SpaceParser;
016import org.jsimpledb.parse.expr.Value;
017
018@Function
019public class QueryFunction extends AbstractFunction {
020
021    private final SpaceParser spaceParser = new SpaceParser();
022
023    public QueryFunction() {
024        super("query");
025    }
026
027    @Override
028    public String getHelpSummary() {
029        return "queries an indexed field";
030    }
031
032    @Override
033    public String getUsage() {
034        return "query(type.field)";
035    }
036
037    @Override
038    public String getHelpDetail() {
039        return "Queries a field index and returns a mapping from field value to set of objects containing that value in the field."
040          + " For collection fields, specify the sub-field, e.g., `Person.friends.element' or `Person.grades.key'.";
041    }
042
043    @Override
044    public Integer parseParams(final ParseSession session, final ParseContext ctx, final boolean complete) {
045
046        // Parse indexed field parameter
047        if (ctx.tryLiteral(")"))
048            throw new ParseException(ctx, "indexed field parameter required");
049        final int storageId = new IndexedFieldParser().parse(session, ctx, complete).getField().getStorageId();
050
051        // Finish parse
052        ctx.skipWhitespace();
053        if (!ctx.tryLiteral(")"))
054            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
055
056        // Done
057        return storageId;
058    }
059
060    @Override
061    public Value apply(ParseSession session, Object params) {
062        final int storageId = (Integer)params;
063        return new Value(null) {
064            @Override
065            public Object get(ParseSession session) {
066                return session.hasJSimpleDB() ?
067                  JTransaction.getCurrent().querySimpleField(storageId, null) :
068                  session.getTransaction().querySimpleField(storageId);
069            }
070        };
071    }
072}
