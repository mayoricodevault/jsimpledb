008package org.jsimpledb.parse.func;
009
010import org.jsimpledb.JTransaction;
011import org.jsimpledb.core.ComplexField;
012import org.jsimpledb.core.ListField;
013import org.jsimpledb.core.MapField;
014import org.jsimpledb.core.SimpleField;
015import org.jsimpledb.core.Transaction;
016import org.jsimpledb.parse.IndexedFieldParser;
017import org.jsimpledb.parse.ParseContext;
018import org.jsimpledb.parse.ParseException;
019import org.jsimpledb.parse.ParseSession;
020import org.jsimpledb.parse.SpaceParser;
021import org.jsimpledb.parse.expr.AbstractValue;
022import org.jsimpledb.parse.expr.Value;
023
024@Function
025public class QueryFunction extends AbstractFunction {
026
027    private final SpaceParser spaceParser = new SpaceParser();
028
029    public QueryFunction() {
030        super("query");
031    }
032
033    @Override
034    public String getHelpSummary() {
035        return "queries an indexed field";
036    }
037
038    @Override
039    public String getUsage() {
040        return "query(type.field)";
041    }
042
043    @Override
044    public String getHelpDetail() {
045        return "Queries a field index and returns a mapping from field value to set of objects containing that value in the field."
046          + " For collection fields, specify the sub-field, e.g., `Person.friends.element' or `Person.grades.key'."
047          + " Indexes on list elements and map values also include corresponding list indicies or map keys (respectively).";
048    }
049
050    @Override
051    public IndexedFieldParser.Result parseParams(final ParseSession session, final ParseContext ctx, final boolean complete) {
052
053        // Parse indexed field parameter
054        if (ctx.tryLiteral(")"))
055            throw new ParseException(ctx, "indexed field parameter required");
056        final IndexedFieldParser.Result result = new IndexedFieldParser().parse(session, ctx, complete);
057
058        // Finish parse
059        ctx.skipWhitespace();
060        if (!ctx.tryLiteral(")"))
061            throw new ParseException(ctx, "expected `)'").addCompletion(") ");
062
063        // Done
064        return result;
065    }
066
067    @Override
068    public Value apply(ParseSession session, Object params) {
069        final IndexedFieldParser.Result result = (IndexedFieldParser.Result)params;
070        return new AbstractValue() {
071            @Override
072            public Object get(ParseSession session) {
073                return session.hasJSimpleDB() ?
074                  QueryFunction.this.query(JTransaction.getCurrent(), result) :
075                  QueryFunction.this.query(session.getTransaction(), result);
076            }
077        };
078    }
079
080    private Object query(JTransaction jtx, IndexedFieldParser.Result result) {
081        return jtx.queryIndex(result.getField().getStorageId());
082    }
083
084    private Object query(Transaction tx, IndexedFieldParser.Result result) {
085        final SimpleField<?> field = result.getField();
086        final ComplexField<?> parentField = result.getParentField();
087        if (parentField instanceof ListField<?>)
088            return tx.queryListField(parentField.getStorageId());
089        else if (parentField instanceof MapField<?, ?> && ((MapField<?, ?>)parentField).getValueField().equals(field))
090            return tx.queryMapValueField(parentField.getStorageId());
091        else
092            return tx.querySimpleField(field.getStorageId());
093    }
094}
