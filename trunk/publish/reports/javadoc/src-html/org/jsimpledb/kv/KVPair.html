008package org.jsimpledb.kv;
009
010import java.util.Arrays;
011import java.util.Map;
012
013import org.jsimpledb.util.ByteUtil;
014
015/**
016 * A key/value pair.
017 *
018 * Note: the internal byte arrays are not copied; therefore, values passed to the constructor
019 * or returned from the accessor methods must not be modified if instances are to remain immutable.
020 */
021public class KVPair {
022
023    private final byte[] key;
024    private final byte[] value;
025
026    /**
027     * Constructor. The given arrays are copied.
028     *
029     * @param key key
030     * @param value value
031     * @throws IllegalArgumentException if {@code key} or {@code value} is null
032     */
033    public KVPair(byte[] key, byte[] value) {
034        if (key == null)
035            throw new IllegalArgumentException("null key");
036        if (value == null)
037            throw new IllegalArgumentException("null value");
038        this.key = key;
039        this.value = value;
040    }
041
042    /**
043     * Constructor. The given key and value arrays are copied.
044     *
045     * @param entry map entry
046     * @throws IllegalArgumentException if {@code entry} or its key or value is null
047     */
048    public KVPair(Map.Entry<byte[], byte[]> entry) {
049        if (entry == null)
050            throw new IllegalArgumentException("null entry");
051        this.key = entry.getKey();
052        this.value = entry.getValue();
053        if (this.key == null)
054            throw new IllegalArgumentException("null key");
055        if (this.value == null)
056            throw new IllegalArgumentException("null value");
057    }
058
059    /**
060     * Get the key.
061     */
062    public byte[] getKey() {
063        return this.key;
064    }
065
066    /**
067     * Get the value.
068     */
069    public byte[] getValue() {
070        return this.value;
071    }
072
073    @Override
074    public String toString() {
075        return "KVPair[" + ByteUtil.toString(this.key) + ":" + ByteUtil.toString(this.value) + "]";
076    }
077
078    @Override
079    public boolean equals(Object obj) {
080        if (obj == this)
081            return true;
082        if (obj == null || obj.getClass() != this.getClass())
083            return false;
084        final KVPair that = (KVPair)obj;
085        return Arrays.equals(this.key, that.key) && Arrays.equals(this.value, that.value);
086    }
087
088    @Override
089    public int hashCode() {
090        return Arrays.hashCode(this.key) ^ Arrays.hashCode(this.value);
091    }
092}
