010import java.util.ArrayList;
011import java.util.HashSet;
012import java.util.List;
013import java.util.NavigableSet;
014import java.util.TreeSet;
015import java.util.WeakHashMap;
016
017import org.dellroad.stuff.java.Predicate;
018import org.dellroad.stuff.java.TimedWait;
019import org.jsimpledb.util.ByteUtil;

021/**
022 * Manager of read/write locks on {@code byte[]} key ranges that ensures isolation and serialization while allowing concurrent
023 * access by multiple threads to a single underlying {@code byte[]} key/value store.
024 *
025 * <p>
026 * This implementation is straightforward: read locks can overlap, but write locks may not, and all locks owned
027 * by the same owner remain in force until all are {@linkplain #release released} at the same time.
028 * </p>
029 *
030 * <p>
031 * Two timeout values are supported:
032 * <ul>
033 *  <li>The wait timeout (specified as a parameter to {@link #lock lock()}) limits how long a thread will wait
034 *      on a lock held by another thread before giving up</li>
035 *  <li>The {@linkplain #getHoldTimeout hold timeout} limits how long a thread may hold
036 *      on to a contested lock before being forced to release all its locks; after that, the
037 *      next call to {@link #lock lock} or {@link #release release} will fail</li>
038 * </ul>
039 * </p>
040 */
041public class LockManager {
042
043    private static final long TEN_YEARS_MILLIS = 10L * 365L * 24L * 60L * 60L * 1000L;
044
045    private final WeakHashMap<LockOwner, Long> holdDeadlines = new WeakHashMap<>();         // hold deadline, or null if expired
046    private final TreeSet<Lock> locksByMin = new TreeSet<>(Lock.MIN_COMPARATOR);            // locks ordered by minimum
047    private final TreeSet<Lock> locksByMax = new TreeSet<>(Lock.MAX_COMPARATOR);            // locks ordered by maximum
048    private final long nanoBasis = System.nanoTime();
049
050    private long holdTimeout;
051
052    /**
053     * Get the hold timeout configured for this instance.
054     *
055     * <p>
056     * The hold timeout limits how long a thread may hold on to a contested lock before being forced to release
057     * all of its locks; after that, the next call to {@link #lock lock} or {@link #release release} will fail.
058     * </p>
059     *
060     * @return hold timeout in milliseconds
061     */
062    public long getHoldTimeout() {
063        return this.holdTimeout;
064    }
065
066    /**
067     * Set the hold timeout for this instance. Default is zero (unlimited).
068     *
069     * @param holdTimeout how long a thread may hold a contestested lock before {@link LockResult#HOLD_TIMEOUT_EXPIRED}
070     *  will be returned by {@link #lock lock()} or {@link #release release()} in milliseconds, or zero for unlimited
071     * @throws IllegalArgumentException if {@code holdTimeout} is negative
072     */
073    public void setHoldTimeout(long holdTimeout) {
074        if (holdTimeout < 0)
075            throw new IllegalArgumentException("holdTimeout < 0");
076        this.holdTimeout = Math.min(holdTimeout, TEN_YEARS_MILLIS);             // limit to 10 years to avoid overflow
077    }
078
079    /**
080     * Acquire a lock on behalf of the specified owner.
081     *
082     * <p>
083     * This method will block for up to {@code waitTimeout} milliseconds if the lock is held by
084     * another thread, after which point {@link LockResult#WAIT_TIMEOUT_EXPIRED} is returned.
085     * </p>
086     *
087     * <p>
088     * If the owner's {@linkplain #getHoldTimeout hold timeout} has expired, then
089     * {@link LockResult#HOLD_TIMEOUT_EXPIRED} is returned.
090     * </p>
091     *
092     * <p>
093     * Once a lock is acquired it stays acquired, until all locks are released together via {@link #release release()}.
094     * </p>
095     *
096     * @param owner lock owner
097     * @param minKey minimum key (inclusive), or null for no minimum
098     * @param maxKey maximum key (exclusive), or null for no maximum
099     * @param write true for a write lock, false for a read lock
100     * @param waitTimeout how long to wait before returning {@link LockResult#WAIT_TIMEOUT_EXPIRED}
101     *  in milliseconds, or zero for unlimited
102     * @return a {@link LockResult}
103     * @throws InterruptedException if the current thread is interrupted while waiting for the lock
104     * @throws IllegalArgumentException if {@code owner} or {@code range} is null
105     * @throws IllegalArgumentException if {@code minKey > maxKey}
106     * @throws IllegalArgumentException if {@code waitTimeout} is negative
107     */
108    public synchronized LockResult lock(LockOwner owner, byte[] minKey, byte[] maxKey, boolean write, long waitTimeout)
109      throws InterruptedException {
110
111        // Sanity check
112        if (owner == null)
113            throw new IllegalArgumentException("null owner");
114        if (waitTimeout < 0)
115            throw new IllegalArgumentException("waitTimeout < 0");
116        waitTimeout = Math.min(waitTimeout, TEN_YEARS_MILLIS);                  // limit to 10 years to avoid overflow
117
118        // Create lock
119        Lock lock = new Lock(owner, minKey, maxKey, write);
120
121        // Hold timeout expired? Also calculate our wait deadline
122        if (this.holdTimeout != 0 && this.holdDeadlines.containsKey(owner) && this.holdDeadlines.get(owner) == null)
123            return LockResult.HOLD_TIMEOUT_EXPIRED;
124
125        // Build lock tester
126        final LockChecker lockChecker = new LockChecker(lock);
127
128        // Wait for lockability
129        if (!TimedWait.wait(this, waitTimeout, lockChecker))
130            return LockResult.WAIT_TIMEOUT_EXPIRED;
131
132        // Merge the lock with other locks it can merge with, removing those locks in the process
133        for (Lock that : lockChecker.getMergers()) {
134            final Lock mergedLock = lock.mergeWith(that);
135            if (mergedLock != null) {
136                this.locksByMin.remove(that);
137                this.locksByMax.remove(that);
138                owner.locks.remove(that);
139                lock = mergedLock;
140            }
141        }
142
143        // Add lock
144        this.locksByMin.add(lock);
145        this.locksByMax.add(lock);
146        owner.locks.add(lock);
147
148        // Done
149        return LockResult.SUCCESS;
150    }
151
152    /**
153     * Release all locks held by the specified owner.
154     *
155     * <p>
156     * If the owner's {@linkplain #getHoldTimeout hold timeout} has expired, then
157     * {@link LockResult#HOLD_TIMEOUT_EXPIRED} is returned.
158     * </p>
159     *
160     * @param owner lock owner
161     * @return one of {@link LockResult#HOLD_TIMEOUT_EXPIRED} or {@link LockResult#SUCCESS}
162     * @throws IllegalArgumentException if {@code owner} is null
163     */
164    public synchronized LockResult release(LockOwner owner) {
165
166        // Sanity check
167        if (owner == null)
168            throw new IllegalArgumentException("null owner");
169
170        // Check if hold timeout has alread expired; if not, remove deadline
171        if (this.holdTimeout != 0 && this.holdDeadlines.containsKey(owner)) {
172            final Long holdDeadline = this.holdDeadlines.remove(owner);
173            if (holdDeadline == null)
174                return LockResult.HOLD_TIMEOUT_EXPIRED;         // sorry, released too late
175        }
176
177        // Release all locks
178        this.doRelease(owner);
179
180        // Done
181        return LockResult.SUCCESS;
182    }
183
184    // Release all locks held by owner
185    private /*synchronized*/ void doRelease(LockOwner owner) {
186        for (Lock lock : owner.locks) {
187            this.locksByMin.remove(lock);
188            this.locksByMax.remove(lock);
189        }
190        owner.locks.clear();
191        this.notifyAll();
192    }
193
194    // Check whether we can lock, and fill the list of mergers if so
195    private /*synchronized*/ boolean checkLock(Lock lock, List<Lock> mergers) {
196
197        // Get lock's min & max
198        final byte[] lockMin = lock.getMin();
199        final byte[] lockMax = lock.getMax();
200
201    startOver:
202        while (true) {
203
204            // Get locks whose min is < lockMax
205            final NavigableSet<Lock> lhs = lockMax == null ? this.locksByMin :
206              this.locksByMin.headSet(Lock.getMinKey(lockMax, false), false);
207
208            // Get locks whose max is > lockMin
209            final NavigableSet<Lock> rhs = lockMin == null ? this.locksByMax : this.locksByMax.tailSet(
210              Lock.getMaxKey(ByteUtil.getNextKey(lockMin), false), true);
211
212            // Find overlapping locks and check for conflicts
213            final HashSet<Lock> overlaps = new HashSet<>();
214            for (Lock other : lhs) {
215
216                // Does this lock overlap?
217                if (!rhs.contains(other))
218                    continue;
219
220                // Do this lock & other lock conflict?
221                if (lock.conflictsWith(other)) {
222
223                    // Handle possible hold timeout of other lock's owner
224                    if (this.holdTimeout != 0) {
225                        final long currentTime = System.nanoTime() - this.nanoBasis;
226                        if (this.holdDeadlines.containsKey(other)) {
227                            final long holdDeadline = this.holdDeadlines.get(other.owner);
228                            if (currentTime >= holdDeadline) {                              // hold timeout has expired for 'other'
229                                this.holdDeadlines.put(other.owner, null);
230                                this.doRelease(other.owner);
231                                continue startOver;
232                            }
233                        } else {
234                            final long holdDeadline = currentTime + (this.holdTimeout * 1000000L);
235                            this.holdDeadlines.put(other.owner, holdDeadline);
236                        }
237                    }
238
239                    // Can't acquire lock due to conflict
240                    return false;
241                }
242
243                // Add overlap
244                overlaps.add(other);
245            }
246
247            //System.out.println(Thread.currentThread().getName() + ": LockChecker: BEFORE: lock = " + lock + "\n"
248            //+ "  lockByMin = " + this.locksByMin + "\n"
249            //+ "  lockByMax = " + this.locksByMax + "\n"
250            //+ "   overlaps = " + overlaps + "\n");
251
252            // Find overlaps we can merge with
253            for (Lock other : overlaps) {
254                if (lock.mergeWith(other) != null)
255                    mergers.add(other);
256            }
257
258            //System.out.println(Thread.currentThread().getName() + ": LockChecker: AFTER: lock = " + lock + "\n"
259            //+ "  lockByMin = " + this.locksByMin + "\n"
260            //+ "  lockByMax = " + this.locksByMax + "\n"
261            //+ "    mergers = " + mergers + "\n");
262
263            // Done
264            return true;
265        }
266    }
267
268// LockResult
269
270    /**
271     * Possible return values from {@link LockManager#lock LockManager.lock()}.
272     */
273    public enum LockResult {
274
275        /**
276         * The lock was successfully acquired.
277         */
278        SUCCESS,
279
280        /**
281         * The timeout expired while waiting to acquire the lock.
282         */
283        WAIT_TIMEOUT_EXPIRED,
284
285        /**
286         * The owner's hold timeout expired.
287         */
288        HOLD_TIMEOUT_EXPIRED;
289    }
290
291// LockChecker predicate
292
293    private class LockChecker implements Predicate {
294
295        private final Lock lock;
296        private final ArrayList<Lock> mergers = new ArrayList<>();
297
298        LockChecker(Lock lock) {
299            this.lock = lock;
300        }
301
302        public List<Lock> getMergers() {
303            return this.mergers;
304        }
305
306        @Override
307        public boolean test() {
308
309            // Reset state
310            this.mergers.clear();
311
312            // See if we can lock
313            return LockManager.this.checkLock(this.lock, this.mergers);
314        }
315    }
