008package org.jsimpledb.kv.fdb;
009
010import com.foundationdb.Database;
011import com.foundationdb.FDB;
012import com.foundationdb.FDBException;
013import com.foundationdb.NetworkOptions;
014
015import java.util.concurrent.Executor;
016
017import javax.annotation.PostConstruct;
018import javax.annotation.PreDestroy;
019
020import org.jsimpledb.kv.KVDatabase;
021import org.jsimpledb.kv.KVDatabaseException;
022
023/**
024 * FoundationDB {@link KVDatabase} implementation.
025 *
026 * <p>
027 * Allows specifying a {@linkplain #setKeyPrefix key prefix} for all keys, allowing multiple independent databases.
028 * </p>
029 */
030public class FoundationKVDatabase implements KVDatabase {
031
032    /**
033     * The API version used by this class.
034     */
035    public static final int API_VERSION = 200;
036
037    private final FDB fdb = FDB.selectAPIVersion(API_VERSION);
038    private final NetworkOptions options = this.fdb.options();
039
040    private String clusterFilePath;
041    private byte[] databaseName = new byte[] { (byte)'D', (byte)'B' };
042    private byte[] keyPrefix;
043    private Executor executor;
044
045    private Database database;
046
047    /**
048     * Constructor.
049     *
050     * @throws FDBException if {@link #API_VERSION} is not supported
051     */
052    public FoundationKVDatabase() {
053    }
054
055    /**
056     * Get the {@link NetworkOptions} associated with this instance.
057     * Options must be configured prior to {@link #start}.
058     */
059    public NetworkOptions getNetworkOptions() {
060        return this.options;
061    }
062
063    /**
064     * Configure the {@link Executor} used for the FoundationDB networking event loop.
065     */
066    public void setExecutor(Executor executor) {
067        this.executor = executor;
068    }
069
070    /**
071     * Configure the cluster file path. Default is null, which results in the default fdb.cluster file being used.
072     */
073    public void setClusterFilePath(String clusterFilePath) {
074        this.clusterFilePath = clusterFilePath;
075    }
076
077    /**
078     * Configure the database name. Currently the default value ({@code "DB".getBytes()}) is the only valid value.
079     */
080    public void setDatabaseName(byte[] databaseName) {
081        this.databaseName = databaseName;
082    }
083
084    /**
085     * Get the key prefix for all keys.
086     *
087     * @param keyPrefix key prefix, or null if there is none configured
088     */
089    public byte[] getKeyPrefix() {
090        return this.keyPrefix.clone();
091    }
092
093    /**
094     * Configure a prefix for all keys. The prefix will be added/removed automatically with all access.
095     * The default prefix is null, which is equivalent to an empty prefix.
096     *
097     * <p>
098     * The key prefix may not be changed after this instance has {@linkplain #start started}.
099     * </p>
100     *
101     * @param keyPrefix new prefix, or null for none
102     * @throws IllegalStateException if this instance has already been {@linkplain #start started}
103     */
104    public void setKeyPrefix(byte[] keyPrefix) {
105        if (this.database != null)
106            throw new IllegalStateException("already started");
107        this.keyPrefix = keyPrefix != null && keyPrefix.length > 0 ? keyPrefix.clone() : null;
108    }
109
110    /**
111     * Get the underlying {@link Database} associated with this instance.
112     *
113     * @throws IllegalStateException if this instance has not yet been {@linkplain #start started}
114     */
115    public Database getDatabase() {
116        if (this.database == null)
117            throw new IllegalStateException("not started");
118        return this.database;
119    }
120
121    /**
122     * Start this instance. This method must be called prior to creating any transactions.
123     *
124     * <p>
125     * This method is idempotent.
126     * </p>
127     *
128     * @throws IllegalStateException if this instance has already been {@linkplain #start started}
129     * @throws IllegalStateException if this instance has already been {@linkplain #stop stopped}
130     */
131    @PostConstruct
132    public synchronized void start() {
133        if (this.database != null)
134            throw new IllegalStateException("already started");
135        this.database = this.fdb.open(this.clusterFilePath, this.databaseName);
136        if (this.executor != null)
137            this.fdb.startNetwork(this.executor);
138        else
139            this.fdb.startNetwork();
140    }
141
142    /**
143     * Stop this instance. Invokes {@link FDB#stopNetwork}. Does nothing if not {@linkplain #start started}.
144     *
145     * @throws FDBException if an error occurs
146     */
147    @PreDestroy
148    public synchronized void stop() {
149        if (this.database == null)
150            return;
151        this.fdb.stopNetwork();
152    }
153
154    /**
155     * Create a new transaction.
156     *
157     * @throws IllegalStateException if this instance has not yet been {@linkplain #start started}
158     * @throws IllegalStateException if this instance has already been {@linkplain #stop stopped}
159     * @throws KVDatabaseException if an unexpected error occurs
160     */
161    @Override
162    public FoundationKVTransaction createTransaction() {
163        if (this.database == null)
164            throw new IllegalStateException("not started");
165        try {
166            return new FoundationKVTransaction(this, this.keyPrefix);
167        } catch (FDBException e) {
168            throw new KVDatabaseException(this, e);
169        }
170    }
171}
