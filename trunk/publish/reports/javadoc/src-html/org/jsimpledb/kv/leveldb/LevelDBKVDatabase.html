033/**
034 * {@link KVDatabase} implementation based on LevelDB and providing linearizable ACID semantics.
035 *
036 * <p>
037 * Instances implement a simple optimistic locking scheme for MVCC using LevelDB snapshots. Concurrent transactions
038 * do not contend for any locks until commit time. Each transaction's reads are noted and its writes are batched up.
039 * At commit time, if any other transactions have committed writes since the transaction's snapshot was created that
040 * conflict with any of the transaction's reads, a {@link RetryTransactionException} is thrown. Otherwise, the
041 * transaction is committed and its writes are applied.
042 * </p>
043 *
044 * <p>
045 * A {@linkplain #setDirectory database directory} is the only required configuration property.
046 * Instances may be stopped and (re)started multiple times.
047 * </p>
048 *
049 * <p>
050 * The LevelDB Java implementation is configurable by specifying the {@link DBFactory} implementation.
051 * </p>
052 *
053 * @see <a href="https://github.com/dain/leveldb">leveldb</a>
054 * @see <a href="https://github.com/fusesource/leveldbjni">leveldbjni</a>
055 */
056public class LevelDBKVDatabase implements KVDatabase {

119    private static DBFactory getDefaultDBFactory() {
120
121        // Get class names to try
122        final ArrayList<String> classNames = new ArrayList<>(3);
123        final String configuredFactoryClass = System.getProperty(DB_FACTORY_PROPERTY, null);
124        if (configuredFactoryClass != null)
125            classNames.add(configuredFactoryClass);
126        classNames.add(LEVELDBJNI_CLASS_NAME);
127        classNames.add(LEVELDB_CLASS_NAME);
128
129        // Find class
130        final ClassLoader loader = Thread.currentThread().getContextClassLoader();
131        final Logger log = LoggerFactory.getLogger(LevelDBKVDatabase.class);
132        for (String className : classNames) {
133            try {
134                return Class.forName(className, false, loader).asSubclass(DBFactory.class).newInstance();
135            } catch (Exception e) {
136                if (log.isDebugEnabled())
137                    log.debug("can't load factory class `" + className + "': " + e);
138                continue;
139            }
140        }
141        throw new RuntimeException("no " + DBFactory.class.getName() + " implementation found; tried: " + classNames);
142    }

146    /**
147     * Start this instance. This method must be called prior to creating any transactions.
148     *
149     * <p>
150     * This method is idempotent.
151     * </p>
152     *
153     * @throws IllegalStateException if this instance is not properly configured
154     * @throws IOException if an I/O error occurs
155     */
156    @PostConstruct
157    public synchronized void start() throws IOException {
158
159        // Already started?
160        if (this.db != null)
161            return;
162        assert this.versionInfoMap.isEmpty();
163        this.log.info("starting " + this);
164
165        // Check configuration
166        if (this.directory == null)
167            throw new IllegalStateException("no directory configured");
168
169        // Open database
170        if (this.log.isDebugEnabled())
171            this.log.debug("opening " + this + " LevelDB database");
172        this.db = this.factory.open(this.directory, this.options);
173
174        // Add shutdown hook so we don't leak native resources
175        if (this.shutdownHookRegistered.compareAndSet(false, true)) {
176            Runtime.getRuntime().addShutdownHook(new Thread() {
177                public void run() {
178                    LevelDBKVDatabase.this.stop();
179                }
180            });
181        }
182    }
183
184    /**
185     * Stop this instance. Does nothing if not {@linkplain #start started} or already stopped.
186     */
187    @PreDestroy
188    public void stop() {
189
190        // Set stopping flag to prevent new transactions from being created
191        synchronized (this) {
192            if (this.db == null || this.stopping)
193                return;
194            this.log.info("stopping " + this);
195            this.stopping = true;
196        }
197
198        // Grab all remaining open transactions
199        final ArrayList<LevelDBKVTransaction> openTransactions = new ArrayList<>();
200        synchronized (this) {
201            for (VersionInfo versionInfo : this.versionInfoMap.values())
202                openTransactions.addAll(versionInfo.getOpenTransactions());
203        }
204
205        // Close them (but not while holding my lock, to avoid lock order reversal)
206        for (LevelDBKVTransaction tx : openTransactions) {
207            try {
208                tx.rollback();
209            } catch (Throwable e) {
210                this.log.debug("caught exception closing open transaction during shutdown (ignoring)", e);
211            }
212        }
213
214        // Finish up
215        synchronized (this) {
216
217            // Sanity check
218            assert this.db != null;
219            assert this.versionInfoMap.isEmpty();
220
221            // Shut down LevelDB database
222            try {
223                if (this.log.isDebugEnabled())
224                    this.log.info("closing " + this + " LevelDB database");
225                this.db.close();
226            } catch (Throwable e) {
227                this.log.error("caught exception closing database during shutdown (ignoring)", e);
228            }
229
230            // Reset state
231            this.db = null;
232            this.stopping = false;
233        }
234    }

385    /**
386     * Create a new transaction.
387     *
388     * @throws IllegalStateException if this instance is not {@linkplain #start started}
389     */
390    @Override
391    public synchronized LevelDBKVTransaction createTransaction() {
392
393        // Check we're open
394        if (this.db == null)
395            throw new IllegalStateException("not started");
396
397        // Check not stopping
398        if (this.stopping)
399            throw new IllegalStateException("stop in progress");
400
401        // Get info for the current version
402        final VersionInfo versionInfo = this.getCurrentVersionInfo();
403
404        // Create the new transaction
405        final LevelDBKVTransaction tx = new LevelDBKVTransaction(this, versionInfo);
406        versionInfo.getOpenTransactions().add(tx);
407        if (this.log.isDebugEnabled())
408            this.log.debug("created new transaction " + tx);
409        if (this.log.isTraceEnabled())
410            this.log.trace("updated current version info: " + versionInfo);
411
412        // Done
413        return tx;
414    }

416// Transactions

418    /**
419     * Commit a transaction.
420     */
421    synchronized void commit(LevelDBKVTransaction tx) {
422        try {
423            this.doCommit(tx);
424        } finally {
425            this.cleanupTransaction(tx);
426        }
427    }
428
429    private synchronized void doCommit(LevelDBKVTransaction tx) {
430
431        // Get current and transaction's version info
432        final VersionInfo currentVersionInfo = this.getCurrentVersionInfo();
433        final VersionInfo transactionVersionInfo = tx.getVersionInfo();
434        final long transactionVersion = transactionVersionInfo.getVersion();
435        assert this.currentVersion - transactionVersion >= 0;
436        assert transactionVersionInfo.getOpenTransactions().contains(tx);
437
438        // Debug
439        if (this.log.isDebugEnabled()) {
440            this.log.debug("committing transaction " + tx + " based on version "
441              + transactionVersion + " (current version is " + this.currentVersion + ")");
442        }
443
444        // If the current version has advanced past the transaction's version, we must check for conflicts from intervening commits
445        for (long version = transactionVersion; version != this.currentVersion; version++) {
446            final LevelDBKVTransaction committedTransaction = this.versionInfoMap.get(version).getCommittedTransaction();
447            final boolean conflict = tx.getMutableView().isAffectedBy(committedTransaction.getMutableView());
448            if (this.log.isDebugEnabled()) {
449                this.log.debug("ordering " + tx + " after " + committedTransaction + " (version " + version + ") results in "
450                  + (conflict ? "conflict" : "no conflict"));
451                if (this.log.isTraceEnabled()) {
452                    this.log.trace("transaction view: {} committed view: {}",
453                      tx.getMutableView(), committedTransaction.getMutableView());
454                }
455            }
456            if (conflict) {
457                throw tx.logException(new RetryTransactionException(tx, "transaction is based on MVCC version "
458                  + transactionVersionInfo.getVersion() + " but the transaction committed at MVCC version "
459                  + version + " contains conflicting writes"));
460            }
461        }
462
463        // Apply the transaction's mutations
464        if (this.log.isDebugEnabled())
465            this.log.debug("applying mutations of " + tx + " to LevelDB database");
466        try (WriteBatch writeBatch = this.db.createWriteBatch()) {
467            try (LevelDBKVStore tempKV = new LevelDBKVStore(this.db, null/*doesn't matter*/, writeBatch)) {
468                tx.getMutableView().applyTo(tempKV);
469            }
470            this.db.write(writeBatch, new WriteOptions().sync(true));
471        } catch (IOException e) {
472            throw tx.logException(new KVTransactionException(tx, "error applying changes to LevelDB", e));
473        }
474        currentVersionInfo.setCommittedTransaction(tx);
475
476        // Discard transaction's reads - we only need writes from now on
477        tx.getMutableView().disableReadTracking();
478
479        // Update to the next MVCC version
480        if (this.log.isDebugEnabled())
481            this.log.debug("updating current version from " + this.currentVersion + " -> " + (this.currentVersion + 1));
482        this.currentVersion++;
483    }
484
485    /**
486     * Rollback a transaction.
487     */
488    synchronized void rollback(LevelDBKVTransaction tx) {
489        if (this.log.isDebugEnabled())
490            this.log.debug("rolling back transaction " + tx);
491        this.cleanupTransaction(tx);
492    }
493
494    private void cleanupTransaction(LevelDBKVTransaction tx) {
495
496        // Debug
497        if (this.log.isTraceEnabled())
498            this.log.trace("cleaning up transaction " + tx);
499
500        // Remove open transaction from version
501        tx.getVersionInfo().getOpenTransactions().remove(tx);
502
503        // Discard all versions older than all remaining open transactions
504        for (Iterator<Map.Entry<Long, VersionInfo>> i = this.versionInfoMap.entrySet().iterator(); i.hasNext(); ) {
505            final VersionInfo versionInfo = i.next().getValue();
506            if (!versionInfo.getOpenTransactions().isEmpty())
507                break;
508            if (this.log.isDebugEnabled()) {
509                this.log.debug("discarding obsolete version " + versionInfo + " committed by "
510                  + versionInfo.getCommittedTransaction());
511            }
512            versionInfo.close();
513            i.remove();
514        }
515    }
516
517    // Get VersionInfo for the current MVCC version, creating on demand if necessary
518    private VersionInfo getCurrentVersionInfo() {
519        VersionInfo versionInfo = this.versionInfoMap.get(this.currentVersion);
520        if (versionInfo == null) {
521            versionInfo = new VersionInfo(this.db, this.currentVersion, this.db.getSnapshot(), this.options.verifyChecksums());
522            this.versionInfoMap.put(this.currentVersion, versionInfo);
523            if (this.log.isTraceEnabled())
524                this.log.trace("created new version " + versionInfo);
525        }
526        return versionInfo;
527    }
