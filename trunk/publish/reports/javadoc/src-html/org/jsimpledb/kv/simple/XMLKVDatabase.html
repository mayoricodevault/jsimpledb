008package org.jsimpledb.kv.simple;
009
010import java.io.BufferedInputStream;
011import java.io.File;
012import java.io.FileNotFoundException;
013import java.io.IOException;
014import java.io.OutputStream;
015
016import javax.xml.stream.XMLStreamException;
017
018import org.dellroad.stuff.io.AtomicUpdateFileOutputStream;
019import org.dellroad.stuff.io.FileStreamRepository;
020import org.dellroad.stuff.io.StreamRepository;
021import org.jsimpledb.kv.KVDatabaseException;
022import org.jsimpledb.kv.RetryTransactionException;
023import org.jsimpledb.kv.util.XMLSerializer;
024
025/**
026 * Simple persistent {@link org.jsimpledb.kv.KVDatabase} backed by an XML file stored in a {@link StreamRepository}.
027 * The data is kept in memory, and the XML file is rewritten in its entirety after each successful commit.
028 * In normal usage, the XML file is stored in a regular {@link File} using a {@link FileStreamRepository}, which
029 * guarantees (via the use of {@link AtomicUpdateFileOutputStream}) that a partially written XML file can never exist.
030 *
031 * <p>
032 * If a {@link FileNotFoundException} is caught when trying to read the XML file, we assume that the underlying file has
033 * not yet been created and the database will initially be empty.
034 * </p>
035 *
036 * <p>
037 * When a {@link FileStreamRepository} is used, instances support "out-of-band" updates of the XML file. In that case,
038 * each time a transaction is accessed the modification timestamp of the XML file is examined. If the XML file has been
039 * updated by some external process since the time the transaction was created, the database will be reloaded from
040 * the XML file and the transaction will fail with a {@link RetryTransactionException}.
041 * </p>
042 *
043 * <p>
044 * Note that two different processes modifying the XML file at the same time is not without race conditions: e.g., it's possible
045 * for an external process to update the XML file just as a transaction associated with this instance is being committed
046 * and written to the file, which will result in overwriting the external process' changes.
047 * </p>
048 *
049 * @see XMLSerializer
050 * @see AtomicUpdateFileOutputStream
051 */
052public class XMLKVDatabase extends SimpleKVDatabase {
053
054    private final StreamRepository repository;
055    private final XMLSerializer serializer;
056    private final File file;
057
058    private int generation;
059    private long timestamp;
060
061    /**
062     * Normal constructor. Uses a {@link FileStreamRepository} backed by the specified file.
063     *
064     * @param file persistent XML file
065     * @throws IllegalArgumentException if {@code file} is null
066     */
067    public XMLKVDatabase(File file) {
068        this(new FileStreamRepository(file));
069    }
070
071    /**
072     * Constructor allowing storage in any user-supplied {@link StreamRepository}.
073     *
074     * @param repository XML file storage
075     * @throws IllegalArgumentException if {@code file} is null
076     */
077    public XMLKVDatabase(StreamRepository repository) {
078        if (repository == null)
079            throw new IllegalArgumentException("null repository");
080        this.repository = repository;
081        this.serializer = new XMLSerializer(this.kv);
082        this.reload();
083        this.file = repository instanceof FileStreamRepository ? ((FileStreamRepository)repository).getFile() : null;
084    }
085
086    @Override
087    public synchronized XMLKVTransaction createTransaction() {
088        this.checkForOutOfBandUpdate();
089        return new XMLKVTransaction(this, this.getWaitTimeout(), this.generation);
090    }
091
092    /**
093     * Forcibly reload this database by re-reading the XML file.
094     *
095     * <p>
096     * Any transactions that are in-progress when this method is called immediately become unusable.
097     * </p>
098     */
099    public synchronized void reload() {
100        this.readXML();
101    }
102
103    /**
104     * Get the generation number associated with the XML file.
105     * The generation number is incremented every time the database is wholesale updated by reading the file into memory,
106     * e.g., by invoking {@link #reload}.
107     *
108     * @see XMLKVTransaction#getGeneration
109     */
110    public synchronized int getGeneration() {
111        return this.generation;
112    }
113
114    /**
115     * Check the XML file's timestamp and reload it if it has been modified since the most recent
116     * read or write by this instance.
117     *
118     * @return true if file was updated and re-read, otherwise false
119     */
120    public synchronized boolean checkForOutOfBandUpdate() {
121        if (this.file == null)
122            return false;
123        final long fileTime = this.file.lastModified();
124        if (fileTime == 0)
125            return false;
126        if (this.timestamp != 0) {
127            if (fileTime <= this.timestamp)
128                return false;
129            this.log.info("detected out-of-band update of XMLKVDatabase file `" + this.file + "'; reloading");
130        }
131        this.readXML();
132        return true;
133    }
134
135    @Override
136    protected void checkState(SimpleKVTransaction tx) {
137        this.checkForOutOfBandUpdate();
138        final int txGeneration = ((XMLKVTransaction)tx).getGeneration();
139        if (txGeneration != this.generation) {
140            throw new RetryTransactionException(tx, "XML file changed since transaction started (generation number changed from "
141              + txGeneration + " to " + this.generation + ")");
142        }
143    }
144
145    @Override
146    protected void postCommit(SimpleKVTransaction tx, boolean successful) {
147
148        // If something weird happened, reload from storage
149        if (!successful) {
150            this.readXML();
151            return;
152        }
153
154        // Persist data to file
155        this.writeXML();
156    }
157
158    protected synchronized void readXML() {
159        this.generation++;
160        this.kv.removeRange(null, null);
161        if (this.file != null)
162            this.timestamp = this.file.lastModified();
163        try {
164            final BufferedInputStream input = new BufferedInputStream(this.repository.getInputStream());
165            this.serializer.read(input);
166        } catch (FileNotFoundException e) {
167            // no problem, we'll create a new file
168        } catch (IOException e) {
169            throw new KVDatabaseException(this, "error reading XML content", e);
170        } catch (XMLStreamException e) {
171            throw new KVDatabaseException(this, "error reading XML content", e);
172        }
173    }
174
175    protected synchronized void writeXML() {
176        boolean successful = false;
177        try {
178            final OutputStream output = this.repository.getOutputStream();
179            try {
180                this.serializer.write(output, true);
181                if (output instanceof AtomicUpdateFileOutputStream)
182                    ((AtomicUpdateFileOutputStream)output).getFD().sync();
183                output.close();
184                if (this.file != null)
185                    this.timestamp = this.file.lastModified();
186                successful = true;
187            } finally {
188                if (!successful && output instanceof AtomicUpdateFileOutputStream)
189                    ((AtomicUpdateFileOutputStream)output).cancel();
190            }
191        } catch (IOException e) {
192            throw new KVDatabaseException(this, "error writing XML content", e);
193        } catch (XMLStreamException e) {
194            throw new KVDatabaseException(this, "error writing XML content", e);
195        }
196    }
197}
