008package org.jsimpledb.kv.sql;
009
010import java.sql.Connection;
011import java.sql.SQLException;
012import java.sql.SQLRecoverableException;
013import java.sql.SQLTimeoutException;
014import java.sql.SQLTransientException;
015
016import javax.sql.DataSource;
017
018import org.jsimpledb.kv.KVDatabase;
019import org.jsimpledb.kv.KVDatabaseException;
020import org.jsimpledb.kv.KVTransactionException;
021import org.jsimpledb.kv.RetryTransactionException;
022import org.jsimpledb.kv.TransactionTimeoutException;
023
024/**
025 * Support superclass for SQL {@link KVDatabase} implementations.
026 */
027public class SQLKVDatabase implements KVDatabase {
028
029    /**
030     * Default table name ({@value #DEFAULT_TABLE_NAME}).
031     */
032    public static final String DEFAULT_TABLE_NAME = "KV";
033
034    /**
035     * Default key column name ({@value #DEFAULT_KEY_COLUMN_NAME}).
036     */
037    public static final String DEFAULT_KEY_COLUMN_NAME = "kv_key";
038
039    /**
040     * Default value column name ({@value #DEFAULT_VALUE_COLUMN_NAME}).
041     */
042    public static final String DEFAULT_VALUE_COLUMN_NAME = "kv_value";
043
044    protected DataSource dataSource;
045
046    /**
047     * The name of the key/value table. Default value is {@value #DEFAULT_TABLE_NAME}.
048     */
049    protected String tableName = DEFAULT_TABLE_NAME;
050
051    /**
052     * The name of the key column. Default value is {@value #DEFAULT_KEY_COLUMN_NAME}.
053     */
054    protected String keyColumnName = DEFAULT_KEY_COLUMN_NAME;
055
056    /**
057     * The name of the value column. Default value is {@value #DEFAULT_VALUE_COLUMN_NAME}.
058     */
059    protected String valueColumnName = DEFAULT_VALUE_COLUMN_NAME;
060
061    /**
062     * Get the {@link DataSource} used with this instance.
063     */
064    public DataSource getDataSource() {
065        return this.dataSource;
066    }
067
068    /**
069     * Configure the {@link DataSource} used with this instance.
070     *
071     * <p>
072     * Required property.
073     * </p>
074     */
075    public void setDataSource(DataSource dataSource) {
076        this.dataSource = dataSource;
077    }
078
079    /**
080     * Get the name of the table containing keys and values.
081     *
082     * <p>
083     * Default value is {@value #DEFAULT_TABLE_NAME}.
084     * </p>
085     */
086    public String getTableName() {
087        return this.tableName;
088    }
089
090    /**
091     * Set the name of the key/value table.
092     *
093     * @param tableName the name of the key/value table
094     * @throws IllegalArgumentException if {@code table} is null
095     */
096    public void setTableName(String tableName) {
097        if (tableName == null)
098            throw new IllegalArgumentException("null tableName");
099        this.tableName = tableName;
100    }
101
102    /**
103     * Get the name of the column containing keys.
104     *
105     * <p>
106     * Default value is {@value #DEFAULT_KEY_COLUMN_NAME}.
107     * </p>
108     */
109    public String getKeyColumnName() {
110        return this.keyColumnName;
111    }
112
113    /**
114     * Configure the name of the column containing keys.
115     *
116     * @param keyColumnName the name of the key column
117     * @throws IllegalArgumentException if {@code keyColumnName} is null
118     */
119    public void setKeyColumnName(String keyColumnName) {
120        if (keyColumnName == null)
121            throw new IllegalArgumentException("null keyColumn");
122        this.keyColumnName = keyColumnName;
123    }
124
125    /**
126     * Get the name of the column containing values.
127     *
128     * <p>
129     * Default value is {@value #DEFAULT_VALUE_COLUMN_NAME}.
130     * </p>
131     */
132    public String getValueColumnName() {
133        return this.valueColumnName;
134    }
135
136    /**
137     * Configure the name of the column containing values.
138     *
139     * @param valueColumnName the name of the value column
140     * @throws IllegalArgumentException if {@code valueColumnName} is null
141     */
142    public void setValueColumnName(String valueColumnName) {
143        if (valueColumnName == null)
144            throw new IllegalArgumentException("null valueColumn");
145        this.valueColumnName = valueColumnName;
146    }
147
148    /**
149     * Create a new transaction.
150     *
151     * @throws KVDatabaseException if an unexpected error occurs
152     * @throws IllegalStateException if no {@link DataSource} is {@linkplain #setDataSource configured}
153     */
154    @Override
155    public SQLKVTransaction createTransaction() {
156        if (this.dataSource == null)
157            throw new IllegalStateException("no DataSource configured");
158        try {
159            return this.createSQLKVTransaction(this.createConnection());
160        } catch (SQLException e) {
161            throw new KVDatabaseException(this, e);
162        }
163    }
164
165    /**
166     * Create a {@link Connection} for a new transaction.
167     *
168     * <p>
169     * Note: this method is responsible for beginning the SQL transaction.
170     * </p>
171     *
172     * <p>
173     * The implementation in {@link SQLKVDatabase} invokes {@link DataSource#getConnection()} on the
174     * {@linkplain #dataSource configured DataSource} and then {@link Connection#setAutoCommit Connection.setAutoCommit(false)}
175     * to open the transaction.
176     * </p>
177     */
178    protected Connection createConnection() throws SQLException {
179        final Connection connection = this.dataSource.getConnection();
180        connection.setAutoCommit(false);
181        return connection;
182    }
183
184    /**
185     * Create a new {@link SQLKVTransaction} for a new transaction given the specified {@link Connection}.
186     *
187     * <p>
188     * The implementation in {@link SQLKVDatabase} just invokes
189     * {@link SQLKVTransaction#SQLKVTransaction new SQLKVTransaction(this, connection)}.
190     * </p>
191     */
192    protected SQLKVTransaction createSQLKVTransaction(Connection connection) throws SQLException {
193        return new SQLKVTransaction(this, connection);
194    }
195
196    /**
197     * Create an SQL statement that reads the value column associated with key <code>&#63;1<?code>.
198     */
199    public String createGetStatement() {
200        return "SELECT " + this.quote(this.keyColumnName) + " FROM "
201          + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " = ?1";
202    }
203
204    /**
205     * Create an SQL statement that reads the key and value columns (in that order) associated
206     * with the smallest key greater than or equal to <code>&#63;1<?code>, if any.
207     */
208    public String createGetAtLeastStatement() {
209        return "SELECT " + this.quote(this.keyColumnName) + ", " + this.quote(this.valueColumnName)
210          + " FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " >= ?1";
211    }
212
213    /**
214     * Create an SQL statement that reads the key and value columns (in that order) associated with the first key, if any.
215     */
216    public String createGetFirstStatement() {
217        return "SELECT " + this.quote(this.keyColumnName) + ", " + this.quote(this.valueColumnName)
218          + " FROM " + this.quote(this.tableName);
219    }
220
221    /**
222     * Create an SQL statement that reads the key and value columns (in that order)
223     * associated with the greatest key strictly less than <code>&#63;1<?code>, if any.
224     */
225    public String createGetAtMostStatement() {
226        return "SELECT " + this.quote(this.keyColumnName) + ", " + this.quote(this.valueColumnName)
227          + " FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName)
228          + " < ?1 ORDER BY " + this.quote(this.keyColumnName) + " DESC";
229    }
230
231    /**
232     * Create an SQL statement that reads the key and value columns (in that order) associated with the last key, if any.
233     */
234    public String createGetLastStatement() {
235        return "SELECT " + this.quote(this.keyColumnName) + ", " + this.quote(this.valueColumnName)
236          + " FROM " + this.quote(this.tableName) + " ORDER BY " + this.quote(this.keyColumnName) + " DESC";
237    }
238
239    /**
240     * Create an SQL statement that inserts the key/value pair with key <code>&#63;1<?code> and value <code>&#63;2<?code>.
241     * Note that a row with key <code>&#63;1<?code> may already exist; if so, the value should be updated to <code>&#63;2<?code>.
242     */
243    public String createPutStatement() {
244        return "INSERT INTO " + this.quote(this.tableName) + " (" + this.quote(this.keyColumnName)
245          + ", " + this.quote(this.valueColumnName) + ") VALUES (?1, ?2) ON DUPLICATE KEY UPDATE";
246    }
247
248    /**
249     * Create an SQL statement that deletes the row associated with key <code>&#63;1<?code>, if any.
250     * Note that the key may or may not exist prior to this method being invoked.
251     */
252    public String createRemoveStatement() {
253        return "DELETE FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " = ?1";
254    }
255
256    /**
257     * Create an SQL statement that deletes all rows with keys in the range <code>&#63;1<?code> (inclusive}
258     * to <code>&#63;2<?code> (exclusive).
259     */
260    public String createRemoveRangeStatement() {
261        return "DELETE FROM " + this.quote(this.tableName)
262          + " WHERE " + this.quote(this.keyColumnName) + " >= ?1 AND " + this.quote(this.keyColumnName) + " < ?2";
263    }
264
265    /**
266     * Create an SQL statement that deletes all rows with keys greater than or equal to <code>&#63;1<?code>.
267     */
268    public String createRemoveAtLeastStatement() {
269        return "DELETE FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " >= ?1";
270    }
271
272    /**
273     * Create an SQL statement that deletes all rows with keys strictly less than <code>&#63;1<?code>.
274     */
275    public String createRemoveAtMostStatement() {
276        return "DELETE FROM " + this.quote(this.tableName) + " WHERE " + this.quote(this.keyColumnName) + " < ?2";
277    }
278
279    /**
280     * Create an SQL statement that deletes all rows.
281     */
282    public String createRemoveAllStatement() {
283        return "DELETE FROM " + this.quote(this.tableName);
284    }
285
286    /**
287     * Enquote a table or column name as necessary.
288     *
289     * <p>
290     * The implementation in {@link SQLKVDatabase} returns its parameter unchanged.
291     * </p>
292     *
293     * @param name table or column name
294     * @return {@code name} enquoted as necessary for this database type
295     * @throws IllegalArgumentException if {@code name} is null
296     */
297    public String quote(String name) {
298        return name;
299    }
300
301    /**
302     * Wrap the given {@link SQLException} in the appropriate {@link KVTransactionException}.
303     *
304     * @param transaction the {@link SQLKVTransaction} in which the exception occured
305     * @param e SQL exception
306     * @return appropriate {@link KVTransactionException} with chained exception {@code e}
307     * @throws NullPointerException if {@code e} is null
308     */
309    public KVTransactionException wrapException(SQLKVTransaction transaction, SQLException e) {
310        if (e instanceof SQLTimeoutException)
311            return new TransactionTimeoutException(transaction, e);
312        if (e instanceof SQLRecoverableException)
313            return new RetryTransactionException(transaction, e);
314        if (e instanceof SQLTransientException)
315            return new RetryTransactionException(transaction, e);
316        return new KVTransactionException(transaction, e);
317    }
318}
