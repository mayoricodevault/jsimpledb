008package org.jsimpledb.kv.util;
009
010import java.io.InputStream;
011import java.io.OutputStream;
012import java.util.Iterator;
013
014import javax.xml.namespace.QName;
015import javax.xml.stream.XMLInputFactory;
016import javax.xml.stream.XMLOutputFactory;
017import javax.xml.stream.XMLStreamException;
018import javax.xml.stream.XMLStreamReader;
019import javax.xml.stream.XMLStreamWriter;
020
021import org.dellroad.stuff.xml.IndentXMLStreamWriter;
022import org.jsimpledb.kv.KVPair;
023import org.jsimpledb.kv.KVStore;
024import org.jsimpledb.util.AbstractXMLStreaming;
025import org.jsimpledb.util.ByteUtil;
026
027/**
028 * Utility methods for serializing and deserializing the contents of a {@link org.jsimpledb.kv.KVStore} to/from XML.
029 *
030 * <p>
031 * The XML has a simple format; empty values may be omitted:
032 *  <pre>
033 *  &lt;xml version="1.0" encoding="UTF-8"?&gt;
034 *
035 *  &lt;entries&gt;
036 *      &lt;entry&gt;
037 *          &lt;key&gt;013f7b&lt;/key&gt;
038 *          &lt;value&gt;5502&lt;/value&gt;
039 *      &lt;/entry&gt;
040 *      &lt;entry&gt;
041 *          &lt;key&gt;ee7698&lt;/key&gt;
042 *      &lt;/entry&gt;
043 *      ...
044 *  &lt;/entries&gt;
045 *  </pre>
046 * </p>
047 */
048public class XMLSerializer extends AbstractXMLStreaming {
049
050    public static final QName ENTRIES_TAG = new QName("entries");
051    public static final QName ENTRY_TAG = new QName("entry");
052    public static final QName KEY_TAG = new QName("key");
053    public static final QName VALUE_TAG = new QName("value");
054
055    private final KVStore kv;
056
057    /**
058     * Constructor.
059     *
060     * @param kv key/value store on which to operate
061     */
062    public XMLSerializer(KVStore kv) {
063        if (kv == null)
064            throw new IllegalArgumentException("null kv");
065        this.kv = kv;
066    }
067
068    /**
069     * Import key/value pairs into the {@link KVStore} associated with this instance from the given XML input.
070     *
071     * @param input XML input
072     * @throws XMLStreamException if an error occurs
073     */
074    public void read(InputStream input) throws XMLStreamException {
075        this.read(XMLInputFactory.newFactory().createXMLStreamReader(input));
076    }
077
078    /**
079     * Import key/value pairs into the {@link KVStore} associated with this instance from the given XML input.
080     * This method expects to see an opening {@code <entries>} as the next event (not counting whitespace, comments, etc.),
081     * which is then consumed up through the closing {@code </entries>} event. Therefore this tag could be part of a
082     * larger XML document.
083     *
084     * @param reader XML reader
085     * @throws XMLStreamException if an error occurs
086     */
087    public void read(XMLStreamReader reader) throws XMLStreamException {
088        this.expect(reader, false, ENTRIES_TAG);
089        while (this.expect(reader, true, ENTRY_TAG)) {
090            this.expect(reader, false, KEY_TAG);
091            byte[] key;
092            try {
093                key = ByteUtil.parse(reader.getElementText());
094            } catch (IllegalArgumentException e) {
095                throw new XMLStreamException("invalid hexadecimal key", reader.getLocation(), e);
096            }
097            if (!this.expect(reader, true, VALUE_TAG)) {
098                this.kv.put(key, new byte[0]);
099                continue;                           // closing </entry> tag alread read
100            }
101            byte[] value;
102            try {
103                value = ByteUtil.parse(reader.getElementText());
104            } catch (IllegalArgumentException e) {
105                throw new XMLStreamException("invalid hexadecimal value", reader.getLocation(), e);
106            }
107            this.kv.put(key, value);
108            this.expect(reader, true);              // read closing </entry> tag
109        }
110    }
111
112    /**
113     * Export all key/value pairs from the {@link KVStore} associated with this instance to the given XML output.
114     *
115     * @param output XML output; will not be closed by this method
116     * @param indent true to indent output, false for all on one line
117     * @throws XMLStreamException if an error occurs
118     */
119    public void write(OutputStream output, boolean indent) throws XMLStreamException {
120        XMLStreamWriter writer = XMLOutputFactory.newInstance().createXMLStreamWriter(output, "UTF-8");
121        if (indent)
122            writer = new IndentXMLStreamWriter(writer);
123        writer.writeStartDocument("UTF-8", "1.0");
124        this.write(writer, null, null);
125    }
126
127    /**
128     * Export a range of key/value pairs from the {@link KVStore} associated with this instance to the given XML output.
129     *
130     * <p>
131     * This method writes a start element as its first action, allowing the output to be embedded into a larger XML document.
132     * Callers not embedding the output may with to precede invocation of this method with a call to
133     * {@link XMLStreamWriter#writeStartDocument writer.writeStartDocument()}.
134     * </p>
135     *
136     * @param writer XML writer; will not be closed by this method
137     * @param minKey minimum key (inclusive), or null for none
138     * @param maxKey maximum key (exclusive), or null for none
139     * @throws XMLStreamException if an error occurs
140     */
141    public void write(XMLStreamWriter writer, byte[] minKey, byte[] maxKey) throws XMLStreamException {
142        writer.setDefaultNamespace(ENTRIES_TAG.getNamespaceURI());
143        writer.writeStartElement(ENTRIES_TAG.getNamespaceURI(), ENTRIES_TAG.getLocalPart());
144        for (Iterator<KVPair> i = this.kv.getRange(minKey, maxKey, false); i.hasNext(); ) {
145            writer.writeStartElement(ENTRY_TAG.getNamespaceURI(), ENTRY_TAG.getLocalPart());
146            final KVPair pair = i.next();
147            this.writeElement(writer, KEY_TAG, pair.getKey());
148            final byte[] value = pair.getValue();
149            if (value.length > 0)
150                this.writeElement(writer, VALUE_TAG, value);
151            writer.writeEndElement();
152        }
153        writer.writeEndElement();
154        writer.flush();
155    }
156
157// Internal methods
158
159    private void writeElement(XMLStreamWriter writer, QName element, byte[] value) throws XMLStreamException {
160        this.writeElement(writer, element, ByteUtil.toString(value));
161    }
162}
