008package org.jsimpledb.kv;
009
010/**
011 * Implemented by {@link KVStore}s that support encoding a 64 bit counter in a key/value pair such that the counter
012 * can be efficiently {@linkplain #adjustCounter adjusted} by concurrent transactions without conflict.
013 * In practice this means no locking is required to increment or decrement the counter by some amount, as long as
014 * it's not necessary to actually directly read or write the counter value in the same transaction.
015 *
016 * <p>
017 * How counters are encoded is specific to the implementation. Clients needing to read or write counter values directly
018 * should use {@link #decodeCounter decodeCounter()} and {@link #encodeCounter encodeCounter()}, respectively.
019 * Counters are removed using the normal methods (i.e., {@link #remove remove()} and {@link #removeRange removeRange()}).
020 * </p>
021 */
022public interface CountingKVStore extends KVStore {
023
024    /**
025     * Encode a counter value into a {@code byte[]} value suitable for use with {@link #decodeCounter decodeCounter()}
026     * and/or {@link #adjustCounter adjustCounter()}.
027     *
028     * @param value desired counter value
029     * @return encoded counter value
030     */
031    byte[] encodeCounter(long value);
032
033    /**
034     * Decode a counter value.
035     *
036     * @param value encoded counter value
037     * @return decoded counter value
038     * @throws IllegalArgumentException if {@code value} is not a valid counter value
039     * @throws NullPointerException if {@code value} is null
040     */
041    long decodeCounter(byte[] value);
042
043    /**
044     * Adjust the counter at the given key by the given amount.
045     *
046     * <p>
047     * This operation should behave in a lock-free manner, so that concurrent transactions can invoke it without
048     * conflict. However, if {@code key} is read or written to in this transaction, then this behavior may no
049     * longer be guaranteed.
050     * </p>
051     *
052     * <p>
053     * If there is no value associated with {@code key}, or {@code key}'s value is not a valid counter encoding as
054     * would be acceptable to {@link #decodeCounter decodeCounter()}, then how this operation affects {@code key}'s
055     * value, if at all, is undefined.
056     * </p>
057     *
058     * @param key key
059     * @param amount amount to adjust counter value by
060     * @throws NullPointerException if {@code key} is null
061     */
062    void adjustCounter(byte[] key, long amount);
063}
