008package org.jsimpledb.kv;
009
010import java.util.ArrayList;
011import java.util.Arrays;
012
013import org.jsimpledb.util.ByteUtil;
014
015/**
016 * Utility methods for dealing with {@link KeyFilter}s.
017 *
018 * @see KeyFilter
019 */
020public final class KeyFilterUtil {
021
022    private KeyFilterUtil() {
023    }
024
025    /**
026     * Create a {@link KeyFilter} that represents the union of the given instances.
027     *
028     * @param keyFilters instances to unify
029     * @return the union of the given {@code keyFilters}
030     * @throws IllegalArgumentException if {@code keyFilters} is empty
031     * @throws IllegalArgumentException if {@code keyFilters} or any element in {@code keyFilters} is null
032     */
033    public static KeyFilter union(KeyFilter... keyFilters) {
034        if (keyFilters == null)
035            throw new IllegalArgumentException("null keyFilters");
036        switch (keyFilters.length) {
037        case 0:
038            throw new IllegalArgumentException("empty keyFilters");
039        case 1:
040            return keyFilters[0];
041        default:
042            break;
043        }
044
045        // Optimize when all are KeyRanges instances
046        boolean allKeyRanges = true;
047        for (KeyFilter keyFilter : keyFilters) {
048            if (keyFilter == null)
049                throw new IllegalArgumentException("null keyFilter");
050            if (!(keyFilter instanceof KeyRanges))
051                allKeyRanges = false;
052        }
053        if (allKeyRanges) {
054            KeyRanges union = (KeyRanges)keyFilters[0];
055            for (int i = 1; i < keyFilters.length; i++)
056                union = union.union((KeyRanges)keyFilters[i]);
057            return union;
058        }
059
060        // Can't optimize
061        return new UnionKeyFilter(keyFilters);
062    }
063
064    /**
065     * Create a {@link KeyFilter} that represents the intersection of the given instances.
066     *
067     * @param keyFilters instances to intersect
068     * @return the intersection of the given {@code keyFilters}
069     * @throws IllegalArgumentException if {@code keyFilters} is empty
070     * @throws IllegalArgumentException if {@code keyFilters} or any element in {@code keyFilters} is null
071     */
072    public static KeyFilter intersection(KeyFilter... keyFilters) {
073        if (keyFilters == null)
074            throw new IllegalArgumentException("null keyFilters");
075        switch (keyFilters.length) {
076        case 0:
077            throw new IllegalArgumentException("empty keyFilters");
078        case 1:
079            return keyFilters[0];
080        default:
081            break;
082        }
083
084        // Optimize when all are KeyRanges instances
085        boolean allKeyRanges = true;
086        for (KeyFilter keyFilter : keyFilters) {
087            if (keyFilter == null)
088                throw new IllegalArgumentException("null keyFilter");
089            if (!(keyFilter instanceof KeyRanges))
090                allKeyRanges = false;
091        }
092        if (allKeyRanges) {
093            KeyRanges intersection = (KeyRanges)keyFilters[0];
094            for (int i = 1; i < keyFilters.length; i++)
095                intersection = intersection.intersection((KeyRanges)keyFilters[i]);
096            return intersection;
097        }
098
099        // Can't optimize
100        return new IntersectionKeyFilter(keyFilters);
101    }
102
103    /**
104     * Determine if the given instance is empty, i.e., contains no keys.
105     *
106     * @param keyFilter instance to check
107     * @throws IllegalArgumentException if {@code keyFilter} is null
108     */
109    public static boolean isEmpty(KeyFilter keyFilter) {
110        if (keyFilter == null)
111            throw new IllegalArgumentException("null keyFilter");
112        return keyFilter.seekHigher(ByteUtil.EMPTY) == null;
113    }
114
115    private static byte[] seek(KeyFilter[] keyFilters, byte[] key, boolean seekHigher, boolean preferHigher) {
116        if (key == null)
117            throw new IllegalArgumentException("null key");
118        assert keyFilters.length > 0;
119        final boolean preferNull = seekHigher == preferHigher;
120        byte[] best = null;
121        for (int i = 0; i < keyFilters.length; i++) {
122            final KeyFilter keyFilter = keyFilters[i];
123            final byte[] next = seekHigher ? keyFilter.seekHigher(key) : keyFilter.seekLower(key);
124            if (i == 0)
125                best = next;
126            if (next == null) {
127                if (preferNull)
128                    return null;
129                continue;
130            }
131            assert next.length != 0 || key.length == 0;
132            if (i > 0 && best != null) {
133                final int diff = (!seekHigher && next.length == 0) ? 1 : ByteUtil.compare(next, best);
134                if (preferHigher ? diff < 0 : diff > 0)
135                    continue;
136            } else
137                assert i == 0 || !preferNull;
138            best = next;
139        }
140        return best;
141    }
142
143// UnionKeyFilter
144
145    private static class UnionKeyFilter implements KeyFilter {
146
147        private final KeyFilter[] keyFilters;
148
149        UnionKeyFilter(KeyFilter[] keyFilters) {
150            assert keyFilters.length >= 2;
151            final ArrayList<KeyFilter> list = new ArrayList<>();
152            for (KeyFilter keyFilter : keyFilters) {
153                if (keyFilter instanceof UnionKeyFilter)
154                    list.addAll(Arrays.asList(((UnionKeyFilter)keyFilter).keyFilters));
155                else
156                    list.add(keyFilter);
157            }
158            this.keyFilters = list.toArray(new KeyFilter[list.size()]);
159        }
160
161        @Override
162        public boolean contains(byte[] key) {
163            for (KeyFilter keyFilter : this.keyFilters) {
164                if (keyFilter.contains(key))
165                    return true;
166            }
167            return false;
168        }
169
170        @Override
171        public byte[] seekHigher(byte[] key) {
172            return KeyFilterUtil.seek(this.keyFilters, key, true, false);
173        }
174
175        @Override
176        public byte[] seekLower(byte[] key) {
177            return KeyFilterUtil.seek(this.keyFilters, key, false, true);
178        }
179    }
180
181// IntersectionKeyFilter
182
183    private static class IntersectionKeyFilter implements KeyFilter {
184
185        private final KeyFilter[] keyFilters;
186
187        IntersectionKeyFilter(KeyFilter[] keyFilters) {
188            assert keyFilters.length >= 2;
189            final ArrayList<KeyFilter> list = new ArrayList<>();
190            for (KeyFilter keyFilter : keyFilters) {
191                if (keyFilter instanceof IntersectionKeyFilter)
192                    list.addAll(Arrays.asList(((IntersectionKeyFilter)keyFilter).keyFilters));
193                else
194                    list.add(keyFilter);
195            }
196            this.keyFilters = list.toArray(new KeyFilter[list.size()]);
197        }
198
199        @Override
200        public boolean contains(byte[] key) {
201            for (KeyFilter keyFilter : this.keyFilters) {
202                if (!keyFilter.contains(key))
203                    return false;
204            }
205            return true;
206        }
207
208        @Override
209        public byte[] seekHigher(byte[] key) {
210            return KeyFilterUtil.seek(this.keyFilters, key, true, true);
211        }
212
213        @Override
214        public byte[] seekLower(byte[] key) {
215            return KeyFilterUtil.seek(this.keyFilters, key, false, false);
216        }
217    }
