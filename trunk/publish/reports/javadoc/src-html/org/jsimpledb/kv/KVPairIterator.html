008package org.jsimpledb.kv;
009
010import java.util.Iterator;
011import java.util.NoSuchElementException;
012
013import org.jsimpledb.util.ByteUtil;
014
015/**
016 * An {@link Iterator} that iterates over all key/value pairs in a {@link KVStore} within a contiguous range of keys,
017 * without using the {@link KVStore#getRange KVStore.getRange()} method. Therefore, it can be used to implement
018 * {@link KVStore#getRange KVStore.getRange()} in {@link KVStore} implementaions that don't natively support iteration.
019 *
020 * <p>
021 * The iteration is instead implemented using {@link KVStore#getAtLeast KVStore.getAtLeast()},
022 * {@link KVStore#getAtMost KVStore.getAtMost()}, and {@link KVStore#remove KVStore.remove()}.
023 * Instances support {@linkplain #setNextTarget repositioning}, forward or reverse iteration, and {@link #remove removal}.
024 * </p>
025 */
026public class KVPairIterator implements Iterator<KVPair> {
027
028    protected final KVStore kv;
029    protected final boolean reverse;
030
031    private final byte[] prefix;
032    private final byte[] minKey;
033    private final byte[] maxKey;
034
035    private KVPair currPair;            // cached value to return from next()
036    private byte[] nextKey;             // next key lower/upper bound to go fetch, or null for none
037    private byte[] removeKey;           // next key to remove if remove() invoked
038    private boolean finished;
039
040    /**
041     * Convenience constructor for forward iteration and arbitrary minimum and maximum keys. Equivalent to:
042     *  <blockquote><code>
043     *  KVPairIterator(kv, minKey, maxKey, true)
044     *  </code></blockquote>
045     *
046     * @param kv underlying {@link KVStore}
047     * @param minKey minimum key (inclusive), or null for no minimum
048     * @param maxKey maximum key (exclusive), or null for no maximum
049     * @throws IllegalArgumentException if {@code kv} is null
050     * @throws IllegalArgumentException if {@code minKey > maxKey}
051     */
052    public KVPairIterator(KVStore kv, byte[] minKey, byte[] maxKey) {
053        this(kv, minKey, maxKey, true);
054    }
055
056    /**
057     * Convenience constructor for forward iteration when the range is defined as all keys having a given prefix. Equivalent to:
058     *  <blockquote><code>
059     *  KVPairIterator(kv, prefix, true)
060     *  </code></blockquote>
061     *
062     * @param kv underlying {@link KVStore}
063     * @param prefix range prefix
064     * @throws IllegalArgumentException if any parameter is null
065     */
066    public KVPairIterator(KVStore kv, byte[] prefix) {
067        this(kv, prefix, true);
068    }
069
070    /**
071     * Primary constructor handling arbitrary minimum and maximum keys.
072     *
073     * @param kv underlying {@link KVStore}
074     * @param minKey minimum key (inclusive), or null for no minimum
075     * @param maxKey maximum key (exclusive), or null for no maximum
076     * @param reverse true to iterate in a reverse direction, false to iterate in a forward direction
077     * @throws IllegalArgumentException if {@code kv} is null
078     * @throws IllegalArgumentException if {@code minKey > maxKey}
079     */
080    public KVPairIterator(KVStore kv, byte[] minKey, byte[] maxKey, boolean reverse) {
081        this(kv, minKey, maxKey, null, reverse);
082    }
083
084    /**
085     * Primary constructor for when the range is defined as all keys having a given prefix.
086     *
087     * @param kv underlying {@link KVStore}
088     * @param prefix range prefix
089     * @param reverse true to iterate in a reverse direction, false to iterate in a forward direction
090     * @throws IllegalArgumentException if any parameter is null
091     */
092    public KVPairIterator(KVStore kv, byte[] prefix, boolean reverse) {
093        this(kv, prefix, KVPairIterator.getMaxKey(prefix), prefix, reverse);
094    }
095
096    private KVPairIterator(KVStore kv, byte[] minKey, byte[] maxKey, byte[] prefix, boolean reverse) {
097        if (kv == null)
098            throw new IllegalArgumentException("null kv");
099        if (minKey != null && maxKey != null && ByteUtil.compare(minKey, maxKey) > 0)
100            throw new IllegalArgumentException("minKey > maxKey");
101        this.kv = kv;
102        this.minKey = minKey != null ? minKey.clone() : null;
103        this.maxKey = maxKey != null ? maxKey.clone() : null;
104        this.prefix = prefix != null ? prefix.clone() : null;
105        this.reverse = reverse;
106        this.nextKey = reverse ? this.maxKey : this.minKey;
107    }
108
109    /**
110     * Get the {@link KVStore} associated with this instance.
111     *
112     * @return this instance's transaction
113     */
114    public KVStore getKVStore() {
115        return this.kv;
116    }
117
118    /**
119     * Get the prefix associated with this instance, if any.
120     *
121     * @return a modifiable copy of this instance's range prefix if a prefix constructor was used, otherwise null
122     */
123    public byte[] getPrefix() {
124        return this.prefix.clone();
125    }
126
127    /**
128     * Determine if this instance is going forward or backward.
129     */
130    public boolean isReverse() {
131        return this.reverse;
132    }
133
134    /**
135     * Determine if the given key is within this iterator's original range of keys.
136     *
137     * @return true if key is greater than or equal to this instance's minimum key (if an)
138     *  and strictly less than this instance's maximum key (if any)
139     */
140    public boolean inRange(byte[] key) {
141        return (this.minKey == null || ByteUtil.compare(key, this.minKey) >= 0)
142          && (this.maxKey == null || ByteUtil.compare(key, this.maxKey) < 0);
143    }
144
145    /**
146     * Set the next target key.
147     *
148     * <p>
149     * This is the key we will use to find the next element via {@link KVStore#getAtLeast}
150     * (or {@link KVStore#getAtMost} if this is a reverse iterator). Note that in the forward case, this is an
151     * inclusive lower bound on the next key, while in the reverse case it is an exclusive upper bound on the next key.
152     * </p>
153     *
154     * @param targetKey next lower bound (exclusive) if going forward, or upper bound (exclusive) if going backward,
155     *  or null to immediately end the iteration
156     */
157    public void setNextTarget(byte[] targetKey) {
158        this.nextKey = targetKey.clone();
159    }
160
161// Iterator
162
163    @Override
164    public boolean hasNext() {
165
166        // Already have next element?
167        if (this.currPair != null)
168            return true;
169
170        // No more elements?
171        if (this.finished)
172            return false;
173
174        // Find next element
175        final KVPair pair = this.reverse ?
176          this.kv.getAtMost(this.nextKey) : this.kv.getAtLeast(this.nextKey != null ? this.nextKey : ByteUtil.EMPTY);
177        if (pair == null) {
178            this.finished = true;
179            return false;
180        }
181
182        // Check if within range
183        if (!this.inRange(pair.getKey()))
184            return false;
185
186        // Save it (pre-fetch)
187        this.currPair = pair;
188        return true;
189    }
190
191    @Override
192    public KVPair next() {
193
194        // Check there is a next element
195        if (this.currPair == null && !this.hasNext())
196            throw new NoSuchElementException();
197
198        // Get next element
199        final KVPair pair = this.currPair;
200        final byte[] key = pair.getKey().clone();
201        this.removeKey = key;
202
203        // Set up next advance
204        this.nextKey = this.reverse ? key : ByteUtil.getNextKey(key);
205        this.currPair = null;
206
207        // Done
208        return pair;
209    }
210
211    @Override
212    public void remove() {
213        if (this.removeKey == null)
214            throw new IllegalStateException();
215        this.kv.remove(this.removeKey);
216        this.removeKey = null;
217    }
218
219// Internal methods
220
221    private static byte[] getMaxKey(byte[] prefix) {
222        if (prefix == null)
223            throw new IllegalArgumentException("null prefix");
224        try {
225            return ByteUtil.getKeyAfterPrefix(prefix);
226        } catch (IllegalArgumentException e) {
227            return null;
228        }
229    }
230}
