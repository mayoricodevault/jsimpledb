021/**
022 * Manager of read/write locks on {@code byte[]} key ranges that ensures isolation and serialization while allowing concurrent
023 * access by multiple threads to a single underlying {@code byte[]} key/value store.
024 *
025 * <p>
026 * This implementation is straightforward: read locks can overlap, but write locks may not, and all locks owned
027 * by the same owner remain in force until all are {@linkplain #release released} at the same time.
028 * </p>
029 *
030 * <p>
031 * Instances are configured with a monitor object which is used for all internal locking and inter-thread wait/notify
032 * handshaking (by default, this instance). A user-supplied monitor object may be provided via the constructor.
033 * </p>
034 *
035 * <p>
036 * Two timeout values are supported:
037 * <ul>
038 *  <li>The wait timeout (specified as a parameter to {@link #lock lock()}) limits how long a thread will wait
039 *      on a lock held by another thread before giving up</li>
040 *  <li>The {@linkplain #getHoldTimeout hold timeout} limits how long a thread may hold
041 *      on to a contested lock before being forced to release all its locks; after that, the
042 *      next call to {@link #lock lock} or {@link #release release} will fail</li>
043 * </ul>
044 * </p>
045 *
046 * <p>
047 * Note that if the hold timeout is set to zero (unlimited), then an application bug that leaks locks will result
048 * in those locks never being released.
049 * </p>
050 */
051public class LockManager {
052
053    private static final long TEN_YEARS_MILLIS = 10L * 365L * 24L * 60L * 60L * 1000L;
054
055    private final Object lockObject;
056
057    // Contains each owner's lock time, or null if hold timeout has expired.
058    // Invariant: if owner has any locks, then owner is a key in this map and value is not null.
059    // In the case owner's hold timeout expired and another owner forced it to release all of its locks, the expired
060    // owner will own no locks but still exist in this map (with a null value), until its next lock() or release().
061    private final HashMap<LockOwner, Long> lockTimes = new HashMap<>();
062
063    private final TreeSet<Lock> locksByMin = new TreeSet<>(Lock.MIN_COMPARATOR);            // locks ordered by minimum
064    private final TreeSet<Lock> locksByMax = new TreeSet<>(Lock.MAX_COMPARATOR);            // locks ordered by maximum
065    private final long nanoBasis = System.nanoTime();
066
067    private long holdTimeout;
068
069    /**
070     * Convenience constructor. Equivalent to <code>LockManager(null)</code>.
071     */
072    public LockManager() {
073        this(null);
074    }
075
076    /**
077     * Primary constructor.
078     *
079     * @param lockObject Java object used to synchronize field access and inter-thread wait/notify handshake,
080     *  or null to use this instance
081     */
082    public LockManager(Object lockObject) {
083        this.lockObject = lockObject != null ? lockObject : this;
084    }
085
086    /**
087     * Get the hold timeout configured for this instance.
088     *
089     * <p>
090     * The hold timeout limits how long a thread may hold on to a contested lock before being forced to release
091     * all of its locks; after that, the next call to {@link #lock lock} or {@link #release release} will fail.
092     * </p>
093     *
094     * @return hold timeout in milliseconds
095     */
096    public long getHoldTimeout() {
097        synchronized (this.lockObject) {
098            return this.holdTimeout;
099        }
100    }
101
102    /**
103     * Set the hold timeout for this instance. Default is zero (unlimited).
104     *
105     * @param holdTimeout how long a thread may hold a contestested lock before {@link LockResult#HOLD_TIMEOUT_EXPIRED}
106     *  will be returned by {@link #lock lock()} or {@link #release release()} in milliseconds, or zero for unlimited
107     * @throws IllegalArgumentException if {@code holdTimeout} is negative
108     */
109    public void setHoldTimeout(long holdTimeout) {
110        synchronized (this.lockObject) {
111            if (holdTimeout < 0)
112                throw new IllegalArgumentException("holdTimeout < 0");
113            this.holdTimeout = Math.min(holdTimeout, TEN_YEARS_MILLIS);             // limit to 10 years to avoid overflow
114        }
115    }
116
117    /**
118     * Acquire a lock on behalf of the specified owner.
119     *
120     * <p>
121     * This method will block for up to {@code waitTimeout} milliseconds if the lock is held by
122     * another thread, after which point {@link LockResult#WAIT_TIMEOUT_EXPIRED} is returned.
123     * The configured locking object will be used for inter-thread wait/notify handshaking.
124     * </p>
125     *
126     * <p>
127     * If {@code owner} already holds one or more locks, but the {@linkplain #getHoldTimeout hold timeout} has expired,
128     * then {@link LockResult#HOLD_TIMEOUT_EXPIRED} is returned and all of the other locks are will have already been
129     * automatically released.
130     * </p>
131     *
132     * <p>
133     * Once a lock is successfully acquired, it stays acquired until all locks are released together via {@link #release release()}.
134     * </p>
135     *
136     * @param owner lock owner
137     * @param minKey minimum key (inclusive), or null for no minimum
138     * @param maxKey maximum key (exclusive), or null for no maximum
139     * @param write true for a write lock, false for a read lock
140     * @param waitTimeout how long to wait before returning {@link LockResult#WAIT_TIMEOUT_EXPIRED}
141     *  in milliseconds, or zero for unlimited
142     * @return a {@link LockResult}
143     * @throws InterruptedException if the current thread is interrupted while waiting for the lock
144     * @throws IllegalArgumentException if {@code owner} or {@code range} is null
145     * @throws IllegalArgumentException if {@code minKey > maxKey}
146     * @throws IllegalArgumentException if {@code waitTimeout} is negative
147     */
148    public LockResult lock(LockOwner owner, byte[] minKey, byte[] maxKey, boolean write, long waitTimeout)
149      throws InterruptedException {
150        synchronized (this.lockObject) {
151
152            // Sanity check
153            if (owner == null)
154                throw new IllegalArgumentException("null owner");
155            if (waitTimeout < 0)
156                throw new IllegalArgumentException("waitTimeout < 0");
157            waitTimeout = Math.min(waitTimeout, TEN_YEARS_MILLIS);                  // limit to 10 years to avoid overflow
158
159            // Check hold timeout
160            final long lockerRemaining = this.checkHoldTimeout(owner);
161            if (lockerRemaining == -1)
162                return LockResult.HOLD_TIMEOUT_EXPIRED;
163
164            // Create lock
165            Lock lock = new Lock(owner, minKey, maxKey, write);
166
167            // Wait for lockability, until the first one of:
168            //  - Wait timeout
169            //  - Locker's hold timeout
170            //  - Lock owner's hold timeout
171            final LockChecker lockChecker = new LockChecker(lock);
172            if (!lockChecker.test()) {
173                long timeToWait = Math.min(waitTimeout, lockChecker.getTimeRemaining());
174                if (lockerRemaining != 0)
175                    timeToWait = Math.min(timeToWait, lockerRemaining);
176                if (!TimedWait.wait(this.lockObject, timeToWait, lockChecker))
177                    return LockResult.WAIT_TIMEOUT_EXPIRED;
178            }
179
180            // Check hold timeout again
181            if (this.checkHoldTimeout(owner) == -1)
182                return LockResult.HOLD_TIMEOUT_EXPIRED;
183
184            // Merge the lock with other locks it can merge with, removing those locks in the process
185            for (Lock that : lockChecker.getMergers()) {
186                final Lock mergedLock = lock.mergeWith(that);
187                if (mergedLock != null) {
188                    this.locksByMin.remove(that);
189                    this.locksByMax.remove(that);
190                    owner.locks.remove(that);
191                    lock = mergedLock;
192                }
193            }
194
195            // Add lock
196            this.locksByMin.add(lock);
197            this.locksByMax.add(lock);
198            owner.locks.add(lock);
199
200            // Set hold timeout (if not already set)
201            if (!this.lockTimes.containsKey(owner)) {
202                final long currentTime = System.nanoTime() - this.nanoBasis;
203                this.lockTimes.put(owner, currentTime);
204            } else
205                assert this.lockTimes.get(owner) != null;
206
207            // Done
208            return LockResult.SUCCESS;
209        }
210    }
211
212    /**
213     * Release all locks held by the specified owner.
214     *
215     * <p>
216     * If the owner's {@linkplain #getHoldTimeout hold timeout} has already expired, then all locks will have
217     * already been released and false is returned.
218     * </p>
219     *
220     * <p>
221     * Does nothing (and returns true) if {@code owner} does not own any locks.
222     * </p>
223     *
224     * @param owner lock owner
225     * @return true if successful, false if {@code owner}'s hold timeout expired
226     * @throws IllegalArgumentException if {@code owner} is null
227     */
228    public boolean release(LockOwner owner) {
229        synchronized (this.lockObject) {
230
231            // Sanity check
232            if (owner == null)
233                throw new IllegalArgumentException("null owner");
234
235            // Check if hold timeout has alread expired; in any case, remove lock time
236            if (this.lockTimes.containsKey(owner)) {
237                final Long lockTime = this.lockTimes.remove(owner);
238                if (lockTime == null)
239                    return false;
240            }
241
242            // Release all locks
243            this.doRelease(owner);
244
245            // Done
246            return true;
247        }
248    }
249
250    /**
251     * Check whether the {@linkplain #getHoldTimeout hold timeout} has expired for the given lock owner
252     * and if not return the amount of time remaining.
253     *
254     * <p>
255     * If the owner's hold timeout has expired, then {@code -1} is returned and any locks previously held by {@code owner}
256     * will have been automatically released.
257     * </p>
258     *
259     * @param owner lock owner
260     * @return milliseconds until {@code owner}'s hold timeout expires, zero if {@code owner} has no hold timeout
261     *  (e.g., nothing is locked or hold timeout disabled), or -1 if {@code owner}'s hold timeout has expired
262     * @throws IllegalArgumentException if {@code owner} is null
263     */
264    public long checkHoldTimeout(LockOwner owner) {
265        synchronized (this.lockObject) {
266            if (this.holdTimeout == 0)
267                return 0;
268            if (!this.lockTimes.containsKey(owner))
269                return 0;
270            final Long lockTime = this.lockTimes.get(owner);
271            if (lockTime == null)
272                return -1;
273            final long currentTime = System.nanoTime() - this.nanoBasis;
274            final long holdDeadline = lockTime + this.holdTimeout * 1000000L;
275            final long remaining = holdDeadline - currentTime;
276            if (remaining <= 0) {
277                this.lockTimes.put(owner, null);
278                this.doRelease(owner);
279                return -1;
280            }
281            return (remaining + 999999L) / 1000000L;
282        }
283    }
284
285    // Release all locks held by owner. Assumes synchronized already on this.lockObject.
286    private void doRelease(LockOwner owner) {
287        for (Lock lock : owner.locks) {
288            this.locksByMin.remove(lock);
289            this.locksByMax.remove(lock);
290        }
291        owner.locks.clear();
292        this.lockObject.notifyAll();
293    }
294
295    // Check whether we can lock, and fill the list of mergers and return zero if so. If not, return time remaining.
296    // Assumes synchronized already on this.lockObject.
297    private long checkLock(Lock lock, List<Lock> mergers) {
298
299        // Get lock's min & max
300        final byte[] lockMin = lock.getMin();
301        final byte[] lockMax = lock.getMax();
302
303    startOver:
304        while (true) {
305
306            // Get locks whose min is < lockMax
307            final NavigableSet<Lock> lhs = lockMax == null ? this.locksByMin :
308              this.locksByMin.headSet(Lock.getMinKey(lockMax, false), false);
309
310            // Get locks whose max is > lockMin
311            final NavigableSet<Lock> rhs = lockMin == null ? this.locksByMax : this.locksByMax.tailSet(
312              Lock.getMaxKey(ByteUtil.getNextKey(lockMin), false), true);
313
314            // Find overlapping locks and check for conflicts
315            final HashSet<Lock> overlaps = new HashSet<>();
316            for (Lock other : lhs) {
317
318                // Does this lock overlap?
319                if (!rhs.contains(other))
320                    continue;
321
322                // Do this lock & other lock conflict?
323                if (lock.conflictsWith(other)) {
324
325                    // See if other lock's owner's hold timeout has expired
326                    assert this.lockTimes.containsKey(other.owner);
327                    final long remaining = this.checkHoldTimeout(other.owner);
328                    if (remaining == -1)
329                        continue startOver;
330
331                    // Return time remaining until conflicting owner's hold timeout
332                    return Math.max(remaining, 1);
333                }
334
335                // Add overlap
336                overlaps.add(other);
337            }
338
339            //System.out.println(Thread.currentThread().getName() + ": LockChecker: BEFORE: lock = " + lock + "\n"
340            //+ "  lockByMin = " + this.locksByMin + "\n"
341            //+ "  lockByMax = " + this.locksByMax + "\n"
342            //+ "   overlaps = " + overlaps + "\n");
343
344            // Find overlaps we can merge with
345            for (Lock other : overlaps) {
346                if (lock.mergeWith(other) != null)
347                    mergers.add(other);
348            }
349
350            //System.out.println(Thread.currentThread().getName() + ": LockChecker: AFTER: lock = " + lock + "\n"
351            //+ "  lockByMin = " + this.locksByMin + "\n"
352            //+ "  lockByMax = " + this.locksByMax + "\n"
353            //+ "    mergers = " + mergers + "\n");
354
355            // Done
356            return 0;
357        }
358    }

362    /**
363     * Possible return values from {@link LockManager#lock LockManager.lock()}.
364     */
365    public enum LockResult {
366
367        /**
368         * The lock was successfully acquired.
369         */
370        SUCCESS,
371
372        /**
373         * The timeout expired while waiting to acquire the lock.
374         */
375        WAIT_TIMEOUT_EXPIRED,
376
377        /**
378         * The owner's hold timeout expired.
379         */
380        HOLD_TIMEOUT_EXPIRED;
381    }
382
383// LockChecker predicate
384
385    private class LockChecker implements Predicate {
386
387        private final Lock lock;
388        private final ArrayList<Lock> mergers = new ArrayList<>();
389
390        private long timeRemaining;
391
392        LockChecker(Lock lock) {
393            this.lock = lock;
394        }
395
396        public List<Lock> getMergers() {
397            return this.mergers;
398        }
399
400        public long getTimeRemaining() {
401            return this.timeRemaining;
402        }
403
404        @Override
405        public boolean test() {
406
407            // Reset state
408            this.mergers.clear();
409
410            // See if we can lock
411            this.timeRemaining = LockManager.this.checkLock(this.lock, this.mergers);
412            return this.timeRemaining == 0;
413        }
414    }
