021/**
022 * A fixed set of {@link KeyRange} instances that can be treated as a unified whole, in particular as a {@link KeyFilter}.
023 *
024 * <p>
025 * Instances are immutable.
026 * </p>
027 *
028 * @see KeyRange
029 */
030public class KeyRanges implements KeyFilter {
031
032    /**
033     * The empty instance containing zero ranges.
034     */
035    public static final KeyRanges EMPTY = new KeyRanges(Collections.<KeyRange>emptyList());
036
037    /**
038     * The "full" instance containing a single {@link KeyRange} that contains all keys.
039     */
040    public static final KeyRanges FULL = new KeyRanges(Arrays.asList(KeyRange.FULL));
041
042    private final List<KeyRange> ranges;
043
044    private volatile KeyRange lastContainingKeyRange;           // used for optimization
045
046// Constructors
047
048    /**
049     * Constructor.
050     *
051     * <p>
052     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
053     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
054     * will normalize them.
055     * </p>
056     *
057     * @param ranges individual key ranges
058     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
059     */
060    public KeyRanges(Iterable<? extends KeyRange> ranges) {
061        if (ranges == null)
062            throw new IllegalArgumentException("null ranges");
063        this.ranges = KeyRanges.minimize(Lists.<KeyRange>newArrayList(ranges));
064    }
065
066    /**
067     * Constructor.
068     *
069     * <p>
070     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
071     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
072     * will normalize them.
073     * </p>
074     *
075     * @param ranges individual key ranges
076     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
077     */
078    public KeyRanges(KeyRange... ranges) {
079        this(ranges != null ? Arrays.asList(ranges) : null);
080    }
081
082    /**
083     * Constructor for an instance containing a single range.
084     *
085     * @param range single range
086     * @throws IllegalArgumentException if {@code range} is null
087     */
088    public KeyRanges(KeyRange range) {
089        this(Collections.singletonList(range));
090    }
091
092    /**
093     * Constructor for an instance containing a single range.
094     *
095     * @param min minimum key (inclusive), or null for no minimum
096     * @param max maximum key (exclusive), or null for no maximum
097     * @throws IllegalArgumentException if {@code min > max}
098     */
099    public KeyRanges(byte[] min, byte[] max) {
100        this(Collections.singletonList(new KeyRange(min, max)));
101    }
102
103    /**
104     * Construct an instance containing a single range corresponding to all keys with the given prefix.
105     *
106     * @param prefix prefix of all keys in the range
107     * @throws IllegalArgumentException if {@code prefix} is null
108     */
109    public static KeyRanges forPrefix(byte[] prefix) {
110        return new KeyRanges(KeyRange.forPrefix(prefix));
111    }

113// Instance methods
114
115    /**
116     * Get {@link KeyRange}s underlying with this instance.
117     *
118     * <p>
119     * The returned {@link KeyRange}s will be listed in order.
120     * </p>
121     *
122     * @return minimal, unmodifiable list of {@link KeyRange}s sorted by key range
123     */
124    public List<KeyRange> getKeyRanges() {
125        return Collections.unmodifiableList(this.ranges);
126    }
127
128    /**
129     * Determine whether this instance is empty, i.e., contains no keys.
130     */
131    public boolean isEmpty() {
132        return this.ranges.isEmpty();
133    }
134
135    /**
136     * Determine whether this instance is "full", i.e., contains all keys.
137     */
138    public boolean isFull() {
139        return this.ranges.size() == 1 && this.ranges.get(0).isFull();
140    }
141
142    /**
143     * Get the minimum key contained by this instance (inclusive).
144     *
145     * @return minimum key contained by this instance (inclusive),
146     *  or null if there is no lower bound, or this instance {@link #isEmpty}
147     */
148    public byte[] getMin() {
149        return !this.ranges.isEmpty() ? this.ranges.get(0).getMin() : null;
150    }
151
152    /**
153     * Get the maximum key contained by this instance (exclusive).
154     *
155     * @return maximum key contained by this instance (exclusive),
156     *  or null if there is no upper bound, or this instance {@link #isEmpty}
157     */
158    public byte[] getMax() {
159        final int numRanges = this.ranges.size();
160        return numRanges > 0 ? this.ranges.get(numRanges - 1).getMax() : null;
161    }
162
163    /**
164     * Create a new instance all of whose individual {@link KeyRange}s are have the given byte sequence prepended.
165     *
166     * @throws IllegalArgumentException if {@code prefix} is null
167     */
168    public KeyRanges prefixedBy(final byte[] prefix) {
169        return new KeyRanges(Lists.transform(this.ranges, new Function<KeyRange, KeyRange>() {
170            @Override
171            public KeyRange apply(KeyRange keyRange) {
172                return keyRange.prefixedBy(prefix);
173            }
174        }));
175    }
176
177    /**
178     * Create the inverse of this instance. The inverse contains all keys not contained by this instance.
179     *
180     * @return the inverse of this instance
181     */
182    public KeyRanges inverse() {
183        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
184        if (this.ranges.isEmpty())
185            return KeyRanges.FULL;
186        final KeyRange first = this.ranges.get(0);
187        int i = 0;
188        byte[] lastMax;
189        if (first.getMin() == null) {
190            if ((lastMax = first.getMax()) == null) {
191                assert this.ranges.size() == 1;
192                return KeyRanges.EMPTY;
193            }
194            i++;
195        } else
196            lastMax = null;
197        while (true) {
198            if (i == this.ranges.size()) {
199                list.add(new KeyRange(lastMax, null));
200                break;
201            }
202            final KeyRange next = this.ranges.get(i++);
203            assert next.getMin() != null;
204            list.add(new KeyRange(lastMax, next.getMin()));
205            if ((lastMax = next.getMax()) == null) {
206                assert i == this.ranges.size();
207                break;
208            }
209        }
210        return new KeyRanges(list);
211    }
212
213    /**
214     * Determine whether this instance contains the given {@link KeyRanges}, i.e., all keys contained by
215     * the given {@link KeyRanges} are also contained by this instance.
216     *
217     * @param ranges other instance to test
218     * @return true if this instance contains {@code ranges}, otherwise false
219     * @throws IllegalArgumentException if {@code ranges} is null
220     */
221    public boolean contains(KeyRanges ranges) {
222        if (ranges == null)
223            throw new IllegalArgumentException("null ranges");
224        return ranges.equals(this.intersection(ranges));
225    }
226
227    /**
228     * Find the contiguous {@link KeyRange}(s) within this instance containing, or adjacent to, the given key.
229     *
230     * <p>
231     * This method returns an array of length two: if this instance contains {@code key} then both elements are the
232     * same Java object, namely, the {@link KeyRange} that contains {@code key}; otherwise, the first element is
233     * the nearest {@link KeyRange} to the left of {@code key}, or null if none exists, and the second element is
234     * the {@link KeyRange} to the right of {@code key}, or null if none exists. Note if this instance is empty
235     * then <code>{ null, null }</code> is returned.
236     * </p>
237     *
238     * @param key key to find
239     * @return array with the containing {@link KeyRange} or nearest neighbor to the left (or null) and
240     *  the containing {@link KeyRange} or nearest neighbor to the right (or null)
241     * @throws IllegalArgumentException if {@code key} is null
242     */
243    public KeyRange[] findKey(byte[] key) {
244
245        // Sanity check
246        if (key == null)
247            throw new IllegalArgumentException("null key");
248
249        // Optimization: assume previous success is likely to repeat
250        final KeyRange temp = this.lastContainingKeyRange;
251        if (temp != null) {
252            if (temp.contains(key))
253                return new KeyRange[] { temp, temp };
254            this.lastContainingKeyRange = null;
255        }
256
257        // Search for matching range
258        final int i = ~Collections.binarySearch(this.ranges, new KeyRange(key, key), KeyRange.SORT_BY_MIN);
259        assert i >= 0;                                                  // this.ranges should never contain new KeyRange(key, key)
260        KeyRange left = null;
261        if (i > 0) {
262            if ((left = this.ranges.get(i - 1)).contains(key)) {
263                this.lastContainingKeyRange = left;
264                return new KeyRange[] { left, left };
265            }
266        }
267        KeyRange right = null;
268        if (i < this.ranges.size()) {
269            if ((right = this.ranges.get(i)).contains(key)) {
270                this.lastContainingKeyRange = right;
271                return new KeyRange[] { right, right };
272            }
273        }
274
275        // Not contained
276        return new KeyRange[] { left, right };
277    }

279    /**
280     * Create an instance that represents the union of this and the provided instance(s).
281     *
282     * @param others other instances
283     * @return the union of this instance and {@code others}
284     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
285     */
286    public KeyRanges union(KeyRanges... others) {
287        if (others == null)
288            throw new IllegalArgumentException("null others");
289        if (others.length == 0)
290            return this;
291        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
292        list.addAll(this.ranges);
293        for (KeyRanges other : others) {
294            if (other == null)
295                throw new IllegalArgumentException("null other");
296            list.addAll(other.ranges);
297        }
298        return new KeyRanges(list);
299    }
300
301    /**
302     * Create an instance that represents the intersection of this and the provided instance(s).
303     *
304     * @param others other instances
305     * @return the intersection of this instance and {@code others}
306     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
307     */
308    public KeyRanges intersection(KeyRanges... others) {
309        if (others == null)
310            throw new IllegalArgumentException("null others");
311        if (others.length == 0)
312            return this;
313        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
314        list.addAll(this.inverse().ranges);
315        for (KeyRanges other : others) {
316            if (other == null)
317                throw new IllegalArgumentException("null other");
318            list.addAll(other.inverse().ranges);
319        }
320        return new KeyRanges(list).inverse();
321    }

323// KeyFilter

325    @Override
326    public boolean contains(byte[] key) {
327        final KeyRange[] pair = this.findKey(key);
328        return pair[0] == pair[1] && pair[0] != null;
329    }
330
331    @Override
332    public byte[] seekHigher(byte[] key) {
333        final KeyRange[] pair = this.findKey(key);
334        if (pair[0] == pair[1])
335            return pair[0] != null ? key : null;
336        return pair[1] != null ? pair[1].getMin() : null;
337    }
338
339    @Override
340    public byte[] seekLower(byte[] key) {
341        if (key == null)
342            throw new IllegalArgumentException("null key");
343        if (key.length == 0) {
344            if (this.ranges.isEmpty())
345                return null;
346            final byte[] lastMax = this.ranges.get(this.ranges.size() - 1).getMax();
347            return lastMax != null ? lastMax : ByteUtil.EMPTY;
348        }
349        final KeyRange[] pair = this.findKey(key);
350        if (pair[0] == pair[1])
351            return pair[0] != null ? key : null;
352        return pair[0] != null ? pair[0].getMax() : null;
353    }

358    public boolean equals(Object obj) {
359        if (obj == this)
360            return true;
361        if (obj == null || obj.getClass() != this.getClass())
362            return false;
363        final KeyRanges that = (KeyRanges)obj;
364        return this.ranges.equals(that.ranges);
365    }

367    @Override
368    public int hashCode() {

370    }

372    @Override
373    public String toString() {
374        final StringBuilder buf = new StringBuilder();
375        buf.append(this.getClass().getSimpleName()).append("[");
376        boolean first = true;
377        for (KeyRange range : this.ranges) {
378            if (first)
379                first = false;
380            else
381                buf.append(",");
382            buf.append(range);
383        }
384        buf.append("]");
385        return buf.toString();
386    }

388// Internal methods

390    // Return a "minimal" list with these properties:
391    //  - Sorted according to KeyRange.SORT_BY_MIN
392    //  - No overlapping ranges
393    //  - Adjacent ranges consolidated into a single range
394    private static ArrayList<KeyRange> minimize(List<KeyRange> ranges) {
395
396        // For consistency, replace all lower bounds of [] with null (same thing)
397        final ArrayList<KeyRange> sortedRanges = new ArrayList<>(ranges);
398        for (int i = 0; i < sortedRanges.size(); i++) {
399            final KeyRange range = sortedRanges.get(i);
400            final byte[] min = range.getMin();
401            if (min != null && min.length == 0)
402                sortedRanges.set(i, new KeyRange(null, range.getMax()));
403        }
404
405        // Remove empty ranges
406        for (Iterator<KeyRange> i = sortedRanges.iterator(); i.hasNext(); ) {
407            final KeyRange range = i.next();
408            final byte[] min = range.getMin();
409            final byte[] max = range.getMax();
410            if (max == null)                                        // { x, null } never empty
411                continue;
412            if ((min == null && max.length == 0)                    // { null, [] } is empty (special case)
413              || (min != null && ByteUtil.compare(min, max) == 0))  // { x, x } is empty
414                i.remove();
415        }
416
417        // Sort remaining ranges by min, then max
418        Collections.sort(sortedRanges, KeyRange.SORT_BY_MIN);
419
420        // Consolidate
421        final ArrayList<KeyRange> list = new ArrayList<>(ranges.size());
422        KeyRange prev = null;
423        for (KeyRange range : sortedRanges) {
424
425            // Sanity check range
426            if (range == null)
427                throw new IllegalArgumentException("null range");
428
429            // Handle first in list
430            if (prev == null) {
431                prev = range;
432                continue;
433            }
434
435            // Compare to previous range
436            final int diff1 = KeyRange.compare(range.getMin(), KeyRange.MIN, prev.getMin(), KeyRange.MIN);
437            assert diff1 >= 0;
438            if (diff1 == 0) {                           // range contains prev -> discard prev
439                assert range.contains(prev);
440                prev = range;
441                continue;
442            }
443            final int diff2 = KeyRange.compare(range.getMin(), KeyRange.MIN, prev.getMax(), KeyRange.MAX);
444            if (diff2 <= 0) {                           // prev and range overlap -> take their union
445                final byte[] max = KeyRange.compare(range.getMax(), KeyRange.MAX, prev.getMax(), KeyRange.MAX) > 0 ?
446                  range.getMax() : prev.getMax();
447                prev = new KeyRange(prev.getMin(), max);
448                continue;
449            }
450
451            // OK add it
452            list.add(prev);                             // prev and range don't overlap -> accept prev
453            prev = range;
454            continue;
455        }
456        if (prev != null)
457            list.add(prev);
458        list.trimToSize();
459        return list;
460    }
