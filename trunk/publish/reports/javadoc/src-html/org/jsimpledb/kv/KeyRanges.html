021/**
022 * A fixed set of {@link KeyRange} instances that can be treated as a unified whole, in particular as a {@link KeyFilter}.
023 *
024 * <p>
025 * Instances are immutable.
026 * </p>
027 *
028 * @see KeyRange
029 */
030public class KeyRanges implements KeyFilter {
031
032    /**
033     * The empty instance containing zero ranges.
034     */
035    public static final KeyRanges EMPTY = new KeyRanges(Collections.<KeyRange>emptyList());
036
037    /**
038     * The "full" instance containing a single {@link KeyRange} that contains all keys.
039     */
040    public static final KeyRanges FULL = new KeyRanges(Arrays.asList(KeyRange.FULL));
041
042    private final List<KeyRange> ranges;
043
044    private volatile KeyRange lastContainingKeyRange;           // used for optimization
045
046// Constructors
047
048    /**
049     * Constructor.
050     *
051     * <p>
052     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
053     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
054     * will normalize them.
055     * </p>
056     *
057     * @param ranges individual key ranges
058     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
059     */
060    public KeyRanges(Iterable<? extends KeyRange> ranges) {
061        if (ranges == null)
062            throw new IllegalArgumentException("null ranges");
063        this.ranges = KeyRanges.minimize(Lists.<KeyRange>newArrayList(ranges));
064    }
065
066    /**
067     * Constructor.
068     *
069     * <p>
070     * Creates an instance that contains all keys contained by any of the {@link KeyRange}s in {@code ranges}.
071     * The given {@code ranges} may be adjacent, overlap, and/or be listed in any order; this constructor
072     * will normalize them.
073     * </p>
074     *
075     * @param ranges individual key ranges
076     * @throws IllegalArgumentException if {@code ranges} or any {@link KeyRange} therein is null
077     */
078    public KeyRanges(KeyRange... ranges) {
079        this(ranges != null ? Arrays.asList(ranges) : null);
080    }

082    /**
083     * Constructor for an instance containing a single range.
084     *
085     * @param range single range
086     * @throws IllegalArgumentException if {@code range} is null
087     */
088    public KeyRanges(KeyRange range) {
089        this(Collections.singletonList(range));
090    }

092    /**
093     * Constructor for an instance containing a single range.
094     *
095     * @param min minimum key (inclusive); must not be null
096     * @param max maximum key (exclusive), or null for no maximum
097     * @throws IllegalArgumentException if {@code min > max}
098     */
099    public KeyRanges(byte[] min, byte[] max) {
100        this(Collections.singletonList(new KeyRange(min, max)));
101    }

103    /**
104     * Construct an instance containing a single range corresponding to all keys with the given prefix.
105     *
106     * @param prefix prefix of all keys in the range
107     * @return instance containing all keys prefixed by {@code prefix}
108     * @throws IllegalArgumentException if {@code prefix} is null
109     */
110    public static KeyRanges forPrefix(byte[] prefix) {
111        return new KeyRanges(KeyRange.forPrefix(prefix));
112    }

114// Instance methods

116    /**
117     * Get the {@link KeyRange}s underlying with this instance as a list.
118     *
119     * <p>
120     * The returned {@link KeyRange}s will be listed in order.
121     * </p>
122     *
123     * @return minimal, unmodifiable list of {@link KeyRange}s sorted by key range
124     */
125    public List<KeyRange> asList() {
126        return Collections.unmodifiableList(this.ranges);
127    }
128
129    /**
130     * Determine whether this instance is empty, i.e., contains no keys.
131     *
132     * @return true if this instance is empty
133     */
134    public boolean isEmpty() {
135        return this.ranges.isEmpty();
136    }
137
138    /**
139     * Determine whether this instance is "full", i.e., contains all keys.
140     *
141     * @return true if this instance is full
142     */
143    public boolean isFull() {
144        return this.ranges.size() == 1 && this.ranges.get(0).isFull();
145    }
146
147    /**
148     * Get the minimum key contained by this instance (inclusive).
149     *
150     * @return minimum key contained by this instance (inclusive), or null if this instance {@link #isEmpty}
151     */
152    public byte[] getMin() {
153        return !this.ranges.isEmpty() ? this.ranges.get(0).getMin() : null;
154    }
155
156    /**
157     * Get the maximum key contained by this instance (exclusive).
158     *
159     * @return maximum key contained by this instance (exclusive),
160     *  or null if there is no upper bound, or this instance {@link #isEmpty}
161     */
162    public byte[] getMax() {
163        final int numRanges = this.ranges.size();
164        return numRanges > 0 ? this.ranges.get(numRanges - 1).getMax() : null;
165    }
166
167    /**
168     * Create a new instance from this one, with each {@link KeyRange} prefixed by the given byte sequence.
169     *
170     * @param prefix prefix to apply to this instance
171     * @return prefixed instance
172     * @throws IllegalArgumentException if {@code prefix} is null
173     */
174    public KeyRanges prefixedBy(final byte[] prefix) {
175        return new KeyRanges(Lists.transform(this.ranges, new Function<KeyRange, KeyRange>() {
176            @Override
177            public KeyRange apply(KeyRange keyRange) {
178                return keyRange.prefixedBy(prefix);
179            }
180        }));
181    }
182
183    /**
184     * Create the inverse of this instance. The inverse contains all keys not contained by this instance.
185     *
186     * @return the inverse of this instance
187     */
188    public KeyRanges inverse() {
189        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
190        if (this.ranges.isEmpty())
191            return KeyRanges.FULL;
192        final KeyRange first = this.ranges.get(0);
193        int i = 0;
194        byte[] lastMax;
195        if (first.getMin().length == 0) {
196            if ((lastMax = first.getMax()) == null) {
197                assert this.ranges.size() == 1;
198                return KeyRanges.EMPTY;
199            }
200            i++;
201        } else
202            lastMax = ByteUtil.EMPTY;
203        while (true) {
204            if (i == this.ranges.size()) {
205                list.add(new KeyRange(lastMax, null));
206                break;
207            }
208            final KeyRange next = this.ranges.get(i++);
209            list.add(new KeyRange(lastMax, next.getMin()));
210            if ((lastMax = next.getMax()) == null) {
211                assert i == this.ranges.size();
212                break;
213            }
214        }
215        return new KeyRanges(list);
216    }
217
218    /**
219     * Determine whether this instance contains the given {@link KeyRanges}, i.e., all keys contained by
220     * the given {@link KeyRanges} are also contained by this instance.
221     *
222     * @param ranges other instance to test
223     * @return true if this instance contains {@code ranges}, otherwise false
224     * @throws IllegalArgumentException if {@code ranges} is null
225     */
226    public boolean contains(KeyRanges ranges) {
227        if (ranges == null)
228            throw new IllegalArgumentException("null ranges");
229        return ranges.equals(this.intersection(ranges));
230    }
231
232    /**
233     * Determine whether this instance contains the given {@link KeyRange}, i.e., all keys contained by
234     * the given {@link KeyRange} are also contained by this instance.
235     *
236     * @param range key range to test
237     * @return true if this instance contains {@code range}, otherwise false
238     * @throws IllegalArgumentException if {@code range} is null
239     */
240    public boolean contains(KeyRange range) {
241        if (range == null)
242            throw new IllegalArgumentException("null range");
243        final KeyRange[] pair = this.findKey(range.getMin());
244        if (pair[0] != pair[1])
245            return false;
246        final KeyRange container = pair[0];
247        if (container == null)
248            return false;
249        return container.contains(range);
250    }
251
252    /**
253     * Find the contiguous {@link KeyRange}(s) within this instance containing, or adjacent to, the given key.
254     *
255     * <p>
256     * This method returns an array of length two: if this instance contains {@code key} then both elements are the
257     * same Java object, namely, the {@link KeyRange} that contains {@code key}; otherwise, the first element is
258     * the nearest {@link KeyRange} to the left of {@code key}, or null if none exists, and the second element is
259     * the {@link KeyRange} to the right of {@code key}, or null if none exists. Note if this instance is empty
260     * then <code>{ null, null }</code> is returned.
261     * </p>
262     *
263     * @param key key to find
264     * @return array with the containing {@link KeyRange} or nearest neighbor to the left (or null) and
265     *  the containing {@link KeyRange} or nearest neighbor to the right (or null)
266     * @throws IllegalArgumentException if {@code key} is null
267     */
268    public KeyRange[] findKey(byte[] key) {
269
270        // Sanity check
271        if (key == null)
272            throw new IllegalArgumentException("null key");
273
274        // Optimization: assume previous success is likely to repeat
275        final KeyRange temp = this.lastContainingKeyRange;
276        if (temp != null) {
277            if (temp.contains(key))
278                return new KeyRange[] { temp, temp };
279            this.lastContainingKeyRange = null;
280        }
281
282        // Search for matching range
283        final int i = ~Collections.binarySearch(this.ranges, new KeyRange(key, key), KeyRange.SORT_BY_MIN);
284        assert i >= 0;                                                  // this.ranges should never contain new KeyRange(key, key)
285        KeyRange left = null;
286        if (i > 0) {
287            if ((left = this.ranges.get(i - 1)).contains(key)) {
288                this.lastContainingKeyRange = left;
289                return new KeyRange[] { left, left };
290            }
291        }
292        KeyRange right = null;
293        if (i < this.ranges.size()) {
294            if ((right = this.ranges.get(i)).contains(key)) {
295                this.lastContainingKeyRange = right;
296                return new KeyRange[] { right, right };
297            }
298        }
299
300        // Not contained
301        return new KeyRange[] { left, right };
302    }

304    /**
305     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} added.
306     *
307     * @param range range to add
308     * @return this instance with {@code range} added
309     * @throws IllegalArgumentException if {@code range} is null
310     */
311    public KeyRanges add(KeyRange range) {
312        if (range == null)
313            throw new IllegalArgumentException("null range");
314        if (range.isEmpty())
315            return this;
316        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + 1);
317        list.addAll(this.ranges);
318        list.add(range);
319        return new KeyRanges(list);
320    }
321
322    /**
323     * Return a new {@link KeyRanges} instance equal to this instance with all keys in the given {@link KeyRange} removed.
324     *
325     * @param range range to remove
326     * @return this instance with {@code range} removed
327     * @throws IllegalArgumentException if {@code range} is null
328     */
329    public KeyRanges remove(KeyRange range) {
330        if (range == null)
331            throw new IllegalArgumentException("null range");
332        if (range.isEmpty())
333            return this;
334        return this.intersection(new KeyRanges(range).inverse());
335    }
336
337    /**
338     * Create an instance that represents the union of this and the provided instance(s).
339     *
340     * @param others other instances
341     * @return the union of this instance and {@code others}
342     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
343     */
344    public KeyRanges union(KeyRanges... others) {
345        if (others == null)
346            throw new IllegalArgumentException("null others");
347        if (others.length == 0)
348            return this;
349        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
350        list.addAll(this.ranges);
351        for (KeyRanges other : others) {
352            if (other == null)
353                throw new IllegalArgumentException("null other");
354            list.addAll(other.ranges);
355        }
356        return new KeyRanges(list);
357    }
358
359    /**
360     * Create an instance that represents the intersection of this and the provided instance(s).
361     *
362     * @param others other instances
363     * @return the intersection of this instance and {@code others}
364     * @throws IllegalArgumentException if {@code others} or any element in {@code others} is null
365     */
366    public KeyRanges intersection(KeyRanges... others) {
367        if (others == null)
368            throw new IllegalArgumentException("null others");
369        if (others.length == 0)
370            return this;
371        final ArrayList<KeyRange> list = new ArrayList<>(this.ranges.size() + others.length);
372        list.addAll(this.inverse().ranges);
373        for (KeyRanges other : others) {
374            if (other == null)
375                throw new IllegalArgumentException("null other");
376            list.addAll(other.inverse().ranges);
377        }
378        return new KeyRanges(list).inverse();
379    }

381// KeyFilter

383    @Override
384    public boolean contains(byte[] key) {
385        final KeyRange[] pair = this.findKey(key);
386        return pair[0] == pair[1] && pair[0] != null;
387    }
388
389    @Override
390    public byte[] seekHigher(byte[] key) {
391        final KeyRange[] pair = this.findKey(key);
392        if (pair[0] == pair[1])
393            return pair[0] != null ? key : null;
394        return pair[1] != null ? pair[1].getMin() : null;
395    }
396
397    @Override
398    public byte[] seekLower(byte[] key) {
399        if (key == null)
400            throw new IllegalArgumentException("null key");
401        if (key.length == 0) {
402            if (this.ranges.isEmpty())
403                return null;
404            final byte[] lastMax = this.ranges.get(this.ranges.size() - 1).getMax();
405            return lastMax != null ? lastMax : ByteUtil.EMPTY;
406        }
407        final KeyRange[] pair = this.findKey(key);
408        if (pair[0] == pair[1])
409            return pair[0] != null ? key : null;
410        return pair[0] != null ? pair[0].getMax() : null;
411    }

430    @Override
431    public String toString() {
432        final StringBuilder buf = new StringBuilder();
433        buf.append(this.getClass().getSimpleName()).append("[");
434        boolean first = true;
435        for (KeyRange range : this.ranges) {
436            if (first)
437                first = false;
438            else
439                buf.append(",");
440            buf.append(range);
441        }
442        buf.append("]");
443        return buf.toString();
444    }

446// Internal methods

448    // Return a "minimal" list with these properties:
449    //  - Sorted according to KeyRange.SORT_BY_MIN
450    //  - No overlapping ranges
451    //  - Adjacent ranges consolidated into a single range
452    private static ArrayList<KeyRange> minimize(List<KeyRange> ranges) {
453
454        // Create array
455        final ArrayList<KeyRange> sortedRanges = new ArrayList<>(ranges);
456
457        // Remove empty ranges
458        for (Iterator<KeyRange> i = sortedRanges.iterator(); i.hasNext(); ) {
459            if (i.next().isEmpty())
460                i.remove();
461        }
462
463        // Sort remaining ranges by min, then max
464        Collections.sort(sortedRanges, KeyRange.SORT_BY_MIN);
465
466        // Consolidate
467        final ArrayList<KeyRange> list = new ArrayList<>(ranges.size());
468        KeyRange prev = null;
469        for (KeyRange range : sortedRanges) {
470
471            // Sanity check range
472            if (range == null)
473                throw new IllegalArgumentException("null range");
474
475            // Handle first in list
476            if (prev == null) {
477                prev = range;
478                continue;
479            }
480
481            // Compare to previous range
482            final int diff1 = KeyRange.compare(range.getMin(), prev.getMin());
483            assert diff1 >= 0;
484            if (diff1 == 0) {                           // range contains prev -> discard prev
485                assert range.contains(prev);
486                prev = range;
487                continue;
488            }
489            final int diff2 = KeyRange.compare(range.getMin(), prev.getMax());
490            if (diff2 <= 0) {                           // prev and range overlap -> take their union
491                final byte[] max = KeyRange.compare(range.getMax(), prev.getMax()) > 0 ? range.getMax() : prev.getMax();
492                prev = new KeyRange(prev.getMin(), max);
493                continue;
494            }
495
496            // OK add it
497            list.add(prev);                             // prev and range don't overlap -> accept prev
498            prev = range;
499            continue;
500        }
501        if (prev != null)
502            list.add(prev);
503        list.trimToSize();
504        return list;
505    }
