008package org.jsimpledb.kv.sql;
009
010import com.mysql.jdbc.MysqlErrorNumbers;
011
012import java.sql.Connection;
013import java.sql.SQLException;
014import java.sql.Statement;
015
016import org.jsimpledb.kv.KVTransactionException;
017import org.jsimpledb.kv.RetryTransactionException;
018
019/**
020 * MySQL variant of {@link SQLKVDatabase}.
021 *
022 * <p>
023 * An idempotent script for initializing a MySQL database is available at
024 * {@code classpath:org/jsimpledb/kv/sql/createTable-mysql.sql}
025 * (see also {@link org.dellroad.stuff.schema.UpdatingDataSource}).
026 * </p>
027 *
028 */
029public class MySQLKVDatabase extends SQLKVDatabase {
030
031    @Override
032    protected Connection createConnection() throws SQLException {
033        final Connection connection = super.createConnection();
034        final Statement statement = connection.createStatement();
035        statement.execute("SET SESSION sql_mode = 'TRADITIONAL'");      // force error if key or value is too long
036        return connection;
037    }
038
039    /**
040     * Encloses the given {@code name} in backticks.
041     */
042    @Override
043    public String quote(String name) {
044        return "`" + name + "`";
045    }
046
047    /**
048     * Appends {@code LIMIT 1} to the statement.
049     */
050    @Override
051    public String limitSingleRow(String sql) {
052        return sql + " LIMIT 1";
053    }
054
055    @Override
056    public KVTransactionException wrapException(SQLKVTransaction tx, SQLException e) {
057        switch (e.getErrorCode()) {
058        case MysqlErrorNumbers.ER_LOCK_WAIT_TIMEOUT:
059            return new RetryTransactionException(tx, e);
060        case MysqlErrorNumbers.ER_LOCK_DEADLOCK:
061            return new RetryTransactionException(tx, e);
062        default:
063            return super.wrapException(tx, e);
064        }
065    }
066}
