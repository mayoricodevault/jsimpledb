008package org.jsimpledb.kv.util;
009
010import org.jsimpledb.kv.KVStore;
011import org.jsimpledb.util.ByteReader;
012import org.jsimpledb.util.ByteUtil;
013import org.jsimpledb.util.ByteWriter;
014
015/**
016 * {@link KVStore} support superclass that implements the {@link #encodeCounter encodeCounter()},
017 * {@link #decodeCounter decodeCounter()}, and {@link #adjustCounter adjustCounter()} methods.
018 * This class uses a simple big-endian encoding and of course provides no lock-free behavior.
019 */
020public abstract class AbstractCountingKVStore implements KVStore {
021
022    protected AbstractCountingKVStore() {
023    }
024
025    @Override
026    public byte[] encodeCounter(long value) {
027        final ByteWriter writer = new ByteWriter(8);
028        ByteUtil.writeLong(writer, value);
029        return writer.getBytes();
030    }
031
032    @Override
033    public long decodeCounter(byte[] value) {
034        if (value.length != 8)
035            throw new IllegalArgumentException("invalid encoded counter value: length = " + value.length + " != 8");
036        return ByteUtil.readLong(new ByteReader(value));
037    }
038
039    @Override
040    public void adjustCounter(byte[] key, long amount) {
041        if (key == null)
042            throw new NullPointerException("null key");
043        byte[] previous = this.get(key);
044        if (previous == null)
045            previous = new byte[8];
046        this.put(key, this.encodeCounter(this.decodeCounter(previous) + amount));
047    }
048}
