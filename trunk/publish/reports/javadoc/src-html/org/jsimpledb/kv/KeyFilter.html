008package org.jsimpledb.kv;
009
010/**
011 * A predicate for filtering {@code byte[]} keys that is also capable of efficiently jumping over
012 * contiguous regions of uncontained keys.
013 *
014 * <p>
015 * Instances assume {@code byte[]} keys are ordered in unsigned lexicographical order.
016 * </p>
017 */
018public interface KeyFilter {
019
020    /**
021     * Determine whether this instance contains the given key.
022     *
023     * @param key key to test
024     * @return true if {@code key} is contained by this instance, otherwise false
025     * @throws IllegalArgumentException if {@code key} is null
026     */
027    boolean contains(byte[] key);
028
029    /**
030     * Skip over the largest possible uncontained region in an upward direction.
031     *
032     * <p>
033     * This method should return an inclusive lower bound on all keys greater than or equal to {@code key}
034     * that are contained by this instance. The bound does not have to be tight, but the tighter the better.
035     * </p>
036     *
037     * <p>
038     * A value of null may be returned to indicate that no key greater than or equal to {@code key} is contained by this instance.
039     * </p>
040     *
041     * <p>
042     * If {@code key} is contained by this instance, this method must return {@code key};
043     * if {@code key} is not contained by this instance, this method must return a key strictly higher than {@code key} or null.
044     * </p>
045     *
046     * @param key starting key
047     * @return a lower bound (inclusive) for contained keys greater than or equal to {@code key},
048     *  or null if no key greater than or equal to {@code key} is contained by this instance
049     * @throws IllegalArgumentException if {@code key} is null
050     */
051    byte[] seekHigher(byte[] key);
052
053    /**
054     * Skip over the largest possible uncontained region in a downward direction.
055     *
056     * <p>
057     * This method should return an exclusive upper bound on all keys strictly less than {@code key}
058     * that are contained by this instance. The bound does not have to be tight, but the tighter the better.
059     * </p>
060     *
061     * <p>
062     * A value of null may be returned to indicate that no key strictly less than {@code key} is contained by this instance.
063     * </p>
064     *
065     * <p>
066     * For the purposes of this method, an empty {@code byte[]} array represents an upper bound greater than all
067     * {@code byte[]} keys. This interpretation applies both to the {@code key} parameter and returned value. Note that
068     * this implies an empty array cannot be returned to indicate that no keys exist (instead, return null).
069     * </p>
070     *
071     * @param key starting key, or an empty array to indicate a maximal upper bound
072     * @return an upper bound (exclusive) for contained keys strictly less that {@code key},
073     *  null if no key strictly less than {@code key} is contained by this instance, or an empty {@code byte[]} array
074     *  to indicate an upper bound greater than all {@code byte[]} keys (which implies {@code key} was also empty)
075     * @throws IllegalArgumentException if {@code key} is null
076     */
077    byte[] seekLower(byte[] key);
078}
