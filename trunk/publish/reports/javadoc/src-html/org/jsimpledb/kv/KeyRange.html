008package org.jsimpledb.kv;
009
010import com.google.common.primitives.Bytes;
011
012import java.util.Arrays;
013import java.util.Comparator;
014
015import org.jsimpledb.util.ByteUtil;
016
017/**
018 * Represents a range of {@code byte[]} keys sorted in unsigned lexical order.
019 * Either bound may be specified as null to represent no minimum or no maximum (respectively).
020 *
021 * <p>
022 * Instances are immutable: the minimum and maximum {@code byte[]} arrays are copied during
023 * construction and when accessed by {@link #getMin} and {@link #getMax}.
024 * </p>
025 */
026public class KeyRange {
027
028    /**
029     * The {@link KeyRange} containing the full range (i.e., all keys).
030     */
031    public static final KeyRange FULL = new KeyRange(null, null);
032
033    /**
034     * Constant used by {@link #compare KeyRange.compare()} to distinguish null values.
035     */
036    public static final boolean MIN = false;
037
038    /**
039     * Constant used by {@link #compare KeyRange.compare()} to distinguish null values.
040     */
041    public static final boolean MAX = true;
042
043    /**
044     * Sorts instances by {@linkplain KeyRange#getMin min value}, then {@linkplain KeyRange#getMax max value}.
045     */
046    public static final Comparator<KeyRange> SORT_BY_MIN = new Comparator<KeyRange>() {
047        @Override
048        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
049            int diff = KeyRange.compare(keyRange1.min, KeyRange.MIN, keyRange2.min, KeyRange.MIN);
050            if (diff != 0)
051                return diff;
052            diff = KeyRange.compare(keyRange1.max, KeyRange.MAX, keyRange2.max, KeyRange.MAX);
053            if (diff != 0)
054                return diff;
055            return 0;
056        }
057    };
058
059    /**
060     * Sorts instances by {@linkplain KeyRange#getMax max value}, then {@linkplain KeyRange#getMin min value}.
061     */
062    public static final Comparator<KeyRange> SORT_BY_MAX = new Comparator<KeyRange>() {
063        @Override
064        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
065            int diff = KeyRange.compare(keyRange1.max, KeyRange.MAX, keyRange2.max, KeyRange.MAX);
066            if (diff != 0)
067                return diff;
068            diff = KeyRange.compare(keyRange1.min, KeyRange.MIN, keyRange2.min, KeyRange.MIN);
069            if (diff != 0)
070                return diff;
071            return 0;
072        }
073    };
074
075    /**
076     * Lower bound (inclusive), or null for no minimum. Subclasses must <b>not</b> modify the array (to preserve immutability).
077     */
078    protected final byte[] min;
079
080    /**
081     * Upper bound (exclusive), or null for no maximum. Subclasses must <b>not</b> modify the array (to preserve immutability).
082     */
083    protected final byte[] max;
084
085// Constructors
086
087    /**
088     * Constructor.
089     *
090     * @param min minimum key (inclusive), or null for no minimum
091     * @param max maximum key (exclusive), or null for no maximum
092     * @throws IllegalArgumentException if {@code min > max}
093     */
094    public KeyRange(byte[] min, byte[] max) {
095        if (KeyRange.compare(min, KeyRange.MIN, max, KeyRange.MAX) > 0)
096            throw new IllegalArgumentException("min = " + ByteUtil.toString(min) + " > max = " + ByteUtil.toString(max));
097        this.min = min == null ? null : min.clone();
098        this.max = max == null ? null : max.clone();
099    }
100
101    /**
102     * Construct key range containing a single key.
103     *
104     * @param key the key contained in the range
105     * @throws IllegalArgumentException if {@code key} is null
106     */
107    public KeyRange(byte[] key) {
108        if (key == null)
109            throw new IllegalArgumentException("null key");
110        this.min = key.clone();
111        this.max = ByteUtil.getNextKey(this.min);
112    }
113
114    /**
115     * Construct an instance containing all keys with the given prefix.
116     *
117     * @param prefix prefix of all keys in the range
118     * @throws IllegalArgumentException if {@code prefix} is null
119     */
120    public static KeyRange forPrefix(byte[] prefix) {
121        if (prefix == null)
122            throw new IllegalArgumentException("null prefix");
123        if (prefix.length == 0)
124            return new KeyRange(null, null);
125        /*final*/ byte[] maxKey;
126        try {
127            maxKey = ByteUtil.getKeyAfterPrefix(prefix);
128        } catch (IllegalArgumentException e) {
129            maxKey = null;
130        }
131        return new KeyRange(prefix, maxKey);
132    }
133
134// Instance Methods
135
136    /**
137     * Get range minimum (inclusive), or null if there is no lower bound.
138     */
139    public byte[] getMin() {
140        return this.min == null ? null : this.min.clone();
141    }
142
143    /**
144     * Get range maximum (exclusive), or null if there is no upper bound.
145     */
146    public byte[] getMax() {
147        return this.max == null ? null : this.max.clone();
148    }
149
150    /**
151     * Determine if this key range overlaps the specified key range.
152     *
153     * @param range other instance
154     * @throws IllegalArgumentException if {@code range} is null
155     */
156    public boolean overlaps(KeyRange range) {
157        if (range == null)
158            throw new IllegalArgumentException("null range");
159        return KeyRange.compare(this.min, MIN, range.max, MAX) < 0 && KeyRange.compare(range.min, MIN, this.max, MAX) < 0;
160    }
161
162    /**
163     * Determine if this key range fully contains the specified key range.
164     *
165     * @param range other instance
166     * @throws IllegalArgumentException if {@code range} is null
167     */
168    public boolean contains(KeyRange range) {
169        if (range == null)
170            throw new IllegalArgumentException("null range");
171        return KeyRange.compare(this.min, MIN, range.min, MIN) <= 0 && KeyRange.compare(this.max, MAX, range.max, MAX) >= 0;
172    }
173
174    /**
175     * Determine if this key range contains the specified key.
176     *
177     * @throws IllegalArgumentException if {@code key} is null
178     */
179    public boolean contains(byte[] key) {
180        return this.compareTo(key) == 0;
181    }
182
183    /**
184     * Determine whether this instance contains the full range covering all keys.
185     */
186    public boolean isFull() {
187        return this.min == null && this.max == null;
188    }
189
190    /**
191     * Determine whether this instance contains zero keys (implying {@link #getMin}{@code == }{@link #getMax}).
192     */
193    public boolean isEmpty() {
194        return this.min != null && this.max != null && ByteUtil.compare(this.min, this.max) == 0;
195    }
196
197    /**
198     * Create a new instance whose minimum and maximum keys are the same as this instance's
199     * but with the given byte sequence prepended.
200     *
201     * @throws IllegalArgumentException if {@code prefix} is null
202     */
203    public KeyRange prefixedBy(byte[] prefix) {
204        if (prefix == null)
205            throw new IllegalArgumentException("null prefix");
206        final byte[] prefixedMin = this.min != null ? Bytes.concat(prefix, this.min) : prefix;
207        final byte[] prefixedMax = this.max != null ? Bytes.concat(prefix, this.max) : ByteUtil.getNextKey(prefix);
208        return new KeyRange(prefixedMin, prefixedMax);
209    }
210
211    /**
212     * Determine if this range is left of, contains, or is right of the given key.
213     *
214     * @return -1 if this range is left of {@code key},
215     *  0 if this range contains {@code key}, or 1 if this range is right of {@code key},
216     * @throws IllegalArgumentException if {@code key} is null
217     */
218    public int compareTo(byte[] key) {
219        if (key == null)
220            throw new IllegalArgumentException("null key");
221        if (KeyRange.compare(this.min, MIN, key, MIN) > 0)
222            return 1;
223        if (KeyRange.compare(this.max, MAX, key, MAX) <= 0)
224            return -1;
225        return 0;
226    }
227
228    /**
229     * Compare two {@code byte[]} keys using unsigned lexical ordering, while also accepting
230     * null values that represent "negative infinity" and "positive infinity".
231     *
232     * @param key1 first key
233     * @param type1 how to interpret {@code key1} if value is null:
234     *  {@link #MIN} (for "negative infinity") or {@link #MAX} (for "positive infinity")
235     * @param key2 second key
236     * @param type2 how to interpret {@code key2} if value is null:
237     *  {@link #MIN} (for "negative infinity") or {@link #MAX} (for "positive infinity")
238     * @return -1 if {@code key1 < key2}, 1 if {@code key1 > key2}, or zero if {@code key1 = key2}
239     */
240    public static int compare(byte[] key1, boolean type1, byte[] key2, boolean type2) {
241        if (key1 == null && key2 == null)
242            return type1 == type2 ? 0 : type1 == MIN ? -1 : 1;
243        if (key1 == null)
244            return type1 == MIN ? -1 : 1;
245        if (key2 == null)
246            return type2 == MAX ? -1 : 1;
247        return ByteUtil.compare(key1, key2);
248    }
249
250// Object
251
252    @Override
253    public boolean equals(Object obj) {
254        if (obj == this)
255            return true;
256        if (obj == null || obj.getClass() != this.getClass())
257            return false;
258        final KeyRange that = (KeyRange)obj;
259        return (this.min == null ? that.min == null : Arrays.equals(this.min, that.min))
260          && (this.max == null ? that.max == null : Arrays.equals(this.max, that.max));
261    }
262
263    @Override
264    public int hashCode() {
265        return (this.min != null ? Arrays.hashCode(this.min) : 0)
266          ^ (this.max != null ? Arrays.hashCode(this.max) : 0);
267    }
268
269    @Override
270    public String toString() {
271        return this.getClass().getSimpleName() + "[" + ByteUtil.toString(this.min) + "," + ByteUtil.toString(this.max) + "]";
272    }
273}
