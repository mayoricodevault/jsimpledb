008package org.jsimpledb.kv;
009
010import java.util.Arrays;
011import java.util.Comparator;
012
013import org.jsimpledb.util.ByteUtil;
014
015/**
016 * Represents a range of {@code byte[]} keys sorted in unsigned lexical order.
017 * Either bound may be specified as null to represent no minimum or no maximum (respectively).
018 *
019 * <p>
020 * Instances are immutable: the minimum and maximum {@code byte[]} arrays are copied during
021 * construction and when accessed by {@link #getMin} and {@link #getMax}.
022 * </p>
023 */
024public class KeyRange {
025
026    /**
027     * Constant used by {@link #compare KeyRange.compare()} to distinguish null values.
028     */
029    public static final boolean MIN = false;
030
031    /**
032     * Constant used by {@link #compare KeyRange.compare()} to distinguish null values.
033     */
034    public static final boolean MAX = true;
035
036    /**
037     * Sorts instances by {@linkplain KeyRange#getMin min value}, then {@linkplain KeyRange#getMax max value}.
038     */
039    public static final Comparator<KeyRange> SORT_BY_MIN = new Comparator<KeyRange>() {
040        @Override
041        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
042            int diff = KeyRange.compare(keyRange1.min, KeyRange.MIN, keyRange2.min, KeyRange.MIN);
043            if (diff != 0)
044                return diff;
045            diff = KeyRange.compare(keyRange1.max, KeyRange.MAX, keyRange2.max, KeyRange.MAX);
046            if (diff != 0)
047                return diff;
048            return 0;
049        }
050    };
051
052    /**
053     * Sorts instances by {@linkplain KeyRange#getMax max value}, then {@linkplain KeyRange#getMin min value}.
054     */
055    public static final Comparator<KeyRange> SORT_BY_MAX = new Comparator<KeyRange>() {
056        @Override
057        public int compare(KeyRange keyRange1, KeyRange keyRange2) {
058            int diff = KeyRange.compare(keyRange1.max, KeyRange.MAX, keyRange2.max, KeyRange.MAX);
059            if (diff != 0)
060                return diff;
061            diff = KeyRange.compare(keyRange1.min, KeyRange.MIN, keyRange2.min, KeyRange.MIN);
062            if (diff != 0)
063                return diff;
064            return 0;
065        }
066    };
067
068    /**
069     * Lower bound (inclusive), or null for no minimum. Subclasses must <b>not</b> modify the array (to preserve immutability).
070     */
071    protected final byte[] min;
072
073    /**
074     * Upper bound (exclusive), or null for no maximum. Subclasses must <b>not</b> modify the array (to preserve immutability).
075     */
076    protected final byte[] max;
077
078    /**
079     * Constructor.
080     *
081     * @param min minimum key (inclusive), or null for no minimum
082     * @param max maximum key (exclusive), or null for no maximum
083     * @throws IllegalArgumentException if {@code min > max}
084     */
085    public KeyRange(byte[] min, byte[] max) {
086        if (KeyRange.compare(min, KeyRange.MIN, max, KeyRange.MAX) > 0)
087            throw new IllegalArgumentException("min = " + ByteUtil.toString(min) + " > max = " + ByteUtil.toString(max));
088        this.min = min == null ? null : min.clone();
089        this.max = max == null ? null : max.clone();
090    }
091
092    /**
093     * Construct key range containing a single key.
094     *
095     * @param key the key contained in the range
096     * @throws IllegalArgumentException if {@code key} is null
097     */
098    public KeyRange(byte[] key) {
099        if (key == null)
100            throw new IllegalArgumentException("null key");
101        this.min = key.clone();
102        this.max = ByteUtil.getNextKey(this.min);
103    }
104
105    /**
106     * Get range minimum (inclusive), or null if there is no lower bound.
107     */
108    public byte[] getMin() {
109        return this.min == null ? null : this.min.clone();
110    }
111
112    /**
113     * Get range maximum (exclusive), or null if there is no upper bound.
114     */
115    public byte[] getMax() {
116        return this.max == null ? null : this.max.clone();
117    }
118
119    /**
120     * Determine if this key range overlaps the specified key range.
121     *
122     * @param range other instance
123     * @throws IllegalArgumentException if {@code range} is null
124     */
125    public boolean overlaps(KeyRange range) {
126        if (range == null)
127            throw new IllegalArgumentException("null range");
128        return KeyRange.compare(this.min, MIN, range.max, MAX) < 0 && KeyRange.compare(range.min, MIN, this.max, MAX) < 0;
129    }
130
131    /**
132     * Determine if this key range fully contains the specified key range.
133     *
134     * @param range other instance
135     * @throws IllegalArgumentException if {@code range} is null
136     */
137    public boolean contains(KeyRange range) {
138        if (range == null)
139            throw new IllegalArgumentException("null range");
140        return KeyRange.compare(this.min, MIN, range.min, MIN) <= 0 && KeyRange.compare(this.max, MAX, range.max, MAX) >= 0;
141    }
142
143    /**
144     * Determine if this key range contains the specified key.
145     *
146     * @throws IllegalArgumentException if {@code key} is null
147     */
148    public boolean contains(byte[] key) {
149        return this.compareTo(key) == 0;
150    }
151
152    /**
153     * Determine if this range is left of, contains, or is right of the given key.
154     *
155     * @return -1 if this range is left of {@code key},
156     *  0 if this range contains {@code key}, or 1 if this range is right of {@code key},
157     * @throws IllegalArgumentException if {@code key} is null
158     */
159    public int compareTo(byte[] key) {
160        if (key == null)
161            throw new IllegalArgumentException("null key");
162        if (KeyRange.compare(this.min, MIN, key, MIN) > 0)
163            return 1;
164        if (KeyRange.compare(this.max, MAX, key, MAX) <= 0)
165            return -1;
166        return 0;
167    }
168
169    /**
170     * Compare two {@code byte[]} keys using unsigned lexical ordering, while also accepting
171     * null values that represent "negative infinity" and "positive infinity".
172     *
173     * @param range1 first key
174     * @param type1 how to interpret {@code range1} if value is null:
175     *  {@link #MIN} (for "negative infinity") or {@link #MAX} (for "positive infinity")
176     * @param range2 second key
177     * @param type2 how to interpret {@code range2} if value is null:
178     *  {@link #MIN} (for "negative infinity") or {@link #MAX} (for "positive infinity")
179     */
180    public static int compare(byte[] range1, boolean type1, byte[] range2, boolean type2) {
181        if (range1 == null && range2 == null)
182            return type1 == type2 ? 0 : type1 == MIN ? -1 : 1;
183        if (range1 == null)
184            return type1 == MIN ? -1 : 1;
185        if (range2 == null)
186            return type2 == MAX ? -1 : 1;
187        return ByteUtil.compare(range1, range2);
188    }
189
190    @Override
191    public boolean equals(Object obj) {
192        if (obj == this)
193            return true;
194        if (obj == null || obj.getClass() != this.getClass())
195            return false;
196        final KeyRange that = (KeyRange)obj;
197        return (this.min == null ? that.min == null : Arrays.equals(this.min, that.min))
198          && (this.max == null ? that.max == null : Arrays.equals(this.max, that.max));
199    }
200
201    @Override
202    public int hashCode() {
203        return (this.min != null ? Arrays.hashCode(this.min) : 0)
204          ^ (this.max != null ? Arrays.hashCode(this.max) : 0);
205    }
206
207    @Override
208    public String toString() {
209        return this.getClass().getSimpleName() + "[" + ByteUtil.toString(this.min) + "," + ByteUtil.toString(this.max) + "]";
210    }
211}
