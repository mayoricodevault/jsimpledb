008package org.jsimpledb.kv.sql;
009
010import java.sql.Connection;
011import java.sql.PreparedStatement;
012import java.sql.ResultSet;
013import java.sql.SQLException;
014
015import org.jsimpledb.kv.KVPair;
016import org.jsimpledb.kv.KVTransaction;
017import org.jsimpledb.kv.StaleTransactionException;
018
019/**
020 * {@link SQLKVDatabase} transaction.
021 */
022public class SQLKVTransaction implements KVTransaction {
023
024    protected final SQLKVDatabase database;
025    protected final Connection connection;
026
027    private PreparedStatement getStatement;
028    private PreparedStatement getAtLeastStatement;
029    private PreparedStatement getFirstStatement;
030    private PreparedStatement getAtMostStatement;
031    private PreparedStatement getLastStatement;
032    private PreparedStatement putStatement;
033    private PreparedStatement removeStatement;
034    private PreparedStatement removeRangeStatement;
035    private PreparedStatement removeAtLeastStatement;
036    private PreparedStatement removeAtMostStatement;
037    private PreparedStatement removeAllStatement;
038
039    private boolean stale;
040
041    /**
042     * Constructor.
043     */
044    public SQLKVTransaction(SQLKVDatabase database, Connection connection) throws SQLException {
045        if (database == null)
046            throw new IllegalArgumentException("null database");
047        if (connection == null)
048            throw new IllegalArgumentException("null connection");
049        this.database = database;
050        this.connection = connection;
051    }
052
053    @Override
054    public SQLKVDatabase getKVDatabase() {
055        return this.database;
056    }
057
058    /**
059     * Set transaction timeout. This method is not supported by {@link SQLKVTransaction}.
060     *
061     * @throws UnsupportedOperationException always
062     */
063    @Override
064    public void setTimeout(long timeout) {
065        throw new UnsupportedOperationException();
066    }
067
068    @Override
069    public synchronized byte[] get(byte[] key) {
070        if (this.stale)
071            throw new StaleTransactionException(this);
072        if (key == null)
073            throw new IllegalArgumentException("null key");
074        try {
075            if (this.getStatement == null)
076                this.getStatement = this.connection.prepareStatement(this.database.createGetStatement());
077            this.getStatement.setBytes(1, key);
078            final ResultSet rs = this.getStatement.executeQuery();
079            final byte[] value = rs.next() ? rs.getBytes(1) : null;
080            rs.close();
081            return value;
082        } catch (SQLException e) {
083            throw this.database.wrapException(this, e);
084        }
085    }
086
087    @Override
088    public synchronized KVPair getAtLeast(byte[] minKey) {
089        if (this.stale)
090            throw new StaleTransactionException(this);
091        try {
092            final ResultSet rs;
093            if (minKey != null) {
094                if (this.getAtLeastStatement == null)
095                    this.getAtLeastStatement = this.connection.prepareStatement(this.database.createGetAtLeastStatement());
096                this.getAtLeastStatement.setBytes(1, minKey);
097                rs = this.getAtLeastStatement.executeQuery();
098            } else {
099                if (this.getFirstStatement == null)
100                    this.getFirstStatement = this.connection.prepareStatement(this.database.createGetFirstStatement());
101                rs = this.getFirstStatement.executeQuery();
102            }
103            final KVPair pair = rs.next() ? new KVPair(rs.getBytes(1), rs.getBytes(2)) : null;
104            rs.close();
105            return pair;
106        } catch (SQLException e) {
107            throw this.database.wrapException(this, e);
108        }
109    }
110
111    @Override
112    public synchronized KVPair getAtMost(byte[] maxKey) {
113        if (this.stale)
114            throw new StaleTransactionException(this);
115        try {
116            final ResultSet rs;
117            if (maxKey != null) {
118                if (this.getAtMostStatement == null)
119                    this.getAtMostStatement = this.connection.prepareStatement(this.database.createGetAtMostStatement());
120                this.getAtMostStatement.setBytes(1, maxKey);
121                rs = this.getAtMostStatement.executeQuery();
122            } else {
123                if (this.getLastStatement == null)
124                    this.getLastStatement = this.connection.prepareStatement(this.database.createGetLastStatement());
125                rs = this.getLastStatement.executeQuery();
126            }
127            final KVPair pair = rs.next() ? new KVPair(rs.getBytes(1), rs.getBytes(2)) : null;
128            rs.close();
129            return pair;
130        } catch (SQLException e) {
131            throw this.database.wrapException(this, e);
132        }
133    }
134
135    @Override
136    public synchronized void put(byte[] key, byte[] value) {
137        if (key == null)
138            throw new IllegalArgumentException("null key");
139        if (value == null)
140            throw new IllegalArgumentException("null value");
141        if (this.stale)
142            throw new StaleTransactionException(this);
143        try {
144            if (this.putStatement == null)
145                this.putStatement = this.connection.prepareStatement(this.database.createPutStatement());
146            this.putStatement.setBytes(1, key);
147            this.putStatement.setBytes(2, value);
148            this.putStatement.executeUpdate();
149        } catch (SQLException e) {
150            throw this.database.wrapException(this, e);
151        }
152    }
153
154    @Override
155    public synchronized void remove(byte[] key) {
156        if (key == null)
157            throw new IllegalArgumentException("null key");
158        if (this.stale)
159            throw new StaleTransactionException(this);
160        try {
161            if (this.removeStatement == null)
162                this.removeStatement = this.connection.prepareStatement(this.database.createRemoveStatement());
163            this.removeStatement.setBytes(1, key);
164            this.removeStatement.executeUpdate();
165        } catch (SQLException e) {
166            throw this.database.wrapException(this, e);
167        }
168    }
169
170    @Override
171    public synchronized void removeRange(byte[] minKey, byte[] maxKey) {
172        if (this.stale)
173            throw new StaleTransactionException(this);
174        try {
175            if (minKey != null && maxKey != null) {
176                if (this.removeRangeStatement == null)
177                    this.removeRangeStatement = this.connection.prepareStatement(this.database.createRemoveRangeStatement());
178                this.removeRangeStatement.setBytes(1, minKey);
179                this.removeRangeStatement.setBytes(2, maxKey);
180                this.removeRangeStatement.executeUpdate();
181            } else if (minKey != null) {
182                if (this.removeAtLeastStatement == null)
183                    this.removeAtLeastStatement = this.connection.prepareStatement(this.database.createRemoveAtLeastStatement());
184                this.removeAtLeastStatement.setBytes(1, minKey);
185                this.removeAtLeastStatement.executeUpdate();
186            } else if (minKey == null) {
187                if (this.removeAtMostStatement == null)
188                    this.removeAtMostStatement = this.connection.prepareStatement(this.database.createRemoveAtMostStatement());
189                this.removeAtMostStatement.setBytes(1, maxKey);
190                this.removeAtMostStatement.executeUpdate();
191            } else {
192                if (this.removeAllStatement == null)
193                    this.removeAllStatement = this.connection.prepareStatement(this.database.createRemoveAllStatement());
194                this.removeAllStatement.executeUpdate();
195            }
196        } catch (SQLException e) {
197            throw this.database.wrapException(this, e);
198        }
199    }
200
201    @Override
202    public synchronized void commit() {
203        if (this.stale)
204            throw new StaleTransactionException(this);
205        this.stale = true;
206        try {
207            this.connection.commit();
208        } catch (SQLException e) {
209            throw this.database.wrapException(this, e);
210        }
211    }
212
213    @Override
214    public synchronized void rollback() {
215        if (this.stale)
216            throw new StaleTransactionException(this);
217        this.stale = true;
218        try {
219            this.connection.rollback();
220        } catch (SQLException e) {
221            throw this.database.wrapException(this, e);
222        }
223    }
224}
