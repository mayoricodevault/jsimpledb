008package org.jsimpledb.kv;
009
010import java.util.Iterator;
011
012/**
013 * General API into a key/value store where the keys are sorted lexicographically as unsigned bytes.
014 *
015 * <p>
016 * Implementations are not required to support accessing keys that start with {@code 0xff}.
017 * </p>
018 *
019 * <p><b>Lock-free Counters</b></p>
020 *
021 * <p>
022 * Implementations are encouraged to include support for encoding a 64 bit counter in a key/value pair such that the counter
023 * can be efficiently {@linkplain #adjustCounter adjusted} by concurrent transactions without conflict.
024 * In practice this means no locking is required to increment or decrement the counter by some amount, as long as
025 * it's not necessary to actually directly read or write the counter value in the same transaction.
026 * Whether counter adjustments are actually lock-free is implementation dependent, however, the counter methods
027 * {@link #encodeCounter encodeCounter()}, {@link #decodeCounter decodeCounter()}, and {@link #adjustCounter adjustCounter()}
028 * must function correctly as specified in all cases.
029 * </p>
030 *
031 * <p>
032 * How counters are encoded is specific to the implementation. Clients needing to read or write counter values directly
033 * should use {@link #decodeCounter decodeCounter()} and {@link #encodeCounter encodeCounter()}, respectively.
034 * Counters are removed using the normal methods (i.e., {@link #remove remove()} and {@link #removeRange removeRange()}).
035 * </p>
036 */
037public interface KVStore {
038
039    /**
040     * Get the value associated with the given key, if any.
041     *
042     * @param key key
043     * @return value associated with key, or null if not found
044     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
045     * @throws NullPointerException if {@code key} is null
046     */
047    byte[] get(byte[] key);
048
049    /**
050     * Get the key/value pair having the smallest key greater than or equal to the given minimum, if any.
051     *
052     * <p>
053     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
054     * then this method returns null.
055     * </p>
056     *
057     * @param minKey minimum key (inclusive), or null for no minimum (get the smallest key)
058     * @return smallest key/value pair with {@code key >= minKey}, or null if none exists
059     */
060    KVPair getAtLeast(byte[] minKey);
061
062    /**
063     * Get the key/value pair having the largest key strictly less than the given maximum, if any.
064     *
065     * <p>
066     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
067     * then this method behaves as if {@code maxKey} were null.
068     * </p>
069     *
070     * @param maxKey maximum key (exclusive), or null for no maximum (get the largest key)
071     * @return largest key/value pair with {@code key < maxKey}, or null if none exists
072     */
073    KVPair getAtMost(byte[] maxKey);
074
075    /**
076     * Iterate the key/value pairs in the specified range. The returned {@link Iterator}'s {@link Iterator#remove remove()}
077     * must be supported and should have the same effect as invoking {@link #remove remove()} on the corresponding key.
078     *
079     * <p>
080     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
081     * then this method returns an empty iteration.
082     * </p>
083     *
084     * <p>
085     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
086     * then this method behaves as if {@code maxKey} were null.
087     * </p>
088     *
089     * <p>
090     * The returned {@link Iterator} must not throw {@link java.util.ConcurrentModificationException};
091     * however, whether or not a "live" {@link Iterator} reflects any modifications made after its creation is
092     * implementation dependent. Implementations that do update automatically, even if the update occurs after some delay,
093     * must appear to preserve the order in which the modifications actually occurred.
094     * </p>
095     *
096     * @param minKey minimum key (inclusive), or null for no minimum (start at the smallest key)
097     * @param maxKey maximum key (exclusive), or null for no maximum (end at the largest key)
098     * @param reverse true to return key/value pairs in reverse order (i.e., keys descending)
099     * @return iteration of key/value pairs in the range {@code minKey} (inclusive) to {@code maxKey} (exclusive)
100     * @throws IllegalArgumentException if {@code minKey > maxKey}
101     */
102    Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse);
103
104    /**
105     * Set the value associated with the given key.
106     *
107     * @param key key
108     * @param value value
109     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
110     * @throws StaleTransactionException if this transaction is no longer usable
111     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
112     * @throws NullPointerException if {@code key} or {@code value} is null
113     */
114    void put(byte[] key, byte[] value);
115
116    /**
117     * Remove the key/value pair with the given key, if it exists.
118     *
119     * @param key key
120     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
121     * @throws NullPointerException if {@code key} is null
122     */
123    void remove(byte[] key);
124
125    /**
126     * Remove all key/value pairs whose keys are in a given range.
127     *
128     * <p>
129     * The {@code minKey} must be less than or equal to {@code maxKey}; if they equal (and not null)
130     * then nothing happens; if they are both null then all entries are deleted.
131     * </p>
132     *
133     * <p>
134     * If keys starting with {@code 0xff} are not supported by this instance, then:
135     * <ul>
136     *  <li>If {@code minKey} starts with {@code 0xff}, then no change occurs</li>
137     *  <li>If {@code maxKey} starts with {@code 0xff}, then this method behaves as if {@code maxKey} were null</li>
138     * </ul>
139     *
140     * @param minKey minimum key (inclusive), or null for no minimum
141     * @param maxKey maximum key (exclusive), or null for no maximum
142     * @throws IllegalArgumentException if {@code minKey > maxKey}
143     */
144    void removeRange(byte[] minKey, byte[] maxKey);
145
146    /**
147     * Encode a counter value into a {@code byte[]} value suitable for use with {@link #decodeCounter decodeCounter()}
148     * and/or {@link #adjustCounter adjustCounter()}.
149     *
150     * @param value desired counter value
151     * @return encoded counter value
152     */
153    byte[] encodeCounter(long value);
154
155    /**
156     * Decode a counter value previously encoded by {@link #encodeCounter encodeCounter()}.
157     *
158     * @param value encoded counter value
159     * @return decoded counter value
160     * @throws IllegalArgumentException if {@code value} is not a valid counter value
161     * @throws StaleTransactionException if this transaction is no longer usable
162     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
163     * @throws NullPointerException if {@code value} is null
164     */
165    long decodeCounter(byte[] value);
166
167    /**
168     * Adjust the counter at the given key by the given amount.
169     *
170     * <p>
171     * Ideally this operation should behave in a lock-free manner, so that concurrent transactions can invoke it without
172     * conflict. However, when lock-free behavior occurs (if at all) depends on the implementation.
173     * </p>
174     *
175     * <p>
176     * If there is no value associated with {@code key}, or {@code key}'s value is not a valid counter encoding as
177     * would be acceptable to {@link #decodeCounter decodeCounter()}, then how this operation affects {@code key}'s
178     * value is undefined.
179     * </p>
180     *
181     * @param key key
182     * @param amount amount to adjust counter value by
183     * @throws StaleTransactionException if this transaction is no longer usable
184     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
185     * @throws NullPointerException if {@code key} is null
186     */
187    void adjustCounter(byte[] key, long amount);
188}
