008package org.jsimpledb.kv;
009
010import java.util.Iterator;
011
012/**
013 * General API into a key/value store where the keys are sorted lexicographically as unsigned bytes.
014 *
015 * <p>
016 * Implementations are not required to support accessing keys that start with {@code 0xff}.
017 * </p>
018 *
019 * <p>
020 * Implementations are encouraged to also implement {@link CountingKVStore}.
021 * </p>
022 */
023public interface KVStore {
024
025    /**
026     * Get the value associated with the given key, if any.
027     *
028     * @param key key
029     * @return value associated with key, or null if not found
030     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
031     * @throws NullPointerException if {@code key} is null
032     */
033    byte[] get(byte[] key);
034
035    /**
036     * Get the key/value pair having the smallest key greater than or equal to the given minimum, if any.
037     *
038     * <p>
039     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
040     * then this method returns null.
041     * </p>
042     *
043     * @param minKey minimum key (inclusive), or null for no minimum (get the smallest key)
044     * @return smallest key/value pair with {@code key >= minKey}, or null if none exists
045     */
046    KVPair getAtLeast(byte[] minKey);
047
048    /**
049     * Get the key/value pair having the largest key strictly less than the given maximum, if any.
050     *
051     * <p>
052     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
053     * then this method behaves as if {@code maxKey} were null.
054     * </p>
055     *
056     * @param maxKey maximum key (exclusive), or null for no maximum (get the largest key)
057     * @return largest key/value pair with {@code key < maxKey}, or null if none exists
058     */
059    KVPair getAtMost(byte[] maxKey);
060
061    /**
062     * Iterate the key/value pairs in the specified range. The returned {@link Iterator}'s {@link Iterator#remove remove()}
063     * must be supported and should have the same effect as invoking {@link #remove remove()} on the corresponding key.
064     *
065     * <p>
066     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
067     * then this method returns an empty iteration.
068     * </p>
069     *
070     * <p>
071     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
072     * then this method behaves as if {@code maxKey} were null.
073     * </p>
074     *
075     * <p>
076     * The returned {@link Iterator} must not throw {@link java.util.ConcurrentModificationException};
077     * however, whether or not a "live" {@link Iterator} reflects any modifications made after its creation is
078     * implementation dependent. Implementations that do update automatically, even if the update occurs after some delay,
079     * must appear to preserve the order in which the modifications actually occurred.
080     * </p>
081     *
082     * @param minKey minimum key (inclusive), or null for no minimum (start at the smallest key)
083     * @param maxKey maximum key (exclusive), or null for no maximum (end at the largest key)
084     * @param reverse true to return key/value pairs in reverse order (i.e., keys descending)
085     * @return iteration of key/value pairs in the range {@code minKey} (inclusive) to {@code maxKey} (exclusive)
086     */
087    Iterator<KVPair> getRange(byte[] minKey, byte[] maxKey, boolean reverse);
088
089    /**
090     * Set the value associated with the given key.
091     *
092     * @param key key
093     * @param value value
094     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
095     * @throws StaleTransactionException if this transaction is no longer usable
096     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
097     * @throws NullPointerException if {@code key} or {@code value} is null
098     */
099    void put(byte[] key, byte[] value);
100
101    /**
102     * Remove the key/value pair with the given key, if it exists.
103     *
104     * @param key key
105     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
106     * @throws NullPointerException if {@code key} is null
107     */
108    void remove(byte[] key);
109
110    /**
111     * Remove all key/value pairs whose keys are in a given range.
112     *
113     * <p>
114     * The {@code minKey} must be less than or equal to {@code maxKey}; if they equal (and not null)
115     * then nothing happens; if they are both null then all entries are deleted.
116     * </p>
117     *
118     * <p>
119     * If keys starting with {@code 0xff} are not supported by this instance, then:
120     * <ul>
121     *  <li>If {@code minKey} starts with {@code 0xff}, then no change occurs</li>
122     *  <li>If {@code maxKey} starts with {@code 0xff}, then this method behaves as if {@code maxKey} were null</li>
123     * </ul>
124     * </p>
125     *
126     * @param minKey minimum key (inclusive), or null for no minimum
127     * @param maxKey maximum key (exclusive), or null for no maximum
128     * @throws IllegalArgumentException if {@code minKey > maxKey}
129     */
130    void removeRange(byte[] minKey, byte[] maxKey);
131}
