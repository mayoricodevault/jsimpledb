008package org.jsimpledb.kv;
009
010/**
011 * General API into a key/value store where the keys are sorted lexicographically as unsigned bytes.
012 *
013 * <p>
014 * For some implementations, accessing keys that start with {@code 0xff} is not supported and doing so
015 * will result in {@link IllegalArgumentException}.
016 * </p>
017 */
018public interface KVStore {
019
020    /**
021     * Get the value associated with the given key, if any.
022     *
023     * @param key key
024     * @return value associated with key, or null if not found
025     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
026     * @throws NullPointerException if {@code key} is null
027     */
028    byte[] get(byte[] key);
029
030    /**
031     * Get the key/value pair having the smallest key greater than or equal to the given minimum, if any.
032     *
033     * <p>
034     * If keys starting with {@code 0xff} are not supported by this instance, and {@code minKey} starts with {@code 0xff},
035     * then this method returns null.
036     * </p>
037     *
038     * @param minKey minimum key (inclusive), or null for no minimum (get the smallest key)
039     * @return smallest key/value pair with {@code key >= minKey}, or null if none exists
040     */
041    KVPair getAtLeast(byte[] minKey);
042
043    /**
044     * Get the key/value pair having the largest key strictly less than the given maximum, if any.
045     *
046     * <p>
047     * If keys starting with {@code 0xff} are not supported by this instance, and {@code maxKey} starts with {@code 0xff},
048     * then this method behaves as if {@code maxKey} were null.
049     * </p>
050     *
051     * @param maxKey maximum key (exclusive), or null for no maximum (get the largest key)
052     * @return largest key/value pair with {@code key < maxKey}, or null if none exists
053     */
054    KVPair getAtMost(byte[] maxKey);
055
056    /**
057     * Set the value associated with the given key.
058     *
059     * @param key key
060     * @param value value
061     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
062     * @throws StaleTransactionException if this transaction is no longer usable
063     * @throws RetryTransactionException if this transaction must be retried and is no longer usable
064     * @throws NullPointerException if {@code key} or {@code value} is null
065     */
066    void put(byte[] key, byte[] value);
067
068    /**
069     * Remove the key/value pair with the given key, if it exists.
070     *
071     * @param key key
072     * @throws IllegalArgumentException if {@code key} starts with {@code 0xff} and such keys are not supported
073     * @throws NullPointerException if {@code key} is null
074     */
075    void remove(byte[] key);
076
077    /**
078     * Remove all key/value pairs whose keys are in a given range.
079     *
080     * <p>
081     * The {@code minKey} must be less than or equal to {@code maxKey}; if they equal (and not null)
082     * then nothing happens; if they are both null then all entries are deleted.
083     * </p>
084     *
085     * <p>
086     * If keys starting with {@code 0xff} are not supported by this instance, then:
087     * <ul>
088     *  <li>If {@code minKey} starts with {@code 0xff}, then no change occurs</li>
089     *  <li>If {@code maxKey} starts with {@code 0xff}, then this method behaves as if {@code maxKey} were null</li>
090     * </ul>
091     * </p>
092     *
093     * @param minKey minimum key (inclusive), or null for no minimum
094     * @param maxKey maximum key (exclusive), or null for no maximum
095     * @throws IllegalArgumentException if {@code minKey > maxKey}
096     */
097    void removeRange(byte[] minKey, byte[] maxKey);
098}
