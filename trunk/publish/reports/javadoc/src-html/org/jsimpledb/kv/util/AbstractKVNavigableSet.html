008package org.jsimpledb.kv.util;
009
010import java.util.Iterator;
011import java.util.NavigableSet;
012
013import org.jsimpledb.kv.KVPair;
014import org.jsimpledb.kv.KVStore;
015import org.jsimpledb.kv.KeyFilter;
016import org.jsimpledb.kv.KeyFilterUtil;
017import org.jsimpledb.kv.KeyRange;
018import org.jsimpledb.util.AbstractNavigableSet;
019import org.jsimpledb.util.Bounds;
020import org.jsimpledb.util.ByteReader;
021import org.jsimpledb.util.ByteUtil;
022import org.jsimpledb.util.ByteWriter;
023
024/**
025 * {@link java.util.NavigableSet} support superclass for sets backed by elements encoded as {@code byte[]}
026 * keys in a {@link KVStore} and whose sort order is consistent with their {@code byte[]} key encoding.
027 * There must be an equivalence between elements and {@code byte[]} key encodings (i.e., there must be
028 * only one valid encoding per set element).
029 *
030 * <p><b>Subclass Methods</b></p>
031 *
032 * <p>
033 * Subclasses must implement the {@linkplain #encode(ByteWriter, Object) encode()} and {@linkplain #decode decode()}
034 * methods to convert elements to/from {@code byte[]} keys (associated values are ignored), and
035 * {@link #createSubSet(boolean, KeyRange, KeyFilter, Bounds) createSubSet()}
036 * to allow creating reversed and restricted range sub-sets.
037 * </p>
038 *
039 * <p>
040 * Subclasses must also implement {@link #comparator comparator()}, and the resulting sort order must be consistent with
041 * the sort order of the encoded {@code byte[]} keys (possibly {@link #reversed}).
042 * </p>
043 *
044 * <p>
045 * This class provides a read-only implementation; for a mutable implementation, subclasses should also implement
046 * {@link #add add()} (if appropriate), {@link #remove remove()}, and {@link #clear}; note, these methods must verify
047 * the key {@link #isVisible isVisible()} before making any changes.
048 * </p>
049 *
050 * <p>
051 * Additional subclass notes:
052 * <ul>
053 *  <li>{@link #iterator} returns an {@link Iterator} whose {@link Iterator#remove Iterator.remove()} method delegates
054 *      to this instance's {@link #remove remove()} method.
055 * </ul>
056 * </p>
057 *
058 * <p><b>Prefix Mode</b></p>
059 *
060 * <p>
061 * Instances support a "prefix mode" where the {@code byte[]} keys may have arbitrary trailing garbage, which is ignored,
062 * and so by definition no key can be a prefix of any other key. The length of the prefix is determined implicitly by the
063 * number of bytes produced by {@link #encode encode()} or consumed by {@link #decode decode()}.
064 * When not in prefix mode, {@link #decode decode()} <b>must</b> consume the entire key to preserve correct semantics.
065 * </p>
066 *
067 * <p><b>Key Restrictions</b></p>
068 *
069 * <p>
070 * Instances are configured with an (optional) {@link KeyRange}; when {@linkplain #bounds range restriction} is in
071 * effect, this key range corresponds to the bounds.
072 * </p>
073 *
074 * <p>
075 * Instances also support filtering visible values using a {@link KeyFilter}; see {@link #filterKeys filterKeys()}.
076 * To be {@linkplain #isVisible} in the set, keys must both be in the {@link KeyRange} and pass the {@link KeyFilter}.
077 * </p>
078 *
079 * <p><b>Concurrent Modifications</b></p>
080 *
081 * <p>
082 * This implementation never throws {@link java.util.ConcurrentModificationException}; instead, iterators always
083 * see the most up-to-date state of the associated {@link KVStore}.
084 * </p>
085 *
086 * @see AbstractKVNavigableMap
087 * @param <E> element type
088 */
089public abstract class AbstractKVNavigableSet<E> extends AbstractNavigableSet<E> {
090
091    /**
092     * The underlying {@link KVStore}.
093     */
094    protected final KVStore kv;
095
096    /**
097     * Whether we are in "prefix" mode.
098     */
099    protected final boolean prefixMode;
100
101    /**
102     * Whether the ordering of this instance is reversed.
103     */
104    protected final boolean reversed;
105
106    /**
107     * Key range, or null for the entire range.
108     */
109    protected final KeyRange keyRange;
110
111    /**
112     * Key filter, or null if all keys in the key range should be visible.
113     */
114    protected final KeyFilter keyFilter;
115
116// Constructors
117
118    /**
119     * Convenience constructor for when there are no range restrictions.
120     *
121     * @param kv underlying {@link KVStore}
122     * @param prefixMode whether to allow keys to have trailing garbage
123     * @throws IllegalArgumentException if {@code kv} is null
124     */
125    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode) {
126        this(kv, prefixMode, (KeyRange)null);
127    }
128
129    /**
130     * Convenience constructor for when the range of visible {@link KVStore} keys is all keys sharing a given {@code byte[]} prefix.
131     *
132     * @param kv underlying {@link KVStore}
133     * @param prefixMode whether to allow keys to have trailing garbage
134     * @param prefix prefix defining minimum and maximum keys
135     * @throws IllegalArgumentException if {@code kv} is null
136     * @throws IllegalArgumentException if {@code prefix} is null or empty
137     */
138    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, byte[] prefix) {
139        this(kv, prefixMode, KeyRange.forPrefix(prefix));
140    }
141
142    /**
143     * Primary constructor.
144     *
145     * @param kv underlying {@link KVStore}
146     * @param prefixMode whether to allow keys to have trailing garbage
147     * @param keyRange key range restriction, or null for none
148     * @throws IllegalArgumentException if {@code kv} is null
149     */
150    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, KeyRange keyRange) {
151        this(kv, prefixMode, false, keyRange, null, new Bounds<E>());
152    }
153
154    /**
155     * Internal constructor. Used for creating sub-sets and reversed views.
156     *
157     * <p>
158     * Note: if {@code bounds} are set, then {@code keyRange} must exclude all keys outside of those bounds.
159     * </p>
160     *
161     * @param kv underlying {@link KVStore}
162     * @param prefixMode whether to allow keys to have trailing garbage
163     * @param reversed whether ordering is reversed (implies {@code bounds} are also inverted, but <i>not</i> {@code keyRange})
164     * @param keyRange key range restriction, or null for none
165     * @param keyFilter key filter, or null for none
166     * @param bounds range restriction
167     * @throws IllegalArgumentException if {@code kv} or {@code bounds} is null
168     */
169    protected AbstractKVNavigableSet(KVStore kv, boolean prefixMode, boolean reversed,
170      KeyRange keyRange, KeyFilter keyFilter, Bounds<E> bounds) {
171        super(bounds);
172        if (kv == null)
173            throw new IllegalArgumentException("null kv");
174        this.kv = kv;
175        this.prefixMode = prefixMode;
176        this.reversed = reversed;
177        this.keyRange = keyRange;
178        this.keyFilter = keyFilter;
179    }
180
181    @Override
182    @SuppressWarnings("unchecked")
183    public boolean contains(Object obj) {
184
185        // Encode key and check visibility
186        final byte[] key = this.encodeVisible(obj, false);
187        if (key == null)
188            return false;
189
190        // Find key, or some longer key with the same prefix in prefix mode
191        final KVPair pair;
192        if (this.prefixMode) {
193            pair = this.kv.getAtLeast(key);
194            return pair != null && ByteUtil.isPrefixOf(key, pair.getKey());
195        } else
196            return this.kv.get(key) != null;
197    }
198
199    @Override
200    public Iterator<E> iterator() {
201        return new AbstractKVIterator<E>(this.kv, this.prefixMode, this.reversed, this.keyRange, this.keyFilter) {
202
203            @Override
204            protected E decodePair(KVPair pair, ByteReader keyReader) {
205                return AbstractKVNavigableSet.this.decode(keyReader);
206            }
207
208            @Override
209            protected void doRemove(E value, KVPair pair) {
210                AbstractKVNavigableSet.this.remove(value);
211            }
212        };
213    }
214
215    /**
216     * Create a view of this instance with additional filtering applied to the underlying {@code byte[]} encoded keys.
217     * Any set element for which the corresponding key does not pass {@code keyFilter} will be effectively hidden from view.
218     *
219     * <p>
220     * The restrictions of the given {@link KeyFilter} will be added to any current {@link KeyFilter} restrictions on this instance.
221     * The {@link #bounds} associated with this instance will not change.
222     * </p>
223     *
224     * @param keyFilter additional key filtering to apply
225     * @throws IllegalArgumentException if {@code keyFilter} is null
226     */
227    public NavigableSet<E> filterKeys(KeyFilter keyFilter) {
228        if (keyFilter == null)
229            throw new IllegalArgumentException("null keyFilter");
230        if (this.keyFilter != null)
231            keyFilter = KeyFilterUtil.intersection(keyFilter, this.keyFilter);
232        return this.createSubSet(this.reversed, this.keyRange, keyFilter, this.bounds);
233    }
234
235    @Override
236    protected boolean isWithinLowerBound(E elem) {
237        if (!super.isWithinLowerBound(elem))
238            return false;
239        if (this.keyRange == null)
240            return true;
241        final ByteWriter writer = new ByteWriter();
242        this.encode(writer, elem);
243        return KeyRange.compare(writer.getBytes(), KeyRange.MIN, this.keyRange.getMin(), KeyRange.MIN) >= 0;
244    }
245
246    @Override
247    protected boolean isWithinUpperBound(E elem) {
248        if (!super.isWithinUpperBound(elem))
249            return false;
250        if (this.keyRange == null)
251            return true;
252        final ByteWriter writer = new ByteWriter();
253        this.encode(writer, elem);
254        return KeyRange.compare(writer.getBytes(), KeyRange.MAX, this.keyRange.getMax(), KeyRange.MAX) < 0;
255    }
256
257    @Override
258    protected final NavigableSet<E> createSubSet(boolean reverse, Bounds<E> newBounds) {
259
260        // Determine the direction of the new sub-set
261        final boolean newReversed = this.reversed ^ reverse;
262
263        // Determine new min and max keys
264        final KeyRange newKeyRange = this.buildKeyRange(newReversed ? newBounds.reverse() : newBounds);
265
266        // Create subset
267        return this.createSubSet(newReversed, newKeyRange, this.keyFilter, newBounds);
268    }
269
270    /**
271     * Create a (possibly reversed) view of this instance with (possibly) tighter lower and/or upper bounds and
272     * the given {@link KeyFilter}, if any.
273     * The bounds are consistent with the reversed ordering (i.e., reversed if {@code reverse} is true)
274     * and have already been range-checked against this instance's bounds.
275     *
276     * @param newReversed whether the new set's ordering should be reversed (implies {@code newBounds} are also inverted,
277     *  but <i>not</i> {@code keyRange}); note: means "absolutely" reversed, not relative to this instance
278     * @param newKeyRange new key range, or null for none; will be consistent with {@code bounds}, if any
279     * @param newKeyFilter new key filter, or null for none
280     * @param newBounds new bounds
281     * @throws IllegalArgumentException if {@code newBounds} is null
282     */
283    protected abstract NavigableSet<E> createSubSet(boolean newReversed,
284      KeyRange newKeyRange, KeyFilter newKeyFilter, Bounds<E> newBounds);
285
286    /**
287     * Encode the given object into a {@code byte[]} key.
288     * Note that this method must throw {@link IllegalArgumentException}, not {@link ClassCastException}
289     * or {@code NullPointerException}, if {@code obj} does not have the correct type or is an illegal null value.
290     *
291     * @param writer output for encoded {@code byte[]} key corresponding to {@code obj}
292     * @param obj set element
293     * @throws IllegalArgumentException if {@code obj} is not of the required Java type supported by this set
294     * @throws IllegalArgumentException if {@code obj} is null and this set does not support null elements
295     */
296    protected abstract void encode(ByteWriter writer, Object obj);
297
298    /**
299     * Decode an element from a {@code byte[]} key.
300     *
301     * <p>
302     * If not in prefix mode, all of {@code reader} must be consumed; otherwise, the consumed portion
303     * is the prefix and any following keys with the same prefix are ignored.
304     * </p>
305     *
306     * @param reader input for encoded bytes
307     * @return decoded set element
308     */
309    protected abstract E decode(ByteReader reader);
310
311    /**
312     * Determine if the given {@code byte[]} key is visible in this set according to the configured
313     * {@link KeyRange} and/or {@link KeyFilter}, if any.
314     *
315     * @see #filterKeys filterKeys()
316     */
317    protected boolean isVisible(byte[] key) {
318        return (this.keyRange == null || this.keyRange.contains(key))
319          && (this.keyFilter == null || this.keyFilter.contains(key));
320    }
321
322    /**
323     * Encode the given object, if possible, and verify corresponding {@code byte[]} key is visible,
324     * otherwise return null or throw an exception.
325     * Delegates to {@link #encode(ByteWriter, Object)} to attempt the actual encoding.
326     *
327     * @param obj object to encode, possibly null
328     * @param fail whether, if {@code obj} can't be encoded, to throw an exception (true) or return null (false)
329     * @return encoed key for {@code obj}, or null if {@code fail} is false and {@code obj} has the wrong type or is out of bounds
330     * @throws IllegalArgumentException if {@code fail} is true and {@code obj} has the wrong type
331     * @throws IllegalArgumentException if {@code fail} is true and the resulting key is not {@linkplain #isVisible visible}
332     */
333    protected byte[] encodeVisible(Object obj, boolean fail) {
334        final ByteWriter writer = new ByteWriter();
335        try {
336            this.encode(writer, obj);
337        } catch (IllegalArgumentException e) {
338            if (!fail)
339                return null;
340            throw e;
341        }
342        final byte[] key = writer.getBytes();
343        if (this.keyRange != null && !this.keyRange.contains(key)) {
344            if (fail)
345                throw new IllegalArgumentException("value is out of bounds: " + obj);
346            return null;
347        }
348        if (this.keyFilter != null && !this.keyFilter.contains(key)) {
349            if (fail)
350                throw new IllegalArgumentException("value is filtered out: " + obj);
351            return null;
352        }
353        return key;
354    }
355
356    /**
357     * Derive a new {@link KeyRange} from (possibly) new element bounds. The given bounds must <i>not</i> ever be reversed.
358     */
359    private KeyRange buildKeyRange(Bounds<E> bounds) {
360        final byte[] minKey = this.keyRange != null ? this.keyRange.getMin() : null;
361        final byte[] maxKey = this.keyRange != null ? this.keyRange.getMax() : null;
362        byte[] newMinKey;
363        byte[] newMaxKey;
364        switch (bounds.getLowerBoundType()) {
365        case NONE:
366            newMinKey = minKey;
367            break;
368        default:
369            final ByteWriter writer = new ByteWriter();
370            this.encode(writer, bounds.getLowerBound());
371            newMinKey = writer.getBytes();
372            if (!bounds.getLowerBoundType().isInclusive())
373                newMinKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMinKey) : ByteUtil.getNextKey(newMinKey);
374            if (minKey != null)
375                newMinKey = ByteUtil.max(newMinKey, minKey);
376            if (maxKey != null)
377                newMinKey = ByteUtil.min(newMinKey, maxKey);
378            break;
379        }
380        switch (bounds.getUpperBoundType()) {
381        case NONE:
382            newMaxKey = maxKey;
383            break;
384        default:
385            final ByteWriter writer = new ByteWriter();
386            this.encode(writer, bounds.getUpperBound());
387            newMaxKey = writer.getBytes();
388            if (bounds.getUpperBoundType().isInclusive())
389                newMaxKey = this.prefixMode ? ByteUtil.getKeyAfterPrefix(newMaxKey) : ByteUtil.getNextKey(newMaxKey);
390            if (maxKey != null)
391                newMaxKey = ByteUtil.min(newMaxKey, maxKey);
392            if (minKey != null)
393                newMaxKey = ByteUtil.max(newMaxKey, minKey);
394            break;
395        }
396        return new KeyRange(newMinKey, newMaxKey);
397    }
