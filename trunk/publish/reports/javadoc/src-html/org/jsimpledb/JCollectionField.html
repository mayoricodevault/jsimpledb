008package org.jsimpledb;
009
010import java.lang.reflect.Method;
011import java.util.Collection;
012import java.util.Collections;
013import java.util.List;
014
015import org.jsimpledb.core.CollectionField;
016import org.jsimpledb.schema.CollectionSchemaField;
017
018/**
019 * Represents a collection field in a {@link JClass}.
020 */
021public abstract class JCollectionField extends JComplexField {
022
023    final JSimpleField elementField;
024
025    JCollectionField(JSimpleDB jdb, String name, int storageId, JSimpleField elementField, String description, Method getter) {
026        super(jdb, name, storageId, description, getter);
027        if (elementField == null)
028            throw new IllegalArgumentException("null elementField");
029        this.elementField = elementField;
030    }
031
032    /**
033     * Get the element sub-field.
034     */
035    public JSimpleField getElementField() {
036        return this.elementField;
037    }
038
039    @Override
040    public abstract Collection<?> getValue(JObject jobj);
041
042    @Override
043    public List<JSimpleField> getSubFields() {
044        return Collections.<JSimpleField>singletonList(this.elementField);
045    }
046
047    @Override
048    public JSimpleField getSubField(String name) {
049        if (CollectionField.ELEMENT_FIELD_NAME.equals(name))
050            return this.elementField;
051        throw new IllegalArgumentException("unknown sub-field `"
052          + name + "' (did you mean `" + CollectionField.ELEMENT_FIELD_NAME + "' instead?)");
053    }
054
055    @Override
056    String getSubFieldName(JSimpleField subField) {
057        if (subField == this.elementField)
058            return CollectionField.ELEMENT_FIELD_NAME;
059        throw new IllegalArgumentException("unknown sub-field");
060    }
061
062    @Override
063    abstract CollectionSchemaField toSchemaItem(JSimpleDB jdb);
064
065    void initialize(JSimpleDB jdb, CollectionSchemaField schemaField) {
066        super.initialize(jdb, schemaField);
067        schemaField.setElementField(this.elementField.toSchemaItem(jdb));
068    }
069}
