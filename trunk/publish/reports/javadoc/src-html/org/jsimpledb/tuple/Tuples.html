008package org.jsimpledb.tuple;
009
010import com.google.common.collect.Lists;
011
012import java.util.Arrays;
013
014/**
015 * Tuple utility methods.
016 */
017public final class Tuples {
018
019    private Tuples() {
020    }
021
022    /**
023     * Create a {@link Tuple} of the appropriate cardinality for the given number of objects.
024     *
025     * @param values tuple values
026     * @throws IllegalArgumentException if {@code values} is null or empty
027     * @throws UnsupportedOperationException if {@code values} has more elements than exist concrete {@link Tuple} classes
028     */
029    public static Tuple createTuple(Iterable<?> values) {
030        if (values == null)
031            throw new IllegalArgumentException("null values");
032        return Tuples.createTuple(Lists.newArrayList(values).toArray());
033    }
034
035    /**
036     * Create a {@link Tuple} of the appropriate cardinality for the given number of objects.
037     *
038     * @param values tuple values
039     * @throws IllegalArgumentException if {@code values} is null or empty
040     * @throws UnsupportedOperationException if {@code values} has more elements than exist concrete {@link Tuple} classes
041     */
042    public static Tuple createTuple(Object... values) {
043        if (values == null)
044            throw new IllegalArgumentException("null values");
045        final Class<? extends Tuple> tupleClass = Tuples.tupleClassForSize(values.length);
046        final Class<?>[] parameterTypes = new Class<?>[values.length];
047        Arrays.fill(parameterTypes, Object.class);
048        try {
049            return tupleClass.getConstructor(parameterTypes).newInstance(values);
050        } catch (Exception e) {
051            throw new UnsupportedOperationException("can't instantiate " + tupleClass, e);
052        }
053    }
054
055    /**
056     * Get the {@link Tuple} subclass having the specified cardinality.
057     *
058     * @param size number of objects in the tuple
059     * @throws IllegalArgumentException if {@code size} is less than one
060     * @throws UnsupportedOperationException if {@code size} is more than exist concrete {@link Tuple} classes
061     */
062    public static Class<? extends Tuple> tupleClassForSize(int size) {
063        if (size < 1)
064            throw new IllegalArgumentException("invalid size " + size);
065        final String name = Tuple.class.getName() + size;
066        try {
067            return Class.forName(name, false, Tuple.class.getClassLoader()).asSubclass(Tuple.class);
068        } catch (Exception e) {
069            throw new UnsupportedOperationException("can't find class `" + name + "' of size " + size, e);
070        }
071    }
072
073    /**
074     * Get the cardinality of the given {@link Tuple} subclass.
075     *
076     * @param tupleClass tuple class
077     * @return cardinality of {@code tupleClass}
078     * @throws IllegalArgumentException if {@code tupleClass} is not a recognized tuple class
079     */
080    public static int tupleSizeOf(Class<? extends Tuple> tupleClass) {
081        for (int size = 1; true; size++) {
082            try {
083                if (Tuples.tupleClassForSize(size).isAssignableFrom(tupleClass))
084                    return size;
085            } catch (UnsupportedOperationException e) {
086                throw new IllegalArgumentException("not a tuple (sub)class: " + tupleClass);
087            }
088        }
089    }
090}
