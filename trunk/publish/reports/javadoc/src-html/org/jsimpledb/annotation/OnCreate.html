008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotates JSimpleDB model class methods that are to be invoked whenever a database object is newly created.
018 *
019 * <p>
020 * Note that there is a subtle distinction between (a) the creation of database objects in the database, and
021 * (b) the instantiation of Java model objects that represent database objects (i.e., {@link org.jsimpledb.JObject}s).
022 * These two events do not necessarily always occur at the same time. Methods that are annotated with
023 * {@link OnCreate &#64;OnCreate} are invoked only for event (a). In particular, Java model objects are often
024 * instantiated to represent database objects that may not actually exist in the database.
025 * </p>
026 *
027 * <p>
028 * As a consequence, for any database fields that require default initialization, this initialization should be
029 * performed not in a Java constructor but rather in an {@link OnCreate &#64;OnCreate}-annotated method. Otherwise,
030 * {@link org.jsimpledb.core.DeletedObjectException}s (and {@link org.jsimpledb.core.ReadOnlyTransactionException}s
031 * within read-only transactions) may unexpectedly occur.
032 * </p>
033 *
034 * <p>
035 * For example, instead of this:
036 * <pre>
037 *  &#64;JSimpleClass(storageId = 10)
038 *  public abstract class Event {
039 *
040 *      protected Event() {
041 *          this.setUUID(UUID.randomUUID());
042 *      }
043 *
044 *      &#64;JSetField(storageId = 11)
045 *      public abstract UUID getUUID();
046 *      public abstract void setUUID(UUID uuid);
047 *
048 *      ...
049 * </pre>
050 * do this:
051 * <pre>
052 *  &#64;JSimpleClass(storageId = 10)
053 *  public abstract class Event {
054 *
055 *      &#64;OnCreate(snapshotTransactions = true)
056 *      private void initialize() {
057 *          this.setUUID(UUID.randomUUID());
058 *      }
059 *
060 *      &#64;JSetField(storageId = 11)
061 *      public abstract UUID getUUID();
062 *      public abstract void setUUID(UUID uuid);
063 *
064 *      ...
065 * </pre>
066 * </p>
067 *
068 * <p>
069 * Notifications are delivered in the same thread that created the object, immediately after the object is created.
070 * </p>
071 *
072 * <p>
073 * The annotated method must be an instance method (i.e., not static), return void, and take zero parameters.
074 * It may have any level of access, including {@code private}.
075 * </p>
076 */
077@Retention(RetentionPolicy.RUNTIME)
078@Target(ElementType.METHOD)
079@Documented
080public @interface OnCreate {
081
082    /**
083     * Determines whether this annotation should also be enabled for
084     * {@linkplain org.jsimpledb.SnapshotJTransaction snapshot transaction} objects.
085     * If unset, notifications will only be delivered to non-snapshot (i.e., normal) database instances.
086     *
087     * @see org.jsimpledb.SnapshotJTransaction
088     */
089    boolean snapshotTransactions() default false;
090}
