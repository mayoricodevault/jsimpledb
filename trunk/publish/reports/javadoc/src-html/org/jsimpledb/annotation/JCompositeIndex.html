008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.Retention;
012import java.lang.annotation.RetentionPolicy;
013import java.lang.annotation.Target;
014
015/**
016 * Java annotation defining a composite index on a {@link JSimpleClass}-annotated class.
017 *
018 * <p>
019 * A composite index is an index on two or more fields (to define a single-field index,
020 * just set {@link JField#indexed} to true). All fields indexed in a composite index
021 * must be (a) simple and (b) not a sub-field of a complex field.
022 * </p>
023 *
024 * @see JSimpleClass
025 */
026@Retention(RetentionPolicy.RUNTIME)
027@Target({})
028@Documented
029public @interface JCompositeIndex {
030
031    /**
032     * The name of this composite index. Must be globally unique.
033     *
034     * @return the index name
035     */
036    String name();
037
038    /**
039     * The storage ID for this composite index. Value should be positive; if zero, the configured
040     * {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value.
041     *
042     * @return the index storage ID
043     * @see org.jsimpledb.StorageIdGenerator#generateCompositeIndexStorageId StorageIdGenerator.generateCompositeIndexStorageId()
044     */
045    int storageId() default 0;
046
047    /**
048     * The names of the indexed fields, in the desired order. At least two fields must be specified.
049     *
050     * @return the names of the indexed fields
051     */
052    String[] fields();
053}
