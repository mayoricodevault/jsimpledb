008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotation for methods that are to be invoked whenever an object's schema version is changed.
018 *
019 * <p>
020 * The annotated method must be an instance method (i.e., not static), return void, and
021 * take one, two, or three of the following parameters in this order:
022 *  <ol>
023 *  <li>{@code int oldVersion} - previous schema version; should be present only if {@link #oldVersion} is zero</li>
024 *  <li>{@code int newVersion} - new schema version (always equal to
025 *      {@link org.jsimpledb.core.Transaction}.{@link org.jsimpledb.core.Transaction#getSchemaVersion getSchemaVersion()});
026 *      should be present only if {@link #newVersion} is zero</li>
027 *  <li>{@code Map<Integer, Object> oldFieldValues} - contains all field values from the previous version of the object,
028 *      indexed by storage ID. As a special case, for {@link Enum} fields the old values are represented as
029 *      {@link org.jsimpledb.core.EnumValue} objects rather than {@link Enum} values (this allows for arbitrary
030 *      changes to the {@link Enum} Java model type).</li>
031 *  </ol>
032 * </p>
033 *
034 * <p>
035 * If a class has multiple {@link OnVersionChange &#64;OnVersionChange}-annotated methods, methods with a non-zero
036 * {@link #oldVersion} or {@link #newVersion} (i.e., more specific constraint) will be invoked before methods having
037 * no constraint when possible.
038 * </p>
039 */
040@Retention(RetentionPolicy.RUNTIME)
041@Target(ElementType.METHOD)
042@Documented
043public @interface OnVersionChange {
044
045    /**
046     * Required old schema version.
047     *
048     * <p>
049     * If this property is set to a positive value, only version changes
050     * for which the previous schema version equals the specified version will result in notification,
051     * and the annotated method must have the corresponding parameter omitted. Otherwise notifications
052     * are delivered for any previous schema version and the {@code oldVersion} method parameter is required.
053     * </p>
054     *
055     * <p>
056     * Negative values are not allowed.
057     */
058    int oldVersion() default 0;
059
060    /**
061     * Required new schema version.
062     *
063     * <p>
064     * If this property is set to a positive value, only version changes
065     * for which the new schema version equals the specified version will result in notification,
066     * and the annotated method must have the corresponding parameter omitted. Otherwise notifications
067     * are delivered for any new schema version and the {@code newVersion} method parameter is required.
068     * </p>
069     *
070     * <p>
071     * Negative values are not allowed.
072     */
073    int newVersion() default 0;
074}
