008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotation for methods that are to be invoked whenever a targe field in some target object changes,
018 * where the target object containing the changed field is found at the end of a path of references
019 * starting from the object to be notified.
020 * See {@link org.jsimpledb.ReferencePath} for more information about reference paths.
021 *
022 * <p><b>Method Parameter Types</b></p>
023 *
024 * <p>
025 * In all cases the annotated method must return void and take a single parameter, which is a
026 * sub-type of {@link org.jsimpledb.change.FieldChange} suitable for the field being watched.
027 * The parameter type can be used to restrict which notifications are delivered. For example, an annotated method
028 * taking a {@link org.jsimpledb.change.SetFieldChange} will receive notifications about all changes to a set field,
029 * while a method taking a {@link org.jsimpledb.change.SetFieldAdd} will receive notification only when an element
030 * is added to the set.
031 * </p>
032 *
033 * <p><b>Instance vs. Static Methods</b></p>
034 *
035 * <p>
036 * If the method is an instance method, then {@link #startType} must be left unset; if the instance is a static
037 * method, then {@link #startType} may be explicitly set, or if left unset it defaults to the class containing
038 * the annotated method.
039 * </p>
040 *
041 * <p>
042 * For an instance method, the method will be invoked on <i>each object</i> for which the changed field is
043 * found at the end of the specified reference path <i>starting from that object</i>. Notifications are delivered from
044 * within the thread the made the change, after the change is made and just prior to returning to the original caller.
045 * Additional changes made within an {@link OnChange &#64;OnChange} handler that themselves result in new notifications
046 * are also handled prior to returning to the original caller. Therefore, infinite loops are possible if an
047 * {@link OnChange &#64;OnChange} handler method modifies the field it's monitoring (directly, or indirectly via
048 * other {@link OnChange &#64;OnChange} handler methods).
049 * </p>
050 *
051 * <p>
052 * If the annotated method is a static method, the method is invoked <i>once</i> if any instance exists for which the
053 * changed field is found at the end of the specified reference path (no matter how many such instances there are).
054 * Otherwise the behavior is the same.
055 * </p>
056 *
057 * <p><b>Other Notes</b></p>
058 *
059 * <p>
060 * No notifications are delivered for "changes" that do not actually change anything (e.g., setting a simple field to
061 * the value already contained in that field, or adding an element to a set which is already contained in the set).
062 * </p>
063 *
064 * <p>
065 * For any given field change, only one notification will be delivered per recipient object, even if the changed field is seen
066 * through the path in multiple ways (e.g., via reference path {@code "mylist.element.myfield"} where the changed object
067 * containing {@code myfield} appears multiple times in {@code mylist}).
068 * </p>
069 *
070 * <p>
071 * See {@link org.jsimpledb.core.Transaction#addSimpleFieldChangeListener Transaction.addSimpleFieldChangeListener()}
072 * for further information on other special corner cases.
073 * </p>
074 *
075 * @see org.jsimpledb.ReferencePath
076 * @see org.jsimpledb.change
077 */
078@Retention(RetentionPolicy.RUNTIME)
079@Target(ElementType.METHOD)
080@Documented
081public @interface OnChange {
082
083    /**
084     * Specifies the path to the target field to watch for changes.
085     * See {@link org.jsimpledb.ReferencePath} for information on the proper syntax.
086     *
087     * @see org.jsimpledb.ReferencePath
088     */
089    String value();
090
091    /**
092     * Specifies the starting type for the {@link org.jsimpledb.ReferencePath} specified by {@code #value}.
093     *
094     * <p>
095     * This property must be left unset for instance methods. For static methods, if this property is left unset,
096     * then then class containing the annotated method is assumed.
097     * </p>
098     *
099     * @see org.jsimpledb.ReferencePath
100     */
101    Class<?> startType() default void.class;
102}
