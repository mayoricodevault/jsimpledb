008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016import org.jsimpledb.core.DeleteAction;
017
018/**
019 * Java annotation for creating simple fields, including reference fields that refer to other Java model object types.
020 *
021 * <p>
022 * This annotation is used in two scenarios:
023 * <ul>
024 *  <li>To describe a <b>simple</b> database field by annotating the corresponding abstract Java bean property `getter' method</li>
025 *  <li>To describe the <b>sub-field(s)</b> of a <b>complex</b> database field (i.e., set, list, or map),
026 *      that is, the collection element or map key and value</li>
027 * </ul>
028 * </p>
029 *
030 * <p>
031 * Note that this annotation can be applied to superclass and interface methods to have the corresponding
032 * field defined in all sub-types.
033 * </p>
034 *
035 * <p>
036 * This annotation is not required when auto-generation of properties is enabled; see {@link JSimpleClass#autogenFields}.
037 * </p>
038 *
039 * <p><b>Non-Reference Fields</b></p>
040 *
041 * <p>
042 * If the field is not a reference field, the property type is inferred from the type of the annotated method or,
043 * in the case of complex sub-fields, the generic type of the collection class. The name of the property type
044 * must be registered in the {@link org.jsimpledb.core.FieldTypeRegistry} (perhaps via {@link JFieldType &#64;JFieldType}),
045 * and the corresponding {@link org.jsimpledb.core.FieldType} is then used to encode/decode field values.
046 * The type name may also be specified explicitly by {@link #name}.
047 * </p>
048 *
049 * <p>
050 * Simple fields may be {@link #indexed}; see {@link org.jsimpledb.index} for information on querying indexes.
051 * </p>
052 *
053 * <p><b>Reference Fields</b></p>
054 *
055 * <p>
056 * If the type of the field is (assignable to) a {@link JSimpleClass &#64;JsimpleClass}-annotated Java model object type,
057 * then the field is a reference field.
058 * </p>
059 *
060 * <p>
061 * Reference fields have configurable behavior when the containing object or the referred-to object is deleted;
062 * see {@link #onDelete} and {@link #cascadeDelete}.
063 * </p>
064 *
065 * <p>
066 * Reference fields are always indexed; the value of {@link #indexed} is ignored.
067 * </p>
068 *
069 * <p><b>Uniqueness Constraints</b></p>
070 *
071 * <p>
072 * Fields that are not complex sub-fields may be marked as {@link #unique} to impose a uniqueness constraint on the value.
073 * Fields with uniqueness constraints must be indexed. Uniqueness constraints are handled at the JSimpleDB layer and function as
074 * an implicit validation constraint. In other words, the constraint is verified when the validation queue is processed
075 * and is affected by the transaction's configured {@link org.jsimpledb.ValidationMode}.
076 * </p>
077 *
078 * <p>
079 * Optionally, specific field values may be marked as excluded from the uniqueness constraint via {@link #uniqueExclude}.
080 * If so, the specified values may appear in multiple objects without violating the constraint. Because null values
081 * are not allowed in annotations, {@link #uniqueExcludeNull} is provided to exclude the null value.
082 * </p>
083 *
084 * <p>
085 * In {@link org.jsimpledb.ValidationMode#AUTOMATIC}, any upgraded {@link org.jsimpledb.JObject}s are automatically
086 * added to the validation queue, so a uniqueness constraint added in a new schema version will be automatically verified
087 * when any object is upgraded.
088 * However, simpley adding, removing, or changing uniqueness constraints on an existing field doesn't force a schema
089 * version change. Therefore, existing database objects that were previously valid could be in violation of a uniqueness
090 * constraint that is added without one. To avoid this possibililty, change the schema version number as well.
091 * </p>
092 */
093@Retention(RetentionPolicy.RUNTIME)
094@Target(ElementType.METHOD)
095@Documented
096public @interface JField {
097
098    /**
099     * The name of this field.
100     *
101     * <p>
102     * If equal to the empty string (default value), the name is inferred from the name of the annotated Java bean getter method.
103     * </p>
104     *
105     * <p>
106     * For sub-fields of complex fields, this property must be left unset.
107     * </p>
108     */
109    String name() default "";
110
111    /**
112     * Optional override for the type of this field.
113     *
114     * <p>
115     * If set, this must equal the name of a type registered in the {@link org.jsimpledb.core.FieldTypeRegistry}
116     * associated with the {@link org.jsimpledb.core.Database} instance, and the annotated method's return type must match the
117     * {@link org.jsimpledb.core.FieldType}'s {@linkplain org.jsimpledb.core.FieldType#getTypeToken supported Java type}.
118     *
119     * <p>
120     * If equal to the empty string (default value), then the Java type is inferred from the return type of the getter method
121     * and the {@link org.jsimpledb.core.FieldType} is found via
122     * {@link org.jsimpledb.core.FieldTypeRegistry#getFieldType(com.google.common.reflect.TypeToken)
123     * FieldTypeRegistry.getFieldType()}.
124     * </p>
125     *
126     * <p>
127     * For reference fields (i.e., methods with return value equal to a {@link JSimpleClass &#64;JSimpleClass}-annotated class),
128     * this property must be left unset.
129     * </p>
130     *
131     * <p>
132     * For sub-fields of complex fields, this property can be used to force a primitive sub-field type instead of a
133     * primitive wrapper type. In that case, the complex field will disallow null values. For example:
134     * <pre>
135     *  &#64;JSetField(element = &#64;JField(<b>type = "float"</b>)) // nulls will be disallowed
136     *  public abstract List&lt;<b>Float</b>&gt; getScores();
137     * </pre>
138     * </p>
139     */
140    String type() default "";
141
142    /**
143     * Storage ID for this field. Value should be positive and unique within the contained class.
144     * If zero, the configured {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value
145     * unless {@link JSimpleClass#autogenFields} is false (in which case an error occurs).
146     *
147     * @see org.jsimpledb.StorageIdGenerator#generateFieldStorageId StorageIdGenerator.generateFieldStorageId()
148     * @see org.jsimpledb.StorageIdGenerator#generateSetElementStorageId StorageIdGenerator.generateSetElementStorageId()
149     * @see org.jsimpledb.StorageIdGenerator#generateListElementStorageId StorageIdGenerator.generateListElementStorageId()
150     * @see org.jsimpledb.StorageIdGenerator#generateMapKeyStorageId StorageIdGenerator.generateMapKeyStorageId()
151     * @see org.jsimpledb.StorageIdGenerator#generateMapValueStorageId StorageIdGenerator.generateMapValueStorageId()
152     */
153    int storageId() default 0;
154
155    /**
156     * Whether this field is indexed or not.
157     *
158     * <p>
159     * Setting this property to true creates a simple index on this field. To have this field participate in
160     * a composite index on multiple fields, use {@link JSimpleClass#compositeIndexes}.
161     * </p>
162     *
163     * <p>
164     * Note: reference fields are always indexed (for reference fields, this property is ignored).
165     * </p>
166     */
167    boolean indexed() default false;
168
169    /**
170     * For reference fields, configure the behavior when the referred-to object is
171     * {@linkplain org.jsimpledb.JObject#delete deleted}.
172     *
173     * <p>
174     * For non-reference fields this property must be equal to its default value.
175     * </p>
176     *
177     * @see #cascadeDelete
178     * @see org.jsimpledb.JObject#delete
179     */
180    DeleteAction onDelete() default DeleteAction.EXCEPTION;
181
182    /**
183     * For reference fields, configure cascading behavior when the containing object is
184     * {@linkplain org.jsimpledb.JObject#delete deleted}. If set to true, the referred-to object
185     * is automatically deleted as well.
186     *
187     * <p>
188     * For non-reference fields this property must be equal to its default value.
189     * </p>
190     *
191     * @see #onDelete
192     * @see org.jsimpledb.JObject#delete
193     */
194    boolean cascadeDelete() default false;
195
196    /**
197     * Require this field's value to be unique among all database objects.
198     *
199     * <p>
200     * This property must be false for sub-fields of complex fields.
201     * </p>
202     *
203     * @see #uniqueExclude
204     * @see #uniqueExcludeNull
205     */
206    boolean unique() default false;
207
208    /**
209     * Specify field value(s) which are excluded from the uniqueness constraint.
210     *
211     * <p>
212     * The specified values must be valid {@link String} encodings of the associated field. For example:
213     * <pre>
214     *  &#64;JField(indexed = true, unique = true, uniqueExclude = { "Infinity", "-Infinity" })
215     *  public abstract float getPriority();
216     * </pre>
217     * </p>
218     *
219     * <p>
220     * This property must be left empty when {@link #unique} is false.
221     * </p>
222     *
223     * @see #unique
224     * @see #uniqueExcludeNull
225     */
226    String[] uniqueExclude() default {};
227
228    /**
229     * Specify that the null value is excluded from the uniqueness constraint.
230     *
231     * <p>
232     * This property must be left false when {@link #unique} is false or the field has primitive type.
233     * </p>
234     *
235     * @see #unique
236     * @see #uniqueExclude
237     */
238    boolean uniqueExcludeNull() default false;
239}
