008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016import org.jsimpledb.core.DeleteAction;
017
018/**
019 * Java annotation for creating simple fields, including reference fields that refer to other Java model object types.
020 *
021 * <p>
022 * This annotation is used in two scenarios:
023 * <ul>
024 *  <li>To describe a simple database field by annotating the corresponding abstract Java bean property `getter' method</li>
025 *  <li>To describe the sub-field(s) of a complex database field (i.e., set, list, or map),
026 *      i.e., the collection element or map key and value</li>
027 * </ul>
028 * </p>
029 *
030 * <p>
031 * Note that this annotation can be applied to superclass and interface methods to have the corresponding
032 * field defined in all sub-types.
033 * </p>
034 *
035 * <p>
036 * This annotation is not required when auto-generation of properties is enabled; see {@link JSimpleClass#autogenFields}.
037 * </p>
038 *
039 * <p><b>Non-Reference Fields</b></p>
040 *
041 * <p>
042 * If the field is not a reference field, the property type is inferred from the type of the annotated method or,
043 * in the case of complex sub-fields, the generic type of the collection class. The name of the property type
044 * must be registered in the {@link org.jsimpledb.core.FieldTypeRegistry} (perhaps via {@link JFieldType &#64;JFieldType}),
045 * and the corresponding {@link org.jsimpledb.core.FieldType} is then used to encode/decode field values.
046 * The type name may also be specified explicitly by {@link #name}.
047 * </p>
048 *
049 * <p>
050 * Simple fields may be {@link #indexed}; see {@link org.jsimpledb.index} for information on querying indexes.
051 * </p>
052 *
053 * <p><b>Reference Fields</b></p>
054 *
055 * <p>
056 * If the type of the field is (assignable to) a {@link JSimpleClass &#64;JsimpleClass}-annotated Java model object type,
057 * then the field is a reference field.
058 * </p>
059 *
060 * <p>
061 * Reference fields have configurable behavior when the containing object or the referred-to object is deleted;
062 * see {@link #onDelete} and {@link #cascadeDelete}.
063 * </p>
064 *
065 * <p>
066 * Reference fields are always indexed; the value of {@link #indexed} is ignored.
067 * </p>
068 */
069@Retention(RetentionPolicy.RUNTIME)
070@Target(ElementType.METHOD)
071@Documented
072public @interface JField {
073
074    /**
075     * The name of this field.
076     *
077     * <p>
078     * If equal to the empty string (default value), the name is inferred from the name of the annotated Java bean getter method.
079     * </p>
080     *
081     * <p>
082     * For sub-fields of complex fields, this property must be left unset.
083     * </p>
084     */
085    String name() default "";
086
087    /**
088     * The type of this field.
089     *
090     * <p>
091     * If set, this must equal the name of a type registered in the {@link org.jsimpledb.core.FieldTypeRegistry}
092     * associated with the {@link org.jsimpledb.core.Database} instance, and the annotated method's return type must match the
093     * {@link org.jsimpledb.core.FieldType}'s {@linkplain org.jsimpledb.core.FieldType#getTypeToken supported Java type}.
094     *
095     * <p>
096     * If equal to the empty string (default value), then the Java type is inferred from the return type of the getter method
097     * and the {@link org.jsimpledb.core.FieldType} is found via
098     * {@link org.jsimpledb.core.FieldTypeRegistry#getFieldType(com.google.common.reflect.TypeToken)
099     * FieldTypeRegistry.getFieldType()}.
100     * </p>
101     *
102     * <p>
103     * For reference fields (i.e., methods with return value equal to a {@link JSimpleClass &#64;JSimpleClass}-annotated class),
104     * this property must be left unset.
105     * </p>
106     *
107     * <p>
108     * For sub-fields of complex fields, this property can be used to force a primitive sub-field type instead of a
109     * primitive wrapper type. In that case, the complex field will disallow nulls.
110     * </p>
111     *
112     * <p>
113     * For example:
114     * <pre>
115     * &#64;JSimpleClass(storageId = 10)
116     * public class Team {
117     *
118     *     &#64;JSetField(storageId = 11,
119     *       element = &#64;JField(storageId = 12, <b>type = "float"</b>)) // nulls will be disallowed
120     *     public abstract List&lt;<b>Float</b>&gt; getScores();
121     * }
122     * </pre>
123     * </p>
124     */
125    String type() default "";
126
127    /**
128     * Storage ID for this field. Value should be positive and unique within the contained class.
129     * If zero, the configured {@link org.jsimpledb.StorageIdGenerator} will be consulted to auto-generate a value
130     * unless {@link JSimpleClass#autogenFields} is false (in which case an error occurs).
131     *
132     * @see org.jsimpledb.StorageIdGenerator#generateFieldStorageId StorageIdGenerator.generateFieldStorageId()
133     * @see org.jsimpledb.StorageIdGenerator#generateSetElementStorageId StorageIdGenerator.generateSetElementStorageId()
134     * @see org.jsimpledb.StorageIdGenerator#generateListElementStorageId StorageIdGenerator.generateListElementStorageId()
135     * @see org.jsimpledb.StorageIdGenerator#generateMapKeyStorageId StorageIdGenerator.generateMapKeyStorageId()
136     * @see org.jsimpledb.StorageIdGenerator#generateMapValueStorageId StorageIdGenerator.generateMapValueStorageId()
137     */
138    int storageId() default 0;
139
140    /**
141     * Whether this field is indexed or not.
142     *
143     * <p>
144     * Setting this property to true creates a simple index on this field. To have this field participate in
145     * a composite index on multiple fields, use {@link JSimpleClass#compositeIndexes}.
146     * </p>
147     *
148     * <p>
149     * Note: reference fields are always indexed (for reference fields, this property is ignored).
150     * </p>
151     */
152    boolean indexed() default false;
153
154    /**
155     * For reference fields, configure the behavior when the referred-to object is
156     * {@linkplain org.jsimpledb.JObject#delete deleted}.
157     *
158     * <p>
159     * This field is ignored for non-reference fields.
160     * </p>
161     *
162     * @see #cascadeDelete
163     * @see org.jsimpledb.JObject#delete
164     */
165    DeleteAction onDelete() default DeleteAction.EXCEPTION;
166
167    /**
168     * For reference fields, configure cascading behavior when the containing object is
169     * {@linkplain org.jsimpledb.JObject#delete deleted}. If set to true, the referred-to object
170     * is automatically deleted as well.
171     *
172     * <p>
173     * This field is ignored for non-reference fields.
174     * </p>
175     *
176     * @see #onDelete
177     * @see org.jsimpledb.JObject#delete
178     */
179    boolean cascadeDelete() default false;
180}
