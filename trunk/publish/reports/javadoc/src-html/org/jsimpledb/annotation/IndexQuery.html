008package org.jsimpledb.annotation;
009
010import java.lang.annotation.Documented;
011import java.lang.annotation.ElementType;
012import java.lang.annotation.Retention;
013import java.lang.annotation.RetentionPolicy;
014import java.lang.annotation.Target;
015
016/**
017 * Annotates JSimpleDB model class methods that query an indexed field.
018 *
019 * <p>
020 * The annotated method must take zero parameters and return
021 * {@link java.util.NavigableMap NavigableMap}{@code <V, }{@link java.util.NavigableSet NavigableSet}{@code <T>>}, where
022 * {@code V} is the type of the indexed field's values, and {@code T} is the type of the Java model object
023 * containing the field.
024 * </p>
025 *
026 * <p>
027 * For example:
028 * <pre>
029 * &#64;JSimpleClass(storageId = 10)
030 * public class Person {
031 *
032 *     &#64;JSimpleSetField(storageId = 11, <b>indexed = true</b>)
033 *     public abstract Set&lt;String&gt; getNicknames();
034 *
035 *     <b>&#64;IndexQuery("nicknames")</b>
036 *     private abstract NavigableMap&lt;String, NavigableSet&lt;Person&gt;&gt; queryNicknames();
037 *
038 *     // Get all people with the given nickname
039 *     public Set&lt;Person&gt; getPeopleWithNickame(String nickname) {
040 *         final Set&lt;Person&gt; people = this.queryNicknames().get(nickname);
041 *         return people != null ? people : Collections.&lt;Person&gt;emptySet();
042 *     }
043 *
044 *     // Get all known nicknames, sorted by nickname
045 *     public NavigableSet&lt;String&gt; getAllNickames() {
046 *         return this.queryNicknames().navigableKeySet();
047 *     }
048 *
049 *     // Do an "inner join" on two nicknames
050 *     public NavigableSet&lt;Person&gt; getPeopleWithBothNicknames(String nickname1, String nickname2) {
051 *         return NaviableSets.intersection(this.getPeopleWithNickame(nickname1), this.getPeopleWithNickame(nickname2));
052 *     }
053 * }
054 * </pre>
055 * </p>
056 *
057 * <p>
058 * The returned {@link java.util.NavigableMap NavigableMap} contains an entry for every value that is actually present
059 * in the field in some object, with the corresponding value being the set of those objects. It represents a "live" view:
060 * if an indexed field is modified, any previously returned {@link java.util.NavigableMap NavigableMap} updates itself
061 * immediately.
062 * </p>
063 *
064 * <p>
065 * For list and map fields, you may also query the index for additional information beyond just which objects contain
066 * the queried value in the indexed field: in the case of a list, you can also get the
067 * list index, and in the case of a map, you can also get the corresponding key or value. For these kinds of query, specify the
068 * type {@code T} to be either {@link org.jsimpledb.ListIndexEntry},
069 * {@link org.jsimpledb.MapKeyIndexEntry}, or {@link org.jsimpledb.MapValueIndexEntry} as appropriate. For example:
070 * <pre>
071 * public class Company {
072 *
073 *     &#64;JSimpleMapField(storageId = 11,
074 *       key = &#64;JSimpleField(storageId = 12, <b>indexed = true</b>),
075 *       value = &#64;JSimpleField(storageId = 13, type = "float")) // primitive type to disallow nulls
076 *     public abstract Map&lt;Vendor, Float&gt; getAccountsPayable();
077 *
078 *     // Query to get mapping by Vendor of which Company's owe that vendor and how much they owe
079 *     <b>&#64;IndexQuery("accountsPayable.key")</b>
080 *     private abstract NavigableMap&lt;Vendor, NavigableSet&lt;MapKeyIndexEntry&lt;Company, Float&gt;&gt;&gt; queryVendorDebts();
081 * }
082 * </pre>
083 * </p>
084 *
085 * <p>
086 * The annotation {@link #value} is a {@link org.jsimpledb.ReferencePath} specifying the indexed
087 * field to query. The path is assumed to start at the type containing the annotated method; however, a different
088 * starting type may be specified via {@link #startType}.
089 * </p>
090 *
091 * <p>
092 * The annotated method must be an instance method. Note, static methods would be more appropriate (the index query is
093 * not specific to any instance), but static methods cannot be overridden by generated subclasses. For this reason
094 * it is sometimes useful to have a singleton Java model object type that only provides methods for querying indexes.
095 * <p>
096 */
097@Retention(RetentionPolicy.RUNTIME)
098@Target(ElementType.METHOD)
099@Documented
100public @interface IndexQuery {
101
102    /**
103     * Specifies the target field to watch for changes.
104     *
105     * <p>
106     * The value must be a {@linkplain org.jsimpledb.ReferencePath reference path} with
107     * zero references, e.g., {@code "MyClass.myfield"} or {@code "MyClass.mymap.key"}. The path cannot end on a
108     * complex field itself; it must end on some specific sub-field. So for example {@code "MyClass.mymap"} would be invalid.
109     * <p>
110     *
111     * <p>
112     * See {@link org.jsimpledb.ReferencePath} for information on the proper syntax for reference paths.
113     * </p>
114     *
115     * @see org.jsimpledb.ReferencePath
116     */
117    String value();
118
119    /**
120     * Specifies the starting type for the {@link org.jsimpledb.ReferencePath} specified by {@code #value}.
121     *
122     * <p>
123     * If this property is left unset, then then class containing the annotated method is assumed.
124     * </p>
125     *
126     * @see org.jsimpledb.ReferencePath
127     */
128    Class<?> startType() default void.class;
129}
