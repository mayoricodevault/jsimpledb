008package org.jsimpledb;
009
010import com.google.common.reflect.TypeToken;
011
012import java.lang.reflect.Method;
013
014import org.dellroad.stuff.java.EnumUtil;
015import org.jsimpledb.schema.EnumSchemaField;
016import org.jsimpledb.schema.SimpleSchemaField;
017
018/**
019 * Represents an enum field in a {@link JClass}.
020 */
021public class JEnumField extends JSimpleField {
022
023    JEnumField(String name, int storageId, Class<? extends Enum<?>> enumType,
024      boolean indexed, String description, Method getter, Method setter) {
025        super(name, storageId, TypeToken.of(enumType.asSubclass(Enum.class)),
026          enumType.getName(), indexed, description, getter, setter);
027    }
028
029    @Override
030    public <R> R visit(JFieldSwitch<R> target) {
031        return target.caseJEnumField(this);
032    }
033
034    @Override
035    @SuppressWarnings("unchecked")
036    public TypeToken<? extends Enum<?>> getType() {
037        return (TypeToken<? extends Enum<?>>)this.typeToken;
038    }
039
040    @Override
041    EnumSchemaField toSchemaItem(JSimpleDB jdb) {
042        final EnumSchemaField schemaField = new EnumSchemaField();
043        this.initialize(jdb, schemaField);
044        return schemaField;
045    }
046
047    @SuppressWarnings("unchecked")
048    void initialize(JSimpleDB jdb, SimpleSchemaField schemaField0) {
049        super.initialize(jdb, schemaField0);
050        final EnumSchemaField schemaField = (EnumSchemaField)schemaField0;
051        schemaField.getIdentifiers().clear();
052        for (Enum<?> value : EnumUtil.getValues((Class<Enum<?>>)this.getType().getRawType()))
053            schemaField.getIdentifiers().add(value.name());
054    }
055
056    @Override
057    JEnumFieldInfo toJFieldInfo(int parentStorageId) {
058        return new JEnumFieldInfo(this, parentStorageId);
059    }
060}
