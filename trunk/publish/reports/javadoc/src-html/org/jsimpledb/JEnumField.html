008package org.jsimpledb;
009
010import com.google.common.base.Converter;
011import com.google.common.reflect.TypeToken;
012
013import java.lang.reflect.Method;
014
015import org.dellroad.stuff.java.EnumUtil;
016import org.jsimpledb.core.EnumFieldType;
017import org.jsimpledb.core.EnumValue;
018import org.jsimpledb.schema.EnumSchemaField;
019import org.jsimpledb.schema.SimpleSchemaField;
020
021/**
022 * Represents an {@link Enum} field in a {@link JClass}.
023 */
024public class JEnumField extends JSimpleField {
025
026    final EnumConverter<?> converter;
027
028    @SuppressWarnings({ "unchecked", "rawtypes" })
029    JEnumField(JSimpleDB jdb, String name, int storageId, Class<? extends Enum<?>> enumType,
030      org.jsimpledb.annotation.JField annotation, String description, Method getter, Method setter) {
031        super(jdb, name, storageId, TypeToken.of(enumType.asSubclass(Enum.class)),
032          new EnumFieldType((Class)enumType), annotation.indexed(), annotation, description, getter, setter);
033        this.converter = EnumConverter.createEnumConverter(enumType);
034    }
035
036    @Override
037    public <R> R visit(JFieldSwitch<R> target) {
038        return target.caseJEnumField(this);
039    }
040
041    @Override
042    @SuppressWarnings("unchecked")
043    public TypeToken<? extends Enum<?>> getTypeToken() {
044        return (TypeToken<? extends Enum<?>>)this.typeToken;
045    }
046
047    @Override
048    public Converter<EnumValue, ? extends Enum<?>> getConverter(JTransaction jtx) {
049        return this.converter.reverse();
050    }
051
052    @Override
053    EnumSchemaField toSchemaItem(JSimpleDB jdb) {
054        final EnumSchemaField schemaField = new EnumSchemaField();
055        this.initialize(jdb, schemaField);
056        return schemaField;
057    }
058
059    @SuppressWarnings("unchecked")
060    void initialize(JSimpleDB jdb, SimpleSchemaField schemaField0) {
061        super.initialize(jdb, schemaField0);
062        final EnumSchemaField schemaField = (EnumSchemaField)schemaField0;
063        schemaField.getIdentifiers().clear();
064        for (Enum<?> value : EnumUtil.getValues((Class<Enum<?>>)this.getTypeToken().getRawType()))
065            schemaField.getIdentifiers().add(value.name());
066    }
067
068    @Override
069    JEnumFieldInfo toJFieldInfo(int parentStorageId) {
070        return new JEnumFieldInfo(this, parentStorageId);
071    }
072}
